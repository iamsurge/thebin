using System;

namespace ConsoleApp6
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int countRub = 10000, countUsd = 0, countEur = 0;
            int rubToUsd = 36;
            int rubToEur = 40;
            int countCurrency;
            bool enoughMoney;
            bool pressExit = false;

            Console.WriteLine($"Курс доллара: {rubToUsd} | Курс евро: {rubToEur}");
            Console.WriteLine("---------------------------------------------------");
            while (!pressExit)
            {
                Console.WriteLine($"Ваш баланс - Рубли: {countRub} | Доллары: {countUsd} | Евро: {countEur}");
                Console.WriteLine("---------------------------------------------------");
                Console.Write("Направление конвертации: (1) в доллары (2) в евро (3) выход из обменника: ");
                int conversionDirection = Convert.ToInt32(Console.ReadLine());

                switch (conversionDirection)
                {
                    case 1:
                        Console.Write("Запрашиваемое кол-во иностанной валюты: ");
                        countCurrency = Convert.ToInt32(Console.ReadLine());

                        Console.WriteLine("Проверяем достаточно ли на вашем рублевом счете денег для обмена на ДОЛЛАРЫ.");
                        enoughMoney = countRub >= countCurrency * rubToUsd;

                        if (enoughMoney)
                        {
                            Console.ForegroundColor = ConsoleColor.Green;
                            Console.WriteLine($"Проверка пройдена...\nВыдано {countCurrency} долларов.\n");
                            Console.ForegroundColor = ConsoleColor.White;
                            countRub -= countCurrency * rubToUsd;
                            countUsd += countCurrency;
                        }
                        else
                        {
                            Console.ForegroundColor = ConsoleColor.Red;
                            Console.WriteLine("Недостаточно денег для конвертации...\n");
                            Console.ForegroundColor = ConsoleColor.White;
                        }
                        break;
                    case 2:
                        Console.Write("Запрашиваемое кол-во иностанной валюты: ");
                        countCurrency = Convert.ToInt32(Console.ReadLine());

                        Console.WriteLine("Проверяем достаточно ли на вашем рублевом счете денег для обмена на ЕВРО.");
                        enoughMoney = countRub >= countCurrency * rubToEur;

                        if (enoughMoney)
                        {
                            Console.ForegroundColor = ConsoleColor.Green;
                            Console.WriteLine($"Проверка пройдена...\nВыдано {countCurrency} евро.");
                            Console.ForegroundColor = ConsoleColor.White;
                            countRub -= countCurrency * rubToEur;
                            countEur += countCurrency;
                        }
                        else
                        {
                            Console.WriteLine("Недостаточно денег для конвертации...");
                        }
                        break;
                    case 3:
                        pressExit = true;
                        break;
                }
            }
        }
    }
}
