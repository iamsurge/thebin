//deklarasi variable
int wait = 0;
const int trigger=8; 
const int echo=7; 
float jarak;

void setup(){
  //set pinMode MotorDC
  pinMode(LED_BUILTIN, OUTPUT);
  pinMode(10, OUTPUT);
  pinMode(11, OUTPUT);
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  
  //set pinMode Sensor Ultrasonik
  Serial.begin(9600);
  pinMode(trigger,OUTPUT);
  pinMode(echo,INPUT);
}

void loop(){
  digitalWrite(trigger,LOW);
  delayMicroseconds(5);        
  digitalWrite(trigger,HIGH);  
  delayMicroseconds(10);      
  digitalWrite(trigger,LOW); 
  jarak = pulseIn(echo,HIGH);      
  jarak = jarak/58;     		// konversi jarak ke cm
  Serial.print ("Jarak = ");
  Serial.print (jarak);         
  Serial.print (" cm");         
  Serial.write (10);            
  delay (200);  
  
  if(jarak > 100){
    //Bergerak Maju
    digitalWrite(LED_BUILTIN, LOW);
    digitalWrite(10, LOW);
    digitalWrite(11, LOW);
    digitalWrite(3, HIGH);
    digitalWrite(4, HIGH);
  }else{
    //Berhenti
    digitalWrite(LED_BUILTIN, HIGH);
    digitalWrite(10, LOW);
    digitalWrite(11, LOW);
    digitalWrite(3, LOW);
    digitalWrite(4, LOW);
  }
}
