def square(n, s, l, c):
    n -= 1
    l -= n
    for i in range(n):
        l += 1
        c += 1
        s[l][c] = 1
    for i in range(n):
        l += 1
        c -= 1
        s[l][c] = 1
    for i in range(n):
        l -= 1
        c -= 1
        s[l][c] = 1
    for i in range(n):
        l -= 1
        c += 1
        s[l][c] = 1
    return s


a = int(input())
s = []
for i in range(a):
    s.append([])
    for j in range(a):
        s[i].append(0)
n = 2
c = a // 2
if a <= 5:
    s = square(n, s, c, c)
else:
    k = 1
    while k + n <= c:
        s = square(n, s, k, k)
        s = square(n, s, k, -k-1)
        s = square(n, s, -k-1, k)
        s = square(n, s, -k-1, -k-1)
        n += 1
        k += n
    k -= n
    if k + n < c or a in [9, 87]:
        s = square(n, s, c, c)
for i in range(a):
    print(*s[i], sep='')
