#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <windows.h>
#include <math.h>

int main() {
    SetConsoleOutputCP(CP_UTF8);
    srand(time(NULL));
    printf("Ex.1\n");
    printf("Enter n&m-size of array:\n");
    int n,m;
    scanf("%d %d",&n,&m);
    int x[m];
    int a[n][m];
    for (int i = 0; i < m; ++i) {
        x[i]=rand()%21;
    }
    printf("Array x\n");
    for (int i = 0; i < m; ++i) {
        printf("%d ",x[i]);
    }
    printf("\nmatrix:\n");
    for (int i = 0; i < n; ++i) {
        if(i%2==0)
        {
            memcpy(a[i],x,sizeof a[i]);
        }
        else {
            for (int j = 0; j < m; ++j) {
                a[i][j]=rand()%31;
            }
        }
        for (int j = 0; j < m; ++j) {
            printf("%d ",a[i][j]);
        }
        printf("\n");

    }
    int A[n][n];
    printf("Ex.2\n");
    printf("matrix A\n");
    int mi=0;
    int X,y;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            A[i][j]=rand()%31-10;
            if(i+j+1==n)
            {
              if(mi>A[i][j] || mi==0)
                {
                  mi=A[i][j];
                  X=i;
                  y=j;
                }
            }
            printf("%d ",A[i][j]);
        }
        printf("\n");
    }
    printf("min in A= %d %d",X,y);

    printf("\nEx.3\n");
    int s=1;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if(A[i][j]>0 && j>i)
            {
                s*=A[i][j];
            }
        }
    }
    printf("sr.Geom= %.3f",cbrt (s));
    printf("\nEx.4\n");
    int b[n][n+m];
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n+m; ++j) {
            if (j<m)
            {
                b[i][j]=a[i][j];
            }
            else
            {
                b[i][j]=A[i][j-m];
            }
        }
    }

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j <n+m ; ++j) {
            printf("%d ", b[i][j]);
        }
        printf("\n");
    }
    return 0;
}
