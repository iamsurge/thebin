# Script que implementa una calculadora bàsica

def ingresar_numero():
    while True:
        try:
            n = float(input("Ingrese un nro: "))
        except ValueError:
            print("No ha ingresado un nro")
        else:
            return n
            
def menu():
    return """
    Menu de opciones
    1. Suma de 2 nros
    2. Resta de dos nros
    3. Producto de 2 nros
    4. Cociente de 2 nros
    5. Salir
    """

def suma():
    a = ingresar_numero()
    b = ingresar_numero()
    return a + b

def resta():
    a = ingresar_numero()
    b = ingresar_numero()
    return a - b

def producto():
    a = ingresar_numero()
    b = ingresar_numero()
    return a * b
    
def cociente():
    a = ingresar_numero()
    b = ingresar_numero()
    if not b:
        return "No se puede dividir por cero"
    return a / b
    
    
if __name__ == '__main__':
    print("Calculadora básica")
    while True:
        print(menu())
        opcion = input("Seleccione una opción: ")
        if opcion == "1":
            print(suma())
        
        elif opcion == "2":
            print(resta())
    
        elif opcion == "3":
            print(producto())
    
        elif opcion == "4":
            print(cociente())
        
        elif opcion == "5":
            print("Hasta luego")
            break
    
        else:
            print("Opción incorrecta")
