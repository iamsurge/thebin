import tkinter as tk

def clic():
    nombre = caja_nombre.get()
    print(f"El nombre ingresado es {nombre}")
    # borro el contenido de la caja
    caja_nombre.delete(0,tk.END)


# creo una ventana
ventana = tk.Tk()
ventana.config(width=810, height=540)
ventana.title("Mi primera app")
# prohibo redimensionar la ventana
# ventana.resizable(False, False)

# agrego un boton
boton = tk.Button(text="GUARDAR", command=clic)
boton.place(x=70, y=70, width=120, height=40)

# agrego una etiqueta
etiqueta = tk.Label(text="Nombre")
etiqueta.place(x=20, y=10)

# agrego una caja de texto
caja_nombre = tk.Entry()
caja_nombre.place(x=120, y=10, width=200, height=25)


# evito que se cierre la ventana con un bucle infinito
ventana.mainloop()
