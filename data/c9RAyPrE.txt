close all
clear all
clc

plansza=imread('plansza.tif');
planszaconv=conv2(3,3,plansza,'same');

avr=fspecial('average',3);
plansza=im2uint8(plansza);
fplansza=imfilter(plansza,avr,'replicate');

figure('Name','Filtr average');
subplot(1,3,1);
imshow(plansza);
title('Oryginal');

subplot(1,3,2);
imshow(fplansza);
title('Po filtracji');

subplot(1,3,3);
fplabs=imabsdiff(plansza,fplansza);
imshow(fplabs,[]);
title('Modul z roznicy');
%Moduł z różnicy zawiera krawędzie z obrazu, na którym zostały
%wykonane operacje. Filtracja dolnoprzepustowa poprzez eliminacje
%elementów o wysokiej częstotliwości pozbywa się zakłóceń
