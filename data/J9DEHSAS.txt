using System;

namespace WorkingHours
{
	class Program
	{
		static void Main(string[] args)
		{
			int hour = int.Parse(Console.ReadLine());
			string day = Console.ReadLine();

			bool flagh = false;
          	bool flagd = false;

			if ((hour >= 10) && (hour <= 18))
				flagh = true;

			switch (day)
			{
				case "Monday":
				case "Tuesday":
				case "Wednesday":
				case "Thursday":
				case "Friday":
				case "Saturday":
					flagd = true;
					break;
				default:
					flagd = false;
					break;
			}

			if (flagh && flagd)
				Console.WriteLine("open");
			else
				Console.WriteLine("closed");
		}
	}
}
