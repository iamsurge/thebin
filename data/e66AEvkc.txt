#include<bits/stdc++.h>
using namespace std;
/*
            একটা array দেওয়া থাকবে এবং q সংখ্যক query থাকবে। প্রত্যেক query তে l, r, v দেওয়া থাকবে। এখন array
            এর l-index থেকে r-index পর্যন্ত যতগুলো element থাকবে প্রত্যেক element এর সাথে v যোগ করতে হবে।
*/
int update_sum[6], point_update[6];

int main() {
    int a[] = {1, 5, 10, 12, 8, 16};

    int q;
    cin >> q;

    while(q--) {
        int l, r, v;
        cin >> l >> r >> v;

        update_sum[l] += v;
        update_sum[r+1] -= v;
    }


    point_update[0] = update_sum[0];

    for(int i=1; i<6; i++) {
        point_update[i] = point_update[i-1] + update_sum[i];
    }

    for(int i=0; i<6; i++) {
        cout << a[i] + point_update[i] << " ";
    }
    return 0;
}
