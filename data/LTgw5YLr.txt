#include <iostream>

using namespace std;


int main() {
    int n;
    cin >> n;
    
    int niza[n];
    
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
    }
    
    int najgolem_broj = niza[0];
    int najmal_broj = niza[0];
    
    for(int i = 0; i < n; i++) {
        if(niza[i] > najgolem_broj) {
            najgolem_broj = niza[i];
        }
        if(niza[i] < najmal_broj) {
            najmal_broj = niza[i];
        }
    }
    cout << "Najgolem broj: " << najgolem_broj << endl;
    cout << "Najmal broj: " << najmal_broj << endl;
 
    return 0;
}
