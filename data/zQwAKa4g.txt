#include <iostream>

using namespace std;


int main() {
    int n;
    cin >> n;
    
    string iminja[n];
    
    int visina[n];
    
    for(int i = 0; i < n; i++) {
        cin >> iminja[i];
        cin >> visina[i];
    }
    int barana_visina;
    cin >> barana_visina;
    
    for(int i = 0; i < n; i++) {
        if(barana_visina == visina[i]) {
            cout << iminja[i] << endl;
        }
    }
    
    
    return 0;
}
