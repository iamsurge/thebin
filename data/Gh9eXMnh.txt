create trigger Trig_After_DI_Install8
after insert OR delete on Install
for each row begin
    
    -- Insert action
    IF INSERTING THEN

        UPDATE Workstation SET nbsoft = nbsoft+1
        WHERE NWStation = :NEW.NWStation;

        UPDATE Software SET nbinstall = nbinstall+1
        where NSoft = :NEW.NSoft;
    
    --DELETE action
    ELSE IF DELETING THEN
        UPDATE Workstation SET nbsoft = nbsoft-1
        WHERE NWStation = :OLD.NWStation;

        UPDATE Software SET nbinstall = nbinstall-1
        where NSoft = :OLD.NSoft;
    END IF;
    END IF;
    

end;
--DELIMITER ;
