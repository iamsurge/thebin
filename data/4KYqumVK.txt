    /*
    * restituisce il valore del pricing perl'abbonamento mensile ADVANCED PRO
    * eventuali abbonamenti su più mesi, vanno moltiplicati solo per il numero di mesi, esempio: il trimestrale sono gli stessi prezzi, moltiplicati per 3.
    * @param int sliderValue
    * @return ?int 
    */
    function advancedMonthlyPricing(sliderValue) {
        if (sliderValue <= 1) {
            return 117;
        }
        
        if (sliderValue <= 3) {
            return 149;
        }
    
        if (sliderValue <= 7) {
            return 274;
        }
    
        if (sliderValue <= 15) {
            return 450;
        }
    
        if (sliderValue <= 30) {
            return 720;
        }
    
        return null;
    }
    
    /*
    * restituisce il valore del pricing perl'abbonamento mensile BASIC PRO
    * eventuali abbonamenti su più mesi, vanno moltiplicati solo per il numero di mesi, esempio: il trimestrale sono gli stessi prezzi, moltiplicati per 3.
    * @param int sliderValue
    * @return ?int 
    */
    function basicMonthlyPricing(sliderValue) {
        if (sliderValue <= 1) {
            return 69;
        }
        
        if (sliderValue <= 3) {
            return 99;
        }
    
        if (sliderValue <= 7) {
            return 196;
        }
    
        if (sliderValue <= 15) {
            return 360;
        }
    
        if (sliderValue <= 30) {
            return 600;
        }
    
        return null;
    }

    // Update the current prices values (each time you change in select)
    priceSelector1.onchange = function() {

        var PriAdv1 = document.getElementById("PriAdv1");
        var PriAdv1Iva = document.getElementById("PriAdv1Iva");
        var PriBas1 = document.getElementById("PriBas1");
        var PriBas1Iva = document.getElementById("PriBas1Iva");
        
        var PriAdv1Href = document.getElementById("PriAdv1Href");
        var PriBas1Href = document.getElementById("PriBas1Href");

        var PriceAdv = advancedMonthlyPricing(this.value);
        var PriceBas = basicMonthlyPricing(this.value);

        if (PriceBas != null) {

            PriBas1.innerHTML = "€ " + PriceBas;
            PriBas1Iva.innerHTML = "/ mese+IVA";
            PriBas1Href.innerHTML = "<span class="elementor-button-content-wrapper"><span class="elementor-button-text">PROVALO GRATIS</span></span>";
            PriBas1Href.href = "https://webapp.trovabando.it/";

        } else {

            PriBas1.innerHTML = "Richiedi preventivo";
            PriBas1Iva.innerHTML = "";
            PriBas1Href.innerHTML = "<span class="elementor-button-content-wrapper"><span class="elementor-button-text">CONTATTACI</span></span>";
            PriBas1Href.href = "https://www.trovabando.it/contatti/";

        }

        if (PriceAdv != null) {

            PriAdv1.innerHTML = "€ " + PriceAdv;
            PriAdv1Iva.innerHTML = "/ mese+IVA";
            PriAdv1Href.innerHTML = "<span class="elementor-button-content-wrapper"><span class="elementor-button-text">PROVALO GRATIS</span></span>";
            PriAdv1Href.href = "https://webapp.trovabando.it/";

            PriAdv1.innerHTML = "Richiedi preventivo";
            PriBas1Iva.innerHTML = "";
            PriAdv1Href.innerHTML = "<span class="elementor-button-content-wrapper"><span class="elementor-button-text">CONTATTACI</span></span>";
            PriAdv1Href.href = "https://www.trovabando.it/contatti/";

        }
        
    }
