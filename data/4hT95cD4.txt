void __fastcall TMainForm::CreateNewGame()
{
	TimerAnimation->Enabled = false;
	ListBoxActions->Clear();
	ListBoxActions->ClearSelection();

	ImagePay->Visible = false;
	ImageBuy->Visible = false;
	ImageCancel->Visible = false;

	ImageTurn->Visible = true;
	ImagePlayer1->Top = 800;
	ImagePlayer1->Left = 805;

	ImagePlayer2->Top = 800;
	ImagePlayer2->Left = 815;

	ImagePlayer3->Top = 800;
	ImagePlayer3->Left = 825;

	ImagePlayer4->Top = 800;
	ImagePlayer4->Left = 835;

	ImagePause->Visible = true;
	counterOfPlayer = 0;
	currPosition = 0;

    isDouble = false;
	isPause = false;
    isPause ? ImagePause->Picture->LoadFromFile("assets\\play.png") : ImagePause->Picture->LoadFromFile("assets\\pause.png");

	for (int i = 0; i < amountOfCells; i++)
	{
		arrOfCells[i]->setOwner(NULL);
	}

	for (int i = 1; i < amountOfCells; i++)
	{
		if (i != 10 && i != 20 && i != 30)
			arrOfCells[i]->setShapeColor(DEFAULT_COLOR);
	}

	arrOfPlayers.clear();
    for (int i = 0; i < amountOfPlayers; i++)
	{
		arrOfPlayers.push_back(new Player(startCash, 0, i + 1, false, false));
	}

    ImagePlayer1->Visible = true;
	ImagePlayer2->Visible = true;

	if (amountOfPlayers > 2) {
		ImagePlayer3->Visible = true;
		ImageIconPlayer3->Visible = true;
		LabelPlayerCash3->Visible = true;
	}
	else
	{
		ImagePlayer3->Visible = false;
		ImageIconPlayer3->Visible = false;
		LabelPlayerCash3->Visible = false;
	}

	if (amountOfPlayers > 3) {
		ImagePlayer4->Visible = true;
		ImageIconPlayer4->Visible = true;
		LabelPlayerCash4->Visible = true;
	}
	else
	{
		ImagePlayer4->Visible = false;
		ImageIconPlayer4->Visible = false;
		LabelPlayerCash4->Visible = false;
	}


	UpdateCash();
	ListBoxActions->Visible = true;

	if (personalPlayer == counterOfPlayer)
		ListBoxActions->Items->Insert(0, "Ваш ход");
	else
		ListBoxActions->Items->Insert(0, "Ход игрока №" + IntToStr(counterOfPlayer + 1));
}
