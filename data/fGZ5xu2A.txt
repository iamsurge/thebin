import java.util.Scanner;

public class Latihan06 {
    private static void myMethod(String nama){
        System.out.println("Hai, " + nama + " Semangat pagi !");
    }
    private static void myMethodSuhu(int suhuMinimal, int suhuMaximal){
        int suhuRataRata;
        suhuRataRata =  suhuMinimal + suhuMaximal / 2;
        System.out.println("Suhu minimal C = " + suhuMinimal);
        double farenhitMin = (9.0/5.0 * suhuMinimal) + 32;
        System.out.println("Suhu minimal F = " + farenhitMin);
        System.out.println("Suhu rata - rata C = " + suhuRataRata);
        double farenhitRata = (9.0/5.0 * suhuRataRata) + 32;
        System.out.println("Suhu rata - rata F = " + farenhitRata);
        System.out.println("Suhu maximal C = " + suhuMaximal);
        double farenhitMax = (9.0/5.0 * suhuMaximal) + 32;
        System.out.println("Suhu maximal F = " + farenhitMax);

    }
    public static void main(String[] args) {
        String names;
        Scanner input = new Scanner(System.in);
        names = input.nextLine();
        myMethod(names);
        names = input.nextLine();
        myMethod(names);

        int suhuMin, suhuMax;
        suhuMin = input.nextInt();
        suhuMax = input.nextInt();

        myMethodSuhu(suhuMin, suhuMax);
    }
}
