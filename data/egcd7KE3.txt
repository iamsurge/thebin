local pos = 0

function main()
  while true do
    move()
	os.sleep(10)
  end
end

function move()
	while pos < 16 do
	  turtle.forward()
	  pos = pos + 1
	end
	
	turtle.up()
	turtle.up()

	while pos > 0 do
	  turtle.back()
	  pos = pos - 1
	end
	
	turtle.down()
	turtle.down()
end

main()
