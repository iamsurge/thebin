namespace MyApp
{
    internal class Program
    {
        static double[] _studioPrices = { 50, 75.2, 76 };
        static double[] _apartmentPrices = { 65, 68.7, 77 };
        static string[] _months = { "May", "October", "June", "September", "July", "August" };

        static void Main(string[] args)
        {
            var month = Console.ReadLine();
            var overnightStays = int.Parse(Console.ReadLine());

            int index = Array.IndexOf(_months, month) / 2;
            Console.WriteLine($"Apartment: {Math.Round(ApartmentCalculations(_apartmentPrices[index], overnightStays), 2).ToString("N2").Replace(",", "")} lv.");
            Console.WriteLine($"Studio: {Math.Round(StudioCalculations(_studioPrices[index], month, overnightStays), 2).ToString("N2").Replace(",", "")} lv.");
        }

        static double StudioCalculations(double price, string month, int overnightStays)
        {
            var priceForAll = price * overnightStays;
            switch (month)
            {
                case "May":
                case "October":
                    if (overnightStays > 7 && overnightStays < 14)
                        return priceForAll * (1 - 0.05);
                    else if (overnightStays > 14)
                        return priceForAll * (1 - 0.3);
                    else
                        return priceForAll;
                case "June":
                case "September":
                    if (overnightStays > 14)
                        return priceForAll * (1 - 0.2);
                    else
                        return priceForAll;
                default: return priceForAll;
            }
        }

        static double ApartmentCalculations(double price, int overnightStays)
        {
            return overnightStays > 14 ? price * overnightStays * (1 - 0.1) : price * overnightStays;
        }
    }
}
