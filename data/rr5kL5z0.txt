import functools


def suggest_people_you_may_know(graph, user):
    visited = set()
    suggestions = []
    friends_list = graph[user]

    def dfs(node):
        if node not in visited:
            visited.add(node)
            if node not in friends_list and node != user:
                suggestions.append(node)

            def reducer(acc, elem):
                dfs(elem)
                pass
            functools.reduce(reducer, graph[node], None)
        else:
            return

    dfs(user)
    return suggestions

}
