local plr = game.Players.LocalPlayer
local hum = plr.Character:WaitForChild("HumanoidRootPart")
local mouse = plr:GetMouse()

legacy = false
new = false

function LTP()
	for i,v in pairs(game.Workspace:GetDescendants()) do
		if v:IsA("Trail") then
			v.Parent.CFrame = hum.CFrame
		end
	end
end

function LGoTo()
	for i,v in pairs(game.Workspace:GetDescendants()) do
		if v:IsA("Trail") then
			hum.CFrame = v.Parent.CFrame
		end
	end
end

function TP()
	hum.CFrame = game.Workspace:FindFirstChild("CSF").CFrame 
end

function GoTo()
	game.Workspace:FindFirstChild("CSF").CFrame = hum.CFrame
end

-- legacy: https://www.roblox.com/games/10937521050/MPS-4-a-side-Legacy
-- new: https://www.roblox.com/games/10704630570/MPS-4-a-side

mouse.KeyDown:Connect(function(key)
	if key == "z" then
		if legacy == true then
			LGoTo()
		elseif new == true then
			GoTo()
		elseif key == "x" then
			if legacy == true then
				LTP()
			elseif new == true then
				TP()	
			end
		end
	end
end)
