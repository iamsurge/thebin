# bestand inlezen:
bestand = open('input.txt', 'r')
alles = bestand.readlines()
for x in range(len(alles)):
  alles[x] = alles[x].replace('\n', '')

#lijsten aanmaken:
koppels = []
elfjes = []

#strings verknippen:
for x in range(len(alles)):
  koppels.append(alles[x].split(','))
for x in range(len(koppels)):
  deze2 = []
  for y in range(2):
    deze2.append(koppels[x][y].split('-'))
  elfjes.append(deze2)


#vraag A:
antwoord = 0

for x in range(len(elfjes)):
  if (int(elfjes[x][0][0]) <= int(elfjes[x][1][0]) and int(elfjes[x][0][1]) >= int(elfjes[x][1][1])) or (int(elfjes[x][0][0]) >= int(elfjes[x][1][0]) and int(elfjes[x][0][1]) <= int(elfjes[x][1][1])):
    antwoord += 1

print(antwoord)
