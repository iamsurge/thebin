import java.time.Year

fun main() {
    val birthYear = "2001"
    val currentYear = Year.now()
    val currentAge = currentYear.compareTo(Year.parse(birthYear))
    println("Tanggal lahir : $birthYear")
    println("Umur saat ini : $currentAge")
    if (birthYear.takeLast(2).toInt() > currentAge) {
        println("Lebih")
    } else {
        println("Kurang")
    }
}
