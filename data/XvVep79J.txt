#include <stdio.h>
#include <stdlib.h>

int main(){
    int x;
    
    while(scanf("%d", &x)) {
        if(x >= 10) {
            int pom = x;
            int zigzag = 0;
            int ok = 1;
            while(pom > 9) {
                int posledna_cifra = pom % 10;
                int pretposledna_cifra = (pom / 10) % 10;
                if(zigzag == 0) {
                    if(posledna_cifra >= pretposledna_cifra) {
                        ok = 0;
                        break;
                    }
                    zigzag = 1;
                }
                else if(zigzag == 1) {
                    if(posledna_cifra <= pretposledna_cifra) {
                        ok = 0;
                        break;
                    }
                    zigzag = 0;
                }
                pom /= 10;
            }
            pom = x;
            zigzag = 1;
            int ok2 = 1;
            while(pom > 9) {
                int posledna_cifra = pom % 10;
                int pretposledna_cifra = (pom / 10) % 10;
                if(zigzag == 0) {
                    if(posledna_cifra >= pretposledna_cifra) {
                        ok2 = 0;
                        break;
                    }
                    zigzag = 1;
                }
                else if(zigzag == 1) {
                    if(posledna_cifra <= pretposledna_cifra) {
                        ok2 = 0;
                        break;
                    }
                    zigzag = 0;
                }
                pom /= 10;
            }
            
            if(ok == 1 || ok2 == 1) {
                printf("%d\n", x);
            }
        }
    }
    
    return 0;

}
