Make a JS script and create an object called 'config' and declare it with following variables and values: Bet (0.01), Target (2) using the following script as a template:
`
var originalBet = config.Bet;
engine.on('game_ready', () => {
    engine.placeBet(config.Bet, config.Target).then((result) => {
        Action(((result.crash < config.Target) ? "Lose" : "Win"));        
    });
});
function Action(Response){
    switch (Response){
        case "Win":
	config.Bet = originalBet;
            break;
        case "Lose":
	config.Bet *= 2;
            break;
    }
}
`
