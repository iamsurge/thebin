n = int(input())
primeNumbers = []
for x in range(n + 1):
    primeNumbers.append(x)
primeNumbers[1] = 0
i = 2
while i <= n:
    if primeNumbers[i] != 0:
        j = i + i
        while j <= n:
            primeNumbers[j] = 0
            j = j + i
    i += 1
primeNumbers = set(primeNumbers)
primeNumbers.remove(0)
print(*primeNumbers)
