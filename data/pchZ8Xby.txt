module = peripheral.wrap("back")
displayName = "minecraft:totem_of_undying"


local function moveTotem()
    print(module.getEquipment.getItem(2).getMetadata().name ~= displayName)
    sleep(2)
    if (module.getEquipment().getItem(2).getMetadata().name ~= displayName) then
        for k, v in pairs(module.getInventory().list()) do
            if (v.name == displayName and module.getEquipment().getItem(2).getMetadata().name ~= displayName) then
                module.getInventory.pushItems("equipment",k)
            else
                print(v.name)
                sleep(2)
            end
        end
    else
        print("sadge")
    end
end
moveTotem()
