#! /bin/bash
# dependency : fortune , cowsay
# Android termux version  

whatsup(){
        local comic count pick
        comic=($(cowsay -l))
        count=${#comic[@]}
        pick="${comic[$((RANDOM%count-1))]}"
        echo "$pick"
}

until false ; do
        clear
        if ! cowsay -f "$(whatsup)"  "$(date +%H:%M) $(fortune)" 2> /dev/null ; then
                cowsay "$(date +%H:%M) $(fortune)"
        fi
        sleep 30
done
