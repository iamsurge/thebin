  num: int = 0
  def __init__(self, num):
    self.num = num
  
  def __repr__(self):
    return f'{self.num}'
    
  def __add__(self, other):
    return self.num + other.num

  @property
  def plus(self):
    return(Num_plus(self.num))
  
  @property
  def minus(self):
    return(Num_minus(self.num))

  @property
  def times(self):
    return(Num_times(self.num))
