import random

# Función que permite al usuario poner la bala en su posición inicial.
def cargar():
	while True:
		# Declaramos una lista de huecos vacíos.
		tam = ['clic', 'clic', 'clic', 'clic', 'clic', 'clic']
		print('\nEl tambor está vacío')
		bala = input('En que posición quieres colocar la bala? (1-6): ')
		# Comprueba que lo que introduce el usuario es un valor numerico.
		if bala.isnumeric() != True:
			print('\nTe quedas con la bala en la mano balbucenando cosas sin sentido.')
			print('Debes seleccionar uno de los 6 huecos.')
		# Comprueba que lo que introduce el usuario es un valor entre 1 y 6.
		elif (int(bala) < 1) or (int(bala) > 6):
			print('\nPones la bala en el aire y la sueltas, cae al suelo con un "clinc".')
			print('Solo hay 6 huecos para poner la bala numerados del 1 al 6, debes escoger uno.')
		# Si todo va bien se asigna el valor '¡¡PUM!!' a esa posición de la lista, la bala está colocada.
		else:
			tam[int(bala) - 1] = '¡¡PUM!!'
			print(f'\nDecides colocar la bala en el hueco nº{(tam.index("¡¡PUM!!")) + 1}.')
			return tam

# La variable tambor recibe el valor de la lista resultante de la función cargar. Esta lista tiene una bala.
tambor = cargar()
