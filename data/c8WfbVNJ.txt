class Game
{
	_hud : MyHudClass;
	_someOtherStuff : any;  // just representing any other elements
	constructor()
	{
		this._hud = new MyHudClass( this );
	}
}

class MyHudClass
{
	_panel : PanelClass;
	_otherHudStuff : any;
	constructor( g : Game )
	{
		_panel = new PanelClass( this.panelCallback );
		this.panelCallback = this.panelCallback.bind( this );
	}
	
	private panelCallBack()
	{
		this._panel.DoSomething();  // error cannot read properties of undefined "DoSomething"
	}
}

class PanelClass
{
	_callbackFunc : Function;
	_otherPanelStuff : any;

	constructor( callback : Function )
	{
		this._callbackFunc = callback;
        this.on('pointertap', this.onPointerTap);
	}
	public DoSomething()
	{
		_otherPanelStuff.x = 50;
	}

    private onPointerTap() 
    {
        if( this._callbackFunc )
        {
            this._callbackFunc();            
        }
    }

}
