#include <iostream>
using namespace std;

struct mhs{
	int noMhs, nilaiAngka;
	char kelas, nilaiHuruf;
	string nama;
};

mhs data[6];

int main(){
	for(int i=1; i<=5;i++){
		cout<<"\nData Mahasiswa ke-"<<i;
		cout<<"\nNama\t: ";cin>>data[i].nama;
		cout<<"NIM\t: ";cin>>data[i].noMhs;
		cout<<"Kelas\t: ";cin>>data[i].kelas;
		cout<<"Nilai\t: ";cin>>data[i].nilaiAngka;
		
		if(data[i].nilaiAngka < 20)
			data[i].nilaiHuruf = 'E';
		else if(data[i].nilaiAngka < 40 && data[i].nilaiAngka >= 20)
			data[i].nilaiHuruf = 'D';
		else if(data[i].nilaiAngka < 60 && data[i].nilaiAngka >= 40)
			data[i].nilaiHuruf = 'C';
		else if(data[i].nilaiAngka < 75 && data[i].nilaiAngka >= 60)
			data[i].nilaiHuruf = 'B';
		else if(data[i].nilaiAngka < 100 && data[i].nilaiAngka >= 75)
			data[i].nilaiHuruf = 'A';
	}
	cout<<string(30,'=');
	for(int i=1; i<=5;i++){
		cout<<"\n\nData Mahasiswa ke-"<<i;
		cout<<"\nNama\t: "<<data[i].nama;
		cout<<"\nNIM\t: "<<data[i].noMhs;
		cout<<"\nKelas\t: "<<data[i].kelas;
		cout<<"\nNilai\t: "<<data[i].nilaiAngka;
		
		cout<<"\n\nGrade\t: "<<data[i].nilaiHuruf;
	}
}
