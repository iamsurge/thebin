from kivy.app import App
from kivy.lang import Builder
from kivy.uix.label import Label

import webbrowser

kv = """
<MarkupLabel>:
    markup: True
    on_ref_press: self.action(args[1])

AnchorLayout:
    BoxLayout:
        orientation: 'vertical'
        size_hint_y: None
        height: 60
        MarkupLabel:
            text: 'Email: [ref=mailto:email_adr@mail.com][u]email_adr@mail.com[/u][/ref]'
        MarkupLabel: 
            text: 'Phone: [ref=tel:+1(123)456-7890][u]Call me![/u][/ref]'
"""


class MarkupLabel(Label):
    def action(self, url):
        print(url)
        webbrowser.open(url)


class EmailPhoneApp(App):
    def build(self):
        return Builder.load_string(kv)

EmailPhoneApp().run()
