int n, k;
vector<int> a;
vector<vector<int>> dp;
int fix(int x)
{
    return (x % k + k) % k;
}
bool isdiv(int idx, int sum)
{
    if (idx == n - 1)
        return sum == 0;
    int &ret = dp[idx][sum];
    if (~ret)
        return ret;
    return (ret = isdiv(idx + 1, fix(sum - a[idx])) || isdiv(idx + 1, fix(sum + a[idx])));
}
void solve()
{
    cin >> n >> k;
    a.resize(n);
    cin(a);
    dp.assign(n, vector<int>(k, -1));
    cout << isdiv(1, a[0] % k) << nl;
}
int main()
{
    sherry();
    int t = 1;
    cin >> t;
    while (t--)
        solve();
    return 0;
