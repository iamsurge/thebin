RegExMatch(selected, ".:\\.+\\", vPath)

	}

; function from Masonjar on ahk forum
getSelected(){
    cO:=clipboardAll
    clipboard:=
    send !c
    clipWait
    selected:=clipboard
    clipboard:=cO
    return selected
}
