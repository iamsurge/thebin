  #define DAC_PIN A0
  int i =0;

  void setup()
  {
  analogWriteResolution(10); // Set analog out resolution to max, 10-bits
  analogReadResolution(12); // Set analog input resolution to max, 12-bits
  }

  void loop()
  {
  for (int i = 0; i <= 950; i = i + 1) 
  {
  analogWrite (A0,i);
  delay (5);
  }
      
  for (int i = 950; i >= 0; i = i - 1) {
  analogWrite (A0,i);
  delay (5);
  } 
  
  }
