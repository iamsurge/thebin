// Problem 5. Min, Max, Sum and Average of N Numbers
//
// Write a program that reads from the console a sequence of n integer numbers and returns the minimal, the maximal number,
// the sum and the average of all numbers (displayed with 2 digits after the decimal point).
// The input starts by the number n (alone in a line) followed by n lines, each holding an integer number.
// The output is like in the examples below.
//
//   Examples:
// n    output
// 3    min = 1
// 2    max = 5
// 5    sum = 8
// 1 	avg = 2.66

let input = ['3',
    '2',
    '5',
    '1'
];

let print = this.print || console.log;
let gets = this.gets || ((arr, index) => () => arr[index++])(input, 0);

let length = +gets();
let num = '';
let maxNum = '';
let minNum = '';
let sum = 0;
let avg = '';

//go true inputs and collect them in array
for (let i = 0; i < length; i++) {
    num += +gets();
}
for(let i = 0; i < num.length; i++) {
    if (num[i] > num[i + 1]) {
        maxNum += num[i];
    }
    if (num[i] < num[i + 1]) {
        minNum += num[i];
    }
    sum += +num[i];
    avg = (sum / num.length).toFixed(2);
}

console.log(`min = ${minNum}`);
console.log(`max = ${maxNum}`);
console.log(`sum = ${sum}`);
console.log(`avg = ${avg}`);
