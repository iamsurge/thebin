#include <stdio.h>
#include <stdlib.h>
 
double f(double x, double m) {
    double i = 0;
    double sum = 1;
    double tmp = 1;
    while (m != i) {
        tmp *= (m - i) / (i + 1);
        tmp *= x;
        sum += tmp;
        i++;
    }
    return sum;
}
 
int main(int argc, char *argv[]) {
    if(argc != 3){
        printf("Incorrect input, check README.md\n");
        return 0;
    }
    FILE *input = fopen(argv[1], "r");
    FILE *out = fopen(argv[2], "w");
    if((input == NULL) || (out == NULL)){
        printf("Incorrect file\n");
        return 0;
    }
    double x, m;
    fscanf(input, "%lf", &x);
    fscanf(input, "%lf", &m);
    double res = f(x, m);
    fprintf(out, "Integral = %lf\n", res);
    fclose(input);
    fclose(out);
    return 0;
}
