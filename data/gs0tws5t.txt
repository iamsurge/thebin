
add_filter( 'streamtube/core/user/profile/menu/items', function( $items ){
	$items['my-link'] = array(
		'title'			=>	esc_html__( 'My Link', 'streamtube-core' ),
		'icon'			=>	'icon-cog',
		'url'			=>	home_url( 'my-link' ),				
		'priority'		=>	100,
		'private'		=>	false
	);

	return $items;

}, 10, 1 );
