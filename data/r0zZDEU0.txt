#include <iostream>
#include <new>
using namespace std;

int main() {
    int n;
    int *ptr;
    cout << "How many numbers would you like to type : ";
    cin >> n;

    ptr = new (nothrow) int[n];
    if (ptr == nullptr)
        cout << "Error: memory could not be allocated" << '\n';

    else {
        for (int i = 0; i < n; i++) {
            cout << "Enter number: ";
            cin >> ptr[i];
        }

        cout << "You have entered: ";
        for (int i = 0; i < n; i++)
            cout << ptr[i] << ", ";

        delete[] ptr;
    }
    cout << "\n\nAfter delete ptr memori: " << '\n';
    cout << "You have entered: ";
    for (int i = 0; i < n; i++)
        cout << ptr[i] << ", ";

    return 0;
}
