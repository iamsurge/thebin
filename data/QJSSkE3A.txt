-- Interpret rednet signals as movement commands

local moves = {
[32] = turtle.up,
[340] = turtle.down,
[87] = turtle.forward,
[83] = turtle.back,
[65] = turtle.turnLeft,
[68] = turtle.turnRight,
[265] = turtle.digUp,
[264] = turtle.digDown,
[262] = turtle.dig,
[81] = turtle.place,
[90] = turtle.placeDown,
[50] = turtle.placeUp,
[82] = turtle.refuel
};

while true do
	local id, message = rednet.recieve();
    moves[tonumber(message)]();
end
