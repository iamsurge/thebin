struct TT
{
	
	TT            *p;

} TT_Head, *Tail;




int TT_size = 0;


//////////////////////////////////////////////////////////
int add_El(int x)
{
	
	TT *p = (TT*)malloc(sizeof(TT));
	
	TT_size ++;
	
	p->old = x;
	
	if(TT_size == 0)
	{  
		
	   TT_Head.p = p;	
	   Tail	     = p;
	}
	else
	{
	   Tail->p   = p;	
	   Tail	     = p; 	
	}
}
