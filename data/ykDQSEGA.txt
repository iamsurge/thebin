/*Función para WordPress para que los artículos programados se muestren de fecha más actual a posterior en el área de administración de WordPress.*/

function show_scheduled_posts_in_order() {
global $wp_query;
$wp_query->set('orderby', 'date');
$wp_query->set('order', 'ASC');
}
add_action('pre_get_posts', 'show_scheduled_posts_in_order');
