while true do
-- Pulls blazerod chickens from an inventory above it and spawns them into the world I use this for making blazing blood in the smeltry in Stoneblock 3
local function spawn()
	turtle.select(1)
	turtle.place()
end

local function check()
	if turtle.getItemCount(1) >= 8 then
    spawn()
    else
		turtle.select(1)
		turtle.suckUp()
	end
end

check()
sleep (0.5)
 
end
