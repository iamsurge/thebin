`timescale 1ns / 1ps

module testbench();

    reg clk1;
    reg clk2;
    wire res;
    reg [7:0] args;
    reg [0:31] error_reg;
    initial begin
        #1500 $finish;
    end
    initial begin
        clk1 = 0;
        clk2 = 0;
        args = 0;
        error_reg = 0;
    end
   
    always #10 clk1 = ~clk1;
    always #280 clk2 = ~clk2;
frequency_divider #(28) freq(.clock_in(clk1), .clock_out(res));
endmodule
