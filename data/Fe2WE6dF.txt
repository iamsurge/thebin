using System;
namespace LearnRSUE
{
    class secondPers1
    {
        static void Main()
        {
            int num = 10;
            int[] masiv = new int[num];
            Random random = new Random();

            for (int i = 0; i < 10; i++)
            {
                masiv[i] = random.Next(0, 100);
            }

            Array.Sort(masiv);

            foreach (int i in masiv)
            {
                Console.WriteLine(i);
            }
        }
    }
