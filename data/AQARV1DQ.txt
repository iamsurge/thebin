# script que imprime 4 listas

def imprimir(lista):
    for item in lista:
        print(item)
        

numeros =  [2,3,4,5]
nombres = ["Ana","Juana"]
notas = [7, 8, 10, 6, 5]
frutas = ["anana", "pera", "banana"]

imprimir(numeros)
imprimir(nombres)
imprimir(notas)
imprimir(frutas)

def saludar(nombre):
    print(f"Hola {nombre}")
    
saludar("Juan")
saludar("Ana")

# POSIBLE

    return a+b+c+d

# pero si no quiero guardar los datos, tambièn puedo hacerlo
