# Saisie des ventes de billets
nb1 = int(input("Nombre de billets 1er jour ? "))
nb2 = int(input("Nombre de billets 2eme jour ? "))
nb3 = int(input("Nombre de billets 3eme jour ? "))
nb4 = int(input("Nombre de billets 4eme jour ? "))

# Nombre total de billets vendus
nbt = nb1+nb2+nb3+nb4

# Prix d'un seul billet
pb = 3.0

# Chiffre d'affaire des ventes
cha = pb*nbt

# Afficher le chiffre d'affaire des ventes
print("Nombre total de billets vendus =", nbt)
print("Chiffre d'affaire des ventes =", cha)

# Calcul du nombre maximal de billets vendus : m
# et en quelle journée                       : mj
m = nb1
mj = 1
if nb2 > m:
    m = nb2
    mj = 2
if nb3 > m:
    m = nb3
    mj = 3
if nb4 > m:
    m = nb4
    mj = 4

# Calcul du nombre maximal de billets vendus : mn
# et en quelle journée                       : mnj
mn = nb1
mnj = 1
if nb2 < mn:
    mn = nb2
    mnj = 2
if nb3 < mn:
    mn = nb3
    mnj = 3
if nb4 < mn:
    mn = nb4
    mnj = 4

# Affichage des ventes de billets
print("Nombre maximal de billets vendus =", m, "- Jour :", mj)
print("nombre minimal de billets vendus =", mn, "- Jour :", mnj)

# Calcul de la moyenne et de l'écart type
moy = (nb1+nb2+nb3+nb4)/4
ecart = (((nb1*nb1+nb2*nb2+nb3*nb3+nb4*nb4)/4)-moy*moy)**0.5
print("Moyenne des ventes =", moy, "+/-", ecart)
