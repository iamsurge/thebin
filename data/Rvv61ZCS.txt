#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
    int n;
    cin >> n;
    int niza[n];
    for(int i = 0; i < n; i++)
    {
        cin >> niza[i];
    }
    sort(niza, niza + n);
    int sredisna_cifra = niza[n / 2];
    int paricki = 0;
    for(int i = 0; i < n; i++)
    {
        paricki += abs(sredisna_cifra - niza[i]);
    }
    cout << paricki << endl;
    return 0;
}
