#! /bin/bash
# $PROG: fclock - a simple ascii clock
# dependency : figlet
# debian-base OS: sudo apt update && sudo apt upgrade -y ; sudo apt install figlet
# if not installed in system , press Ctrl-C to quit script

while : ; do
        clear
        d=$(date +%T)
        echo -e "\033[34m]$(figlet $d)"
        sleep 1
done
