import tensorflow
from sklearn.datasets import load_iris
from tensorflow.keras.models import Sequential
from tensorflow.keras.layer import Dense

iris = load_iris()
X = iris.data
y = iris.target

model = Sequential()
model.add(Dense(8, input_dim=4, activation = 'relu'))
model.add(Dense(8,activatoin='relu'))
model.add(Dense(3,activation='sofmax'))

model.compile(optimizer='adam', loss='categorical_crossentropy',metrics=['accurarcy'])
model.fit(X,)
