alias wd3 { return $gettok($1,$2,44) }
alias sub.schemes {
  var %y = 1
  while ($ini(mirc.ini, colors, %y)) {
    var %read = $deltok($readini(mirc.ini, colors, n $+ %y),2-,44)
    var %theme %theme $+($wd3(%read,1),$chr(44))
    inc %y
  }
  var %f = $numtok($wd3(%theme,1-),44), %z = 1
  while (%z <= %f) {
    var %t = $wd3(%theme,%z)
    if ($1 == %z) return $iif($readini(mirc.ini,text,theme) == %t,$style(3)) %t :color -s %t
    inc %z
  }
}
