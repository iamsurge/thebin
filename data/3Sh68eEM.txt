d = 9

n = int(input())
A = [0] * n

flag = False
for i in range(n):
    A[i] = int(input())
    if A[i] % 2 == 0:
        flag = True

if flag == False:
    print(-1)
else:
    answer = 0
    max_even = 0
    max_odd = -1
    for i in range(d, n):
        if A[i - d] % 2 == 0:
            max_even = max(max_even, A[i - d])
        else:
            max_odd = max(max_odd, A[i - d])
        
        answer = max(answer, A[i] * max_even)
        if A[i] % 2 == 0:
            answer = max(answer, A[i] * max_odd)
    print(answer)
