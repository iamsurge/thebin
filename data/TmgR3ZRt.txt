// bookmarklet to generate Discord "relative" timestamps, v1, released to public domain, no warranty etc
// use one (or both!) of these as your bookmark URL, place it on the bookmark toolbar, click, enjoy

// relative variant, e.g. "2 years ago" or "in 19 minutes"
javascript:void(function(){var dt = Date.parse(prompt("Desired date, in YYYY-MM-DDTHH:mm format.\nFor example, '2020-01-02T03:04'")); if (!Number.isNaN(dt)) {alert(`Copy this: <t:${dt/1000}:R>`)}}())

// long variant, e.g. "Thursday, March 4, 2021 12:00 PM" or "Monday, December 12, 2022 6:00 PM"
javascript:void(function(){var dt = Date.parse(prompt("Desired date, in YYYY-MM-DDTHH:mm format.\nFor example, '2020-01-02T03:04'")); if (!Number.isNaN(dt)) {alert(`Copy this: <t:${dt/1000}:F>`)}}())
