int findSubstrings(string s) {
 
    map<long long, long long> mp;
    int l=0, r=0, ans = 0, n=s.size();
    for (; r<n; r++) {
        mp[s[r]]++;
        while (mp[s[r]] > 1) {
            mp[s[l]]--;
            if (mp[s[l]] == 0) mp.erase(s[l]);
            l++;
        }
        ans += (r-l+1);
    }

    return ans;
}
