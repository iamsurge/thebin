package ppa1;

public class Zbran {
	public String nazev;
	public int utok;
	public int obrana;
	
	public Zbran(String nazev, int utok, int obrana) {
		this.nazev = nazev;
		this.utok = utok;
		this.obrana = obrana;
	}
	
	public int getUtok() {
		return utok;
	}
	
	public int getObrana() {
		return obrana;
	}
	
	public String toString() {
		return nazev + " (" + utok + "/" + obrana + ")";
	}

}
