package Foxx;

import javax.swing.JOptionPane;

/*
 * Helper - Used as a function library 
 * 
 * @author Foxx Azalea Pinkerton
 * @version 1.1.0
 */
public class Helper {

  /**
   * Shows the user a error message
   *
   * @param e
   */
  private static void showError(Throwable e) {
    final String PREFIX = "ERROR: ";
    String message = PREFIX + e.getMessage();
    JOptionPane.showMessageDialog(null, message, "ERROR", 0);
  }

  /**
   * Shows a message box
   *
   * @param message
   * @param title
   */
  public static void messageBox(String message, String title) {
    JOptionPane.showMessageDialog(null, message, title, 1);
  }
  
  /**
   * Shows the user a problem
   * @param message
   * @param title 
   */
  public static void problemBox(String message, String title) {
    JOptionPane.showMessageDialog(null, message, title, 0);
  }

  /**
   * Gets a string and validates it then returns it
   *
   * @param message
   * @param title
   * @return
   */
  public static String getString(String message, String title) {
    String input = JOptionPane.showInputDialog(null, message, title, 1);
    if (input.equals("")) {
      Throwable error = new NullPointerException("Null Pram Passed");
      showError(error);
      System.exit(0);
    }
    return input;
  }
  
  /**
   * Gets a int from the user
   * 
   * @param message
   * @param title
   * @return 
   */
  public static int getInt(String message, String title) {
    String input = JOptionPane.showInputDialog(null, message, title, 1);
    if (input.equals("")) {
      Throwable error = new NullPointerException("Null Pram Passed");
      showError(error);
      System.exit(0);
    }
    
    return toInt(input);
  }

  /**
   * Converts a string to an int
   *
   * @param input
   * @return
   */
  public static int toInt(String input) {
    return Integer.parseInt(input);
  }

  /**
   * Returns the greatest common denominator
   *
   * @param a
   * @param b
   * @return
   */
  public static long gcd(long a, long b) {
    return b == 0 ? a : gcd(b, a % b);
  }

  /**
   * Returns a reduced fraction
   *
   * @param numerator
   * @param denomerater
   * @return
   */
  public static String reduceFraction(long numerator, long denomerater) {
    long gcd = Helper.gcd(numerator, denomerater);
    return (numerator / gcd) + "/" + (denomerater / gcd);
  }

  /**
   * Asks the user a yes or no question
   * 
   * @param message
   * @param title
   * @return
   */
  public static boolean yesNo(String message, String title) {
    int responce = JOptionPane.showConfirmDialog(null, message, title, JOptionPane.YES_NO_OPTION, 1);
    return responce == JOptionPane.YES_OPTION;
  }

  /**
   * Asks the user if they would like to play again.
   *
   * @param title
   * @return
   */
  public static boolean playAgain(String title) {
    return yesNo("Would you like to play again?", title);
  }
  
  /**
   * Checks if a int is even
   * @param number
   * @return
   */
  public static boolean isEven(int number) {
    return number %2 == 0;
  }
  
  /**
   * Welcomes the user to the program
   * 
   * @param title 
   */
  public static void welcome(String title){
    messageBox("Welcome to " + title, title);
  }
  
  /**
   * Provides an exit message
   * 
   * @param title 
   */
  public static void exit(String title) {
    messageBox("Thank you for using " + title, title);
    System.exit(0);
  }
}
