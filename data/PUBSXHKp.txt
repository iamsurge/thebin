#include <iostream>
using namespace std;
#define ll long long 
   
                                           
                                           
 
ll a, 
   b, 
   c[1000000];   
 
 
int main()
{
    cin >> a;
    
 
    for(int i = 1; i < 1000000; i += 2) c[i] = i;
    
    
    for(int i = 0; i < a; i++)
    {
        cin >> b;
        
        for(int u = 1; u < b; u += 2) 
        
        if(b % c[u])  cout << "NO"  << endl;
                      else cout << "YES" << endl;
                      
    }
}
