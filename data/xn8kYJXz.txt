public class C3 extends C2{
        //Caminhao
    public float comprimento; // em metros
    protected int cargaMaxima; // em Toneladas
    private int qtdEixos;

    public float getComprimento() {
        return comprimento;
    }

    public void setComprimento(float comprimento) {
        this.comprimento = comprimento;
    }

    public int getCargaMaxima() {
        return cargaMaxima;
    }

    public void setCargaMaxima(int cargaMaxima) {
        this.cargaMaxima = cargaMaxima;
    }

    public int getQtdEixos() {
        return qtdEixos;
    }

    public void setQtdEixos(int qtdEixos) {
        this.qtdEixos = qtdEixos;
    }

    public C3(String placa, String marca, int kilometragem, int anoChassi, String tipoPneu, float tipoMotor, float comprimento, int cargaMaxima, int qtdEixos){
        super(placa, marca, kilometragem, anoChassi, tipoPneu, tipoMotor);
        System.out.println("Chamada do construtor C3 com parametros!");
        this.setComprimento(comprimento);
        this.setCargaMaxima(cargaMaxima);
        this.setQtdEixos(qtdEixos);
    }

    public C3(){
        System.out.println("Chamada do construtor C3 sem parametros!");
        this.comprimento = 0.0f;
        this.cargaMaxima = 0;
        this.qtdEixos = 0;
    }

    public void mostrar_atributos(){
        System.out.println("Numero da placa: " + this.getPlaca());
        System.out.println("Marca: " + this.getMarca());
        System.out.println("Kilometragem do veiculo: " + this.getKilometragem() + " Km");
        System.out.println("Ano do Chassi: " + this.getanoChassi());
        System.out.println("Pneu do tipo: " + this.getTipoPneu());
        System.out.println("Motor do tipo: " + this.getTipoMotor());
        System.out.println("Comprimento: " + this.getComprimento() + " metros");
        System.out.println("Carga Maxima: " + this.getCargaMaxima() + " Toneladas");
        System.out.println("Quantidade de eixos: " + this.getQtdEixos());
    }

    public void mostrar_atributos_super() {
        super.mostrar_atributos_super();
        System.out.println("Comprimento: " + this.getComprimento() + " metros");
        System.out.println("Carga Maxima: " + this.getCargaMaxima() + " Toneladas");
        System.out.println("Quantidade de eixos: " + this.getQtdEixos());
    }
}
