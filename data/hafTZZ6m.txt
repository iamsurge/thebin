int getLowerBound(vector<int> &v, int x) {
    int L=0, R = v.size(), mid;

    while(L < R) {
        mid = L + (R-L)/2;

        if(v[mid] < x) {
            L = mid + 1;
        } else {
            R = mid;
        }
    }
    return L;
}
