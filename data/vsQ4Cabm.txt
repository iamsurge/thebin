# Saisie des données
tb = input("Type de billet (AS/AR) ? ")
pb = float(input("Poids du bagage (kg) ? "))

# calculer le poids autorisé
if tb == "AR":
    pa = 30
else:
    pa = 20

# Calculer l'excès de bagages
pe = pb - pa
if pe != int(pe):
    pe = int(pe) + 1

# Calculer le supplément à payer
sap = pe*2

# Afficher le supplément à payer
print("Poids en excès =", pe)
print("Supplément à payer =", sap, "$")

# Saisie de la classe du billet
cl = input("Classe du billet (AF/EC) ? ")

# Calcul de réduction et su supplément à payer après réduction
if cl == "AF":
    red = sap * 25/100
    sap = sap - red
else:
    red = 0.0

# Afficher le supplément à payer après réduction
print("Le voyageur bénéficie d'une réduction de", red, "$")
print("Supplément à payer aprés réduction =", sap, "$")

# Est-ce que le voyageur doit payer le timbre de voyage ?
if cl == "EC":
    print("Le voyageur doit payer le timbre de voyage")
else:
    print("Le timbre de voyage est déjà inclu dans le billet classe affaires")
