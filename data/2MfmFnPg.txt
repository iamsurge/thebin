#include <iostream>

using namespace std;

int main()
{
    int n;

    cout << "Enter the number of elements" << endl;
    cin >> n;

    int A[5];
    int sum = 0;
    int i = 0;


    cout << "Enter the elements" << endl;
    for (i = 0; i < n; i++) {
        cin >> A[i];
    }

    for (i = 0; i < 5; i++) {
        sum = sum + A[i];
    }

    if (sum % 2 == 0) {
        cout << "Sum of array elements is _even_";
    }
    else {
        cout << "Sum of array elements is _odd_";
    }

    return 0;
}
