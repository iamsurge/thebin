// ==UserScript==
// @name            Mastodon: Star → Heart
// @description     Change star to heart icon everywhere
// @version         19-November-2022
// @author          Jurid
// @match           *://twiukraine.com/*
// @run-at          document-start
// ==/UserScript==


(function () {
  function changeIcon() {
    document.querySelectorAll('*').forEach(node => node.className = node.className.replace("fa-star", "fa-heart"))
  }

  new MutationObserver(() => changeIcon()).observe(document, { childList: true, subtree: true })
})()
