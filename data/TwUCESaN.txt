SELECT DISTINCT
post_title
, post_content

/* récupérer la méta 'client' */
,(SELECT meta_value FROM wp_postmeta WHERE wp_postmeta.meta_key = 'client' AND wp_postmeta.post_id = wp_posts.ID) AS "ID client"

/* récupérer la méta 'adresse_email' d'un CPT lié par la méta 'client' */
,(SELECT meta_value FROM mso_postmeta a WHERE a.meta_key = 'adresse_email' AND a.post_id = (SELECT meta_value FROM mso_postmeta b WHERE b.meta_key = 'client' AND b.post_id = mso_posts.ID)) AS "Email client"

FROM mso_posts
WHERE post_type = 'reservation' /* slug du CPT */
ORDER BY
post_title
, post_content
