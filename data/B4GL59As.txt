#
# cliente para cargar datos

import requests
datos = [
    {'nombre':'JOHN', 'cursos':4},
    {'nombre':'HENRY', 'cursos':1},
    {'nombre':'ANNE', 'cursos':6},
    {'nombre':'KEVIN', 'cursos':3},
]

url = "http://localhost:5000/form"

for dato in datos:
    r = requests.post(url, data=dato)
    respuesta = r.content.decode("utf-8")
    if "Datos guardados" in respuesta:
        print(f"Datos del alumno {dato['nombre']} subidos correctamente")
    else:
        print(f"Datos del alumno {dato['nombre']} no subidos")
        print(f"Error {r.status_code}")
