namespace _398
{
	using System;

	class Program
	{
		static void Main(string[] args)
		{
			int n = int.Parse(Console.ReadLine());

			double[] a = SetPrecipitation(n);
			double[] b = SetPrecipitation(n);
			double[] c = SetPrecipitation(n);

			double averageA = AveragePrecipitation(a);
			double averageB = AveragePrecipitation(b);
			double averageC = AveragePrecipitation(c);

			Console.WriteLine(averageA);
			Console.WriteLine(averageB);
			Console.WriteLine(averageC);

			string daysWithExcessPrecipitationA = DaysWithExcessPrecipitation(averageA, a);
			string daysWithExcessPrecipitationB = DaysWithExcessPrecipitation(averageB, b);
			string daysWithExcessPrecipitationC = DaysWithExcessPrecipitation(averageC, c);

			Console.WriteLine(daysWithExcessPrecipitationA);
			Console.WriteLine(daysWithExcessPrecipitationB);
			Console.WriteLine(daysWithExcessPrecipitationC);
		}

		static string DaysWithExcessPrecipitation(double average, double[] arr)
		{
			string s = "";

			for (int i = 0; i < arr.Length; i++)
			{
				if (arr[i] > average)
				{
					s = $"{s}{arr[i]} ";
				}
			}
			return s.TrimEnd();
		}

		static double AveragePrecipitation(double[] arr)
		{
			double sum = 0;

			for (int i = 0; i < arr.Length; i++)
			{
				sum += arr[i];
			}

			return sum / arr.Length;
		}

		static double[] SetPrecipitation(int n)
		{
			double[] result = new double[n];

			for (int i = 0; i < result.Length; i++)
			{
				double input = -1;

				while (input < 0)
				{
					input = double.Parse(Console.ReadLine());
				}

				result[i] = input;
			}

			return result;
		}
	}
}
