#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
 
int f(int x){
    int res = 0;
    if (x == 0){
        return 0;
    }
    while (x % 2 == 0){
        res += 1;
        x /= 2;
    }
    return res;
}
 
int main(){
    int t, a, n;
    cin >> t;
    for (int i = 0; i < t; i++){
        cin >> n;
        int summ = 0;
        for (int j = 0; j < n; j++){
            cin >> a;
            summ += f(a);
        }
        if (summ >= n){
            cout << 0 << "\n";
        }
        else{
            vector <int> b (n);
            for (int j = 1; j <= n; j++){
                b[j - 1] = f(j);
            }
            sort(b.begin(), b.end());
            int j = 0;
            int sch = 0;

            bool tr = true;
            while (tr && j <= n){
                if (n - summ - sch <= 0){
                    cout << j << "\n";
                    tr = false;
                }
                else{
                    j += 1;
                    sch += b[n - j];
                }
            }
            if (tr){
                cout << -1 << "\n";
            }
        }
    }
}
