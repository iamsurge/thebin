import random
randomNumber = random.randint(0,100)


#print(randomNumber) # entry to test the code

print("Startingin a guess game!\n")
user_choice = int(input("Guess the random number between 0 and 100?\n"))

while user_choice != randomNumber:
    user_choice = int(input("Not coreect please try again!\n"))
    if user_choice == randomNumber:
        print("Bravo")
    elif user_choice > randomNumber:
        print("Your number is to high, please try again")
    else:
        print("Your number is to low, please try again")
