#include <stdio.h>
#include <stdint.h>

int getBitsParity(uint64_t x) {
  int parity = 0;
  if (x) {
    do {
      parity ^= 1;
    } while(x &= x - 1);
  }
  return parity;
}

int main() {
  uint64_t x;
  printf("x="); scanf("%llu", &x);
  printf("bits parity: %d\n", getBitsParity(x));
  return 0;
}
