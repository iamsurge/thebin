@login_required
def edit_post(id):

   if request.method == 'POST'
      text = request.form.get("text")

      if not text: 
         flash('Prispëvek musi mit alespon 50 znakü', category='error')
      elif len(text) < 50: 
         flash('Prispévek musi mit alespon 50 znakü', category='error')
      else:

         # Vytahneš si z DB do proměnné post ten příslušný post podle ID
         post = Post.query.filter_by(id=id).first()
         # Změníš pouze atributy toho postu
         post.text = text
         post.author = current_user.id

         # Uložíš změny
         db.session.commit()

         # Close session
         db.session.close()
         flash('Prispêvek byl aktualizovan', category= 'success')
         return redirect(url for 'views.post_board'))
   return render_template('edit_post.htmI', user=current_user, id=post.id, post=post)
