#include <stdio.h>

int rekurzija(int x) {
    int broj_na_deliteli = 0;
    for(int i = 1; i <= x; i++) {
        if(x % i == 0) {
            broj_na_deliteli++;
        }
    }
    if(broj_na_deliteli >= 5) {
        return x;
    }
    return rekurzija(x + 1);
}
int main() {
    int n;
    scanf("%d", &n);
  
    printf("%d\n", rekurzija(n + 1));
    return 0;
}
