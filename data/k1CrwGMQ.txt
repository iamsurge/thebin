turtle.up()
turtle.turnRight()

local function getTurtle()
    for i = 2, 16 do
        turtle.forward()
        turtle.select(i)
        turtle.digDown()
    end
end

local function goBack()
    for i = 1, 15 do
        turtle.back()
    end
    turtle.down()
    turtle.turnLeft()
end

getTurtle()
goBack()
