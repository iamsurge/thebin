import java.util.Scanner;
public class Jurnal03 {
	private static void jurnal03(int n, char batangAwal, char batangPindah, char batangAux){
		if( n == 0){
		    return;
		}
		jurnal03(n - 1, batangAwal, batangAux, batangPindah);
		System.out.println("Cakram " + n + " dipindahkan dari " + batangAwal + " ke " + batangPindah );
		jurnal03(n - 1, batangAux, batangPindah, batangAwal);
    	}

	public static void main(final String[] args) {
		Scanner input= new Scanner(System.in);
		int banyakCakram;
		banyakCakram = input.nextInt();
		jurnal03(banyakCakram, 'A', 'C', 'B');
	}
}
