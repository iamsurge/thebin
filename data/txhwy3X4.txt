#include <stdio.h>

int proverka(int x) {
    int posledna_cifra = x % 10;
    int prva_cifra;
    
    while(x > 0) {
        prva_cifra = x % 10;
        x /= 10;
    }
    if(prva_cifra < posledna_cifra) {
        return 1;
    }
    else {
        return 0;
    }
}
// 154 15 1
int rekurzija(int a, int b) {
    if(a > b) {
        return 0;
    }
    return rekurzija(a + 1, b) + proverka(a);
}
int main() {
    int a, b;
    scanf("%d%d", &a, &b);
    
    printf("%d\n", rekurzija(a, b));
    return 0;
}
