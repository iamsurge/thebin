add_filter( 'fl_theme_builder_current_page_layouts' , 'header_footer_only_on_pages' );

function header_footer_only_on_pages( $layouts ) {

    if ( !is_singular( 'page' ) ) {

        if ( isset( $layouts[ 'header' ] ) ) unset( $layouts[ 'header' ] );
        if ( isset( $layouts[ 'footer' ] ) ) unset( $layouts[ 'footer' ] );

    }

    return $layouts;
}
