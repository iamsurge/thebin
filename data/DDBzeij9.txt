#include <iostream>
using namespace std;
int main(){
    int n, m;
    cin>>n>>m;
    if (n*(n+1)/2<m){
        cout<<0<<endl;
        return 0;
    }
    int l=0;
    int r=m;
    while (r-l>1){
        int k=(r+l)/2;
        if (k*n<=m+(k*(k-1))/2){
            l=k;
        }
        else{
            r=k;
        }
    }
    int x=r;
    if (l*n>=m+(l*(l-1))/2){
        x=l;
    }
    int sum=m;
    for (int i=n; i>n-x+1; i--){
        cout<<i<<"\n";
        sum-=i;
    }
    cout<<sum<<"\n";
    return 0;
}
