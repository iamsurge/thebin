SELECT p.NAZWA_PRZEDMIOT, count(*) as "Liczba egzaminow"
FROM Egzaminy e 
inner join Przedmioty p
on e.ID_PRZEDMIOT = p. ID_PRZEDMIOT
GROUP BY p.NAZWA_PRZEDMIOT
Having count(*) = (select max(count(*)) FROM Egzaminy e 
inner join Przedmioty p
on e.ID_PRZEDMIOT = p. ID_PRZEDMIOT
GROUP BY p.NAZWA_PRZEDMIOT)
ORDER BY Count(*) DESC
