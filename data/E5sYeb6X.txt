#include <stdio.h>
#include <stdlib.h>

int main() {
    int X;
    scanf("%d", &X);
    
    int broj;
    int deliteli = 0;
    int brojac = 0;
    while(scanf("%d", &broj)) {
        brojac++;
        if(X % broj == 0) {
            deliteli++;
        }
        if(deliteli == 2) {
            break;
        }
    }
    printf("%d \n", brojac);
    return 0;
}
