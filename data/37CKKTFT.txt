/*
За дадена низа од случајни броеви кои се внесуваат од стандарден влез, да се направи преместување на сите нули на крајот на низата. На стандарден излез да се испечати трансформираната низа.
*/
import java.util.Scanner;
import java.io.*;

public class PushZero {
    static void pushZerosToend(int arr[], int n) {
        int count = 0;

        for (int i = 0; i < n; i++) {
            if (arr[i] != 0)
                arr[count++] = arr[i];
        }

        while (count < n) {
            arr[count++] = 0;
        }
    }

    public static void main(String[] arg) {
        int arr[] = new int[100];
        int n;

        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();

        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        pushZerosToend(arr, n);

        System.out.println("Transformiranata niza e:");
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}
