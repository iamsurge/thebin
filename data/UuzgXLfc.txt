--PERIPHERALS
br = peripheral.wrap("top")
y = peripheral.getMethods("top")
 
--CONSTANTS
speed = 0
sptosu = 32
sutofe = 0.02197265625
 
 
--FUNCTIONS
function readValues()
    data = br.getBlockData()
    speed = data.Speed
    for k,v in pairs(data.Network) do
        print(k,v)
    end
end
 
function mapValues()
    if speed < 0 then
        speed = speed * -1
    end
    su_drawn = speed * sptosu
    generated = su_drawn * sutofe
    roundedg = math.floor(generated)
end
 
function printValues()
    print("DRAWN STRESS UNITS: ",su_drawn,"SU")
    print("GENERATION RATE: ",roundedg,"FE/T")
end
 
--PROGRAM LOOP
while true do
    readValues()
    mapValues()
    printValues()
    sleep(1)
    term.clear()
    term.setCursorPos(1,1)
end
