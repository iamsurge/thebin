import java.util.*;

class Main
{
    public static void bubbleSortAsc(int a[])
    {
        int size = a.length;
        for (int i = 0; i < size - 1; i++)
            for (int j = 0; j < (size - i - 1); j++)
                    if (a[j] > a[j+1])
                    {
                        int temp = a[j];
                        a[j] = a[j+1];
                        a[j+1] = temp;
                    }
    }
    public static void bubbleSortDesc(int a[])
    {
        int size = a.length;
        for (int i = 0; i < size - 1; i++)
            for (int j = 0; j < (size - i - 1); j++)
                    if (a[j] < a[j+1])
                    {
                        int temp = a[j];
                        a[j] = a[j+1];
                        a[j+1] = temp;
                    }
    }
    
    public static void main(String[] args)
    {
        Scanner s = new Scanner(System.in);
        Map<Integer,String> stud = new HashMap<Integer,String>();
        int n = s.nextInt();
        for (int i = 0; i < n; i++)
            stud.put(s.nextInt(),s.next());
        int rollNums[] = new int[n];
        int idx = 0;
        for (int i: stud.keySet())
        {
            rollNums[idx] = i;
            idx++;
        }
        
        //Ascending
        bubbleSortAsc(rollNums);
        System.out.println("Ascending Order");
        for (int i = 0; i < n; i++)
        {
            System.out.println(rollNums[i]);
            System.out.println(stud.get(rollNums[i]));
        }
        
        //Descending
        bubbleSortDesc(rollNums);
        System.out.println("Descending Order");
        for (int i = 0; i < n; i++)
        {
            System.out.println(rollNums[i]);
            System.out.println(stud.get(rollNums[i]));
        }
    }
}
