REM Program Perulangan untuk mencetak pola segita sama sisi dengan asterisk
dim i as integer 'Deklarasi variabel i dengan tipe data integer'
dim j as integer 'Deklarasi variabel j dengan tipe data integer'
dim k as integer 'Deklarasi variabel k dengan tipe data integer'
dim baris as integer 'Deklarasi variabel baris dengan tipe data integer'

input baris 'Melakukan Input nilai ke variabel baris'
PRINT "Berapa baris? ";baris

for i = 1 to baris  'Menggunakan perulangan for terhadap iterator i dengan nilai awal 1 dan nilai akhir variabel baris dengan nilai yang disimpannya'
    for k = baris to i step -1 'Menggunakan perulangan for terhadap iterator k dengan nilai awal variabel baris dengan nilai yang disimpannya dan nilai akhir i dan STEP setiap iterasinya adalah -1'
        PRINT " "; 'Mencetak spasi'
    NEXT k
        for j = 1 to i ' Menggunakan perulangan for terhadap iterator j dengan nilai awal 1 dan nilai akhir i'
            PRINT "* "; 'Cetak karakter asterisk'
        NEXT j
PRINT " " 'Mencetak spasi'
NEXT i
END
