local bucket = 1
local lava_egg = 2

local function move_up()
    turtle.up()
    turtle.up()
    turtle.forward()
end

local function move_down()
    turtle.back()
    turtle.down()
    turtle.down()
end

local function get_lava()
    turtle.select(lava_egg)
    turtle.placeDown()
    turtle.select(bucket)
    turtle.placeDown()
end

local function put_lava()
    turtle.select(bucket)
    turtle.dropDown()
end

local function loop()
    while true do
        get_lava()
        move_up()
        put_lava()
        move_down()
    end
end

loop()
