class reDateEntry(cDateEntry):
	def __init__(self, master, **kwargs):
		super(reDateEntry, self).__init__(master)
		self.stvar = StringVar() ##defined here
		self.regex = ""
		kw_wid, kw_pak, kw_style = pack_opts(**kwargs)
		if 'text' in kw_wid.keys():
			self.stvar.set(kw_wid['text']) ##set to the value of the input text
			del kw_wid['text']
		if 're' in kw_wid.keys():
			self.regex = kw_wid['re']
			del kw_wid['re']
		else:
			self.regex = '.*'
		kw_wid['textvariable'] = self.stvar ##used here 
		self.bg = self.cget('background')
		self.fg = self.cget('foreground')
		if len(kw_pak) != 0:
			self.pack(kw_pak)
	def insert(self, pos, end, date="01/01/1970"):
		super().insert(date)
		self._isvalid()
	def get(self):
		return self.stvar.get(), self._isvalid() ##used here
	def _isvalid(self):
		instr = self.stvar.get() ##and used here
		match = re.match(self.regex, instr)
		if not match:
			self.configure(bg="red", fg="white")
		else:
			self.configure(bg=self.bg, fg=self.fg)
		return match
