using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

enum Situation { fire, theft}

namespace Program
{
    delegate void MyDelegate(Cafe cafe);

    class Cafe
    {
        public string CafeName { get; set; }
        public int NumberOfWorkers { get; set; }
        public string Address { get; set; }

        public event MyDelegate Fire;
        public event MyDelegate Theft;

        public void GetSituation (Situation situation)
        {
            if(situation == Situation.fire)
            {
                Fire?.Invoke(this);
            }
            else if(situation == Situation.theft)
            {
                Theft?.Invoke(this);
            }
        }

        public Cafe (string name, string address, int numberOfWorkers)
        {
            CafeName = name;
            NumberOfWorkers = numberOfWorkers;
            Address = address;
        }


    }
}
