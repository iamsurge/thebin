from spike import PrimeHub
from spike.control import wait_for_seconds
hub = PrimeHub()

for i in range(5):
    hub.status_light.on('blue')
    wait_for_seconds(1)
    hub.status_light.on('green')
    wait_for_seconds(1)
    hub.status_light.on('azure')
    wait_for_seconds(1)

hub.status_light.off()
