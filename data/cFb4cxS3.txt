// ==UserScript==
// @name         Rx Classic/Pro Swap Buttons
// @namespace    http://tampermonkey.net/
// @version      0.1
// @description  Button to switch between the Classic and Pro Rx views in WELL OSCAR
// @author       Gyula Voros
// @match        */oscar/oscarRx/choosePatient.do*
// @match        */kaiemr/app/components/rx/*
// @grant        none
// ==/UserScript==
// 0.1 Initial Version

let currentView = "";

const PROPath = "/kaiemr/app/components/rx/";
const ClassicPath = "/oscar/oscarRx/choosePatient.do";

if (window.location.pathname==PROPath) {
    currentView = "PRO";
} else if (window.location.pathname==ClassicPath) {
    currentView = "Classic";
}


function swapURL () {
    if (currentView=="PRO") { window.location.pathname=ClassicPath; }
    if (currentView=="Classic") { window.location.pathname=PROPath; }
};


let swapButton = document.createElement('input');
swapButton.type = 'button';
swapButton.value = 'Swap Classic/PRO';
swapButton.onclick = swapURL;
swapButton.setAttribute('style', 'font-size:16px;position:absolute;top:' + (10) + 'px;left:100px;background-color: lightblue;');
document.body.appendChild(swapButton);
