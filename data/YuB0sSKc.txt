#include <iostream>
#include <string>
#include <array>
 
using namespace std;
 
int main(){
    int min, max;
    cout << "Enter initial number: " << endl;
    cin >> min;
    cout << "Enter final number: " <<  endl;
    cin >> max;
    
    string evens;
    string odds;
    
    int test;
    
    if(min > max){
        cout << "Initial number cannot be greater than final number!";
    }else{
        for(int i = min; i <= max; i++){
            test = i % 2;
            
            if(test == 0){
                evens = evens + to_string(i) + " ";
            }else{
                odds = odds + to_string(i) + " ";
            }
        }
        cout << endl;
        cout << "Even Numbers: " << evens << endl;
        cout << "Odds Numbers: " << odds << endl;
    }
}
