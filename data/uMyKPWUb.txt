function solve(input) {
    let num = Number(input);
    
    let result = ' '
    if (num < 100) {
        result = "Less than 100"
    } else if (num >= 100 && num <= 200){
        result = "Between 100 and 200"
    } else {
        result = "Greater than 200"
    }
    console.log(result);
}
