
#include <iostream>
using namespace std;

int main()
{
    int const n = 23;
    int i;
    int x = 0, j = 0;
    float a[n];
    int p = 999999999;
    srand(time(0));
    for (i = 0; i < n; i++) {
        a[i] = (float)rand() / ((float)RAND_MAX / 10);
        cout << a[i] << " ";
    }
    cout << endl;
    for (j = 0; j < 4; j++) {

        for (i = 0; i < n; i++) {

            if (a[i] < p) {
                x = i;
                p = a[i];
            }

        }
        cout << x << " "; //4 меньших числа
        a[x] = 9999;
        x = 0;
        p = 99999;
    }
}
