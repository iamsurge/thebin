#include <stdio.h>
#include <cs50.h>

bool valid_triangle(float a, float b, float c)
{
    if ((a && b && c) > 0 && a + b > c && a + c > b && b + c > a)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

int main(void)
{
    float a = get_float("Enter length of side A: ");
    float b = get_float("Enter length of side B: ");
    float c = get_float("Enter length of side C: ");
    if (valid_triangle(a, b, c) != 0)
    {
        printf("Valid triangle!\n");
    }
    else
    {
        printf("Invalid triangle!\n");
    }
}
