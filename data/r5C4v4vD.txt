#include <stdio.h>

int main() {
    int i;
    scanf("%d", &i);
    int isPositive;
    while (scanf("%d", &i) && i != -9999) {

        if (i < 0) {
            isPositive = 0;
        } else {
            isPositive = 1;
        }
    }
    if (isPositive) {
        printf("YES");
    } else {
        printf("NO");
    }
    return 0;
}
