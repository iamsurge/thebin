import tkinter as tk

def arrojar_dado():
    import random
    caja_resultado.delete(0,tk.END)
    valor = random.randint(1,6)
    caja_resultado.insert(0,valor)
    
    
    
ventana = tk.Tk()
ventana.title("DADO 2.0")
ventana.config(width=250, height=250)

boton = tk.Button(text="arrojar el dado", command=arrojar_dado)
boton.place(x=70, y=50, width=120, height=50)
etiqueta = tk.Label(text="Valor")
etiqueta.place(x=80, y=130)
caja_resultado = tk.Entry()
caja_resultado.place(x=80, y=170, width=70, height=25)
ventana.mainloop()
