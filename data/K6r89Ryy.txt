import logging
from JoycontrolPlugin import JoycontrolPlugin

logger = logging.getLogger(__name__)

class Zoushoku(JoycontrolPlugin):
	async def run(self):
		await self.button_push('left') # Aボタンを押す
		await self.wait(1.0) 		# 1.0秒間待つ

		await self.button_push('a')
		await self.wait(1.0)

		await self.button_push('up')
		await self.wait(1.0)
		await self.button_push('up')
		await self.wait(1.0)

		await self.button_push('a')	# go to rideform
		await self.wait(1.0)
		
		await self.button_push('a')	# question rideform
		await self.wait(1.0)

		await self.button_push('a')	# hai
		await self.wait(3.0)

		await self.button_push('a')	# ridform ni henkashita
		await self.wait(1.0)

		await self.button_push('right') # main menu ni idou
		await self.wait(1.0)

		await self.button_push('down')	# cursol box
		await self.wait(1.0)
		await self.button_push('down')
		await self.wait(1.0)

		await self.button_push('a')	# open box
		await self.wait(5.0)

		await self.button_push('x')	# change battle team
		await self.wait(1.0)
		await self.button_push('x')
		await self.wait(1.0)

		await self.button_push('l')	# change corideon box
		await self.wait(1.0)

		await self.button_push('a')	# open corideon menu
		await self.wait(1.0)

		await self.button_push('down') # down cursol to mochimono shimau
		await self.wait(1.0)
		await self.button_push('down')
		await self.wait(1.0)
		await self.button_push('down')
		await self.wait(1.0)

		await self.button_push('a')	# mochimono wo shimau
		await self.wait(1.0)

		await self.button_push('b')	# close box
		await self.wait(3.0)
