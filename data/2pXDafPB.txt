package ppa1;

import java.util.Scanner;

public class ppa1u05b2 {
	
	static Scanner sc = new Scanner(System.in);
	public static void main(String[] args) {
		
		System.out.println("x1: ");
		double x1 = sc.nextDouble();
		System.out.println("x2: ");
		double x2 = sc.nextDouble();
		System.out.println("Xs: ");
		double xs = sc.nextDouble();
		
		System.out.println("y1: ");
		double y1 = sc.nextDouble();
		System.out.println("y2: ");
		double y2 = sc.nextDouble();
		System.out.println("Ys: ");
		double ys = sc.nextDouble();
		
		System.out.println("Ts: ");
		double ts = sc.nextDouble();
		
		double vysledekX = x1;
		double vysledekY = y1;
		double vysledekT = 0;
		
		System.out.println("x, y, z, t");
		
		for(int i = 0; i < ts; i++) {
					
			for(int j = 0; j < ys; j++) {
						
				for(int k = 0; k < xs; k++) {
					double VysledekZ = Math.sin((Math.sqrt((vysledekY * vysledekY) + (vysledekX * vysledekX)) - ((Math.PI * vysledekT) *2.0)));
					System.out.format("%f, %f, %f, %f %n", vysledekX,vysledekY, VysledekZ, vysledekT);
					
					vysledekX += (Math.abs((x2 - x1) / (xs - 1.0)));
				}
				
				vysledekY += (Math.abs((y2 - y1) / (ys - 1.0)));
				vysledekX = x1;
			}
			
			vysledekT += (1.0 / ts);
			vysledekY = y1;
		}
	}
}
