void solve() {
    int n = (1 << 4), i, ans = 0;
    for (i = 0; i < n; i++) {
        bool x1, x2, x3, x4;
        x1 = n & (1 << 3);
        x2 = n & (1 << 2);
        x3 = n & (1 << 1);
        x4 = n & (1 << 0);
        bool f1, f2, f3, f4;
        f1 = (x1 xor (!x2));
        f2 = (x2 xor (!x3));
        f3 = (x3 xor (!x4));
        f4 = (x4 xor (!x1));
        bool s1, s2;
        s1 = ((!f1) && f2);
        s2 = ((!f3) && f4);
        if (s1 && s2) {
            ans++;
            cout << x1 << " " << x2 << " " << x3 << " " << x4 << endl;
        }
        //cout << i << endl;
    }
    cout << ans << endl;
}
