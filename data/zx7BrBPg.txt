
import java.util.Scanner;
public class Jurnal02 {
	private static int jurnal02(int x, int y){
		if(x > y){
		    if(x % y == 0){
			return y;
		    } else{
			return jurnal02(y, x%y);
		    }
		}else{
		    if(y % x == 0){
			return x;
		    }else {
			return jurnal02(x, y % x);
		    }
		}
    	}

	public static void main(final String[] args) {
		Scanner input= new Scanner(System.in);
		int fpb1, fpb2;
		fpb1 = input.nextInt();
		fpb2 = input.nextInt();
		System.out.println(jurnal02(fpb1, fpb2));
	}
}
