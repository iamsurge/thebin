import java.util.Scanner;

public class Main
{
    public static void main(String[] args)
    {
        Scanner scan = new Scanner(System.in);

        int numeroDeRetangulos;
        float auxX, auxY;

        System.out.print("Quantos retângulos serão calculados: ");
        numeroDeRetangulos = scan.nextInt();

        Retangulo[] oblong = new Retangulo[numeroDeRetangulos];
        Ponto ponto1;
        Ponto ponto2;

        for (int i = 0; i < oblong.length; i++)
        {
            System.out.println("Retângulo " + (i+1));

            System.out.print("Ponto 1 (x): ");
            auxX = scan.nextFloat();

            System.out.print("Ponto 1 (y): ");
            auxY = scan.nextFloat();

            ponto1 = new Ponto(auxX, auxY);

            System.out.print("Ponto 2 (x): ");
            auxX = scan.nextFloat();

            System.out.print("Ponto 2 (y): ");
            auxY = scan.nextFloat();

            ponto2 = new Ponto(auxX, auxY);

            oblong[i] = new Retangulo(ponto1, ponto2);
            System.out.println("Área formada por (" + String.format("%.2f",ponto1.getX()) + ',' + String.format("%.2f",ponto1.getY()) + ") e (" +
                               String.format("%.2f",ponto2.getX()) + ',' + String.format("%.2f",ponto2.getY()) + ") é " + String.format("%.2f", oblong[i].CalcArea()));

        }
    }
}
