import math
height = int(input())
width = int(input())
percent_not_paint = int(input())

total_surface_area = (height * width) * 4
area_for_paint = total_surface_area - (total_surface_area * (percent_not_paint / 100))
meters_can_paint = 0
input_line = input()
while input_line != "Tired!":
    liters = int(input_line)
    meters_can_paint = liters
    area_for_paint = math.ceil(area_for_paint - meters_can_paint)
    if area_for_paint < meters_can_paint:
        diff = abs(area_for_paint - meters_can_paint)
        print(f"All walls are painted and you have {diff} l paint left!")
        break

    input_line = input()

if input_line == "Tired!":
    print(f"{area_for_paint} quadratic m left.")

if meters_can_paint == 0:
    print("All walls are painted! Great job, Pesho!")
