using System;
 
namespace ConsoleApp1
{
    internal class Program
    {
        static void Main(string[] args)
        {
            bool goinghome = false;
 
            int countSteps = 0;
 
 
            while (countSteps < 10000)
            {
                string steps = Console.ReadLine();
                if (steps == "Going home")
                {
                    goinghome = true;
                    continue;
                }
 
 
                int stepsint = int.Parse(steps);
 
 
                countSteps += stepsint;
                if (countSteps >= 10000)
                {
 
                    Console.WriteLine($"Goal reached! Good job!");
                    Console.WriteLine($"{Math.Abs(10000 - countSteps)} steps over the goal!");
                }
                else if (goinghome == true)
                {
                    Console.WriteLine($"{10000 - countSteps} more steps to reach goal.");
                  break;
 
                }
 
            }
 
 
        }
    }
}
