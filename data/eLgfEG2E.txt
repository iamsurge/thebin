public class BST {
    private Node root;

    public Node findNode(int val) {
        return findNode(val, root);
    }

    private Node findNode(int val, Node root) {
        if (root == null) return null;
        if (val > root.val) {
            return findNode(val, root.right);
        } else {
            return findNode(val, root.left);
        }
    }
}
