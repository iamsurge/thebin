package FirstStepsInCoding.MoreExersise;

import java.util.Scanner;

public class UniquePINCodes_01 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        //четем данни от конзолата
        int firstNumUpLevel=Integer.parseInt(scanner.nextLine());
        int secondNumUpLevel=Integer.parseInt(scanner.nextLine());
        int thirdNumUpLevel=Integer.parseInt(scanner.nextLine());


        for (int i = 2; i <= firstNumUpLevel ; i+=2) {

            for (int j = 2; j <= secondNumUpLevel ; j++) {
                boolean isPrime = j == 2 || j == 3 || j == 5 || j == 7;
                if (isPrime){
                    for (int k = 2; k <= thirdNumUpLevel; k+=2) {
                        System.out.printf("%d %d% d%n", i, j, k);

                    }
                }

            }

        }




    }
}
