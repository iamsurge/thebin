        }
        if (n <= 2)
        {
            cout << "YES" << endl;
        }
        else if (a[0] == a[n - 1])
        {
            cout << "YES" << endl;
        }
        else
        {
            bool flag = false;
            for (int i = 0; i < n - 1; i++)
            {
                if (a[i] == a[0] && a[i + 1] == a[n - 1])
                {
                    flag = true;
                    break;
                }
            }
            if (flag)
            {
                cout << "YES" << endl;
            }
            else
            {
                cout << "NO" << endl;
            }
        }
    }
