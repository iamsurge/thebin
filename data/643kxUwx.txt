#include <iostream>
#include <vector>
using namespace std;


class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        for(int i = 0; i< nums.size(); ++i){
            for(int k = i+1; k< nums.size(); ++k){
                if(nums[k]+nums[i]==target){
                    vector<int> pair1{i,k};
                    return pair1;
                }
            }
        }
        return vector<int>(0);
    }
};
