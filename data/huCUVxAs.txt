import random

# Generación de 42 precios aleatorios.
precios = []
for _ in range(0, 42): ## EN CLASE NO SE ENSEÑO EL USO DE _ ##
    precios.append(random.randint(1000, 10000))

# Tomar los 12 más baratos para sumarlos.
total = []
for _ in range(0, 12): ## EN CLASE NO SE ENSEÑO EL USO DE _ ##
    m = min(precios)
    total.append(m)
    precios.remove(m)

print(f"Valor a pagar: {sum(total)}")
