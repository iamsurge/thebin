#!/bin/sh

sudo apt-get update
sudo apt-get -y dist-upgrade

# Required Packages for build debian package
sudo apt-get -y install build-essential binutils lintian debhelper dh-make devscripts

# Required Package for build libskycharge binary
sudo apt-get -y install \
	libelf-dev uuid-dev libserialport-dev libgps-dev libzmq3-dev \
	libczmq-dev libavahi-client-dev libmicrohttpd-dev libmongoc-dev \
	bison flex

git clone https://github.com/Skycharge/libskycharge.git

# Build debian packages
cd libskycharge
dpkg-buildpackage

cd ..
ls -l
