namespace _395
{
	using System;

	class Temperature
	{
		static void Main(string[] args)
		{
			double[] avetageTempDaysF = new double[31];
			double[] avetageTempDaysC = new double[31];

			for (int i = 0; i < avetageTempDaysF.Length; i++)
			{
				bool isCorrect = false;
				double avetageTempDay = 0;

				while (!isCorrect)
				{
					avetageTempDay = double.Parse(Console.ReadLine());

					if (avetageTempDay >= -100 && avetageTempDay <= 100)
					{
						isCorrect = true;
					}
				}

				avetageTempDaysF[i] = avetageTempDay;
			}

			for (int i = 0; i < avetageTempDaysF.Length; i++)
			{
				avetageTempDaysC[i] = CelF(avetageTempDaysF[i]);
			}

			double maxC = MaxTemperature(avetageTempDaysC);
			double maxF = MaxTemperature(avetageTempDaysF);

			Console.WriteLine(maxF);
			Console.WriteLine(maxC);

			for (int i = 0; i < avetageTempDaysC.Length; i++)
			{
				if (avetageTempDaysC[i] < -10)
				{
					Console.WriteLine(avetageTempDaysC[i]);
				}
			}
		}

		static double MaxTemperature(double[] temperatureArray)
		{
			double maxTemp = temperatureArray[0];

			for (int i = 1; i < temperatureArray.Length; i++)
			{
				if (maxTemp < temperatureArray[i])
				{
					maxTemp = temperatureArray[i];
				}
			}

			return maxTemp;
		}

		static double CelF(double f)
		{
			return 5.0 * (f - 32) / 9;
		}
	}
}
