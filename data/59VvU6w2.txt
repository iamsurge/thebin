#include <stdio.h>

#define TRUE 1
#define FALSE 0

int main()
{
    _Bool a = TRUE;
    
    if (TRUE == a)
    {
        printf("TRUE\r\n");
    }
    else if (FALSE == a)
    {
        printf("FALSE\r\n");
    }
    else
    {
        printf("DIFF\r\n");
    }


    return 0;
}
