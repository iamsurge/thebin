def is_prime(n):
	if n == 1:
		return False
	for i in range(2, n//2):
		if n % i == 0:
			return False
	return True

def good_keys(N):
	result = 0
	for i in range (1, N+1):
		for j in range(1, N+1):
			for k in range(1, N+1):
				if is_prime(i) and is_prime(j) and is_prime(k) and (i ^ j ^ k) == 0:
					result += 1
	return result

N = input()
out_ = good_keys(N)
print(out_)
