#include <stdio.h>
int main()
{
	float minutes, payment, extra;
	scanf("%f %f %f", &minutes, &payment, &extra);
	if (extra <= payment / 499.f) {
		printf("extra minutes must be more expensive");
		return 0;
	}
	if (minutes < 499.f)
		printf("%f", payment);
	else
		printf("%f", payment + (minutes - 499) * extra);
	return 0;
}
