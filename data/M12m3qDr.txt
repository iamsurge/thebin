server {

listen                  443 ssl http2;
    listen                  [::]:443 ssl http2;
    server_name             noa.monkeypower.co.uk;
    root                    /srv/http//noa;

    # SSL
    ssl_certificate         /etc/letsencrypt/live/monkeypower.co.uk/fullchain.pem;
    ssl_certificate_key     /etc/letsencrypt/live/monkeypower.co.uk/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/monkeypower.co.uk/chain.pem;

    # security
    include                 nginxconfig.io/security.conf;

    # logging
    access_log              /var/log/nginx/noa.monkeypower.co.uk.access.log;
    error_log               /var/log/nginx/noa.monkeypower.co.uk.error.log warn;

    # index.html
    index                   index.html;

    location /tools/ {
        proxy_pass          http://localhost:8080/tools/;
        proxy_set_header    X-Real-IP  $remote_addr;
        proxy_set_header    X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header    X-Forwarded-Proto $scheme;
        proxy_set_header    Host $host;
        proxy_pass_header   Content-Type;
    }    

# additional config
    include nginxconfig.io/general.conf;
}
