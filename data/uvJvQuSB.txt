// Программа 1
#include <iostream>

int main()
{
    int goods;    // количество товаров
    double price; // цена за один товар
    int discount; // размер скидки (в процентах)

    std::cout << "Введите количество товаров: ";
    std::cin >> goods;
    std::cout << "Введите цену за один товар: ";
    std::cin >> price;
    std::cout << "Введите размер скидки (в процентах): ";
    std::cin >> discount;

    // Рассчитаем итоговую стоимость
    double final_price = goods * price * (100-discount)/100.0;
    std::cout << "Итоговая стоимость: " << final_price << std::endl;

    return 0;
}
