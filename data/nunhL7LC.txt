{
    // take out the item from contents or general inventory
    let removedItem : Item = this._contents.splice( i )[ 0 ];
    let removedEquip : Item = null;

    // depending on the type
    switch( slot )
    {
        case ItemTypes.Head:
        {
            if( this._equipped.Head != null )
            {
                // save the item we're going to replace
                removedEquip = this._equipped.Head;
            }

            // put the inventory item into the slot
            this._equipped.Head = removedItem;
        }
        break;
            
    }  // end switch

    // re-add it back to inventory if it's there
    if( removedEquip != null )
        this._contents.push( removedEquip );

    return;
}
