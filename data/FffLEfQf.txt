// z04
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int digit(char d)
{
    if (('1' <= d) && (d <= '9')) return d - '0';
        return d - 'a' + 10; //ako ima a i iskame da stane 10, zatva -'a' + 10;
}

string s; // predpochitat se globalni promenlivi za da ne se kopira

int p_th(int p)
{
    int b = digit(s[0]);
    for (int i = 1; i < s.length(); i++)
        b = digit(s[i]) + b*p;
    return b;
}

int main()
{
    while (cin >> s)
    {
        sort(s.begin(), s.end());
  //      reverse(s.begin(), s.end());
   //     cout << s << endl;
        int minp = digit(s[s.length() - 1]) + 1;
   //     cout << "minp = " << minp << endl;
        for (int p = 36; p >= minp; p--) cout << p_th(p) << " ";
        cout << endl;
    }
    return 0;
}
