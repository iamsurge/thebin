/*In un supermercato c’è la promozione 3x2. Se di un prodotto sono acquistati tre pezzi si deve pagare
l’importo relativo a due pezzi. Per le quantità inferiori a tre il prezzo rimane invariato.*/

public class Main {


    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);


        String nomeProdotto1 = "scatola_biscotti";
        double prezzoProdotto1 = 2.50;
        String nomeProdotto2 = "latte";
        double prezzoProdotto2 = 1.50;
        String nomeProdotto3 = "coca cola";
        double prezzoProdotto3 = 3.50;


        System.out.println("Gli articoli disponibili sono i seguenti: ");
        System.out.println("Nome prodotto: " +nomeProdotto1 + " "+"Prezzo: "+prezzoProdotto1 + "cod. prodotto: 1");
        System.out.println("Nome prodotto: " +nomeProdotto2 + " "+"Prezzo: "+prezzoProdotto2 + "cod. prodotto: 2");
        System.out.println("Nome prodotto: " +nomeProdotto3 + " "+"Prezzo: "+prezzoProdotto3 + "cod. prodotto: 3");


        int input = 0;
        int numero_pezzi_acquistati_prodotto_1 = 0;
        int numero_pezzi_acquistati_prodotto_2 = 0;
        int numero_pezzi_acquistati_prodotto_3 = 0;

        double totale = 0;

        System.out.println("Inserisci il codice del prodotto che vuoi acquistare. Per interrompere scrivi \"stop\" ");


        do{
            System.out.println("codice prodotto: ");
            input = sc.nextInt();

            switch (input){
                case 1: {
                    numero_pezzi_acquistati_prodotto_1++;
                    break;
                }
                case 2: {
                    numero_pezzi_acquistati_prodotto_2++;
                    break;
                }
                case 3: {
                    numero_pezzi_acquistati_prodotto_3++;
                    break;
                }
                default:
                    System.out.println("codice prodotto errato!");
            }
        } while(input != 0);

        if (numero_pezzi_acquistati_prodotto_1 >=3 ){
            totale = totale + (numero_pezzi_acquistati_prodotto_1 - 1) * prezzoProdotto1;
        } else {
            totale = totale + numero_pezzi_acquistati_prodotto_1 * prezzoProdotto1;
        }

        if (numero_pezzi_acquistati_prodotto_2 >=3 ){
            totale = totale + (numero_pezzi_acquistati_prodotto_2 - 1) * prezzoProdotto2;
        } else{
            totale = totale + numero_pezzi_acquistati_prodotto_2 * prezzoProdotto2;
        }

        if (numero_pezzi_acquistati_prodotto_3 >=3 ){
            totale = totale + (numero_pezzi_acquistati_prodotto_3 - 1) * prezzoProdotto2;
        } else {
            totale = totale + numero_pezzi_acquistati_prodotto_3 * prezzoProdotto3;
        }

        System.out.println("Hai acquistato: \n" + numero_pezzi_acquistati_prodotto_1 + " " + nomeProdotto1 + "\n" + numero_pezzi_acquistati_prodotto_2 + " " + nomeProdotto2 + "\n" + numero_pezzi_acquistati_prodotto_3 + " " + nomeProdotto3);
        System.out.println("Totale da pagare: "+ totale + " Euro");
        
    }
}
