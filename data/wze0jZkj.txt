SELECT
  psp.userID,
  pspID,
  userIsProPlan,
  pspPassword,
  pspURLKey,
  pspCustomDomain


FROM
  `uptimerobot-001.ur.psp` psp
LEFT JOIN
  `uptimerobot-001.ur.users` u 

ON u.userID = psp.userID

WHERE
pspCustomDomain <> ''
AND pspPassword IS NULL

ORDER BY 
  userID,
  userIsProPlan
