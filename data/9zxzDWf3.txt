void cs(vector <vector<char>>& vec, int k, int ind) {
    vector <int> c(k + 1 - 93);
    vector <char> b(vec.size() + 1);

    for (int i = 0; i < k + 1; ++i)
        c[i] = 0;

    for (int i = 0; i < vec.size(); ++i)
        c[vec[i]] = c[vec[i]] + 1;

    for (int i = 1; i < k + 1; ++i)
        c[i] += c[i - 1];

    for (int i = vec.size(); i > 0; --i) {
        b[c[vec[i - 1]] - 1] = vec[i - 1];
        c[vec[i - 1]]--;
    }
    for (int i = 0; i < vec.size(); ++i)
        vec[i] = b[i];
}
