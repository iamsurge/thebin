#! /bin/bash
# dependency : fortune , cowsay

whatsup(){
        local comic count pick
        comic=($(cowsay -l))
        count=${#comic[@]}
        pick="${comic[$((RANDOM%count-1))]}"
        echo "$pick"
}

until false ; do
        clear
        cowsay -f "$(whatsup)"  "$(date +%H:%M) $(fortune -s)"
        sleep 30
done
