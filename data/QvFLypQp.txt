package SoftUni.Practice.FirstStepsInCoding.Exercise.Week5;
 
import java.util.Scanner;
 
public class ExamPrep {
    public static void main(String[] args) {


        Scanner scr = new Scanner(System.in);

        int maxBadGrades = Integer.parseInt(scr.nextLine());
        String exerciseName = scr.nextLine();

        int counter = 0;
        int sum = 0;
        int counter2 = 0;
        String beforeLast = "";

        while (!exerciseName.equals("Enough")) {
            int grade = Integer.parseInt(scr.nextLine());
            if (grade <= 4) {
                ++counter;
                if (counter == maxBadGrades) {
                    System.out.printf("You need a break, %d poor grades.", counter);
                    break;
                }
            }
            
            beforeLast = exerciseName;
            exerciseName = scr.nextLine();
            sum += grade;
            
            ++counter2;

        }
        double average = 1.0 * sum / counter2;
        if (counter < maxBadGrades) {
            System.out.printf("Average score: %.2f\nNumber of problems: %d\nLast problem: %s", average, counter2, beforeLast);
        }


    }
}
