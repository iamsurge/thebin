package ppa1;

public class Postava {
	public String jmeno;
	public int sila;
	public int hbitost;
	public int vitalita;
	public int zdravi;
	public boolean levaRuka;
	public boolean pravaRuka;
	
	public Postava(String jmeno, int sila, int hbitost, int vitalita) {
		this.jmeno = jmeno;
		this.sila = sila;
		this.hbitost = hbitost;
		this.vitalita = vitalita;
		this.zdravi = vitalita;
		this.levaRuka = false;
		this.pravaRuka = false;
	}
	
	public boolean vezmiZbran(Ruka ruka, Zbran zbran) {
		if(ruka == Ruka.LEVA && levaRuka == false) {
			int zbranSila = zbran.getUtok();
			int zbranObrana = zbran.getObrana();
			sila = sila + zbranSila;
			hbitost = hbitost + zbranObrana;
			
			return (levaRuka = true);
		}
		
		if (ruka == Ruka.PRAVA && pravaRuka == false) {
			int zbranSila = zbran.getUtok();
			int zbranObrana = zbran.getObrana();
			sila = sila + zbranSila;
			hbitost = hbitost + zbranObrana;
			
			return (pravaRuka = true);
		}
		
		return false;
	}
	
	public int branSe(int utok) {
		if ((utok - hbitost) > 0) {
			zdravi = zdravi - (utok - hbitost);
			
			return utok - hbitost;
		}
		
		return 0;
	}
	
	public int zautoc() {
		return sila;
	}
	
	public boolean jeZiva() {
		if (zdravi>0) {
			return true;
		}
		
		return false;
	}
	
	public String toString() {
		return jmeno + " [" + zdravi + "/" + vitalita + "] ("+ sila + "/" + hbitost + ")";
	}

}
