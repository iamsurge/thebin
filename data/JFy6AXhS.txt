void solve()
{
    ll n;
    cin >> n;
    ll sum =0,odd=LLONG_MAX;
    for(ll i=0;i<n;i++){
        ll num;
        cin>>num;
        sum+=num;
        if(num%2){
            odd=min(odd,num);
        }
    }
    if(odd<LLONG_MAX){
        if(sum%2){
            cout<<sum-odd<<" "<<sum;
        }else{
            cout << sum<<" "<<sum-odd;
        }
    }else
        cout<<sum<<" 0";
}
