public class Kata
{
	public static string DuplicateEncode(string word)
	{
    word = word.ToLower();
		
		string result = string.Empty;
		
		for (int i = 0; i < word.Length; i++)
		{
			if(word.IndexOf(word[i]) == word.LastIndexOf(word[i]))
			{
				result += '(';
			}
			else
			{
				result += ')';
			}
		}
		
		return result;
		}
}
