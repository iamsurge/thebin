using System.Linq;
using Turbo.Plugins.Default;

namespace Turbo.Plugins.User
{
    public class MonkS27WaveOfLightSancDecoratorPlugin : BasePlugin, IInGameWorldPainter
    {
	public WorldDecoratorCollection WaveOfLightSancDecorator { get; set; }

	public MonkS27WaveOfLightSancDecoratorPlugin()
	{
	    Enabled = true;
	}

	public override void Load(IController hud)
	{
	    base.Load(hud);

	    WaveOfLightSancDecorator = new WorldDecoratorCollection(
		new GroundCircleDecorator(Hud)
		{
		    Brush = Hud.Render.CreateBrush(200, 255, 128, 0, 4f),
		    Radius = -1
		});

	}
	public void PaintWorld(WorldLayer layer)
	{
	    var actors = Hud.Game.Actors;
	    foreach (var actor in actors)
	    {
		switch (actor.SnoActor.Sno)
		{
			case ActorSnoEnum._p74_monk_bell_waveoflight_runeb:
			{
			    WaveOfLightSancDecorator.Paint(layer, actor, actor.FloorCoordinate, null);
			}
			break;
		}
	    }
	}
    }
}
