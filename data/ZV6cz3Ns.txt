function solve(arg) {

    let minuti = Number(arg[0]);
    let secundi = Number(arg[1]);

    let start = minuti * 60 + secundi + 15;
    let chas = Math.floor(start / 60);
    let min = start % 60;

    if(chas > 23){
        chas = 0 ;
    } 

    if(min < 10){
        console.log(`${chas}:0${min}`);
    }else{
        console.log(`${chas}:${min}`);
    }


}
