local it = 100
local coins = 30
local mwidth = 10
local mheight = 2
local pos = owner.Character.HumanoidRootPart.Position * Vector3.new(1, 0, 1)
local tab = {}
local parts = {}

for i = 0,coins do
    tab[i] = 0
    local part = Instance.new("Part", script)
    part.Anchored = true
    part.Size = Vector3.new(mwidth/coins, 1, 1)
    part.Position = Vector3.new(i * mwidth/coins, 5, 0) + pos
    part.Material = "SmoothPlastic"
    parts[i] = part
end

for i = 1,it do
    for j = 1,coins do
        local sum = 0
        for k = 1,coins do
            sum += math.random(0, 1)
        end
        tab[sum] += 1 
    end
    
    for j = 0,#parts do
        local max = math.max(table.unpack(tab))
        parts[j].Size = Vector3.new(mwidth/coins, (tab[j])/max*mheight, 1)
        parts[j].Position = Vector3.new(j * mwidth/coins, 5 + parts[j].Size.Y/2, 0) + pos
    end
    task.wait(.5)
end
