function stringMerge(str)
{
    token = 'nz'
    length = str.length;
    half_length = (str.length - 1) / 2;
    combine = ''
    result = ''
    for (let i = 0; i < half_length; i++)
    {
        combine += str[i] + str[half_length+1+i];
    }
    for (let i = 0; i < token.length; i++)
    {
        result += combine[i] + token[i];
    }
    return result;
}
