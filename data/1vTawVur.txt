Make a JS script and create an object called 'Settings' and declare it with following variables and values: Wager (1), Target (2) using the following script as a template:
`
Settings.Wager *= 100;
Settings.Target *= 100;
let Profit = 0,
    InitialBalance = engine.getBalance(),
    InitialWager = Settings.Wager,
    InitialTarget = Settings.Target,
    Balance = engine.getBalance();
engine.on('game_starting', () => {
    if (Settings.Wager <= engine.getBalance() && Settings.Wager <= engine.getMaxBet()) engine.placeBet(Math.round(Settings.Wager / 100) * 100, Math.round(Settings.Target / 1) * 1);
    else engine.stop();
    if (engine.lastGamePlay() == 'LOST') Settings.Wager *= 2;
    else if (engine.lastGamePlay() == 'WON') Reset();
    Profit = GetProfit();
    console.log('Profit: ' + (Profit / 100).toFixed(2));
});
function GetProfit() {
    return engine.getBalance() - InitialBalance;
}
function Reset() {
    Balance = Math.floor(engine.getBalance());
    Settings.Wager = InitialWager;
    Settings.Target = InitialTarget;
}
`
