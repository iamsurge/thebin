void doSpecialSpell(int** &main_maze) {
    int hero_num = mainHero.getCharNumber();
    switch(hero_num) {
        case 7:
            doTankSpell(main_maze);
            break;
        case 8:
            doBeatleSpell();
            break;
        default:
            doRestSpell(main_maze);
            break;
    }
}
