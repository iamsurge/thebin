public class Pilha
{
    private int topo;
    private final char[] elementos;
    private final int tamanho;

    public Pilha(int tamanho)
    {
        this.tamanho = tamanho;
        topo = -1;
        elementos = new char[tamanho];
    }

    public void destruir()
    {
        topo = -1;
    }

    public boolean vazia()
    {
        return (topo == -1);
    }

    public boolean cheia()
    {
        return (topo == tamanho - 1);
    }

    public void empilhar(char letra)
    {
        if(!cheia())
        {
            topo = topo + 1;
            elementos[topo] = letra;
        }
    }

    public char desempilhar()
    {
        if(!vazia())
        {
            int aux = topo;
            topo = topo - 1;

            return elementos[aux];

        }
        return 1;
    }
}
