//////////////////////////////////////////////////sphere.cpp
#include "sphere.h"

double Pi = 3.14;

double RToL(int R)
{
	return 2 * Pi * R;
}

double RToS(int R)
{
	return Pi * R * R;
}

double RToSp(int R)
{
	return 4 * Pi * R * R;
}

double RToV(int R)
{
	return Pi * R * R * R * 1 / 3;
}
