#include <stdio.h>
#include <stdlib.h>

int main()
{
  char *f_name = "input.txt";
  FILE *file = fopen(f_name, "r");
  if (file == NULL)
  {
    fprintf(stderr, "Error opening file %s \n", f_name);
    return 1;
  }

  char single;
  int i;
  for (i = 0; (single = fgetc(file)) != EOF; i++)
  {

    if (single >= 'A' && single <= 'Z')
    {
      single += 'a' - 'A';
      printf("%c", single);
    }
    else
      printf("%c", single);
  }

  
  fclose(file);
  return 0;
}
