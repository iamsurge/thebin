<script>
document.addEventListener('DOMContentLoaded', e=>{
    function rio_slider__top(){
      let el = $('.rio-slider__top:visible');
  	  el.fadeOut('slow', function(){
           let next = $(this).next();
            if(next.length){
                next.show();
             } else {
               $('.rio-slider__top:first').show();
            }
        });
    }
    setInterval(rio_slider__top, 5000);

});
</script>
