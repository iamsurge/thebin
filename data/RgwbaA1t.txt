apt install git gcc cmake libpcre3 libpcre3-dev zlib1g zlib1g-dev openssl libssl-dev

wget https://nginx.org/download/nginx-1.14.0.tar.gz
tar zxvf nginx-1.14.0.tar.gz

git clone https://github.com/google/ngx_brotli.git

cd ./nginx-1.14.0
sudo ./configure --with-compat --add-dynamic-module=../ngx_brotli
sudo make modules

sudo cp ./objs/*.so /etc/nginx/modules

echo "" >> /etc/nginx/nginx.conf
echo "load_module \"modules/ngx_http_brotli_filter_module.so\";" >> /etc/nginx/nginx.conf
echo "load_module \"modules/ngx_http_brotli_static_module.so\";" >> /etc/nginx/nginx.conf

systemctl restart nginx
