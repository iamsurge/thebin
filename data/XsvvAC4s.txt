c = int(input())
q = [0 for i in range(5)]
p = 500 - c
while p > 0:
    if p >= 300:
        p -= 300
        q[4] += 1
    elif p >= 60:
        p -= 60
        q[3] += 1
    elif p >= 15:
        p -= 15
        q[2] += 1
    elif p >= 5:
        p -= 5
        q[1] += 1
    else:
        p -= 1
        q[0] += 1
print(q)
