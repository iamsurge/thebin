for (int horizontalIndex = 0; horizontalIndex < _tilesMap.GetLength(0) - FirstOffset; horizontalIndex++)
        {
            if (horizontalIdentical.Count == 0)
                horizontalIdentical.Add(_tilesMap[horizontalIndex, tempVerticalIndex]);

            if (horizontalIndex + FirstOffset <= _tilesMap.GetLength(0) - FirstOffset && _tilesMap[horizontalIndex, 
                tempVerticalIndex].Icon.sprite == _tilesMap[horizontalIndex + FirstOffset, tempVerticalIndex].Icon.sprite)
            {
                horizontalIdentical.Add(_tilesMap[horizontalIndex + FirstOffset, tempVerticalIndex]);
            }
            else
            {
                if (horizontalIdentical.Count < ThirdOffset)
                {
                    horizontalIdentical.Clear();
                }
            }

            if (horizontalIndex == _tilesMap.GetLength(0) - SecondOffset)
            {
                Debug.Log(horizontalIdentical.Count);
                if (horizontalIdentical.Count > SecondOffset)
                    AddNewIdenticalTiles(horizontalIdentical);

                horizontalIdentical.Clear();
                horizontalIndex = ReloadIndexValue;

                if (tempVerticalIndex < _tilesMap.GetLength(1) - FirstOffset)
                    tempVerticalIndex++;
                else
                    break;
            }
        }

        for (int verticalIndex = 0; verticalIndex < _tilesMap.GetLength(1) - FirstOffset; verticalIndex++)
        {
            if (verticalIdentical.Count == 0)
                verticalIdentical.Add(_tilesMap[tempHorizontalIndex, verticalIndex]);

            if (verticalIndex + 1 <= _tilesMap.GetLength(1) - FirstOffset && _tilesMap[tempHorizontalIndex, verticalIndex].Icon.sprite 
                == _tilesMap[tempHorizontalIndex, verticalIndex + FirstOffset].Icon.sprite)
            {
                verticalIdentical.Add(_tilesMap[tempHorizontalIndex, verticalIndex + 1]);
            }
            else
            {
                if (verticalIdentical.Count < ThirdOffset)
                {
                    verticalIdentical.Clear();
                }
            }

            if (verticalIndex == _tilesMap.GetLength(1) - SecondOffset)
            {
                Debug.Log(verticalIdentical.Count);
                if (verticalIdentical.Count > SecondOffset)
                    AddNewIdenticalTiles(verticalIdentical);

                verticalIdentical.Clear();
                verticalIndex = ReloadIndexValue;

                if (tempHorizontalIndex < _tilesMap.GetLength(0) - FirstOffset )
                    tempHorizontalIndex++;
                else
                    break;
            }
        }
