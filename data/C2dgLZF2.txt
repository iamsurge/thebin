#include<iostream>

int main()
{
	int n; // задели ми памет за целочислена променлива n 
	std::cin >> n; //въведи ми n 

	/*   . 1              -> 1 = row, 4 
		 1 2 1            -> 2 = row, 3
	   1 2 3 2 1          -> 3 = row, 2
     1 2 3 4 3 2 1        -> 4, 1
   1 2 3 4 5 4 3 2 1      -> 5, 1
	 1 2 3 4 3 2 1        -> 4 (n - 1)
	   1 2 3 2 1
		 1 2 1
		   1               -> 1

*/
	for (int row = 1; row <= n; row++) // for row in range(1, n + 1, 1) : 
	{
		for (int space = 1; space <= n - row; space++) // for space in range(1, n - row + 1, 1):
		{
			//space за числото 
			//space za razdelitel
			std::cout << "  "; // print(" ")
		}

		for (int number = 1; number <= row; number++)// for number in range(1, row + 1, 1): number++ == number = number + 1
		{
			std::cout << number << " ";
		}

		for (int number = row - 1; number >= 1; number--) //for number in range (row, 1, -1)
		{
			std::cout << number << " ";
		}

		for (int space = 1; space <= n - row; space++) // for space in range(1, n - row + 1, 1):
		{
			//space за числото 
			//space za razdelitel
			std::cout << "  "; // print(" ")
		}

		std::cout << std::endl;// end of line
	}

	for (int row = n - 1; row >= 1; row--) // for row in range(row, 1, -1)
	{
		for (int space = 1; space <= n - row; space++) // for space in range(1, n - row + 1, 1):
		{
			//space за числото 
			//space za razdelitel
			std::cout << "  "; // print(" ")
		}

		for (int number = 1; number <= row; number++)// for number in range(1, row + 1, 1): number++ == number = number + 1
		{
			std::cout << number << " ";
		}

		for (int number = row - 1; number >= 1; number--) //for number in range (row, 1, -1)
		{
			std::cout << number << " ";
		}

		for (int space = 1; space <= n - row; space++) // for space in range(1, n - row + 1, 1):
		{
			//space за числото 
			//space za razdelitel
			std::cout << "  "; // print(" ")
		}

		std::cout << std::endl;// end of line
	  }

}
