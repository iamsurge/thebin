import java.util.Scanner;
public class Jurnal02 {
	private static void methodDua(String lirikLagu){
        int totalPanjang = lirikLagu.length();
        int vokal = totalPanjang % 5;
        if(vokal == 0){
            lirikLagu = lirikLagu.replaceAll("[AaIiUuEeOo]", "a");
            System.out.println(lirikLagu);
       }else if(vokal == 1){
            lirikLagu = lirikLagu.replaceAll("[AaIiUuEeOo]", "e");
            System.out.println(lirikLagu);
        }else if (vokal == 2) {
            lirikLagu = lirikLagu.replaceAll("[AaIiUuEeOo]", "i");
            System.out.println(lirikLagu);
        }else if(vokal == 3){
            lirikLagu = lirikLagu.replaceAll("[AaIiUuEeOo]", "o");
            System.out.println(lirikLagu);
        }else if(vokal == 4){
            lirikLagu = lirikLagu.replaceAll("[AaIiUuEeOo]", "u");
            System.out.println(lirikLagu);
        }
    }

	public static void main(final String[] args) {
		//Todo : Scanner
		Scanner input = new Scanner(System.in);
		//Todo : Jurnal 2
		String lirikLaguMain;
		lirikLaguMain = input.nextLine();
		methodDua(lirikLaguMain);
	}
}
