#include <bits/stdc++.h> 
int differentBitsSumPairwise(vector<int> &arr, int n)
{
    vector<vector<int>> freq_bits(40, vector<int>(2));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < 32; j++) {
            freq_bits[j][(arr[i] >> j) & 1]++;
        }
    }
    int ans = 0;
    constexpr int MOD = 1e9 + 7;
    for (int i = 0; i < 40; i++) {
        ans = (ans + ((freq_bits[i][0]) % MOD * (freq_bits[i][1]) % MOD) % MOD) % MOD;
    }
    return (ans * 2) % MOD;
    
}
