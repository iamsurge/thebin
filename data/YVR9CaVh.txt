#include <iostream>

using namespace std;
//Napisaæ program, który wczytuje kolejno liczby ca³kowite a¿ do powtórnego
//wczytania liczby takiej, jak pierwsza lub druga wczytana liczba. Wydrukowaæ informacjê,
//ile wczytano liczb parzystych oraz ile razy wczytano po sobie trzy liczby podzielne przez
//sta³¹ D.


const int D=7;
int main()
{
    cout<<"Podaj pierwsza liczbe: ";
    int liczba1;
    cin>>liczba1;
    cout<<"Podaj druga liczbe: ";
    int liczba2;
    cin>>liczba2;
    int sumaLP=0;

    if(liczba1%2==0)
    {
        sumaLP++;
    }
    if(liczba2%2==0)
    {
        sumaLP++;
    }

    int a, b = liczba2, c = liczba1;

    int ile_razy_D = 0;

    do
    {
        cout<<"Podaj liczbe calkowita: ";
        cin>>a;
        if(a%2==0)
        {
            sumaLP++;
        }

        if((a % D == 0) && (b % D == 0) && (c % D == 0))
            ++ile_razy_D;

        c = b;
        b = a;

    }
    while(a!=liczba1 && a!=liczba2);

    cout << "Wczytano " << sumaLP << " liczb parzystych.\n";
    cout << ile_razy_D << " razy zdarzylo sie, aby trzykrotnie z rzedu wpisac liczbe podzielona przez " << D << ".\n";

    return 0;
}
