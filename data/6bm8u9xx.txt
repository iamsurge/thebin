#include <iostream>

using namespace std;



bool tableTrue1(bool x, bool y) 
{
	return !x && !y;
}

bool tableTrue2(bool x, bool y)
{
	return !(x || y);
}

bool tableTrue3(bool x, bool y, bool z)
{
	return (x && !y) || z;
}

bool tableTrue4(bool x, bool y, bool z)
{
	return !(x || y) && (y && x);
}

bool tableTrue5(bool x, bool y, bool z)
{
	return x || !(y && (x || !z));
}



int main()
{
	setlocale(LC_ALL, "RU");

	bool t{ true };
	bool f{ false };


	cout << "!x && !y:\n";
	cout << boolalpha;
	cout << "a\tb\tvalue\n";
	cout << f << '\t' << f << '\t' << tableTrue1(f, f) << '\n';   // Перебираем все возможные варианты с 2-мя перменными 
	cout << f << '\t' << t << '\t' << tableTrue1(f, t) << '\n';
	cout << t << '\t' << f << '\t' << tableTrue1(t, f) << '\n';
	cout << t << '\t' << t << '\t' << tableTrue1(t, t) << '\n';


	cout << "\n!(x || y):\n";
	cout << "a\tb\tvalue\n";
	cout << f << '\t' << f << '\t' << tableTrue2(f, f) << '\n';
	cout << f << '\t' << t << '\t' << tableTrue2(f, t) << '\n';
	cout << t << '\t' << f << '\t' << tableTrue2(t, f) << '\n';
	cout << t << '\t' << t << '\t' << tableTrue2(t, t) << '\n';


	cout << "\n(x && !y) || z:\n";
	cout << "a\tb\tz\tvalue\n";
	cout << f << '\t' << f << '\t' << f << '\t' << tableTrue3(f, f, f) << '\n';   // Перебираем все возможные варианты с 3-мя перменными 
	cout << f << '\t' << f << '\t' << t << '\t' << tableTrue3(f, f, t) << '\n';
	cout << f << '\t' << t << '\t' << f << '\t' << tableTrue3(f, t, f) << '\n';
	cout << t << '\t' << f << '\t' << f << '\t' << tableTrue3(t, f, f) << '\n';
	cout << t << '\t' << t << '\t' << f << '\t' << tableTrue3(t, t, f) << '\n';
	cout << t << '\t' << f << '\t' << t << '\t' << tableTrue3(t, f, t) << '\n';


	cout << "\n!(x || y) && (y && x):\n";
	cout << "a\tb\tz\tvalue\n";
	cout << f << '\t' << f << '\t' << f << '\t' << tableTrue4(f, f, f) << '\n';
	cout << f << '\t' << f << '\t' << t << '\t' << tableTrue4(f, f, t) << '\n';
	cout << f << '\t' << t << '\t' << f << '\t' << tableTrue4(f, t, f) << '\n';
	cout << t << '\t' << f << '\t' << f << '\t' << tableTrue4(t, f, f) << '\n';
	cout << t << '\t' << t << '\t' << f << '\t' << tableTrue4(t, t, f) << '\n';
	cout << t << '\t' << f << '\t' << t << '\t' << tableTrue4(t, f, t) << '\n';


	cout << "\nx || !(y && (x || !z)):\n";
	cout << "a\tb\tz\tvalue\n";
	cout << f << '\t' << f << '\t' << f << '\t' << tableTrue5(f, f, f) << '\n';
	cout << f << '\t' << f << '\t' << t << '\t' << tableTrue5(f, f, t) << '\n';
	cout << f << '\t' << t << '\t' << f << '\t' << tableTrue5(f, t, f) << '\n';
	cout << t << '\t' << f << '\t' << f << '\t' << tableTrue5(t, f, f) << '\n';
	cout << t << '\t' << t << '\t' << f << '\t' << tableTrue5(t, t, f) << '\n';
	cout << t << '\t' << f << '\t' << t << '\t' << tableTrue5(t, f, t) << '\n';


	
	return 0;
}
