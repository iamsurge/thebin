function digForward()
    turtle.dig()
    turtle.forward()
end

function digLineLeft(width)
    turtle.turnLeft()
    for i = 1, width - 1 do
        digForward()
    end
    turtle.turnRight()
end

function digLineRight(width)
    turtle.turnRight()
    for i = width - 1, 1, -1 do
        digForward()
    end
    turtle.turnLeft()
end

function digUp()
    turtle.digUp()
    turtle.up()
end

function digDown()
    turtle.digDown()
    turtle.down()
end

function digPlaneUp(width, height)
    for i = 1, height do
        if i % 2 == 0 then
            digLineRight(width)
        else
            digLineLeft(width)
        end
        if i < height then
            digUp()
        end
    end
end

function digPlaneDown(width, height)
    for i = 1, height do
        if i % 2 == 0 then
            digLineLeft(width)
        else
            digLineRight(width)
        end
        if i < height then
            digDown()
        end
    end
end

local width = 0
local height = 0
local length = 0


term.clear()
term.setCursorPos(1, 1)

print("Enter width")
width = read() + 0

print("Enter height")
height = read() + 0

print("Enter length")
length = read() + 0


print("Area: " .. width .. "x" .. length .. "x" .. height)


for i = 1, length do
    digForward()
    if i % 2 == 0 then
        digPlaneDown(width, height)
    else
        digPlaneUp(width, height)
    end
end
