#include <stdio.h>
#include <stdlib.h>

int main(){
    
    int n;
    int nula = 0, eden = 0, dva = 0, tri = 0, cetiri = 0;
    while(scanf("%d", &n)) {
        int pom = n;
        int pozicija = 0;
        int najgolema_cifra = 0;
        int pozicija_na_najgolema_cifra = 0;
        
        while(pom > 0) {
            int cifra = pom % 10;
            pom /= 10;
            if(cifra > najgolema_cifra) {
                najgolema_cifra = cifra;
                pozicija_na_najgolema_cifra = pozicija;
            }
            pozicija++;
        }
        if(pozicija_na_najgolema_cifra == 0) {
            nula++;
        }
        else if(pozicija_na_najgolema_cifra == 1) {
            eden++;
        }
        else if(pozicija_na_najgolema_cifra == 2) {
            dva++;
        }
        else if(pozicija_na_najgolema_cifra == 3) {
            tri++;
        }
        else if(pozicija_na_najgolema_cifra == 4) {
            cetiri++;
        }
        
    }
    printf("0: %d\n", nula);
    printf("1: %d\n", eden);
    printf("2: %d\n", dva);
    printf("3: %d\n", tri);
    printf("4: %d\n", cetiri);
    
    return 0;

}
