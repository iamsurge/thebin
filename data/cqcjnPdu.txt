Section:NewButton("Blood Room", "Makes the room bloody", function()
    local currentLoadedRoom
    local function setRooms()
        local tb = {}
        table.foreach(workspace.CurrentRooms:GetChildren(), function(_, r)
            if r:FindFirstChild("RoomStart") then
                table.insert(tb, tonumber(r.Name))
            end
        end)
        currentLoadedRoom = workspace.CurrentRooms[tostring(math.max(unpack(tb)) - 1)]
    end
    setRooms()
    require(game.ReplicatedStorage.ClientModules.Module_Events).tryp(currentLoadedRoom, 9e307)
end)

Section:NewButton("Summon Seek", "Summons Seek.", function()
    local file = "https://github.com/Noonie1/storage/blob/ff0cc1054044e7880b2293ba97ad531e26dac0b5/SeekerEeker.rbxm"
    local name = "Seek"
    local player = game.Players.LocalPlayer
    local char = player.Character
    local charhum = char:FindFirstChildWhichIsA("Humanoid")
    local cam = game.Workspace.CurrentCamera
    local seek = game:GetObjects("rbxassetid://11361534304")[1]
    seek.Parent = workspace
    seek.Root.Anchored = true
    local seekroot = seek:FindFirstChild("Root")
    local seeksteps = Instance.new("Sound",seek.Head)
    seeksteps.SoundId = "rbxassetid://1244506786"
    seeksteps.Volume = 2
    seeksteps.RollOffMaxDistance = 250
    seeksteps.Parent = seek.Root
    seeksteps.Looped = true
    seeksteps.Pitch = 1.65
    -----
    local cutscene = false
    local stop = false
    local currentrooms = workspace.CurrentRooms
    local seekhipheight = Vector3.new(0,5,0)
    local clerp = 0.04
    local PLRspeed = 20
    local SeekSpeed = 23
    local LatestRoom = game:GetService("ReplicatedStorage").GameData.LatestRoom
    function GetTime(Distance, Speed)
        -- Time = Distance / Speed
        local Time = Distance / Speed
        return Time
    end
    local CF = CFrame.new
    
    local function speed()
        spawn(function()
            while wait(0.5) do
                charhum.WalkSpeed = PLRspeed
            end
        end)
    end
    
    local function CutsceneSeek() cutscene = true
        local CutscenePart = Instance.new("Part",workspace) CutscenePart.Anchored = true CutscenePart.CanCollide = false CutscenePart.Transparency = 1
        local offset = Vector3.new(0,0,-25)
        CutscenePart.CFrame = seekroot.CFrame*CFrame.new(offset)
        spawn(function()
            cam.CameraType = Enum.CameraType.Scriptable
            while cutscene do game["Run Service"].RenderStepped:Wait()
                if stop then
                    break
                end
                charhum.WalkSpeed = 0
                CutscenePart.CFrame = CutscenePart.CFrame:Lerp(CFrame.lookAt(CutscenePart.Position,seek.Head.Position),clerp)
                cam.CFrame = CutscenePart.CFrame
            end
        end)
        wait(6)
        cutscene = false
        local goback = game.TweenService:Create(cam,TweenInfo.new(2),{CFrame = char.Head.CFrame})
        goback:Play() goback.Completed:Wait() speed()
        CutscenePart:Destroy()
        cam.CameraType = Enum.CameraType.Custom
    end
    
    local addeyes = false
    
    spawn(function()
        while wait(math.random(3,10)) and addeyes do
            local roar = Instance.new("Sound",seek.Head) local speeds = {0.6,0.7,0.8}
            roar.Volume = 1
            roar.SoundId = "rbxassetid://6305809364"
            roar.PlaybackSpeed = speeds[math.random(1,#speeds)]
        end
    end)
    
    currentrooms.ChildAdded:Connect(function()
        if addeyes then
            --	require(game:GetService("ReplicatedStorage").ClientModules.EntityModules.Seek).tease(require(game.Players.LocalPlayer.PlayerGui.MainUI.Initiator.Main_Game), workspace.CurrentRooms[game.Players.LocalPlayer:GetAttribute("CurrentRoom")], 50,1,true)
        end
    end)
    
    local latestroom = game.ReplicatedStorage.GameData.LatestRoom.Value
    local function SeekMove()
        seek.AnimationController:LoadAnimation(seek.AnimRun):Play()
        local gruh = workspace.CurrentRooms
        for i = 1, latestroom do 
            LatestRoom = game.ReplicatedStorage.GameData.LatestRoom.Value
            if gruh:FindFirstChild(i) then
                print("room "..i)
                local room = gruh[i]
                if room:FindFirstChild("Nodes") then
                    local nodes = room:FindFirstChild("Nodes")
                    for v = 1, #nodes:GetChildren() do
                        if nodes:FindFirstChild(v) then
                            local waypoint = nodes[v]
                            local Distance = (seekroot.Position - waypoint.Position).magnitude -- Get the distance between the current position and the next node
                            local fakejays = game.TweenService:Create(seekroot,TweenInfo.new(GetTime(Distance, SeekSpeed), Enum.EasingStyle.Linear,Enum.EasingDirection.Out, 0,false,0),{CFrame = waypoint.CFrame + seekhipheight})
                            fakejays:Play()
                            fakejays.Completed:Wait()
                        end
                    end
                end
            end
            spawn(SeekMove)
            print("looping")
        end
    end
    
    
    local function GetFirstRoom()
        for i = 1, game.ReplicatedStorage.GameData.LatestRoom.Value do 
            local gruh = workspace.CurrentRooms
            if gruh:FindFirstChild(i) then
                print("room "..i)
                local room = gruh[i]
                if room:FindFirstChild("Nodes") then
                    local nodes = room:FindFirstChild("Nodes")
                    return room,nodes
                end
            end
        end
    end
    local function SeekSpawn()
        if currentrooms:FindFirstChildOfClass("Model") then
            local firstroom = GetFirstRoom()
            local firstwaypoint
            if firstroom:FindFirstChild("Nodes") and firstroom:FindFirstChild("Nodes"):FindFirstChildWhichIsA("Part") then
                firstwaypoint = firstroom:FindFirstChild("Nodes"):FindFirstChildWhichIsA("Part")
            end
            seek:SetPrimaryPartCFrame(firstwaypoint.CFrame + seekhipheight)
            seek.AnimationController:LoadAnimation(seek.AnimRaise):Play()
            spawn(CutsceneSeek)
            workspace.Ambience_Seek:Play()
            wait(6)
            seeksteps:Play()
            wait(2)
            addeyes = true
            SeekMove()
        end
    end
    
    SeekSpawn()
end)
