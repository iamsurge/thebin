ll recursive(ll b,ll p,ll m)
{
    if(p==0)
        return 1;
    else if(p%2==0)
    {
      return recursive(((b*b)%m),p/2,m);
    }
    else
    {
        return (b*recursive((b*b)%m,(p-1)/2,m)%m);
    }
}
