WITH dates AS (
  SELECT 
    CAST (date AS date) AS date
  FROM
    (VALUES ('2003/04/01'), ('2003/04/02'), ('2003/04/03'),
           ('2003/04/04'), ('2003/04/05'), ('2003/04/06'),
           ('2003/04/07')) d(date)
)
SELECT 
  dates.date, COUNT(*)
FROM dates
LEFT JOIN trip ON time_out = date
LEFT JOIN pass_in_trip ON pass_in_trip.date = dates.date
WHERE town_from = 'Rostov'
GROUP BY dates.date
