R = -1
while not (0 <= R <= 25000):
	R = int(input("Введите длину радиуса: "))
L = -1
while not (0 <= L <= 100):
	L = int(input("Введите длину клетки: "))
#количество клеток в 1 радиусе
n = R//L
#количество целых клеток в 1/4 окружности
count = 0
for i in range(1,n+1):
	count+=int((R**2 - (i*L)**2)**(0.5))
print("В данной окружности находится %d целых клеток"%(count*4))
