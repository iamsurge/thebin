#include <stdio.h>
#include <stdlib.h>

void Delete0(int arr[], int* ptrSize) {
	int length = *ptrSize;
	for (int k = 0; k < length; ++k) {
		if (arr[k] == 0) {
			for (int i = k; i < length - 1; ++i)
				arr[i] = arr[i + 1];
			arr = (int*)realloc(arr, --length * sizeof(int));
			--k;
		}
	}
	*ptrSize = length;
}

int main()
{
	// инициализация массива:
	int size = 5;
	int* arr = (int*)calloc(size, sizeof(int));
	arr[0] = 0; arr[1] = 3; arr[2] = 0; arr[3] = 9; arr[4] = 0;
	for (int i = 0; i < size; ++i)
		printf("%d ", arr[i]);
	puts("\n");

	// обработка массива:
	Delete0(arr, &size);

	// вывод массива:
	for (int i = 0; i < size; ++i)
		printf("%d ", arr[i]);

	return 0;
}
