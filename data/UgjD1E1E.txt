void word_search(string& s2, string& s3, string slova[], int &k)
{
    s2 += " ";
    string word;
    size_t pos;
    while ((pos = s2.find(" ")) != std::string::npos) {
        word = s2.substr(0, pos);
        if (word.find(s3) != std::string::npos) {
            slova[k++] = word;
        }
        s2.erase(0, pos+1);
    }
}
