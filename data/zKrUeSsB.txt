#!/usr/bin/python3
 
import sys      # gives access to cli arguments
 
chardict = {}

chardict['A'] = 'A'
chardict['B'] = 'D'
chardict['C'] = 'C'
chardict['D'] = 'X'
chardict['E'] = 'V'
chardict['F'] = 'H'
chardict['G'] = 'B'
chardict['H'] = 'G'
chardict['I'] = 'A'
chardict['J'] = 'S'
chardict['K'] = 'F'
chardict['L'] = 'O'
chardict['M'] = 'J'
chardict['N'] = 'K'
chardict['O'] = 'T'
chardict['P'] = 'I'
chardict['Q'] = 'P'
chardict['R'] = 'W'
chardict['S'] = 'N'
chardict['T'] = 'R'
chardict['U'] = 'E'
chardict['V'] = 'L'
chardict['W'] = 'C'
chardict['X'] = 'U'
chardict['Y'] = 'Y'
chardict['Z'] = 'M'

 
def main():
    if len(sys.argv) < 2:
        print("Please input the file.")
        return
    
    filename = sys.argv[1]
    file = open(filename, "r")
    
    while 1:
        
        c = file.read(1);
        if len(c) == 0:
            break
            
        if c in chardict:
            sys.stdout.write(chardict[c])
        else:
            sys.stdout.write(c)
    
    
    
 
if __name__ == '__main__':
    main()
