#include <iostream>
 
using namespace std;
 
int main()
{
    int start = 50;
    int sum = 0;
    int num=0;
    while (start != 70)
    {
        for (int i = 1; i < 100; i++)
        {
            num = start / i;
            
            for (int j = 1; j < 100; j++)
            {
                if (num % j == 0)
                    sum += num;
            }
        }
        start++;
    }
    cout << sum << endl;
    return 0;
}
