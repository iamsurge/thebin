#include<bits/stdc++.h>
using namespace std;

string s;
// Time Complexity: O(n^3)
int isPalindrome(int l, int r) {
    for(int i=l, j=r; i<j; i++, j--) {
        if(s[i] != s[j]) return 0;
    }
    return 1;
}

int main() {
    int max_len = 1;
    cin >> s;

    for(int i=0; i<s.size(); i++) {
        for(int j=i; j<s.size(); j++) {
            if(isPalindrome(i, j)) {
                int len = j-i+1;
                if(len>max_len) max_len = len;
            }
        }
    }
    cout << max_len;
    return 0;
}
