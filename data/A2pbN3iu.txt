Se utiliza la Concatenación "+", para evitar líneas largas.

edad = int(input("introduzca su edad: \n"))

# Se asigna "None" a la variable.

RESPUESTA = None

# Se evalúa sí el comprador es mayor de edad.
# Sí lo es, accede a la compra de alcohol.

if edad >= 18:

    print("Es mayor de edad, puede comprar alcohol. Elija una opción")
    RESPUESTA = input("Ingrese 1 para Ron\n" +
                      "Ingrese 2 para Whisky\n" +
                      "Ingrese 3 para Ginebra\n")

    if RESPUESTA == "1":
        print("Ha elegido comprar Ron.")

    elif RESPUESTA == "2":
        print("Ha elegido comprar Whisky.")

    elif RESPUESTA == "3":
        print("Ha elegido comprar Ginebra.")

    else:
        print("Opción Inválida.")

else:
    print("Es menor de edad. No debe comprar alcohol.")
