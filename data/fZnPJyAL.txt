#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    vector<int> v;
    v.push_back(10);
    
    int x = 15;
    v.push_back(x);

    for(int i = 0; i < v.size(); i++) {
        cout << v[i] << " ";
    }
    sort(v.begin(), v.end());
    return 0;
}
