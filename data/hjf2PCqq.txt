void cv(vector <int> &v) {
    for (auto x : v) cout << x << ' ';
    cout << "\n";
}

void cvv(vector <vector <int> > &v) {

string to(int x) {
    string r;
    while (x > 0) {
        r += x % 4 + '0';
        x /= 4;
    }
    reverse(r.begin(), r.end());
    while (r.size() < 9) {
        r = '0' + r;
    }
    return r;
}


bool ok(string s) {
    if (s.find('0') == string::npos) {
        return 1;
    }
    for (int i = 0; i < s.size() - 1; ++i) {
        if (s[i] == '0' && s[i + 1] == '0') {
            return 1;
        }
    }
    return 0;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int ans = 0;
    for (int i = 0; i < pow(4, 9); ++i) {
        if (ok(to(i))) {
            //cout << to(i) << "\n";
            ans++;
        }
    }
    cout << ans;
}
