public void AddActor(string type, string name)
        {
            ActorAddEventArgs args = new ActorAddEventArgs();
            if (type.Equals("Fighter") || type.Equals("Mage") || type.Equals("Cleric"))
            {
                
                if (type.Equals("Fighter"))
                {
                    Fighter newFighter = new Fighter(name);
                    heroes.Add(newFighter);
                    turnOrder.Add(newFighter);
                }

                else if (type.Equals("Mage"))
                {
                    Mage newMage = new Mage(name);
                    heroes.Add(newMage);
                    turnOrder.Add(newMage);
                }

                
            }
            else
            {
                if (type.Equals("Bandit"))
                {
                    Bandit newBandit = new Bandit(name);
                    villains.Add(newBandit);
                    turnOrder.Add(newBandit);
                }
            }

            args.characterType = type;
            args.characterName = name;
            OnActorAdd(this, args);
        }
