using System;

namespace Pipes_In_Pool
{
	class PipesInPool
	{
		
		static void Main(string[] args)
		{
			
			int daysTournament = int.Parse(Console.ReadLine());

			int dayWin = 0;
			int dayLoss = 0;
			decimal sum = 0.0M;

			for (int i = 1; i <= daysTournament; i++)
			{
				bool isExitLoop = false;
				int currentWin = 0;
				int currentLoss = 0;
				decimal currentSum = 0.0M;

				while (!isExitLoop)
				{
					string s = Console.ReadLine().ToLower();
					
					switch (s)
					{
						case "finish":
							isExitLoop = true;
							break;
						default:
							s = Console.ReadLine().ToLower();
				
							if (s == "win")
							{
								currentWin ++;
								currentSum += 20.0M; 
							}
							else
							{
								currentLoss++;
							}
							break;
					}//switch (s)
					
				}//while (!isExitLoop)
				
				if ( currentWin > currentLoss)
				{
					currentSum *= 1.1M;
				}
				dayWin += currentWin; 
				dayLoss += currentLoss;
				sum += currentSum;  

			}//for (int i = 1; i <= daysTournament; i++)
			
			if (dayWin > dayLoss)
			{
				sum *= 1.2M; 
				Console.WriteLine($"You won the tournament! Total raised money: {sum:F2}");
			}
			else
			{
				Console.WriteLine($"You lost the tournament! Total raised money: {sum:F2}");
			}
		}
	}
}
