import java.util.Scanner;

public class App {
    public static void main(String[] args) {
        Scanner numberScanner = new Scanner(System.in); // The Scanner is what takes the input
        int i, j, sum; // The variables used in the first part of the assignment

        System.out.println("Give i: "); // Prompt for the first variable
        i = numberScanner.nextInt(); // Reading a value in i
        System.out.println("Give j: "); // Prompt for the second variable
        j = numberScanner.nextInt();// Reading a value in j
        sum = sumOfValues(i, j); // Calling the function for those i and j, saving the result
        System.out.println(sum); // Printing to console the result

        int n; // The number we will test for being prime
        boolean prime; // The true/ false answer is store in this variable
        System.out.println("Give n: ");
        n = numberScanner.nextInt();
        prime = isPrime(n);
        System.out.println(prime);

        numberScanner.close(); // Closing the Scanner before stopping the application
    }

    public static int sumOfValues(int i, int j) {
        int sum = 0; // The sum has an initial value of 0
        for (int k = i; k <= j; k++) {
            // We add to sum every number between i and j
            sum += k;
        }
        // We return the result
        return sum;
    }

    public static boolean isPrime(int n) {
        // If a number is going to have divisors (not be prime), those will be
        // between 2 and the square root of n.
        for (int d = 2; d * d <= n; d++) {
            if (n % d == 0) {
                return false; // If a divisor is found, the number is not prime
            }
        }
        return true; // If nothing was found, the number is prime
    }
}
