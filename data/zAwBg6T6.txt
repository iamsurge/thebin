$a = 10;
$b = 3;

// Penjumlahan
$c = $a + $b;
echo "$a + $b = $c";

// Pengurangan
$c = $a - $b;
echo "$a - $b = $c";

// Perkalian
$c = $a * $b;
echo "$a * $b = $c";

// Pembagian
$c = $a / $b;
echo "$a / $b = $c";

// Modulus / sisa bagi
$c = $a % $b;
echo "$a % $b = $c";

// Pangkat
$c = $a ** $b;
echo "$a ** $b = $c";
