import pyautogui
from ast import Num
from asyncio import constants
from xml.etree.ElementTree import tostring
from pynput.keyboard import Key, Controller
import time

keyboard = Controller()


settings = {
    "gamble?": False,
    "delay": 20,
    "gamble amount": 150,
}

doGamble = input("Do gambling? (Y/N): ")

if doGamble == "Y" or doGamble == "y":
    settings["gamble?"] = True
    settings["gamble amount"] = int(input("Gamble amount?: "))

delay = input("Use default delay? (Recommended) (Y/N): ")

if delay == "N" or delay == "n":
    settings["delay"] = int(input("Enter custom delay: "))

time.sleep(5)
pyautogui.write('AUTO HUNT V1.1 | By Matt \n' + "Config: " + str(settings))
keyboard.press(keyboard._Key.enter)
# pyautogui.write('**Unhandled promise at 104::discordapp.com/fetch/api/messages**')
keyboard.press(keyboard._Key.enter)

while True:
    keyboard.type('owoh')
    keyboard.press(keyboard._Key.enter)
    time.sleep(.4)
    keyboard.type('owob')
    keyboard.press(keyboard._Key.enter)
    if settings['gamble?'] == True:
        time.sleep(.4)
        keyboard.type("owos " + str(settings["gamble amount"]))
        keyboard.press(keyboard._Key.enter)
    time.sleep(settings['delay'])
