#include <iostream>
#include <vector>

int main() {
    unsigned n, k;
    std::cin >> n >> k;

    if (k == 1) {
        std::cout << "No" << std::endl;
    } else if ((n * n) % k != 0) {
        std::cout << "No" << std::endl;
    } else {
        std::cout << "Yes" << std::endl;
        unsigned curr = 0;
        std::vector<std::vector<unsigned>> d(n, std::vector<unsigned>(n));
        for (unsigned i = 0; i < n; ++i) {
            for (unsigned j = 0; j < n; ++j) {
                ++curr;
                d[i][(j + i) % n] = curr;
                curr %= k;
            }
        }
        for (unsigned i = 0; i < n; ++i) {
            for (unsigned j = 0; j < n; ++j) {
                std::cout << d[i][j] << " ";
            }
            std::cout << std::endl;
        }
    }

    return 0;
}
