package Foxx;


/*
 * HouseBuilder - A program that will build a simple house with a side walk.
 * 
 * @author Foxx Azalea Pinkerton
 * @since Nov 2, 2022
 * @teacher Mr.Wachs
 */
public class HouseBuilder {

  static final String ROOF_LEFT = "/";
  static final String ROOF_RIGHT = "\\";
  static final String WALL = "|";
  static final String FLOOR = "_";
  static final String SPACE = " ";
  static final String STEPS = "**";
  static final String WALK = "*";

  /**
   * Main method for the project
   *
   * @param args the command line arguments
   */
  public static void main(String[] args) {
    final String TITLE = "House Builder";
    Helper.welcome(TITLE);

    do {
      run(TITLE);
    } while (Helper.playAgain(TITLE));

    Helper.exit(TITLE);
  }

  /**
   * Builds the roof of the house
   * 
   * @param base
   * @param half 
   */
  private static void roof(int base, int half) {
    for (int i = 0; i < half; i++) {
      for (int j = half; j > i; j--) {
        System.out.print(SPACE);
      }

      System.out.print(ROOF_LEFT);

      for (int j = 0; j < i * 2; j++) {
        System.out.print(SPACE);
      }

      System.out.println(ROOF_RIGHT);
    }

    System.out.print(ROOF_LEFT);

    for (int i = 0; i < base - 2; i++) {
      System.out.print(FLOOR);
    }

    System.out.println(ROOF_RIGHT);
  }

  /**
   * Builds the main body of the house
   * 
   * @param base
   * @param half 
   */
  private static void house(int base, int half) {

    for (int i = 0; i < half; i++) {
      System.out.print(WALL);

      for (int j = 0; j < base - 2; j++) {
        System.out.print(SPACE);
      }

      System.out.println(WALL);
    }

    System.out.print(WALL);

    for (int i = 0; i < base - 2; i++) {
      System.out.print(FLOOR);
    }

    System.out.println(WALL);
  }

  /**
   * Builds the side walk
   * 
   * @param base
   * @param half 
   */
  private static void walk(int base, int half) {
    for (int i = 0; i < half; i++) {
      System.out.print(SPACE);
    }
    System.out.println(STEPS);

    for (int i = 0; i < half; i++) {
      System.out.print(SPACE);
    }
    for (int i = 0; i < base; i++) {
      System.out.print(WALK);
    }
    System.out.println("");
  }

  /**
   * Is the main "Game Loop" of the program
   * 
   * @param TITLE 
   */
  private static void run(String TITLE) {
    int base = Helper.getInt("Enter the base of the house.", TITLE);
    
    if (!Helper.isEven(base) || 4 >= base) {
      Helper.problemBox("Cannot draw a house with a base of " + base + ". Make sure its a evan number or grater then 4", TITLE);
    } else {
      int half = base / 2 - 1;
      roof(base, half);
      house(base, half);
      walk(base, half);
    }
  }

}
