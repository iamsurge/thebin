using System;

namespace Durak
{
    enum Mark
    {
        Spades,
        Clubs,
        Diamonds,
        Hearts,
    }

    static class Rules
    {
        static bool Throw = false; // Игра в подкидного дурака?
        static bool Move = false; // Игра в передвижного дурака?
    }

    class Card
    {
        public int Score;
        public Mark Mark;
    }

    class Player
    {
        public string Name;
        public Card[] CardsOnHands;
    }

    class BoxOfCards
    {
        public Card[] Cards;
        public int Count { get => Cards.Length; }
    }

    class Table
    {
        public BoxOfCards BoxOfCards;
    }

    class GameMatch
    {
        public Mark SpecialCard;
        public Player[] Players;
        public Table Table;
        public int Turn;
    }
}
