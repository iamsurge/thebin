#include <stdio.h>
#include <stdlib.h>

int main(){
    int N;
    scanf("%d", &N);
    if(N <= 9) {
        printf("Brojot ne e validen");
        return 0;
    }
     // 0
    int najgolem_broj = 0;
    for(int i = 1; i < N; i++) {
        int broj = i;
        int obraten_broj = 0;
        int broj_na_cifri = 0;
        
        while(broj > 0) {
            int cifra = broj % 10;
            obraten_broj = (obraten_broj * 10) + cifra;
            broj /= 10;
            broj_na_cifri++;
        }
        
        if(obraten_broj % broj_na_cifri == 0) {
            najgolem_broj = i;
        }
    }
    printf("%d\n", najgolem_broj);
    return 0;

}
