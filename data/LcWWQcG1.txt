#include <iostream>
#include <vector>
#include <queue>

using namespace std;

class Nums {
public:
	double num;
	double chis;
	int znm;
	Nums(double num):num(num), chis(num), znm(1) {}
    Nums(double num, int znm): num(num/znm), chis(num), znm(znm){}

};

bool operator<(const Nums& num1, const Nums& num2) {
	return num1.num < num2.num;
}

int main(){
    int sz, times;
    double n;
    priority_queue<Nums, vector<Nums> > pq;
    while(cin >> sz >> times) {
        for (int i = 0; i < sz; i++) {
            cin >> n;
            pq.push(Nums(n));
        }

        for (int i = 0; i < times; i++) {
            Nums o = pq.top();
            pq.pop();
            pq.push(Nums(o.num, o.znm * 2));
            pq.push(Nums(o.num, o.znm * 2));
        }


        while(!pq.empty()){
            Nums o = pq.top();
            cout << o.num << " " << o.chis << " " << o.znm << endl;
            pq.pop();
        }
        cout << endl;
        Nums p = pq.top();
        cout << p.chis << "/" << p.znm << endl;
    }
    return 0;
}
