SELECT
	userSMSRefill,
	count(DISTINCT userID) usersCount

FROM
	`uptimerobot-001.ur.users`

WHERE userID IN (
 
				SELECT
				 userID
				FROM
						`uptimerobot-001.ur.alertcontacts`
						
				WHERE
						alertContactType IN (8,14)
						AND alertContactStatus = 2
						
				GROUP BY
								userID
				ORDER BY 
						userID)

GROUP BY
	userSMSRefill
