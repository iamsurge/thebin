
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
 
        /*Напишете програма, която чете от конзолата плод (banana / apple / orange / grapefruit / kiwi / pineapple / grapes),
        ден от седмицата (Monday / Tuesday / Wednesday / Thursday / Friday / Saturday / Sunday) и количество (десетично число) ,
        въведени от потребителя, и пресмята цената според цените от таблиците по-горе.
        При невалиден ден от седмицата или невалидно име на плод да се отпечата "error".  */

        Scanner scanner = new Scanner(System.in);
        String fruit = scanner.nextLine();
        String day = scanner.nextLine();
        double amount = Double.parseDouble(scanner.nextLine());


        boolean dayOfTheWeek = day.equals("Monday")
                || day.equals("Tuesday")
                || day.equals("Wednesday")
                || day.equals("Thursday")
                || day.equals("Friday");
        
        if ((fruit.equals("banana")
                && dayOfTheWeek)) {
            System.out.println(2.5 * amount);
        } else if (fruit.equals("apple")
                && dayOfTheWeek) {
            System.out.println(amount * 1.20);
        } else if (fruit.equals("orange")
                && dayOfTheWeek) {
            System.out.println(amount * 0.85);
        } else if (fruit.equals("grapefruit")
                && dayOfTheWeek) {
            System.out.println(amount * 1.45);
        } else if (fruit.equals("kiwi")
                && dayOfTheWeek){
            System.out.println(amount * 2.70);
        } else if ((fruit.equals("pineapple")  && dayOfTheWeek)) {
            System.out.println(amount * 5.50);
        } else if (fruit.equals("grapes")
                && dayOfTheWeek) {
            System.out.println(amount * 3.85);
        }
    }
}
