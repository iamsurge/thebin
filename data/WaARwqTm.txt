namespace GoldMine
{
	using System;

	class GoldMine
	{
		static void Main(string[] args)
		{
            int n = int.Parse(Console.ReadLine());

            for (int i = 1; i <= n; i++)
            {
                double averageYield = double.Parse(Console.ReadLine());
                int days = int.Parse(Console.ReadLine());

                double sum = 0;

                for (int j = 1; j <= days; j++)
                {
                    sum += double.Parse(Console.ReadLine());
                }

                double realAverage = sum / days;

                if (realAverage >= averageYield)
                {
                    Console.WriteLine($"Good Job! Average gold per day: {realAverage:F2}.");
                }
                else
                {
                    Console.WriteLine($"You need: {(averageYield - realAverage):F2} gold.");
                }
            }
        }
	}
}
