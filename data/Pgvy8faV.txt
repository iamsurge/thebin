Write program that generate random smooth 2d terrain like in terraria

//To generate a random smooth 2D terrain, we can use Perlin Noise.

//first, we need to include the library that contains the methods for creating a perlin noise map
#include <PerlinNoise.h>

//now, we can define the variables for generating the 2D terrain:
int width = 60; //the width of the terrain
int height = 60; //the height of the terrain
int octaves = 4; //the number of octaves for the noise
float persistence = 0.5; //the persistence of the noise
float scale = 0.05; //the scale of the noise

//now, we can create a blank map to store the terrain data
int terrain[width][height] = {0};

//now, we can generate the noise map
PerlinNoise noise = PerlinNoise(octaves, persistence, scale);

//now, we can loop through the map and set the terrain data
for (int x = 0; x < width; x++)
{
    for (int y = 0; y < height; y++)
    {
        float nx = (float)x/width - 0.5;
        float ny = (float)y/height - 0.5;
        float noiseValue = noise.perlin2D(nx,ny);
        terrain[x][y] = (int)(noiseValue * 64.0);
    }
}

//now, we can print the terrain data
for (int x = 0; x < width; x++)
{
    for (int y = 0; y < height; y++)
    {
        printf("%d ", terrain[x][y]);
    }
    printf("\n");
}
