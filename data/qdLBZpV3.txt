package ForLoopExercise;

import java.util.Scanner;

public class TrekkingMania_07 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int countGroups = Integer.parseInt(scanner.nextLine());

        int musalaCount = 0;
        int montBlancCount = 0;
        int kilimandjaroCount = 0;
        int K2Count = 0;
        int everestCount = 0;
        int sumClimbers = 0;

        for (int i = 1; i <= countGroups ; i++) {

            int countClimbers = Integer.parseInt(scanner.nextLine());
            sumClimbers += countClimbers;

            if (countClimbers <= 5){
                musalaCount +=countClimbers;
            }else if (countClimbers <= 12){
                montBlancCount+=countClimbers;
            }else if(countClimbers <= 25){
                kilimandjaroCount+=countClimbers;
            }else if (countClimbers <= 40){
                K2Count+=countClimbers;
            }else {
                everestCount+=countClimbers;
            }

        }

        double percentClimbersMusala = (musalaCount * 1.0/ sumClimbers) * 100;
        double percentClimbersMontBlanc = (montBlancCount * 1.0/ sumClimbers) * 100;
        double percentClimbersKilimanjaro = (kilimandjaroCount * 1.0/ sumClimbers) * 100;
        double percentClimbersK2 = (K2Count * 1.0/ sumClimbers) * 100;
        double percentClimbersEverest = (everestCount * 1.0/ sumClimbers) * 100;

        System.out.printf("%.2f%%%n", percentClimbersMusala);
        System.out.printf("%.2f%%%n", percentClimbersMontBlanc);
        System.out.printf("%.2f%%%n", percentClimbersKilimanjaro);
        System.out.printf("%.2f%%%n", percentClimbersK2);
        System.out.printf("%.2f%%%n", percentClimbersEverest);

    }
}
