local regionPart = game.Workspace.region
local plr = game.Players.LocalPlayer
local char = plr.Character
local makeForceFieldRemote = game.ReplicatedStorage.MakeForceField

while wait(1) do
	
	local region = Region3.new(regionPart.Position - (regionPart.Size/2), regionPart.Position + (regionPart.Size/2))
	
	local allParts = game.Workspace:FindPartsInRegion3WithWhiteList(region, char:GetDescendants())
	isInRegion = false
	
	for i, part in pairs(allParts) do
		if part:FindFirstAncestor(plr.Name) then
			isInRegion = true
		end
	end
	
	if isInRegion == true then
		
		if not char:FindFirstChild("ForceField") then
			makeForceFieldRemote:FireServer(true)

		end
		
		--[[char:WaitForChild("Humanoid"):UnequipTools()
		game.StarterGui:SetCoreGuiEnabled(Enum.CoreGuiType.Backpack, false)]]
		

	else
		
		if char:FindFirstChild("ForceField") then
			makeForceFieldRemote:FireServer(false)

		end

		--game.StarterGui:SetCoreGuiEnabled(Enum.CoreGuiType.Backpack, true)
		
	end
	
end
