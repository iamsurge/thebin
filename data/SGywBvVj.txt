#include<bits/stdc++.h>
#define endl "\n"

using namespace std;

int main() {
    string s1, s2 = "hello";
    cin >> s1;

    int i=0, j=0;

    while(i<s1.size() && j<s2.size()) {
        if(s1[i] == s2[j]) {
            j++;
        }

        i++;
    }
    if(j==s2.size()) cout << "YES";
    else cout << "NO";
    
    return 0;
}
