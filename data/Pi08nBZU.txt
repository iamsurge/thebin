#include <iostream>
#include <string>

using namespace std;

// https://acmp.ru/index.asp?main=task&id_task=536

typedef long long ll;

ll dp[50001];

int main()
{
    ll n, C, k;
    string s;

    cin >> n >> C >> k;
    cin >> s;

    ll q = 0, C2;
    C2 = C;
    while (C2 > 0) {
        C2 /= 10;
        ++q;
    }

    ll mod = 1;
    for (int i = 0; i < k; i++)
    {
        mod *= 10;
    }

    for (int i = 0; i < n; i++)
    {
        if (i <= q && (stoll(s.substr(0, i + 1)) <= C && s[0] != '0') || (i == 0 && s[0] == '0'))
        {
            dp[i] = (dp[i] + 1) % mod;
        }

        for (int j = i; j > max(0ll, i - q); j--)
        {
            if ((stoll(s.substr(j, i - j + 1)) <= C && s[j] != '0') || (i == j && s[j] == '0')) {
                dp[i] = (dp[i] +  dp[j - 1]) % mod;
            }
        }
    }

    cout << dp[n - 1];
}
