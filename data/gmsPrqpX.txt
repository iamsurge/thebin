-- Enter Accumulator Number
accumID = 10
print("Computer set as ID: " .. accumID)

-- Enter Redstone Side [default = "back"]
redstoneSide = "back"
print("Redstone Input Set: " .. redstoneSide)

-- Enter Modem Side
modemSide = "left"
print("Modem Output Set: " .. modemSide)

-- Modem Init
rednet.open(modemSide)
print("Modem Initiated")

hostID = 1
rednet.send(hostID, "Hello from Id: " .. accumID)

while true do
	redstoneValue = redstone.getAnalogInput(redstoneSide)
	percentFull = math.floor((redstoneValue / 15) * 100)
	rednet.send(hostID, percentFull)\
	print("Sent battery percent " .. percentFull .. "%")
	os.sleep(3)
end
