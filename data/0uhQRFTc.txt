#include <iostream>

using namespace std;


int main() {
    int n;
    cin >> n;
    
    float niza[n];
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
    }
    float x;
    cin >> x;
    
    int dali_brojot_e_najden = 0;
    for(int i = 0; i < n; i++) {
        
        if(niza[i] == x) {
            cout << i << " ";
            dali_brojot_e_najden = 1;
        }
        
    }
    
    if(dali_brojot_e_najden == 0) {
        cout << "Baranata vrednost ne e vo nizata" << endl;
    }
    
    
    return 0;
}
