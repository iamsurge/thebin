ll n, x;
vector<ll> v, dp;
ll coins(ll curr)
{
    if (curr == 0)
        return 0;
    ll &ret = dp[curr];
    if (~ret)
        return ret;
    ret = 1e9;
    for (ll i = 0; i < n; i++)
    {
        if (curr - v[i] >= 0)
        {
            ret = min(ret, 1 + coins(curr - v[i]));
        }
    }
    return ret;
}
void solve()
{
    cin >> n >> x;
    v.resize(n);
    cin >> v;
    dp.assign(1e6, -1);
    int ans = coins(x);
    if (ans == 1e9)
        cout << "-1";
    else
        cout << ans;
}
