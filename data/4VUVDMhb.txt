import java.util.*;
 
public class Question1 {
	public static String getHint(String secret, String guess) {
 
	 /*
  * ********************
  * ADD YOUR CODE HERE
  * *********************
   */
   
    int bulls = 0;
    int cows = 0;
    
    
    ArrayList<Character> charsecret = new ArrayList<Character>();
    for (char c : secret.toCharArray()) {
      charsecret.add(c);
    }
    
    ArrayList<Character> charguess = new ArrayList<Character>();
    for (char c: guess.toCharArray())
    {
        charguess.add(c);       
    }
    
    for(int i = 0; i < charsecret.size(); i++)
    {
        if(charsecret.get(i) == charguess.get(i))
        {
            bulls++;
            charsecret.remove(i);
            charguess.remove(i);
            i--;
        }
        
    }
    
    HashMap<Character, Integer> hmsecret = new HashMap<Character, Integer>(); //making freq map
    HashMap<Character, Integer> hmguess = new HashMap<Character, Integer>();
    
    for(char c: charsecret)
    {
        hmsecret.merge(c, 1, Integer::sum);
    }
    for(char c: charguess)
    {
        hmguess.merge(c, 1, Integer::sum);
    }
    
    //now iterate over the map elements of charguess and check them for presence in charsecret
    
    for(Map.Entry<Character, Integer> me: hmguess.entrySet())
    {
        char key = me.getKey();
        int val = me.getValue();
        if(hmsecret.containsKey(key))
        {
            cows+= Math.min(val, hmsecret.get(key));
        }
    }
    
    // for(char c: charsecret)
    // {
    //     System.out.print(c);
    // }
    // System.out.println();
    // for(char c: charguess)
    // {
    //     System.out.print(c);
    // }
    // System.out.println();
    
    // for(char c: charguess)
    // {
    //     if(charsecret.contains(c))
    //     {
    //         cows++;
    //     }
    // }
    
    return Integer.toString(bulls) + "B" + Integer.toString(cows) + "C" ;
    
 
}
 
 
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String secret = sc.nextLine();
        String guess = sc.nextLine();
 
        System.out.println(getHint(secret, guess));
        sc.close();
    }
}
