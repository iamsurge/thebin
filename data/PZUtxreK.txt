import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner obj=new Scanner(System.in);
        int n=obj.nextInt();
        int ctr1=0;
        int ctr2=0;
        for(int i=n;i>0;i=i/2){
            if(i%2==1){
                ctr1++;
            }
            if(i%2==0){
                ctr2++;
            }
        }
        System.out.println(ctr1);
        System.out.println(ctr2);
    }
}
