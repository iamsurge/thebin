function round(num, zap)
	if zap == 2 then
		zap = zap and zap + 1 or 0
		if string.find(tostring(num), 'e%p') then
			reslt = 0
		else
		reslt = string.sub(tostring(num), 1, (string.find(tostring(num), '.%p') or #tostring(num)) + zap) end
		print(reslt, '--', num)
	end

	zap = zap and zap + 1 or 0
	if string.find(tostring(num), 'e%p') then
		return 0
	end
	return string.sub(tostring(num), 1, (string.find(tostring(num), '.%p') or #tostring(num)) + zap)
end
