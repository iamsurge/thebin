#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod[5] = {2000000153, 2500000199, 2500000033, 2500000117, 1000000007};
const ll base = 397;
string s;
ll lens, hashs[5][1000005], pbase[5][1000005];
void input(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	cin >> s, lens = s.size(), s = '#' + s;
}
void build(){
	ll j;
	for(ll i = 0; i < 5; i++) hashs[i][0] = 0, pbase[i][0] = 1; 
	for(ll i = 1; i <= lens; i++){
		j = i - 1;
		for(ll ind = 0; ind < 5; ind++){
			hashs[ind][i] = (hashs[ind][j] * base + s[i]) % mod[ind];
			pbase[ind][i] = pbase[ind][j] * base % mod[ind];
		}
	}
}
ll gethashs(const ll ind, const ll i, const ll j){
	return (hashs[ind][j] - hashs[ind][i - 1] * pbase[ind][j - i + 1] + mod[ind] * mod[ind]) % mod[ind];
}
bool check(const ll i1, const ll j1, const ll i2, const ll j2){
	for(ll i = 0; i < 5; i++) if(gethashs(i, i1, j1) != gethashs(i, i2, j2)) return false;
	return true;
}
int main(){
	input();
	build();
	for(ll i = 1; i < lens; i++) if(check(1, i, lens - i + 1, lens)) cout << i << " ";
}
