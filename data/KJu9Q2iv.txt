#!/usr/bin/perl

use strict;
use warnings;

my $part1 = 0;
my $part2 = 0;

while (<>) {
    # convert input to majuscle ordinals
    my ($a, $b) = map { ord($_) - ord('A') } split;
    $b -= 23;   # shift from X

    # LDW is (me - you + 1) % 3 (+1 to shift to 0-2), move score is b + 1
    $part1 += ($b - $a + 1) % 3 * 3 + $b + 1;

    # LWD is just 3 * b, move score is (a + b) mod 3, but with residue on 1 to 3
    $part2 += ($a + $b - 1) % 3 + 1 + 3 * $b;
}

print "Part 1: $part1\n";
print "Part 2: $part2\n";
