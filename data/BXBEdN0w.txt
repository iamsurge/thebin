    end

        nextbet = bigbase 
        chance  = bigch 
    else 
        if (chance < 20) then 
            nextbet = nextbet + (nextbet * (6.8 / 100)) 
        else 
            nextbet = previousbet * 1.55 
        end 

        if(currentstreak == -1) then 
            nextbet = smallbase 
            chance  = math.random(310, 340) / 100  
        end

            chance  = math.random(2350, 2450) / 100

        end
