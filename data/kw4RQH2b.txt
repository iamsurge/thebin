function solve(input) {
    let city = input[0]
    let bay = Number(input[1]);

    let sum = 0;

    switch (city) {

         case "Sofia":

              if (bay <= 500) {
                   sum = bay * 0.05;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay <= 1000) {
                   sum = bay * 0.07;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay <= 10000) {
                   sum = bay * 0.08;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay > 10000) {
                   sum = bay * 0.12;
                   console.log(sum.toFixed(2));
                   break;
              }
              if (bay === "- 0.00" && city === "") {
                   console.log("error");
              }


         case "Varna":
              if (bay <= 500) {
                   sum = bay * 0.045;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay <= 1000) {
                   sum = bay * 0.075;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay <= 10000) {
                   sum = bay * 0.10;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay > 10000) {
                   sum = bay * 0.13;
                   console.log(sum.toFixed(2));
                   break;
              }
              if (bay === "- 0.00" && city === "") {
                   console.log("error");
              }

         case "Plovdiv":
              if (bay <= 500) {
                   sum = bay * 0.055;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay <= 1000) {
                   sum = bay * 0.08;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay <= 10000) {
                   sum = bay * 0.12;
                   console.log(sum.toFixed(2));
                   break;
              } else if (bay > 10000) {
                   sum = bay * 0.145;
                   console.log(sum.toFixed(2));
                   break;

              }

              if (bay === "- 0.00" && city === "" ) {
                   console.log("error");
              } 

         default:
              console.log("error");
              break;

    }

}
