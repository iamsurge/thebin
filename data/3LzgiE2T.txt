using System;

namespace Oscars
{
	class Program
	{
		static void Main(string[] args)
		{
		    const double starsPoint = 1250.5;
		    
			string name = Console.ReadLine();
			double points = double.Parse(Console.ReadLine());
			int numOfRaters = int.Parse(Console.ReadLine());
			
			for(int i = 0; i < numOfRaters; i++)
			{
			    if(points > starsPoint)
			    {
			        i = numOfRaters;
			    }
			    else
			    {
			        points += Console.ReadLine().Length * double.Parse(Console.ReadLine()) / 2.0;
			    }
			}
			
			if(points > starsPoint)
			{
			    Console.WriteLine($"Congratulations, {name} got a nominee for leading role with {points:F1}!");
			}
			else
			{
			    Console.WriteLine($"Sorry, {name} you need {(starsPoint - points):F1} more!");
			}
		}
	}
}
