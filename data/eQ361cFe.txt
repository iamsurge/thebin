#!/usr/bin/env python
# -*- coding: utf-8 -*-

# para descargar openpyxl
# python -m pip install openpyxl

from openpyxl import Workbook
# creo un libro de Excel
libro = Workbook()

# creo una hoja
hoja = libro.active

# Agrego el nro 42 a la celda A1
hoja['A1'] = 42

# Agrego 3 nros a las celdas A2,B2,C2
hoja.append([1, 2, 3])

# Imprimo la hora
import datetime
hoja['A3'] = datetime.datetime.now()

# Grabo el libro
libro.save("PYTHON.xlsx")
