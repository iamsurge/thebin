#include <iostream>
#include <vector>
#include <bitset>
#include <algorithm>
#include <map>
#include <unordered_map>

#include <cmath>
#include <iomanip>

using namespace std;

#define int ll

typedef unsigned long long ll;
typedef long double ld;

const int N_PYRAMIDS = 3e7;
ll pyramids[N_PYRAMIDS];

signed main()
{
#ifdef HOME_DIDUK001
    freopen("in", "r", stdin);
    freopen("out", "w", stdout);
#endif

    cout << fixed << setprecision(12);
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    for (int i = 1; i < N_PYRAMIDS; i++)
    {
        pyramids[i] = pyramids[i - 1];
        ll t;
        if (i & 1)
        {
            t = i;
            t *= (i + 1) >> 1;
        }
        else
        {
            t = i + 1;
            t *= i >> 1;
        }
        pyramids[i] += t;
    }

    ll k;
    cin >> k;

    int l = 1;
    for (int r = 2; r < N_PYRAMIDS; r++)
    {
        ll diff = pyramids[r] - pyramids[l];
        while (l < r && diff > k)
        {
            l++;
            diff = pyramids[r] - pyramids[l];
        }
        if (diff == k)
        {
            cout << r << '\n'
                 << l << '\n';
            return 0;
        }
    }
    cout << -1 << '\n';
}
