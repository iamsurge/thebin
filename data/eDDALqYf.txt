#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <queue>

using namespace std;

class Nums {
public:
    double num;
    int chis;
    int znm;
    Nums(double num):num(num), chis(num), znm(1) {}
    Nums(double num, int chis, int znm): num(num), chis(chis), znm(znm){}

};

bool operator<(const Nums& num1, const Nums& num2) {
    return num1.num < num2.num;
}

int main(){
    int sz, times;
    double n;
    priority_queue<Nums, vector<Nums> > pq;
    while(cin >> sz >> times) {
        for (int i = 0; i < sz; i++) {
            cin >> n;
            pq.push(Nums(n));
        }
        for (int i = 0; i < times; i++) {
            Nums o = pq.top();
            pq.pop();
            pq.push(Nums(o.num/2., o.chis, o.znm * 2));
            pq.push(Nums(o.num/2., o.chis, o.znm * 2));
        }

        Nums p = pq.top();
        while(p.chis%2==0){
            p.chis= p.chis/2;
            p.znm = p.znm/2;
        }
        cout << p.chis << "/" << p.znm << endl;
    }
    return 0;
}
