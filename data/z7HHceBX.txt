function flatten(arr){
    let flat = [];
   
    for(let i = 0; i < Array.from(arr).length; i++){
        if(Array.isArray(arr[i])){
            flat = flat.concat((function(subArr){
                return flatten(arr[i]);
            })(arr[i]))
        }
        else{
            flat.push(arr[i]);
        }
    }

    return flat;
}
