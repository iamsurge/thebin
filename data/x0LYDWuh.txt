"//    For the Robotics.org.za  ESP32-C3 Mini-1 development board //    By Pieter from Innovation Instruments //\r\n\r\n#include <stdio.h>\r\n\r\n#define Red_LED 3        // rgb led pins, active low #define Green_LED 4 #define Blue_LED 5 #define LED 21            // onboard blue led\r\n\r\nunsigned int  Loop_Count = 0;\r\n\r\n\r\n//\r\n// Setup\r\n//\r\nvoid setup() // put your setup code here, to run once:\r\n{\r\n   pinMode(LED,OUTPUT);\r\n   pinMode(Red_LED,OUTPUT);\r\n   pinMode(Green_LED,OUTPUT);\r\n   pinMode(Blue_LED,OUTPUT);\r\n   digitalWrite(Red_LED,HIGH);\r\n   digitalWrite(Green_LED,HIGH);\r\n   digitalWrite(Blue_LED,HIGH);\r\n}\r\n//\r\n// main loop\r\n//\r\nvoid loop()   // put your main code here, to run repeatedly:\r\n{\r\n   digitalWrite(LED,HIGH);\r\n   delay(200);\r\n   digitalWrite(LED,LOW);\r\n   delay(200);\r\n\r\n   Loop_Count++;\r\n   if (Loop_Count == 3)\r\n   {\r\n     digitalWrite(Red_LED,LOW);    // on\r\n     digitalWrite(Green_LED,HIGH);\r\n     digitalWrite(Blue_LED,HIGH);\r\n   }\r\n   else if (Loop_Count == 6)\r\n   {\r\n     digitalWrite(Red_LED,HIGH);\r\n     digitalWrite(Green_LED,LOW);   // on\r\n     digitalWrite(Blue_LED,HIGH);\r\n   }\r\n   else if (Loop_Count == 9)\r\n   {\r\n     digitalWrite(Red_LED,HIGH);\r\n     digitalWrite(Green_LED,HIGH);\r\n     digitalWrite(Blue_LED,LOW);   // on\r\n     Loop_Count = 0;\r\n   }\r\n   else                            // all off\r\n   {\r\n     digitalWrite(Red_LED,HIGH);\r\n     digitalWrite(Green_LED,HIGH);\r\n     digitalWrite(Blue_LED,HIGH);\r\n   }\r\n}\r\n"
