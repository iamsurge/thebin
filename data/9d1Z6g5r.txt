import java.util.Scanner;
public class Jurnal03 {
	private static void methodTiga(int radius, int height){
		float sisiMiring;
		float luasKerucut, luasTabung;
		sisiMiring = (height * height) + (radius * radius);
		sisiMiring = (float) Math.sqrt(sisiMiring);
		luasKerucut = (float) ((radius + sisiMiring) * Math.PI * radius);
		System.out.printf("%.3f",luasKerucut);
		System.out.print(" ");
		luasTabung = (float) ((height + radius) * 2 * Math.PI * radius);
		System.out.printf("%.3f",luasTabung);
	}

	public static void main(final String[] args) {
		//Todo : Scanner
		Scanner input = new Scanner(System.in);
		//Todo : Jurnal 3
		int radiusMain, heightMain;
		radiusMain = input.nextInt();
		heightMain = input.nextInt();
		methodTiga(radiusMain, heightMain);
	}
}
