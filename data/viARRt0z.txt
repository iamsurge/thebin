#include <iostream>
using namespace std;

void zamiana(int num) {
    int i = 0, tab[31];

    while (num) {
        tab[i++] = num % 2;
        num /= 2;
    }
    for (int j = i - 1; j >= 0; j--) {
        cout << tab[j];
    }
}
int main()
{
    int liczba;
    cout << "Podaj liczbe (10)";
    cin >> liczba;

    cout << liczba << " na bin: ";
    zamiana(liczba);
    cout << endl;

    return 0;
}
