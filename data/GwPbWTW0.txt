#include<bits/stdc++.h>
using namespace std;
int a[12];
int n,k;
bool recurse(int state,int res){
	if(state == n){
		return (res == k);
	}
	if(recurse(state+1,res+a[state+1]) || recurse(state+1,res-a[state+1]) || recurse(state+1,res*a[state+1]) || recurse(state+1,res/a[state+1]))
		return true;
	else
		return false;
}
