struct AddressHasher {
    // реализуйте структуру
    size_t operator()(const Address& a) const
    {
        size_t x = 113;
        return x * x * shash(a.street) + x * shash(a.street) + a.building;
    }
    hash<string> shash;
};

struct PersonHasher {
    // реализуйте структуру
    size_t operator()(const Person& p)
    {
        size_t x = 151;
        size_t result = ahash(p.address);
        return x * x * x;
    }
    hash<string> shash;
    hash<double> dhash;

};
