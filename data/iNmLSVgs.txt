print("Making staircase now, please hold...")

--Starting sequence--

if turtle.detectDown() then
    turtle.digDown()
    turtle.down()
else
    turtle.down()
end

--If starting block not detected--
if not turtle.detect() then
    --place block--
end

turtle.turnLeft()
if turtle.detect() then
    turtle.dig()
end

--Dig down, move down, if no block, place--
if turtle.detectDown() then
    turtle.digDown()
    turtle.down()
    if not turtle.detectDown() then
        --place block--
    end
end

if turtle.detect() then
    turtle.dig()
end

function Dig --direction is Down,Up,Left,Right. "Blank" for foward.
    if turtle.detect() then
        turtle.dig()
        turtle.forward()
        if not turtle.detect() then
            --place block--
        end
    end
end
