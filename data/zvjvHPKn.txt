#include <iostream>
using namespace std;
 
int   a;
float b, 
      s;
 
///////////////////////////////
int main()                  //
{
    
   cin >> a;
   
   for(int i = 0; i < a; i++)
   {
      cin >> b; 
      s += b; 
   }
   
   s = s / a;
   cout.precision(12);
   cout <<  fixed << s;
}
