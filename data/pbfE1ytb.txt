#include <stdlib.h>
#include <string.h>
#include <stdio.h>
#include <time.h>

double f(double x) {
    return x * x * x * x + 2 * x * x * x - x - 1;
}

double solve(double a, double b, double e) {
    double c;
    while (b - a > e) {
        c = (a + b) / 2;
        if (f(b) * f(c) < 0)
            a = c;
        else
            b = c;
    }
    return (a + b) / 2;
}

int main(int argc, char *argv[]) {
    double x, res;
    FILE *input, *output;
    input = fopen(argv[2], "r");
    output = fopen(argv[3], "w");
    fscanf(input, "%lf", &x);
    clock_t begin = clock();
    res = solve(0, 1, 0.000001);
    clock_t end = clock();
    fprintf(output, "%lf\n", res);
    double time_spent = (double) (end - begin) / CLOCKS_PER_SEC;
    printf("Program execution time: %lf\n", time_spent);
    fclose(input);
    fclose(output);
    return 0;
}
