# hacer una funcion que calcule el area de un circulo y otra que
# calcule su perimetro, ambas en funcion del radio

# area = pi * radio**2  perim = 2*pi*r 

def menu():
    return"""
    Menu de opciones:
    1. calcula el perimetro
    2. calcula el Area
    3. Salir
    """
    
def area_del_circulo():
    radio = float(input("Ingrese el radio: "))
    return f"Area del circulo: {3.1416 * radio**2:.2f}"
    
def perimetro_del_circulo():
    radio = float(input("Ingrese el radio: "))
    return f"Perimetro del circulo {2 * 3.1416 * radio:.2f}"  


print("Calcular area o Perimetro de un circulo")
print("---------------------------------------")

while True:
    print(menu())
    opcion = input("Seleccione su opción: ")
    
    if opcion == "1":
        print(perimetro_del_circulo())
    
    elif opcion == "2":
        print(area_del_circulo())
        
    elif opcion == "3":
        print("Gracias por usar este programa...")
        break
        
    else:
        print("Opción incorrecta...")
