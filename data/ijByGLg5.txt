Extraer Cadenas dentro de otra cadena con delimitadores. Utilizando la librería re y utilizando función creada por el usuario.

#!/usr/bin/python3
import re

def between_all(text: str, starting:str, ending:str ) -> list:
    "Función que extrae contenido de una cadena, con los delimitadores como parametros."
    i = -1
    founded = []
    part = text.replace(ending, ending + chr(1)+ chr(1)).split(chr(1)+ chr(1))
    while i < len(part) - 1:
        s = part[i].find(starting) + len(starting)
        e = part[i].find(ending)
        if len(part[i])> 0 and part[i].find(starting) > -1 and part[i].find(ending) > -1:
            founded.append(part[i][s:e])
        i += 1
    return founded


cadena = "hola <img src=host.com/image0.jpg><img src=host.com/image1.jpg><img src=host.com/image3.jpg><img src=host.com/image4.jpg><img src=host.com/image5.jpg><img src=host.com/image6.jpg>"

print("utilizando función creada\n")
links = between_all(cadena,"<img src=",".jpg>")
for link in links:
    print(link)

print("\nutilizando modulo re\n")
enlaces = re.findall("<img src=(.*?).jpg>", cadena)
for enlace in enlaces:
    print(enlace)
