Tambien muestra un mensaje de error si el valor ingresado no es numerico.
*/ 
import java.util.Scanner;

public class punto02 {

    public static void main(String[] args) {
        int a, b, op = 0;

        Scanner teclado = new Scanner(System.in);
        try {
            do {
                menu();
                op = teclado.nextInt();
                switch (op) {
                    case 1:
                        System.out.println("ingrese un Nro...!!");
                        a = teclado.nextInt();
                        System.out.println("ingrese otro Nro...!!");
                        b = teclado.nextInt();
                        System.out.println("--->Los Nros son: " + a + " y " + b);
                        break;
                    case 2:
                        a = (int) (Math.random() * 100 + 1);
                        b = (int) (Math.random() * 100 + 1);
                        System.out.println("--->Los Nros son: " + a + " y " + b);
                        break;
                    case 3:
                        System.out.println("Adios...");
                        break;
                    default:
                        System.out.println("Error!!....opcion incorrecta");
                        break;
                }
            } while (op != 3);

        } catch (Exception e) {
            System.out.println("Error!!!...deben ser numeros");
        }

    }

    public static void menu() {
        System.out.println("\n1 - N째 por consola" + "\n2 - N째 Aleatorios" + "\n3 - Salir" + "\n\nIngrese opcion...");

    }

}
