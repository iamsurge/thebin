app="Films By Kris"
desktop_file="$HOME/Desktop/fbk.desktop"
icon="$HOME/.fbk_icon.png"

echo "Installing ${app} to Desktop..."

wget -qO "$icon" "https://filmsbykris.com/v7/img/tux.png"

echo "#!/usr/bin/env xdg-open
[Desktop Entry]
Version=1.0
Terminal=false
Type=Application
Name=${app}
Exec=brave-browser --app='https://filmsbykris.com/'
Icon=${icon}" > "$desktop_file"

chmod +x "$desktop_file"
