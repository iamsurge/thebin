#include<bits/stdc++.h>
using namespace std;
const int lim = 2e6 + 1;
string txt, pat, s;
int n, k, l = 0, r = 0, res = 0, z[lim];
void buildZ(){
	for(int i = 1; i < n; i++){
		if(i > r){
			l = r = i;
			while(r < n && s[r] == s[r - l]) r++;
			z[i] = r - l, r--;
		}
		else{
			k = i - l;
			if(z[k] <= r - i) z[i] = z[k];
			else{
				l = i;
				while(r < n && s[r] == s[r - l]) r++;
				z[i] = r - l, r--;
			}
		}
	}
}
void Zalgo(){
	for(int i = pat.size(); i < n; i++) if(z[i] == pat.size()) res++;
}
int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	cin >> txt >> pat;
	s = pat + '&' + txt, n = s.size();
	buildZ();
	Zalgo();
	cout << res;
}
