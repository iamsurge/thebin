/*******************************************************************************
 *  Scrivere un programma che usi l’iterazione per stampare la tabella di      *
 * valori allegata nella mail.                                                 *
 * Usare la sequenza di escape tab (\t) nell’istruzione printf per separare    *
 * le colonne con tabulazioni.                                                 *
 *                                                  Simone Flavio P.           *
 ******************************************************************************/
#include <stdio.h>

int main()
{
    // Fase di inizializzazione delle variabili necessarie
    int n = 1;
    int col2;
    int col3;
    int col4;

    printf("n\t10*n\t100*n\t1000*n\n\n");
    
    while (n<=10){
        col2 = n*10;
        col3 = n*100;
        col4 = n*1000;
        printf("%d\t%d\t%d\t%d\n", n, col2, col3, col4);
        n++;
    }
}
