#Unos proteinske sekvence od strane korisnika
proteinska_sekvenca=input("Unesi proteinsku sekvencu:")
#Unos peptida od strane korisnika
peptid=input("Unesi peptid:")
#count broji koliko peptida se nalazi u preteinskoj sekvenci
count=proteinska_sekvenca.count(peptid)
#ispis koliko puta se uneseni peptid nalazi u preteinskoj sekvenci
print("U proteinskoj sekvenci željeni peptid nalazi se {} puta" .format(count))
#for petlja prolazi kroz cijelu proteinsku sekvencu
for i in range (len(proteinska_sekvenca)):
    #startswith naredbom se provjerava nalazi li se peptid unutar proteinske sekvence. Ako se peptid ne nalazi unutar proteinske sekvence daljnji kod se neće nastaviti i program će stati
    if proteinska_sekvenca[i:].startswith(peptid):
        #ako se if uvjet iz gornjeg reda ostvario ispisati će se pozicija na kojoj se nalazi unešeni peptid
        print("{} je pronađen na {} poziciji" .format(peptid, i+1))
        #program će se nastaviti izvršavati sve dok se ne prođe cijela for petlja, odnosno provjeri cijela proteinska sekvenca
        continue
