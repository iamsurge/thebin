#! /bin/bash
# $PROG: fclock - a simple ascii clock
# dependency : figlet
# debian-base OS: sudo apt update && sudo apt upgrade -y ; sudo apt install figlet
# if not installed in system , press Ctrl-C to quit script
colorpick(){
	local color
	color=(\\033[31m \\033[32m \\033[33m \\033[34m \\033[35m \\033[36m)
	count=${#color[@]}
	pick=$(echo ${color[$((RANDOM%count-1))]})
	echo "$pick"
}



while : ; do
        clear
        d=$(date +%T)
        echo -e "$(colorpick)$(figlet $d)"
        sleep 1
done
