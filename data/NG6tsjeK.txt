# Criptografar senha usando letra e número seguinte a digitada pelo usuário

<?php
    function criptografarSenha($senha) {
        $senhaCriptografada = "";

        for ($posicao = 0; $posicao < strlen($senha); $posicao++) {
            $letra = ord($senha[$posicao]) + 1;
            $senhaCriptografada .= chr($letra);
        }

        return $senhaCriptografada;
    }
?>

----

# Gerar hash da senha

<?php
    function gerarHash($senha) {
        $txt = criptografarSenha($senha);
        $hash = password_hash($txt, PASSWORD_DEFAULT);
        return $hash;
    }
?>

----

# Testar se hash equivale a senha

<?php
    function testarHash($senha, $hash) {
        $resultado = password_verify($senha, $hash);
        return $resultado;
    }
?>
