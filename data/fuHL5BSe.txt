#include <bits/stdc++.h>
using namespace std;

int main() {
	ios::sync_with_stdio(false); cin.tie(0);
	int n, l, x, y, mode; cin >> n >> x >> y >> l >> mode;
	char c; cin >> c;
	char ar[n][n]; memset(ar, '.', sizeof(ar));
	int cw[4][2]={{0,1},{1,0},{0,-1},{-1,0}};
	int pi=y, pj=x;
	for(int i=0;i<l;i++) {
		int di=cw[i%4][0], dj=pow(-1, mode)*cw[i%4][1];
		for(int j=0;j<i+1;j++) {
			if(pi>=0&&pi<n&&pj>=0&&pj<n) ar[pi][pj]=c;
			pi+=di, pj+=dj;
		}
	}
	for(auto& i:ar) {
		for(auto j:i)
			cout << j;
		cout << '\n';
	}
	return 0;
}
