import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        int n;
        char x;
        char y;
        Scanner scanner = new Scanner(System.in);
        n=scanner.nextInt();
        scanner.nextLine();
        x=scanner.next().charAt(0);
        y=scanner.next().charAt(0);
        Boolean flag = true;
        int cnt=0;
        String[] arr = new String[n];

        for(int i =1;i<arr.length;i++){
            arr[i] = Integer.toString(i);
            char[] check = arr[i].toCharArray(); // перевод в массив Char
            if(CheckX(check,x,flag,y)){
                cnt++;
            }



        }
        System.out.println(cnt);
    }
    static Boolean CheckX(char[] check, int x, Boolean flag,int y){
        for(int j =0;j<check.length;j++){ // поразрядный переход по массиву check от 0 до размера числа n
            flag = true;
            if(check[j]==x){ // рассмотрения случая 2 для x
                for(int z=0;z<check.length;z++){ //проверка остальных цифр check массива
                    if(z==j){ continue;} // пропуск числа x  в алгоритме поиска

                    if(check[z]==y){ // 1 случай найден.
                        flag = true;
                        return true;
                    }

                    if(check[z]!=x){ // 3 случай
                        flag = false;
                    }

                }

                if(flag){
                    return true;
                }

            }
            if(check[j]==y){
                for(int z=0;z<check.length;z++){ //проверка остальных цифр check массива
                    if(z==j){ continue;} // пропуск числа x  в алгоритме поиска

                    if(check[z]==x){ // 1 случай найден.
                        flag = true;
                        return true;
                    }

                    if(check[z]!=y){ // 3 случай
                        flag = false;
                    }

                }

                if(flag){
                    return true;
                }
            }

        }
        return false;
    }
}
