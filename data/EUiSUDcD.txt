Move = {}
function Move:forward(amount, safe)
  if amount < 0 then
    Move:uturn()^
    amount = -amount
  end
  while amount >= 1 do
    if(safe) then
      while not turtle.forward() do
        turtle.dig()
      end
    else
       turtle.forward()
    end
    amount = amount - 1
  end
end

function Move:uturn()
  Move:left(2)
end

function Move:left(amount)
  if amount < 0 then
    Move:right(amount)
    return
  end
  while amount >= 1 do
    turtle.turnLeft()
    amount = amount - 1
  end
end

function Move:right(amount)
  if amount < 0 then
    Move:left(amount)
    return
  end
  while amount >= 1 do
    turtle.turnRight()
    amount = amount - 1
  end
end

Move:forward(5, true)
Move:left(2)
Move:right(2)
Move:forward(-5, true)
Move:forward(5, false)
Move:uturn()
