#include <iostream>

using namespace std;

float najgolem_broj(int n, float niza[]) {
    float najgolem = niza[0];
    for(int i = 0; i < n; i++) {
        if(niza[i] > najgolem) {
            najgolem = niza[i];
        }
    }
    return najgolem;
}

int main() {
    int n;
    cin >> n;
    
    float niza[n];
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
    }
    
    cout << najgolem_broj(n, niza) << endl;
    return 0;
}
