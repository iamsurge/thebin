def solve():
	words = ['holy','grail','monty','python','life','of','brian','fool','rain']
	matrix = ['nlifeloofw','liargoyloh','oanairbens','troeasonst','swgndmkdhr']
	matrix = ''.join(matrix)
	found = [word for word in words if word in matrix+matrix[::-1]]
	print('Found: '+', '.join(found))
	print('Not found: ', ', '.join(i for i in set(words)^set(found)))
solve()
