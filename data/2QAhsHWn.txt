using System;

namespace _05.SuitcasesLoad
{
	internal class Program
	{
		static void Main(string[] args)
		{
			double capacityTrunk = double.Parse(Console.ReadLine());

			string input;
			double totalVolumeSuitcases = 0;
			double differentCapacities = 0;
			int cntSuitcases = 0;

			while ((input = Console.ReadLine()) != "End")
			{
				double suitaseVolume = double.Parse(input);

				cntSuitcases++;
				if (cntSuitcases % 3 == 0)
				{
					totalVolumeSuitcases += (suitaseVolume + (suitaseVolume * 0.10));
				}
				else
				{
					totalVolumeSuitcases += suitaseVolume;
				}
				differentCapacities = capacityTrunk - totalVolumeSuitcases;

				if (differentCapacities < 0)
				{
					Console.WriteLine($"No more space!");
					cntSuitcases--;
					break;
				}
			}

			if (differentCapacities >= 0)
			{
				Console.WriteLine($"Congratulations! All suitcases are loaded!");
			}

			Console.WriteLine($"Statistic: {cntSuitcases} suitcases loaded.");
		}
	}
}
