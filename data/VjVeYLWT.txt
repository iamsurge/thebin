import java.util.Scanner;

public class demo {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);



        int daysOfStay = Integer.parseInt(scanner.nextLine());
        String room = scanner.nextLine();
        String evaluation = scanner.nextLine();

        double roomForOnePrice = 0;
        double apartmentPrice = 0;
        double presidentApartmentPrice = 0;

        switch (room) {

            case "room for one person":
                if (daysOfStay <= 10) {
                    roomForOnePrice = (daysOfStay - 1) * 18.00;
                } else {
                    roomForOnePrice = (daysOfStay - 1) * 18.00;



                }
                if ("positive".equals(evaluation)) {
                    roomForOnePrice = roomForOnePrice + roomForOnePrice * 0.25;
                } else if   ("negative".equals(evaluation)) {
                    roomForOnePrice = roomForOnePrice - roomForOnePrice * 0.1;
                    
                }
                System.out.printf("%.2f", roomForOnePrice);

                break;

            case "apartment":
                if (daysOfStay <= 10) {
                    apartmentPrice = (daysOfStay - 1) * 25.00;
                    apartmentPrice = apartmentPrice - apartmentPrice * 0.3;
                } else if (daysOfStay > 10 && daysOfStay <= 15) {
                    apartmentPrice = (daysOfStay - 1) * 25.00;
                    apartmentPrice = apartmentPrice - apartmentPrice * 0.35;
                } else {
                    apartmentPrice = (daysOfStay - 1) * 25.00;
                    apartmentPrice = apartmentPrice - apartmentPrice * 0.5;


                }
                if ("positive".equals(evaluation)){
                    apartmentPrice = apartmentPrice + apartmentPrice * 0.25;
                }else {
                    apartmentPrice = apartmentPrice - apartmentPrice * 0.1;
                    
                }
                System.out.printf("%.2f", apartmentPrice);

                break;

            case "president apartment":
                if (daysOfStay <= 10) {
                    presidentApartmentPrice = (daysOfStay - 1) * 35.00;
                    presidentApartmentPrice = presidentApartmentPrice - presidentApartmentPrice * 0.1;
                } else if (daysOfStay > 10 && daysOfStay <= 15) {
                    presidentApartmentPrice = (daysOfStay - 1) * 35.00;
                    presidentApartmentPrice = presidentApartmentPrice - presidentApartmentPrice * 0.15;
                } else {
                    presidentApartmentPrice = (daysOfStay - 1) * 35.00;
                    presidentApartmentPrice = presidentApartmentPrice - presidentApartmentPrice * 0.2;

                }

                if ("positive".equals(evaluation)) {
                    presidentApartmentPrice = presidentApartmentPrice + presidentApartmentPrice * 0.25;
                    
                } else {
                    presidentApartmentPrice = presidentApartmentPrice - presidentApartmentPrice * 0.1;
                   
                }
                System.out.printf("%.2f", presidentApartmentPrice);
                break;

        }


    }
}
