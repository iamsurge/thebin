#include <stdio.h>
#include <stdlib.h>

int main(){
    int A, B;
    scanf("%d%d", &A, &B);
    // 5 10
    
    
    for(int i = A + 1; i <= B; i++) { // 6
        long long factorial = 1;
        
        for(int j = 1; j <= i; j++) { // j = 6
            factorial *= j;
            //120 * 6 = 720
        }
        
        printf("Faktoriel od %d e %ld\n", i, factorial);
    }
    
    
    
    return 0;

}
