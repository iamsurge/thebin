bool sh = 1;
int N, n;

vector <int> gen() {
    srand(time(0));
    n = rand() % 5 + 2;
    vector <int> a;
    for (int i = 0; i < n; ++i) {
        int x = rand() % 20 + 1;
        a.pb(x);
        N += x;
    }
    sort(a.begin(), a.end());
    reverse(a.begin(), a.end());
    //cv(a);
    return a;
}

bool ok(vector <int> a) {
    for (int i = 0; i < a.size() - 1; ++i) {
        if (a[i] == a[i + 1]) {
            return 0;
        }
    }
    return 1;
}

vector <int> my(vector <int> a) {
    int dif = a[0] - a[1] - 1;
    a[0] -= dif;
    vector <int> ans(N);
    int id = -1, cnt = 0;
    int to = n;
    if (sh) {
        cout << "A\n";
    }
    while (cnt < N) {
        int j = -1;
        if (sh) {
            cout << "B\n";
        }
        while (j + 1 < to) {
            id++;
            if (sh) cout << "id j = " << id << ' ' << j << "\n";
            if (j == -1) {
                ans[id] = 0;
                j = 0;
                a[0]--;
                cnt++;
                continue;
            }
            if (dif > 0) {
                if (id == 0 || id > 0 && ans[id - 1] != 0) {
                    ans[id] = 0;
                    dif--;
                    cnt++;
                    continue;
                }
            }
            j++;
            a[j]--;
            cnt++;
            ans[id] = j;
            if (a[j] == 0) {
                to = min(to, j);
            }
        }
    }
    if (ok(ans)) {
        return ans;
    }
    return {-1};
}
