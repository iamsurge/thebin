from random import uniform
import matplotlib.pyplot as plt


settings = {
    "spacing_min": 0.1,
    "spacing_max": 1,
    "x_size": 50,
    "y_max": 4,
    "y_min": 1
}


def generate_points():
    total_size = 0
    x_list1 = []
    y_list1 = []
    while total_size < settings["x_size"]:
        x_spacing = uniform(settings["spacing_min"], settings["spacing_max"])

        if not y_list1:
            y = uniform(-settings["y_max"], settings["y_max"])
        elif y_list1[-1] > 0:
            y = uniform(-settings["y_max"], -settings["y_min"])
        elif y_list1[-1] <= 0:
            y = uniform(settings["y_min"], settings["y_max"])

        x_list1.append(total_size + x_spacing)
        y_list1.append(y)

        x_list1.append(total_size + (2 * x_spacing))
        y_list1.append(-y)

        total_size += x_spacing * 2

    return x_list1, y_list1


x_list, y_list = generate_points()
plt.plot(x_list, y_list)
plt.axis("off")
plt.show()
