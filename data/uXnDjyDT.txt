pub fn mapper<>(val:i32, word: String<>) -> Box<dyn Fn(i32) -> i32>{
        if val%2 == 0{
            let g = move |val: i32| val+word.len() as i32;
            return Box::new(g);
        }
        else {
            let f = |val: i32| val*2;
            return Box::new(f);
        }
    }
