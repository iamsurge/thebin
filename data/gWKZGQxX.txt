#include <stdio.h>
#include <stdlib.h>


int main() {
    int A, B, K;
    scanf("%d%d%d", &A, &B, &K);
    int rezultat = 0;
    for(int i = A; i <= B; i++) {
        int k_deliv = 1;
        int broj_na_deliteli = 0;
        for(int j = 1; j <= i; j++) {
            if(i % j == 0) {
                broj_na_deliteli++;
            }
        }
        
        for(int j = 1; j <= K; j++) {
            int prethodnik = i - j;
            if(prethodnik >= 1) {
                int broj_na_deliteli_na_prethodnik = 0;
                for(int k = 1; k <= prethodnik; k++) {
                    if(prethodnik % k == 0){
                        broj_na_deliteli_na_prethodnik++;
                    }
                }
                if(broj_na_deliteli <= broj_na_deliteli_na_prethodnik) {
                    k_deliv = 0;
                }
                
            }
        }
        if(k_deliv == 1) {
            rezultat += i;
        }
        
    }
    printf("%d\n", rezultat);
    
    return 0;
}
