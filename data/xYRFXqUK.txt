using System;

namespace _04.CarNumber
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int a = int.Parse(Console.ReadLine());
            int b = int.Parse(Console.ReadLine());

            for (int i = a; i <= b; i++)
            {
                for (int j = a; j <= b; j++)
                {
                    for (int k = a; k <= b; k++)
                    {
                        for (int l = a; l <= b; l++)
                        {
                            bool flag1 = (i % 2 == 0) && (l % 2 != 0);
                            bool flag2 = (i % 2 != 0) && (l % 2 == 0);
                            bool flag3 = (i > l) && ((j + k) % 2 == 0);
                            bool resultFlag = (flag1 || flag2) && flag3;

                            if (resultFlag)
                            {
                                Console.Write($"{i}{j}{k}{l} ");
                            }
                        }
                    }
                }
            }
        }
    }
}
