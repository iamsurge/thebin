using System;

namespace ConsoleApp2
{
    class Program
    {
        static void Main(string[] args)
        {
            CalcDelegate calcDelegate = new CalcDelegate(Calculator.Difference);
            calcDelegate += Calculator.Fraction;
            calcDelegate += Calculator.Product;
            calcDelegate += Calculator.Sum;

            foreach (var item in calcDelegate.GetInvocationList())
            {
                Console.WriteLine($"{item.Method.Name} = {item.DynamicInvoke(10, 5)}");
            }

        }
    }
}



using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp2
{
    delegate double CalcDelegate(double one, double two);
    class Calculator
    {
        public static double Sum (double one, double two)
        {
            return one + two;
        }
        public static double Difference (double one, double two)
        {
            return one - two;
        }
        public static double Product (double one, double two)
        {
            return one * two;
        }
        public static double Fraction (double one, double two)
        {
            return one / two;
        }
    }
}
