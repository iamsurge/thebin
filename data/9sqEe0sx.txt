def showHouse(house):
    print("windows :-", house.windows)
    print("Doors :-", house.doors)
    print("rooms :-", house.rooms)
    print("hasGarage :-", house.hasGarage)
    print("hasGarden :-", house.hasGarden)



class HouseI:
    
    def __init__(self, windows, doors, rooms, hasGarage, hasGarden):
        self.windows = windows
        self.doors = doors
        self.rooms = rooms
        self.hasGarage = hasGarage
        self.hasGarden = hasGarden
        
        
house = HouseI(2, 1, 2, True, False)
showHouse(house)



class HouseII:
    
    def setWindows(self, windows):
        self.windows = windows
        
    def getWindows(self):
        return self.windows
        
        
    def setDoors(self, doors):
        self.doors = doors
        
    def getDoors(self):
        return self.doors
        
        
    def setRooms(self, rooms):
        self.rooms = rooms
        
    def getRooms(self):
        return self.rooms
        
        
        
    def setHasGarage(self, hasGarage):
        self.hasGarage = hasGarage
        
    def getHasGarage(self):
        return self.hasGarage
        
        
    def setHasGarden(self, hasGarden):
        self.hasGarden = hasGarden
        
    def getHasGarden(self):
        return self.hasGarden
        
        
        
        
        
house = HouseII()
showHouse(house)
