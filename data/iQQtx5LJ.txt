text = '''But soft what light through yonder window breaks
It is the east and Juliet is the sun
Arise fair sun and kill the envious moon
Who is already sick and pale with grief'''

words = {}

for w in text.replace('\n', ' ').split(' '):
    try:
        words[w] = words[w] + 1
    except:
        words[w] = 1

print(words)
