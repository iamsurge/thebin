#include <iostream>
using namespace std;

struct Element{
    string data;
    struct Element* next;
};

void push(Element** B, Element** E, string d){

    Element* temp = new Element;
    if ((*B)== nullptr){ //ako e prazna opashkata
        (*B)=temp;
        (*E)=temp;
    }
    temp->data = d;
    temp->next = nullptr;
    (*E)->next = temp;
    (*E) = temp;
}

string pop(Element** B){
    Element * temp;
    temp = (*B)->next;
    string d = (*B)->data;
    delete(*B);
    (*B) = temp;
    return d;
}


int main() {
    Element* begin = nullptr;
    Element* end = nullptr;
    Element* temp = nullptr;

    string s;
    for(int i = 0; i < 3; i++) {
        cin >> s;
        push(&begin, &end, s);
    }

    temp = begin;
    while(temp != nullptr){
        cout << temp->data << " ";
        temp = temp->next;
    } cout << endl;

    pop(&begin);
    temp = begin;
    while(temp != nullptr){
        cout << temp->data << " ";
        temp = temp->next;
    }
    return 0;
}
