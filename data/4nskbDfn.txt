textutils.slowPrint("initializing Sayv2.1...")
local function inet()
    rednet.open("right")
    lel = nil
    while lel == nil do
        event, id, hah = os.pullEvent("rednet_message")
        if hah ~= nil then
            lel = hah
        end
    end
    rednet.close("right")
    return lel
end
--gets rednet input

local function Say()
    while true do
        input = inet()
        label = os.getComputerLabel()
        fuel = turtle.getFuelLevel()
        if fuel == 0 then
            turtle.select(16)
            turtle.refuel(1)
            turtle.select(1)
            --refuels turtle
        end
        if input == (label .. " w") then
            print(input)
            turtle.forward(1)
        elseif input == (label .. " s") then
            print(input)
            turtle.back(1)
        elseif input == (label .. " a") then
            print(input)
            turtle.turnLeft(1)
        elseif input == (label .. " d") then
            print(input)
            turtle.turnRight(1)
        elseif input == (label .. " e") then
            print(input)
            turtle.up(1)
        elseif input == (label .. " q") then
            print(input)
            turtle.down(1)
        elseif input == (label .. " r") then
            print(input)
            turtle.dig("left")
        elseif input == (label .. " f") then
            print(input)
            turtle.digUp("left") 
		elsif input == (label .. " v") then
			print(input)
            turtle.digDown("left")
        end
    end
end
Say()
