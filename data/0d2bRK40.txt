if not turtle.getItemDetail(i) then

end
end
end

local function invTab ()
local inv = {}
for i=1,16 do
local item = turtle.getItemDetail(i)
inv[i]={idx=i,it=item}
end
return inv
end
function invItr ()
local i=0
inven = invTab()
local n=#inven
return function ()
    i=i+1
    if i <= n then return inven[i] end
    end
end
function getInv ()
inv = {}
i=1
for slot in invItr() do
if slot.it then
inv[i]=slot.it
i=i+1
end
end
return inv
end
function findI (name)
for slot in invItr() do
if slot.it then
if slot.it.name == name then
return slot.idx
end
end
end
return 0
end
