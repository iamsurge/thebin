entrance, flour = int(input()), int(input())
for i in range(entrance,0,-1):
    if i == entrance:
        for j in range(flour):
            print(f'L{i}{j}',end=' ')
    elif i%2 == 0:
        for j in range(flour):
            print(f'O{i}{j}',end=' ')
    else:
        for j in range(flour):
            print(f'A{i}{j}',end=' ')
    print('')
