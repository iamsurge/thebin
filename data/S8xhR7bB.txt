lenTunnel = 0

write("Enter len tunnel: ")
lenTunnel = tonumber(read())

turtle.refuel(64)

while lenTunnel > 0 do
	turtle.digUp()
    turtle.up()
    turtle.turnRight()
    turtle.dig()
    turtle.forward()
    
    turtle.digDown()
    
    turtle.dig()
    turtle.forward()    
    
    turtle.digDown()
    turtle.down()
    turtle.digDown()
    turtle.down()
    
    turtle.turnLeft()
    turtle.turnLeft()    
    
    turtle.dig()
    turtle.forward()   
    turtle.dig()
    turtle.forward()       
    
    turtle.turnRight()
    
    turtle.dig()
    turtle.forward()
    
	lenTunnel = lenTunnel-1
end
