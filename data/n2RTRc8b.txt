#include<bits/stdc++.h>
using namespace std;
int a[510][510],dp[510][510];
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			scanf("%d",&a[i][j]);
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			if(j>1 && j<m)
				dp[i%2][j] = min({dp[(i-1)%2][j-1],dp[(i-1)%2][j],dp[(i-1)%2][j+1]}) + a[i][j];
			else if(j == 1)
				dp[i%2][j] = min(dp[(i-1)%2][j],dp[(i-1)%2][j+1]) + a[i][j];
			else if(j == m)
				dp[i][j] = min(dp[(i-1)%2][j-1],dp[(i-1)%2][j]) + a[i][j];
		}
	}
	int mn = 1e9;
	for(int j=1;j<=m;j++)
		mn = min(mn,dp[n%2][j]);
	printf("%d\n",mn);
	return 0;
}
