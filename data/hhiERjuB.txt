#include <iostream>
#include <string>
 
using namespace std;
 
struct list{
    list* prev;
    char let;
};
 
void push(list**L, char ch){
    list* temp;
    temp = new list;
    temp->prev = (*L);
    temp->let = ch;
    (*L) = temp;
}
 
char pop(list**L){
    list * temp;
    temp = (*L)->prev;
    char ch = (*L)->let;
    delete (*L);
    (*L) = temp;
    return ch;
}
int main()
{
    list * L;
    L = NULL;
    char s;
    cin >> s;
    push(&L, s);
    cout << pop(&L);
    return 0;
}
