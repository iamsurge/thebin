package opinionPoll;

public class Person {
    //полета -> характеристики -> име, възраст
    private String name;
    private int age;

    //конструктор -> създаваме обекти от класа
    public Person(String name, int age) {
        //нов обект
        this.name = name;
        this.age = age;
    }

    public Person(String name) {
        //нов обект
        this.name = name;
        this.age = 10;
    }

    public Person (int age) {
        //нов обект
        this.age = age;
    }

    //getter -> метод, чрез който достъпваме стойност на поле
    public String getName() {
        return this.name;
    }

    public int getAge() {
        return this.age;
    }
}

package opinionPoll;

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        List<Person> personList = new ArrayList<>();

        int n = Integer.parseInt(scanner.nextLine()); //бр. на редовете с лична информация
        for (int row = 1; row <= n ; row++) {
            String data = scanner.nextLine(); //"Desi 24"
            //"Desi 24".split(" ") -> ["Desi", "24"]
            String name = data.split(" ")[0];
            int age = Integer.parseInt(data.split(" ")[1]);

            if (age > 30) {
                //създавам запис за човека -> съхранявам запис
                Person person = new Person(name, age);
                personList.add(person);
            }
        }

        for (Person person : personList) {
            //name - възраст
            System.out.println(person.getName() + " - " + person.getAge());
        }
    }
}
