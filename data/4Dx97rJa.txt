#include <iostream>

int main(int argc, char** argv)
{
    int numberToCheck;
    std::cout << "Number: ";
    std::cin >> numberToCheck;

    if (numberToCheck % 2 == 0)
    {
        std::cout << "Even\n";
    }
    else
    {
        std::cout << "Odd\n";
    }

    std::cin.get();
    return 0;
}
