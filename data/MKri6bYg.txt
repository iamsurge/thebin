#include <stdio.h>


int a = 1;
int b = 0;

//////////////////////////////////////////////////////////////
int main()                                                  //
{

    printf("b = %d \n", b); 

  __asm
	{
        mov  eax, a     
        inc  eax 
		add  eax, 1000
		shr  eax, 2
        mov  b, eax
	}

    printf("b = %d \n", b); 
    
return 0;
}
