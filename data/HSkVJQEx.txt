running = false
startTime = os.time()
endTime = os.time()

while true do
  os.pullEvent("redstone")
  if rs.getInput("right") then
    running = true
    startTime = os.time()
  else if rs.getInput("back") then
    running = false
    endTime = os.time()
    print(endTime - startTime)
  end
end
