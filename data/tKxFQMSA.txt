

	public function get_script_depends() {
		//$settings = $this->get_settings_for_display();

		


		
		wp_register_script( 'aefe-testimonial-one', AEFE_URL . "assets/js/aefe-testimonial-one.js", [
			'jquery',
			'owlcarousel',
			'elementor-frontend'
		],'2.2.1', true);


		$testimonial_count = $settings['aefe_testimonial_slider_item_count'] ? $settings['aefe_testimonial_slider_item_count'] : 3;
		wp_localize_script( 'aefe-testimonial-one', 'aefe_testimonial_one_object',
			array( 
				'ajaxurl' => admin_url( 'admin-ajax.php' ),
				'aefe_test_one_count' => $testimonial_count,
			)
		);


		return [
			'aefe-testimonial-one',
		];

	}
