--//Setup the mothem, placed at the bottom
modem = peripheral.wrap("bottom")
channel = 2209
modem.open(channel)
--//Tests modem/channel connetion
if modem.isOpen(channel) then
	print("Modem was succesfully open in channel " ..channel)
end
--//Awaits for an instruction an then excecutes with what is in "message"
while true do
	event, modemSide, senderChannel, replyChannel, message, senderDistance = os.pullEvent("modem_message")
	shell.run(message)
end
