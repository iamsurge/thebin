#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <windows.h>

int main() {
    SetConsoleOutputCP(CP_UTF8);
    srand(time(NULL));
    printf("Enter m&n-size of matrix:\n");
    int n,m;
    scanf("%d %d",&m,&n);
    int K[m][n];
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            K[i][j]=rand()%31-10;
            printf("%d ",K[i][j]);
        }
        printf("\n");
    }
    int s=0;
    for (int i = 0; i < n; ++i) {
        int mi=0;
        for (int j = 0; j <m ; ++j) {
            if(K[j][i]<mi || j==0)
            {
                mi=K[j][i];
            }
        }
        printf("%d ",mi);
        s+=mi;
    }
    printf("Sum of min el in every column=%d",s);
    return 0;
}
