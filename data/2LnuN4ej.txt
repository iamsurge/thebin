
def if_simple(n):
    for i in range(2, n//2+1):
        if n % i == 0:
            return False
    return True

N = int(input())

numbers = []

for i in range(N):
    numbers.append(i+1)

numb = 2

lent = len(numbers)


for i in numbers:
    if (if_simple(numb)):
        for j in range(numb**2, lent+1, numb):
            if j in numbers:
                numbers.remove(j)
    if (if_simple(i)):
        numb += 1


print(numbers)
