vector <vector <int> > m;

vector <int> dx = {1, 0, -1, 0}, dy = {0, 1, 0, -1};

bool ok(int i, int j) {
    return i >= 0 && j >= 0 && i < 8 && j < 8 && m[i][j] == 0;
}

bool sh = 0;

void bld() {
    m.assign(8, vector <int> (8, 0));
    for (int i = 1; i <= 4; ++i) {
        m[i][1] = 1;
    }
    for (int i = 3; i <= 6; ++i) {
        m[i][6] = 1;
    }
}
int cnt;
void go(int i, int j) {
    m[i][j] = 1;
    cnt++;
    if (sh) {
        cout << i << ' ' << j << "\n";
    }
    for (int k = 0; k < 4; ++k) {
        int x = i + dx[k];
        int y = j + dy[k];
        if (ok(x, y)) {
            go(x, y);
            break;
        }
    }
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int ans = 0;
    //sh = 1;
    for (int i = 0; i < 8; ++i) {
        for (int j = 0; j < 8; ++j) {
            bld();
            cnt = 0;
            if (!ok(i, j)) {
                continue;
            }
            if (sh) {
                cout << "GO i j = " << i << ' ' << j << "\n";
            }
            go(i, j);
            if (cnt == 56) {
                cout << i << ' ' << j << "\n";
                ans++;
            }
        }
    }
    cout << "ans =  " << ans << "\n";
}
