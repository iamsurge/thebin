import tkinter as tk

personas = []

def guardar_nombre():
    global personas
    nombre = caja_nombre.get()
    email = caja_email.get()
    nac = caja_nac.get()
    tel = caja_tel.get()
    persona = {"nombre":nombre, "email":email, "nacionalidad":nac, "telefono":tel}
    personas.append(persona)
    print(persona)
    
    # borro las cajas
    nombre = caja_nombre.delete(0,tk.END)
    email = caja_email.delete(0,tk.END)
    nac = caja_nac.delete(0,tk.END)
    tel = caja_tel.delete(0,tk.END)
    
    


ventana = tk.Tk()
ventana.title("FORMULARIO")
ventana.config(width=300, height=300)
# ventana.resizable(False,False)  # evita redimensionar la ventana



# Campo nombre
etiqueta = tk.Label(text="Nombre")
etiqueta.place(x=20, y=20)
caja_nombre = tk.Entry()
caja_nombre.place(x=120, y=20)

# Campo email
etiqueta = tk.Label(text="Email")
etiqueta.place(x=20, y=70)
caja_email = tk.Entry()
caja_email.place(x=120, y=70)

# Campo nacionalidad
etiqueta = tk.Label(text="Nacionalidad")
etiqueta.place(x=20, y=120)
caja_nac = tk.Entry()
caja_nac.place(x=120, y=120)

# Campo telefono
etiqueta = tk.Label(text="Tel√©fono")
etiqueta.place(x=20, y=170)
caja_tel = tk.Entry()
caja_tel.place(x=120, y=170)


boton = tk.Button(text="Guardar", command=guardar_nombre)
boton.place(x=100, y=240, width=80, height=30)



ventana.mainloop()
