#include <iostream>
#include <bits/stdc++.h>
using namespace std;
struct date{
    string date;
    string hour;
};
int main() {
    ifstream soccer("/home/elev/Downloads/matches_European_Championship.csv");
    vector<soccer> dates;
    string line;
    getline(soccer, line);
    while(getline(soccer, line)) {
        stringstream ss(line);
        vector<string> cmps;
        string crt;
        while(getline(ss, crt, ',')) {
            cmps.push_back(crt);
        }
        if(cmps.size() > 5) {
            date d;
            d.date = cmps[6];
            dates.push_back(d);
        }
    }
    return 0;
}
