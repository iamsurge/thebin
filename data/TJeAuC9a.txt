if (prevProps.recMode !== this.props.recMode) {
    if (this.props.recMode === 'mic') {
        this.props.resumeAudioAnalyser();
        if (this.props.activeSnapLens.id !== '') babylonRenderer.pauseLens();
    } else {
        this.props.suspendAudioAnalyser();
        this.state.babylonRenderer.restorePreviousExternalBackground();
        if (this.props.activeSnapLens.id !== '') {
            this.props.toggleSnapLensLoadingOverlay(true);
            await babylonRenderer.setSnapLens(this.props.activeSnapLens);
            this.props.toggleSnapLensLoadingOverlay(false);
        }
    }
    babylonRenderer.toggleMicMode(this.props.recMode === 'mic', this.props.audioAnalyser);
}
