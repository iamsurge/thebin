using System;

namespace ConsoleApp5
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int outputImagesRow = 3; // выводятся картинки, по outputImagesRow в ряд
            int picturesInAlbum = 52; // картинок в альбоме всего
            int fullyFilledRows = picturesInAlbum / outputImagesRow; // полностью заполненных рядов
            int picturesBeyondMeasure = picturesInAlbum % outputImagesRow; // картинок сверх меры
            Console.WriteLine($"Полностью заполненных рядов: {fullyFilledRows}. Сверх меры: {picturesBeyondMeasure}.");
            Console.ReadKey();
        }
    }
}
