#include "LittleFS.h"
 
void setup() {
  Serial.begin(115200);
  
  if(!LittleFS.begin()){
    Serial.println("Beim Mounten von LittleFS ist ein Fehler aufgetreten");
    return;
  }
  
  File file = LittleFS.open("/test_example.txt", "r");
  if(!file){
    Serial.println("Datei konnte nicht zum Lesen ge√∂ffnet werden");
    return;
  }
  
  Serial.println("Inhalt der Datei:");
  while(file.available()){
    Serial.write(file.read());
  }
  file.close();
}
