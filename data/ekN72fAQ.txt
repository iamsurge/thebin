#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main() {
    int k;
    cin >> k;
    int n;
    cin >> n;
    int kompanija_1[n];
    for (int i = 0; i < n; i++)
    {
        cin >> kompanija_1[i];
    }
        int kompanija_2[n];
    for (int i = 0; i < n; i++)
    {
        cin >> kompanija_2[i];
    }
    sort(kompanija_1, kompanija_1 + n);
    sort(kompanija_2, kompanija_2 + n);
    long long kombinacii = 0;
    for (int i = 0; i < n; i++)
    {
        if(kompanija_1[i] >= k) {
            kombinacii += n;
        }
        else {
            int razlika = abs(kompanija_1[i] - k);
            int pozicija = lower_bound(kompanija_2, kompanija_2 + n, razlika) - kompanija_2;
            if(pozicija < n) {
                kombinacii += n - pozicija;
            }
        }
    }
    cout << kombinacii << endl;
    return 0;
}
