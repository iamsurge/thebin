#include <stdio.h>
#include <stdlib.h>

int main() {
    int N;
    scanf("%d", &N);
    
    int treta_cifra = N % 10;
    N /= 10;
    int vtora_cifra = N % 10;
    N /= 10;
    int prva_cifra = N % 10;
  
    int proizvod = prva_cifra * vtora_cifra * treta_cifra;
    printf("%d\n", proizvod);
    return 0;
}
