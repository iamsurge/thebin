"""
Tipos de datos: son los diferentes valores que puedo usar en un script
Existen 4 tipos basicos (1 solo valor)
1) nro entero (int): ....... -4 -3 -2 -1 0 1 2 3 .....
2) nro racional (float):  +25.365  25.365    -25365. +25365. (nro con coma)
3) cadenas (str): "Hola", "sdfsafd", "25+58", '-25.25', 'True'
4) valor logico o booleano (bool): True, False

Algunos tipos se pueden convertir
Ej: int(25.0) da 25, convertí un float a int


# ASIGNACION DE VARIABLES
El signo igual es una ASIGNACION, guarda en un espacio de memoria de la PC
lo que está a su derecha ("Juan"). Lo que está a la izquierda es la etiqueta
que identifica donde buscar ese valor
nombre = "Juan"
print(nombre)

total = 1 + 2  # suma 1 + 2 y lo guarda en la variable total
print(total)  # imprime 3
"""

# Script que suma los numeros enteros de 1 al 5 y los muestra en pantalla
suma = 1 + 2 + 3 + 4 + 5
print(suma)
print("La suma es", suma) # la coma concatena añadiendo un espacio



# Script que pide un nombre y lo saluda
nombre = input("Ingrese su nombre: ")
print("Hola",nombre) 



# Script que pide dos numeros enteros y devuelve su suma
a = int(input("Ingrese un numero: "))
b = int(input("Ingrese otro numero: "))
total = a + b
print("Total:",total) 



# Script que pide dos numeros enteros y devuelve su suma
a = int(input("Ingrese un numero: "))
b = int(input("Ingrese otro numero: "))
print("Total:", a + b) 


# Script que pide una temperatura en ºF y la devuelve en ºC
temperatura = float(input("Ingrese una temperatura (en ºF): "))
temperatura = (temperatura - 32) / 1.8

# forma de impresión casi buena
print("Temperatura:",temperatura,"ºC")

# forma de impresión buena pero fea
print("Temperatura: " + str(temperatura) + "ºC")

# forma ideal
print(f"Temperatura: {temperatura:.1f}ºC")


# Script que pide una edad y un nombre y reacciona bromeando
