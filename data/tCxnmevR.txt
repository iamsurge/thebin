bestand = open('voorbeeld.txt', 'r')
alles = bestand.readlines()
for x in range(len(alles)):
  alles[x] = alles[x].replace('\n', '')

koppels = []
elfjes = []

for x in range(len(alles)):
  koppels.append(alles[x].split(','))
for x in range(len(koppels)):
  deze2 = []
  for y in range(2):
    deze2.append(koppels[x][y].split('-'))
  elfjes.append(deze2)

antwoord = 0

for x in range(len(elfjes)):
  if (elfjes[x][0][0] <= elfjes[x][1][0] and elfjes[x][0][1] >= elfjes[x][1][1]) or (elfjes[x][0][0] >= elfjes[x][1][0] and elfjes[x][0][1] <= elfjes[x][1][1]):
    print(elfjes[x])
    antwoord += 1

print(antwoord)
