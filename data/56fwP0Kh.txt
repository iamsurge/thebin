using System;
using System.IO;

namespace WorkWithFile
{
    class Program
    {
        static void Main(string[] args)
        {
            string fname = @"C:\Users\user\source\repos\Practice\WorkWithFile\MyFile.txt";
            string result = @"C:\Users\user\source\repos\Practice\WorkWithFile\Result.txt";

            using (StreamReader streamReader = new StreamReader(fname))
            {
                using (StreamWriter streamWriter = new StreamWriter(result))
                {
                    char symbol = (char)streamReader.Read();

                    while (!streamReader.EndOfStream)
                    {
                        if (!char.IsPunctuation(symbol))
                        {
                            streamWriter.Write(symbol);
                        }

                        symbol = (char)streamReader.Read();
                    }
                }
            }

            File.Delete(fname);
            File.Move(result, fname);
        }
    }
}
