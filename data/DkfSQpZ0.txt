let arrNum = +gets();
// console.log(arrNum);
let arr = [];
let result = '';

for (let i = 1; i <= arrNum; i++) {
    arr = gets().split(',');

    for (let j = 0; j <= arr.length - 2; j++) {
        if (arr[j] <= arr[j + 1]) {
            result = "true";
        } else {
            result = "false";
            break;
        }

    }
    console.log(result);
}
