Program Task4(output);
    var k, p, s: real;
    years: integer;
begin
    write('Капитал');
    readln(k);

    write('Проценты');
    readln(p);
    
    write('Сумма');
    readln(s);

    years := 0;
    
    while k < s do
        begin
            k := k * (1 + p / 100);
            inc(years);
        end;

    writeln(years);

end.
