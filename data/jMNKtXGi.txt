public void setProductList(List<PPOBProductResponse> list) {
        List<SpinnerLabel> spinnerItems = new ArrayList<>();
        int position = 0;
        for(PPOBProductResponse data : list){
            String text = presenter.session.getProduct().toString();
            if (data.getProductName().equals(text)){
                position = list.indexOf(data);
            }
            SpinnerLabel spinner = new SpinnerLabel(data.getProductId(),data.getProductName(),data);
            spinnerItems.add(spinner);
        }

        productAdapter.addAll(SpinnerModel.generate(spinnerItems.toArray()));
        binding.spinnerProduct.setSelection(position);
    }
