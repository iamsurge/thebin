while true do
 
local function fill()
	if turtle.getItemCount(1) >= 1 then
    turtle.select(1)
    turtle.craft()
    turtle.drop()
	else
	check()
	end
	end

local function check()
    if turtle.getItemCount(2) >= 1 then
    fill()
    else
		turtle.suckUp()
	end
	end

local function run()
        check()
    else
        sleep(0.5)
    end
 
 
run()
 
end
