using System;
using System.Text;

namespace Favorite_Movie
{
	class FavoriteMovie
	{
		static void Main(string[] args)
		{
			bool isExitLoop = false;
			int i = 1;

			StringBuilder bs = new StringBuilder();
			StringBuilder currentString = new StringBuilder();
			int sum = 0;


			while ((!isExitLoop) && (i <= 7))
			{
				currentString.Clear();
				currentString.Append(Console.ReadLine());

				if (currentString.ToString() == "STOP")
				{
					isExitLoop = true;
					i--;
				}
				else
				{
					int currentSum = 0;
					for (int j = 0; j < currentString.Length; j++)
					{
						currentSum += (int)currentString[j];
						if ((currentString[j] >= 'A') && (currentString[j] <= 'Z'))
						{
							currentSum -= currentString.Length;
						}
						else if (((currentString[j] >= 'a') && (currentString[j] <= 'z')))
						{
							currentSum -= (2 * currentString.Length);
						}
					}

					if (currentSum > sum)
					{
						sum = currentSum;
						bs.Clear();
						bs.Append(currentString.ToString());
					}
				}
				i++;
			}

			if (i > 7)
			{
				Console.WriteLine("The limit is reached.");
			}

			Console.WriteLine($"The best movie for you is {bs.ToString()} with {sum} ASCII sum.");
		}
	}
}
