import websockets
import asyncio

print("Hello!")

async def listen():
    url = "ws://192.168.0.199/debug/log"
    async with websockets.connect(url) as ws:
        while True:
            msg = await ws.recv()
            print(msg)

asyncio.get_event_loop().run_until_complete(listen())
