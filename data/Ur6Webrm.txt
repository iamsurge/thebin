import datetime
date = datetime.datetime.strptime(input(), "%d %B %Y")

temperature = int(input())
rain = int(input())
winter = int(input()) #35 = 5

def isleap(year):
    if year % 400 == 0:  # Some centuries are leap years...
        return True
    elif year % 100 == 0: # ... but most are not ...
        return False
    elif year % 4 == 0:   # ... even though other divisibly-by-four years are
        return True
    else:
        return False

# calculate winter days
winterDays = winter // 7
date += datetime.timedelta(days=winterDays)

# Handle temperature drift
if isleap(date.year): temperature += 5  # leap year mod
dt = temperature - 20
date += datetime.timedelta(days=-dt)
 
# Handle rain levels
dt = (rain - 30) // 3
date += datetime.timedelta(days=-dt)


print( date.strftime("%d %B %Y") )
