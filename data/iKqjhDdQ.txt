#include <iostream>
#include <cmath>
#include <iomanip>
#include <clocale>
#include <string>
#include <Windows.h>

using namespace std;
//Ссуда
int main()
{
	SetConsoleOutputCP(1251);
	SetConsoleCP(1251);
	setlocale(LC_ALL, "Rus");
	double m, p, s, n, r, mc,r2;
	cout << "Введите сумму займа s " << endl;
	cin >> s;
	while (s < 0)
	{
		cout << "Введите сумму еще раз! ";
		cin >> s;
	}
	cout << "Введите количество лет n, на которое взят займ  " << endl;
	cin >> n;
	while (n < 0)
	{
		cout << "Введите кол-во лет еще раз! ";
		cin >> n;
	}
	cout << "Введите величину месячной выплаты " << endl;
	cin >> m;
	while (m < 0)
	{
		cout << "Введите величину месячной выплаты еще раз! ";
		cin >> m;
	}
	r = 0;
	r2 = 100;
	if (s > m * n)
	{
		do
		{
			r2 = r2 - 0.00001;
			mc = (s * r2 * (pow(1 + r2, n))) / (12 * ((pow(1 + r2, n)) - 1));
		} while (mc > m);
		cout << r2 * 100.0 << "%" << endl;
	}
	else
	{
		do
		{
			r = r + 0.00001;
			mc = (s * r * (pow(1 + r, n))) / (12 * ((pow(1 + r, n)) - 1));
		} while (mc < m);
		cout << r * 100.0 << "%" << endl;
	}
	if (m == 0)
	{
		cout << "0 %" << endl;
	}
	return 0;
}
