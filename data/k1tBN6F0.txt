#include <iostream>
#include <algorithm>


using namespace std;
int main()
{
    int n, k;
    cin >> n >> k;
    
    int niza[n];
    pair<int, int> p[n];
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
        p[i] = make_pair(niza[i], i);
    }
    sort(p, p + n);
    int l = 0, r = n - 1;
    for(int i = 0; i < k; i++) {
        swap(niza[p[l].second], niza[p[r].second]);
        l++;
        r--;
    }
    for(int i = 0; i < n; i++) {
        cout << niza[i] << " ";
    }
    return 0;
}
