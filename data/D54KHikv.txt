n = int(input())
A = list(map(int, input().split()))

m_length, c_length = 1, 1
m_number = A[0]
for i in range(1, n):
    if A[i] == A[i - 1]:
        c_length += 1
    else:
        if c_length > m_length:
            m_length = c_length
            m_number = A[i - 1]
        c_length = 1
if c_length > m_length:
    m_length = c_length
    m_number = A[i - 1]
print(m_number, m_length)
