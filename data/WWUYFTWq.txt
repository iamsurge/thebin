#include <stdio.h>
#include <stdlib.h>


int main() {
    int x;
    int sum = 0;
    while(scanf("%d", &x)) {
        int broj = x;
        int ok = 1;
        while(broj > 9) {
            int poslednaa_cifra = broj % 10;
            int pretposledna_cifra = (broj / 10) % 10;
            if(poslednaa_cifra >= pretposledna_cifra) {
                ok = 0;
            }
            broj /= 10;
        }
        if(ok == 1) {
            sum += x;
        }
    }
    printf("%d\n", sum);
    return 0;
}
