#include <stdio.h>
#include <stdlib.h>

int proverka(int broj) {
    int najneznacajna = broj % 10;
    int najznacajna = broj % 10;
    
    while(broj > 0) {
        najznacajna = broj % 10;
        broj /= 10;
    }
    if(najznacajna < najneznacajna) {
        return 1;
    }
    return 0;
}
int rekurzija(int a, int b) {
    if(a > b) {
        return 0;
    }
    return rekurzija(a + 1, b) + proverka(a);
}
int main(){
    int A, B;
    scanf("%d%d", &A, &B);
    printf("%d\n", rekurzija(A, B));
    return 0;

}
