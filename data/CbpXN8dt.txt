import java.util.Scanner;

public class Main 
{
	public static void main(String[] args)
	{
		Scanner userInput = new Scanner(System.in);
		
		double temperatureSum = 0.0;
		
		int dayCount = 1;
		int occurrence =0;
		
		double temperatureInput =0.0;
		
		while(temperatureInput <= 45)
		{
			System.out.println("Enter the temperature for day: "+dayCount);
			temperatureInput = userInput.nextDouble();
			if(temperatureInput>45)
				break;
			occurrence++;
			temperatureSum+=temperatureInput;
			dayCount++;
		}
		System.out.println("SUM OF TEMPERATURES: "+temperatureSum);
		System.out.println("AVERAGE TEMPERATURE: "+temperatureSum/occurrence);
		System.out.println(occurrence);
		userInput.close();
	}
}
