#include <iostream>
#include <algorithm>
#define N 25005
using namespace std;
int n,i,m;
int a[N],x;
int st, dr, mij;
bool gasit;
int main()
{
   cin>>n;
   for(i=1;i<=n;++i)
    cin>>a[i];
   cin>>m;
   while(m>0)
   {
     --m;
     cin>>x;///valorea de cautat
     ///cautarea  binara
     st=1, dr=n;
     gasit=0;
     while(st<=dr && !gasit)
     {
         mij=(st+dr)/2;///mij=st+(dr-st)/2;
         if(x==a[mij]) gasit=1;
         else if(x>a[mij]) st=mij+1;
         else dr=mij-1;
     }
     cout<<gasit<<" ";
   }
   return 0;
}
