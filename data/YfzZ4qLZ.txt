protective_nylon = int(input())
leaters_paint = int(input())
leathers_thinner = int(input())
hours_work = int(input())
baggs = 0.40

nylon_price = (protective_nylon + 2) * 1.50
percent_paint = leaters_paint * 0.1
paint_price = (leaters_paint + percent_paint) * 14.50
thinner_price = leathers_thinner * 5
total_price_for_materials = nylon_price + paint_price + thinner_price + baggs
work_per_hour_price = total_price_for_materials * 0.3
hours_work_price = hours_work * work_per_hour_price
total_price_for_all = total_price_for_materials + hours_work_price
print(total_price_for_all)
