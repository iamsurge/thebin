for i in map(lambda x:x**2, list(range(10))):
    print(i)
    
# list is stored in memory with all its values (eager evaluation)
print([x**2 for x in range(10)]) 
print("generators")

# g is a generator and when used in a for loop generates its values on the fly (lazy evalution)
g = (2*x + 1 for x in range(10))
print(g)

print(any([0,1,1,0]))  # returns True if *any* of its elements is 'truthy', false otherwise
print(all([1,1,1,True,[]]))  # returns True if *all* of its elements are 'truthy', false otherwise
