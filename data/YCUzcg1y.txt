"""
Arrojar un dado
"""
import tkinter as tk


def clic():
    import random
    valor = random.randint(1,6)
    caja_dado.delete(0,tk.END)
    caja_dado.insert(0,valor )

    

ventana = tk.Tk()
ventana.config(width=200, height=150)
ventana.title("FORMULARIO")
# prohibo redimensionar la ventana
ventana.resizable(False, False)


etiqueta = tk.Label(text="Valor del dado")
etiqueta.place(x=20, y=10)
caja_dado = tk.Entry()
caja_dado.place(x=120, y=10, width=20, height=25)



# BOTON
boton = tk.Button(text="Arrojar dado", command=clic)
boton.place(x=20, y=80, width=120, height=40)


ventana.mainloop()
