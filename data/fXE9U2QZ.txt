function registration(input) {
    let username = input[0];
    let command = input[1];

    while (command) {
        command = command.split(" ");




        if (command[0] === "Letters") {
            if (command[1] === "Lower") {
                username = username.toLowerCase();
                console.log(username);
            }
            if (command[1] === "Upper") {
                username = username.toUpperCase();
                console.log(username);
            }
        }
        if (command[0] === "Reverse") {
            let startIndex = parseInt(command[1]);
            let endIndex = parseInt(command[2]);
            let substringToBeReversed = "";
            if (0 <= startIndex && startIndex < username.length && 0 <= endIndex && endIndex < username.length) {
                substringToBeReversed = username.substring(startIndex, endIndex + 1);
                substringToBeReversed = substringToBeReversed.split("").reverse().join("");
                console.log(substringToBeReversed);
            } else {
                continue;
            }
        }
        if (command[0] === "Substring") {
            let substr = command[1];
            if (username.includes(substr)) {
                username = username.replace(substr, "");
                console.log(username);
            } else {
                console.log(`The username ${username} doesn't contain ${substr}.`);
            }
        }
        if (command[0] === "Replace") {
            let character = command[1];
            username = username.split(character).join("-");
            console.log(username);
        }
        if (command[0] === "IsValid") {
            let char = command[1];
            if (username.includes(char)) {
                console.log("Valid");
            } else {
                console.log(`${char} must be contained in your username.`);
            }
        }
        if (command === "Registration") {
            break;
        }
        command = prompt();
    }

}

registration(['John', 'Letters Lower', 'Substring SA', 'IsValid @', 'Registration']);
