public class WithArgumentsBuildUp {
     public IContainer Container;

     public WithArgumentsBuildUp() {
         Container = new Container(x => {
                                     x.For<IClass1>().Use<Class1>();
                                     x.Forward<IClass1, Class1>();
                                     x.SetAllProperties(c => c.OfType<Class1>());
                                            });
    }
}
