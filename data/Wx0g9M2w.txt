using System.ComponentModel.DataAnnotations;
using Newtonsoft.Json;

internal class Program {
    static void Main(string[] args)
    {
        string[] lines = File.ReadAllLines("./json.json");
        var json = string.Join("", lines);
        var product = JsonConvert.DeserializeObject<AssortmentWithProduct>(json);
        foreach (var element in product.Products)
        {
            Console.WriteLine(element.Name);
        }
    }
}

public class Product
{
    public Guid Id { get; set; }    
    public string? Name { get; set; }
}
 
public class Assortment
{
    [Key]
    public Guid Id { get; set; }
    public string? Name { get; set; }
    public int Quantity { get; set; }
}

public class AssortmentWithProduct
{
    public List<Assortment>? Assortments { get; set; }
    public List<Product>? Products { get; set; }
}
