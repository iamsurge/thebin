count_joinery = int(input())
type_joinery = str(input())
shipment_method = str(input())

price = 0
discount = 0
total_price = 0
if type_joinery == "90 x 130":
    if 30 < count_joinery <= 60:
        price = count_joinery * 110
        discount = price * 0.5
        total_price -= discount
    elif count_joinery > 60:
        price = count_joinery * 110
        discount = price * 0.8
        total_price -= discount
elif type_joinery == "100 x 150":
    if 40 < count_joinery <= 80:
        price = count_joinery * 140
        discount = price * 0.6
        total_price -= discount
    elif count_joinery > 80:
        price = count_joinery * 140
        discount = price * 0.10
        total_price -= discount
elif type_joinery == "130 x 180":
    if 20 < count_joinery <= 50:
        price = count_joinery * 190
        discount = price * 0.7
        total_price -= discount
    elif count_joinery > 50:
        price = count_joinery * 190
        discount = price * 0.12
        total_price -= discount
elif type_joinery == "200 x 300":
    if 25 < count_joinery <= 50:
        price = count_joinery * 250
        discount = price * 0.9
        total_price -= discount
    elif count_joinery > 50:
        price = count_joinery * 250
        discount = price * 0.14
        total_price -= discount

if shipment_method == "With delivery":
    if count_joinery > 99:
        total_price = (total_price + 60) - (total_price * 0.4)
    elif 10 <= count_joinery <= 99:
        total_price = total_price + 60
    print(f"{total_price:.2f} BGN")
elif shipment_method == "Without delivery":
    if count_joinery > 99:
        total_price = total_price - (total_price * 0.4)
    elif 10 <= count_joinery <= 99:
        total_price = total_price
    print(f"{total_price:.2f} BGN")
if count_joinery < 10:
    print("Invalid order")
