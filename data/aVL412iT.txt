create table dept (deptid int, deptname varchar(10));
insert into dept values(1,'HR');

select * from (
select deptname,ename,salary,row_number() over(partition by deptname order by salary desc) rn 
from emp_big e , dept d
where e.deptid=d.deptid
)  t where rn<=5;
