-- Redstone left means "allow pushing netherack here"
-- redstone right means "fire a weight pulse"

while true do
	-- Nothing there, so we can allow piston pushing 
	redstone.setOutput("left", turtle.detect() == false)

	if (turtle.compare() == true) then
		redstone.setOutput("left", false)
		redstone.setOutput("right", true)
		sleep(0.5)
		redstone.setOutput("right", false)
	end
	sleep(5)
end
