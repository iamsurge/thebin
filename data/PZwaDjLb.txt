#include <stdio.h>
#include <stdlib.h>

int main(){
   
    int N = 156;
    int obraten_broj = 0;
    
    while(N > 0) {
        int cifra = N % 10; // 1
        obraten_broj = (obraten_broj * 10) + cifra; // 65
        N /= 10;
    }
    printf("%d\n", obraten_broj);
    return 0;

}
