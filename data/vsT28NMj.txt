# С помощью функции input получаю строку от пользователя и разделяю её на элементы в список через метод split по запятым
enter_data = input('>>> ').split(',')

# Создаю список для хранения конвертированных значений
nums = []

# Конвертурую str в int и добавляю в новый список эти элементы
for num in enter_data:
    nums.append(int(num))

# Создаю переменную для хранения результата суммы
res = 0

# Суммирую в цикле каждый элемент
for num in nums:
    res += num

# Вывожу результат в консоль
print(res)
