#include <iostream>

using namespace std;

int main() 
{
    
    int n;
    cin>>n;
    int tab[n];
    
    for(int i = 0; i < n; i++)
    {
        cin>>tab[i];
    }
    
    int licznik = 0;
    
    for(int i = 0; i < n; i++)
    {
        if(tab[i]>=tab[i+1])
        {
            licznik++;
        }
    }
    
    cout<<licznik;

    return 0;
}
