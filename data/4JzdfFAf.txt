SELECT table_schema,table_name
FROM information_schema.tables
ORDER BY table_schema,table_name;    

SELECT table_schema,table_name
FROM information_schema.tables
WHERE table_name  LIKE '%' || 'material' || '%'
ORDER BY table_schema,table_name;


SELECT 
   table_name, 
   column_name, 
   data_type 
FROM 
   information_schema.columns
WHERE 
   table_name = 'tasks';


select status, event, purpose, notes
from tasks
where started_at > '2022-10-22' and completed_at < '2022-10-22'
