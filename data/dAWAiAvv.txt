namespace KazePashaMSK
{
    /// <summary>
    /// This class is the container for the modifier and its type, and it also stores an enumeration of all possible modifier types.
    /// </summary>
    public class StatModifier
    {
        // All possible types of modifiers. If you want to add a new type - add it here. Do not forget to initialize this type in KZCharacter.
        public enum StatType { Movement, Acceleration, MaxHealth, CurrentHealth, Attack, HealthDropRate, HealthDropPower, AttackDropRate, GhostAttackDamage, GhostAttackDeceleration};

        // Current type of modifier
        public StatType Type;

        // Current modifier
        public Modificator modificator;

        public StatModifier(StatType _type, Modificator _modificator)
        {
            Type = _type;
            modificator = _modificator;
        }
    }
}
