<script>
  /* Neste código comentado o thead.style.display = ... sobrescreve os estilos da folha de estilo css de thead não somente para display.

let
  tableContainer = document.querySelector(".table-container"),
  thead = document.querySelector("thead");

tableContainer.addEventListener('scroll', () =>
                                {if (tableContainer.scrollTop >= 100 )  {thead.style.display = "none";} else {thead.style.display = "block";}}
);
*/
  let
    tableContainer = document.querySelector(".table-container"),
    thead = document.querySelector("thead");
  
  styleElement = document.createElement("style");
  document.head.appendChild(styleElement)
  
  
  tableContainer.addEventListener('scroll', () => {
    if (tableContainer.scrollTop >= 200) {
      styleElement.innerText = "thead {display: none} h1 {visibility:hidden} h2 {color:red}";
    } else {
      styleElement.innerText = "thead {display: sticky} h1 {visibility:visible} h2 {color:blue}"
    }
  });
</script>
