using System;
using System.Collections.Generic;
using System.Text;

namespace Listas_Jugadores
{
    class Player
    {
        // Atributos
        private int idPlayer;
        private string Name;
        private string Position;
        public DateTime BirthDay { set; get; }
        public DateTime EntryDay { set; get; }
        public double Salary { set; get; }
        public short Goals { set; get; }

        // Constructor vacio
        public Player()
        {

        }
        // Constructor con parámetros
        public Player(int _idPlayer, string _nam, string _pos, DateTime _birthD, DateTime _entryDate, double _sal, short _goals)
        {
            this.idPlayer = _idPlayer;
            this.Name = _nam;
            this.Position = _pos;
            this.BirthDay = _birthD;
            this.EntryDay = _entryDate;
            this.Salary = _sal;
            this.Goals = _goals;
        }

        public void setIdPlayer(int _idPlayer) {
            this.idPlayer = _idPlayer;
        }
        public int getIdPlayer()
        {
            return this.idPlayer;
        }
        public void setName(string _name)
        {
            this.Name = _name;
        }
        public string getName()
        {
            return this.Name;
        }
        public void setPosition(string _pos)
        {
            this.Position = _pos;
        }
        public string getPosition()
        {
            return this.Position;
        }

        // Métodos propios de la clase
        public void showInfo()
        {
            Console.WriteLine("**********************************************");
            Console.WriteLine("Id Player: {0}", this.idPlayer);
            Console.WriteLine("Name: {0}", this.Name);
            Console.WriteLine("Position: {0}", this.Position);
            Console.WriteLine("BirthDay: {0}", this.BirthDay);
            Console.WriteLine("Entry Day: {0}", this.EntryDay);
            Console.WriteLine("Salary: {0}", this.Salary);
            Console.WriteLine("Goals: {0}", this.Goals);
            Console.WriteLine("**********************************************");
        }
    }
}
