DROP DATABASE IF EXISTS Localita;

CREATE DATABASE Localita;

USE Localita;

CREATE TABLE Regioni(
    Codice int PRIMARY KEY AUTO_INCREMENT,
    Nome varchar(30) NOT NULL);


CREATE TABLE Comuni(
    Codice int PRIMARY KEY AUTO_INCREMENT,
    Denominazione varchar(30) NOT NULL,
    NumeroAbitanti int NOT NULL,
    CodiceRegione int,
    FOREIGN KEY (CodiceRegione) REFERENCES Regioni(Codice));
