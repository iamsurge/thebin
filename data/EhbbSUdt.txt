package JavaInputs;

import java.awt.Color;
import java.awt.EventQueue;
import java.awt.Font;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;

import javax.swing.*;
public class JavaInputs {

	public static void main(String[] args) {
		EventQueue.invokeLater(new Runnable() {
			public void run() {
				try {
					JavaInputs frm = new JavaInputs();
					frm.Inputs();
				}catch (Exception ex) {
					ex.printStackTrace();
				}
			}
		});
	}
	
	void Inputs() {
		JFrame frmMain = new JFrame("Enter Letters and Numbers");
		JLabel lblLetters =  new JLabel("Letters", SwingConstants.CENTER); //Align text to CENTER
		JLabel lblNumbers =  new JLabel("Numbers", SwingConstants.CENTER);
		JLabel lblOutput =  new JLabel("Output", SwingConstants.RIGHT); //Align text to RIGHT
		JTextField txtLetters = new JTextField();
		JTextField txtNumbers = new JTextField();
		JTextField txtOutput = new JTextField();
		JButton btnClick = new JButton("Click Me!");
		
		frmMain.setSize(350,350);
		frmMain.setLocationRelativeTo(null);
		frmMain.setVisible(true);
		frmMain.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		frmMain.getContentPane().setLayout(null);
		
		lblLetters.setSize(120,80);
		lblLetters.setLocation(35,0);
		lblLetters.setForeground(new Color(0,0,153));
		frmMain.getContentPane().add(lblLetters);	
		
		lblNumbers.setSize(120,80);
		lblNumbers.setLocation(180,0);
		lblNumbers.setForeground(new Color(0,0,153));
		frmMain.getContentPane().add(lblNumbers);
		
		lblOutput.setSize(150,20);
		lblOutput.setLocation(15,130);
		lblOutput.setForeground(new Color(0,0,153));
		frmMain.getContentPane().add(lblOutput);
		
		txtLetters.setSize(120,50);
		txtLetters.setLocation(35,50);
		txtLetters.setForeground(new Color(0,0,168));
		txtLetters.setHorizontalAlignment(SwingConstants.CENTER);
		frmMain.getContentPane().add(txtLetters);
		txtLetters.addKeyListener(new KeyAdapter() {
			@Override
			public void keyPressed(KeyEvent e) {
				int key = e.getKeyCode();
				
				if (key <= 57 && key >= 48) {
					JOptionPane.showMessageDialog(frmMain, "Invalid input!\nPlease enter alphabets only!", "Error!", JOptionPane.ERROR_MESSAGE);
					txtLetters.setText(null);
				}
			}
		});
		
		txtNumbers.setSize(120,50);
		txtNumbers.setLocation(180,50);
		txtNumbers.setForeground(new Color(0,0,168));
		txtNumbers.setHorizontalAlignment(SwingConstants.CENTER);
		frmMain.getContentPane().add(txtNumbers);
		txtNumbers.addKeyListener(new KeyAdapter() {
			@Override
			public void keyPressed(KeyEvent e) {
				int key = e.getKeyCode();
				
				if (key <= 90 && key >= 65) {
					JOptionPane.showMessageDialog(frmMain, "Invalid input!\nPlease enter numbers only!", "Error!", JOptionPane.ERROR_MESSAGE);
					txtNumbers.setText(null);
				}
			}
		});
		
		txtOutput.setSize(120,50);
		txtOutput.setLocation(180,120);
		txtOutput.setForeground(new Color(0,0,168));
		txtOutput.setHorizontalAlignment(SwingConstants.CENTER);
		txtOutput.setEditable(false);
		frmMain.getContentPane().add(txtOutput);
		
		btnClick.setSize(180,40);
		btnClick.setLocation(80,190);
		btnClick.setForeground(new Color(0,0,168));
		btnClick.setHorizontalAlignment(SwingConstants.CENTER);
		frmMain.getContentPane().add(btnClick);
		btnClick.addActionListener(new ActionListener() {
			public void actionPerformed(ActionEvent e) {
				try {
					int numbers = Integer.parseInt(txtNumbers.getText());
					String letters = txtLetters.getText();
					
					txtOutput.setText(letters + numbers);
				}catch(Exception ex){
					JOptionPane.showMessageDialog(frmMain, "There's something wrong going on!\n" + ex.getStackTrace(), "Error!", JOptionPane.ERROR_MESSAGE);
				}
			}
		});
	}
}
