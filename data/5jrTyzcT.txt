import java.util.Scanner;

class HelloWorld {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Input one line of words (S): ");
        String str = scanner.nextLine();

        scanner.close();
        
        str = str.toLowerCase();

        Data consonant[] = new Data[str.length()];
        Data vowel[] = new Data[str.length()];

        int c = 0;
        int v = 0;

        for(int i = 0; i < str.length(); ++i) {
            if(str.charAt(i) == ' ') continue;

            boolean flag = false;

            if(str.charAt(i) == 'a' || str.charAt(i) ==  'i' || 
               str.charAt(i) == 'u' || str.charAt(i) == 'e' || 
               str.charAt(i) ==  'o') {
                for(int j = 0; j < v; ++j) {
                    if(vowel[j].ch == str.charAt(i)) {
                        vowel[j].count += 1;   
                        flag = true;

                        break;
                    }
                }

                if(!flag) {
                    vowel[v] = new Data();
                    vowel[v].ch = str.charAt(i);
                    vowel[v].count += 1;

                    v += 1;
                }
            } else {
                for(int j = 0; j < c; ++j) {
                    if(consonant[j].ch == str.charAt(i)) {
                        consonant[j].count += 1;   
                        flag = true;

                        break;
                    }
                }

                if(!flag) {
                    consonant[c] = new Data();
                    consonant[c].ch = str.charAt(i);
                    consonant[c].count += 1;

                    c += 1;
                }
            }
        }

        for(int i = 0; i < v; ++i) {
            for(int j = 0; j < vowel[i].count; ++j)
                System.out.print(vowel[i].ch);
        }

        System.out.println();

        for(int i = 0; i < c; ++i) {
            for(int j = 0; j < consonant[i].count; ++j)
                System.out.print(consonant[i].ch);
        }
        
    }

    static class Data {
        public char ch = 0;
        public int count = 0;
    }

    @Override
    public String toString() {
        return "HelloWorld []";
    }
}
