#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <conio.h>
#include <windows.h>
int main() {
    SetConsoleOutputCP(CP_UTF8);
    srand((int)time(NULL));
    int n=100;
    int *a;
    int *b;
    a=(int*)calloc(n , sizeof(int));
    b=(int*)malloc(n * sizeof(int));
    for (int i = 0; i < 10; ++i) {
        a[i]=rand()%21-10;
        b[i]=rand()%21-10;
    }
    printf("\nЗадание 3\n");
    printf("Адрес массива а равен= %p\n",&a);
    printf("Адрес массива b равен= %p\n",&b);

    for (int i = 0; i < n; ++i) {
        printf("%d ",*(a+i));
    }
    printf("\n");
    for (int i = 0; i < n; ++i) {
        printf("%d ",*(b+i));
    }
    return 0;
}
