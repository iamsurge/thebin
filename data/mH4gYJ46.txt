using System;
using System.Linq;

namespace Lab6
{
    class Program
    {
        static void Main(string[] args)
        {
            Random rnd = new Random();
            int n, m, count, count2;
            
            Console.Write("Столбцы: ");
            n = Convert.ToInt32(Console.ReadLine());
            Console.Write("Строки: ");
            m = Convert.ToInt32(Console.ReadLine());
            
            int[,] arr = new int[m,n];
            count = 0;
            count2 = n;
            
            Console.Clear();
            for (int i = 0; i < m; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    int temp = rnd.Next(-20, 10);
                    arr[i,j] = temp;
                    Console.Write($"{arr[i,j]}\t");
                }
            Console.WriteLine();
            }
            
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < m; j++)
                {
                    if (arr[j,i] >= 0) count++;
                }
            if (count > 0) count2--;
            count = 0;
            }
            Console.WriteLine($"Кол-во столбцов без положительных: {count2}");
        }
    }
}
