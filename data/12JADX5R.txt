// Алгоритм нахождения самого короткого слова в тексте.

#include <stdio.h>
#include <string.h>

int main(int argc, char **argv) {
	int short_len = 0, len = 0, continue_counting = 1, iter;
	char *symb = " 0123456789\\/.,-+=_!@#$%%^\"&*()";
	char str1[5000];

	printf("Вставьте свой текст:\n> ");
	scanf(" %[^\n]", &str1);
	fflush(stdin);
	strcat(str1, ".");

	for (int i1 = 0; i1 < strlen(str1); i1++) {
		for (int i2 = 0; i2 < strlen(symb); i2++) {
			if (*(str1 + i1) == *(symb + i2)) {
				if (((len < short_len) || (short_len == 0)) && (len > 0)) {
					short_len = len;
					iter = i1;
				}

				i2 = strlen(symb);
				len = 0;
				continue_counting = 0;
			}
		}

		if (continue_counting) {
			len++;
		} else {
			continue_counting++;
		}
	}

	printf("\nСамое короткое слово в тексте: '");
	for (int i1 = 0; i1 < short_len; i1++) {
		printf("%c", *(str1 + iter - short_len + i1));
	}
	printf("'.\n");

	return 0;
}
