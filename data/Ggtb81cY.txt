public class StringProcessor
{
    public static String copy(String s, int N)
    {
        String result = null;
        if(s == null)
            throw new NullPointerException();
        if(N < 0)
            throw new IllegalArgumentException();
        result = s.repeat(N);
        return result;
    }
    public static int numberOfOccurrences(String str,String sub)
    {
        if(str == null)
            throw new NullPointerException();
        if(str.isEmpty())
            throw new IllegalArgumentException();
        int count = (str.length() - str.replace(sub,"").length())/sub.length();
        return count;
    }

    public static String replaceDigits(String s)
    {
        if(s == null)
            throw new NullPointerException();
        s = s.replace("1","один");
        s = s.replace("2","два");
        s = s.replace("3","три");
        return s;
    }
    public static void removeEven(StringBuilder str)
    {
        if(str == null)
            throw new NullPointerException();
        for (int i = 1; i < str.length(); i+=1)
        {
            str.deleteCharAt(i);
        }
    }
    public static String reverseWithSpace(String str)
    {
        StringBuilder buf = new StringBuilder(str);
        String[] words = str.split(" ");
        StringBuilder result = new StringBuilder();
        int j = words.length - 1;
        boolean flag = true;
        for (int i = 0; i < str.length(); i++)
        {
            if(buf.charAt(i) == ' ')
            {
                result.append(" ");
                flag = true;
            }
            else if(flag && buf.charAt(i) != ' ')
            {
                while(words[j].equals(""))
                    j--;
                result.append(words[j]);
                j--;
                flag = false;
            }
        }
        str = result.toString();
        return str;
    }
    public static String hexNumber(String str)
    {
        String[] words = str.split(" ");
        String buf = null;
        StringBuilder result = new StringBuilder();
        int hex;
        for (int i = 0; i < words.length; i++)
        {
            if(words[i].startsWith("0x"))
            {
                buf = words[i].replaceFirst("0x","");
                if(buf.matches("\\p{XDigit}+") && buf.length() == 8)
                {
                    hex = Integer.parseInt(buf,16);
                    result.append(hex).append(" ");
                }
                else
                    result.append(words[i]).append(" ");
            }
            else
                result.append(words[i]).append(" ");
        }
        str = result.toString();
        str = str.trim();
        return str;
    }
    public static void main(String[] args)
    {
        String str1 = "Васе 0x00000010";
        System.out.println(hexNumber(str1));
    }

}
