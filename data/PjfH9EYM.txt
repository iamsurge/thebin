#include <iostream>
#include <cstdlib>
#include <cmath>
#include <iomanip>
#include<ctime>

using namespace std;

int main() {  
    
    srand(time(NULL));
    const int ROWS=5;
    const int COLS=5;
    int a[ROWS][COLS];
    int b[COLS][ROWS],modulesum[COLS];
    int minum_in_each_columns,i,j;
    int column_sum = 0;
    int column_min;
    int max_sum = 0;
    int column_no = 0;
    int result;
    
    for ( int i=0; i<ROWS;i++){
         for ( int j=0; j<COLS;j++){
             a[i][j]=rand()%50-25;
        }
    }
    
    for ( int i=0; i<ROWS;i++){
         for ( int j=0; j<COLS;j++){
         cout<<setw(4)<<a[i][j];
        }
        cout<<endl;
    }
    
        
    for (int j=0; j<COLS; j++){
        
        column_sum = 0;
        
        for (int i=0; i<ROWS; i++){
           
            column_sum = column_sum + fabs(a[i][j]);
            
        }
        
        modulesum[j] = column_sum;
        
        if (column_sum > max_sum) 
            max_sum = column_sum;
            
    }
    
    //cout<<"Max abs sum of elements among all columns = "<<max_sum<<endl;
    
    result = max_sum;
    
    for (int j=0; j<COLS; j++) {
        //cout<<"Abs sum of elements in column #"<< j+1 <<" = "<<modulesum[j]<<endl;
    
        if (modulesum[j] == max_sum) {
            
            column_min = a[0][j];
            
            for (int i=0; i<ROWS; i++){
               
                if (a[i][j] < column_min) column_min = a[i][j];
                
            }
        
            //cout<<"Min element in column #"<< j+1 <<" = "<<column_min<<endl;
            
            if (result > column_min) {
                
                result = column_min;
                column_no = j;
            }
        }

    }
    
    cout<<"Min element among columns with max abs sum of elements is in the column #"<< column_no+1 <<endl;
    cout<<"Result = "<<result<<endl;

}
