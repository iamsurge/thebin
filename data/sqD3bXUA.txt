int go(int n) {
    int ans = 0, i = 0, c = 2, p = 2;
    vector <int> a(n);
    while (i < n) {
        a[i] = i + 2;
        i += 1;
    }
    while (p < n) {
        if (a[p] != 0) {
            c = 2;
            while (p * c < n) {
                a[p * c] = 0;
                c++;
            }
            ans++;
        }
        p += 1;
        c = 2;
    }
    return ans;
}

void solve() {
    for (int i = 1; i <= 10000; i++) {
        if (go(i) == 14) {
            cout << i << endl;
            break;
        }
    }
}
