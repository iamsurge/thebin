// Collapse spaces

#include <stdio.h>
#include <ctype.h>

int main()
{

    int c;
    while((c=getchar())!=EOF)
    {
        if(c==' ')
        {
            while(isspace(c=getchar()));

            if(c!='\n')
                putchar(' ');
            putchar(c);
        }
        else
            putchar(c);
    }
    return 0;
}
