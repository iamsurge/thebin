# script que pide numeros enteros no nulos. Se deben cargar nros hasta 
# que se ingrese un cero. En ese momento, se concluye la carga y se imprime el 
# promedio de todos los nros ingresados

# i = 1
# while True:
#    print(i)
#    i = i + 1
#    continue
#    break  # break me permite salir del bucle, continue lo reinicia 

numeros = []

while True:
    numero = int(input("Ingrese un nro: "))
    if numero == 0:
        break
    numeros.append(numero)

if len(numeros) != 0:
    print(f"El promedio de los nros ingresados es {sum(numeros) / len(numeros):.2f}")
else:
    print("No se ingresaron nros, promedio cero")
