#include <iostream>
#include <map>
#include <string>

using namespace std;

#define sz(x) (int)x.size()

int main() {
	string cip_text, high_freq_text;
	getline(cin, cip_text);
	getline(cin, high_freq_text);

	map<char, int> m;
	for (int i = 0; i < sz(cip_text); i++)
		m[cip_text[i]]++;
	map<char, int>::iterator it = m.begin();
	int shift_value = (high_freq_text[0] - 'a' - (it->first - 'a') + 26) % 26;
	it++;
	for (int i = 1; i < 5; i++, it++) {
		int temp_shift_value =
		    (high_freq_text[i] - 'a' - (it->first - 'a') + 26) % 26;
		if (temp_shift_value != shift_value) {
			cout << "Not a valid high freq text input!\n";
			return 0;
		}
	}
	cout << "Shift value is: " << shift_value << '\n';

	return 0;
}
