using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class RoomTransition : MonoBehaviour
{
    // Create some 2D vectors
    public Vector2 cameraChange;
    public Vector3 playerChange;

    // Create a private reference to cameraMovement.
    private CameraMovement cameraMovement;

    // Initialization method when the script is first called..
    void Start() {
        // Set cameraMovement equal to the Camera main component.
        cameraMovement = Camera.Main.GetComponent<CameraMovement>();
    }

    // Update method, this updates the game once per frame.
    void Update() {

    }

    
    private void OnTriggerEnter2D(Collider2D col)
    {
        if (col.CompareTag("Player")) {
		    cam.minPosition += cameraChange;
            cam.maxPosition += cameraChange;

            col.transform.position += playerChange;
        }
    }
}
