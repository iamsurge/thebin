#include <stdio.h>
#include <cstring>

using namespace std;

int main()
{
    FILE *in;
    in = fopen("fisier.txt", "rb");

    if (in == NULL)
    {
        perror("GRESIT");
        return -1;
    }

    int contor = 1;

    while(1)
    {
        char c = getc(in);
        if(c == EOF)
            break;
        if(contor % 2 == 1)
            printf("%c", c);
        if(c==10)
            contor++;
    }
    fclose(in);
    return 0;
}
