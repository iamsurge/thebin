#include <iostream>
using namespace std;

int n,t,p;
long long x,sc, sm=0;
int m,i;

int main()
{
  cin>>n>>t>>p;
  if(t==1)
  {
      m=1e9;///minimul
      for(i=1;i<=p;i++)///am parcurs primele p
      {
          cin>>x;
          if(x<m)m=x;
          sm+=x;
      }
      ///suma cifrelor
      sc=0;
      while(m)
      {
          sc+=m%10;
          m/=10;
      }
      cout<<sc<<"\n";
  }
  else ///t=2
  {
    m=-1;///maxim
    for(i=1;i<=p;i++)///maximul din primele p
    {
        cin>>x;
        if(x>m)m=x;
        sm+=x; ///suma primelor
    }
    ///numarul de divizori- descompunerea in factori primi
    long long pu=0, d=2,nrd=1;
    while(m!=1)
    {
        pu=0;
        while(m%d==0){ pu++, m/=d;}
        if(pu>0) nrd*=(pu+1);
        if(d*d<m) d++;
        else d=m;
    }
    cout<<nrd<<"\n";
  }

  for( ;i<=n ;i++ ) ///calculul sumei pana la final
  {
      cin>>x;
      sm+=x;
  }
  cout<<sm;
  return 0;
}
