// avg - среднее количество убийств
// \n - снос строки
Console.Write("сегодня играют Navi vs Faze\n" +
    "какое будет максимальный avg? ");

// считываем ответ из ввода с консоли
double answer = double.Parse(Console.ReadLine());

// рассчитываем avg
// Random.Shared.Next(5, 20) - макс кол-во убийств
// Random.Shared.Next(3, 9) - макс кол-во смертей
double result = Random.Shared.Next(5, 20) / Random.Shared.Next(3, 9);

// result - 0.5 < answer < result + 0.5
if (answer > result - 0.5 && answer < result + 0.5)
    Console.WriteLine($"Правильный ответ, вы были близки, максимальный avg равен {result}");
else
    Console.WriteLine($"К сожалению, но ваш ответ слишком отличается от правильного: {result}");
