using System;

namespace ConsoleApp7
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Введите сколько сейчас золота в кармане: ");
            int amountOfUserGold = Convert.ToInt32(Console.ReadLine());
            int priceOfOneCrystal = 20;
            Console.WriteLine($"Цела одного кристалла - {priceOfOneCrystal}");
            Console.WriteLine("Сколько кристаллов купить? Введите кол-во:");
            int howMuchBuyCrystals = Convert.ToInt32(Console.ReadLine());
            int remainingGold = amountOfUserGold - priceOfOneCrystal * howMuchBuyCrystals;
            int remainingСrystal = howMuchBuyCrystals;
            Console.WriteLine($"Остаток золота - {remainingGold}, кристаллов - {remainingСrystal}");
        }
    }
}
