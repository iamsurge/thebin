--Place in ServerScriptService


--CONFIG
--[[
	You can change any of the values below to get the script to do different things!
]]
local config = {
	["shouldRotate"] = false, --should the block rotate towards the player
	["shouldAnchor"] = true, --should it be anchored (not moving)
	["brickColor"] = BrickColor.new("Gold"), --name of color
	["brickSize"] = Vector3.new(50, 10, 5), --size width, height, length
	["brickMaterial"] = Enum.Material.Neon --material name for block
}



local servEv = Instance.new("RemoteEvent", game.ReplicatedStorage)
servEv.Name = "servEv"
servEv.OnServerEvent:Connect(function(plr, mCF)
	local newPart = Instance.new("Part", workspace)

	newPart.Position = mCF.Position
	newPart.Size = config.brickSize
	if config.shouldRotate then
		newPart.CFrame = mCF
	end
	newPart.Anchored = config.shouldAnchor

	newPart.BrickColor = config.brickColor
	newPart.Material = config.brickMaterial
end)
