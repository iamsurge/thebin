// ==UserScript==
// @name         Make Velocity111 less annoying
// @namespace    http://tampermonkey.net/
// @version      0.1
// @description  try to take over the world!
// @author       You
// @match        https://animemusicquiz.com/
// @icon         https://www.google.com/s2/favicons?sz=64&domain=animemusicquiz.com
// @grant        none
// ==/UserScript==

(function() {
    'use strict';
    if (document.getElementById('startPage')) return;

    const l = new Listener('game chat update');
    l.callback = ({messages}) => {
        messages.forEach(message => {
            if (message.sender === 'Velocity111') {
                setTimeout(() => {
                    const element = document.querySelector(`#gcPlayerMessage-${message.messageId} .gcMessage`)
                    element.innerText = message.message;
                }, 0);
            }
        });
    };
    l.bindListener();

})();
