import java.util.ArrayList;

public class BuscaLinear
{
    private int repete;
    private ArrayList<Integer> repetidos = new ArrayList<Integer>();
    private int[] array;

    public BuscaLinear(int[] array)
    {
        this.array = array;
    }

    public ArrayList<Integer> getRepetidos()
    {
        return repetidos;
    }

    public int getRepete() {
        return repete;
    }

    int buscaLinear(int numero)
    {
        repete = 0;
        int i;
        for(i = 0; i < array.length; i++)
        {
            if(numero == array[i])
            {
                repetidos.add(repete);
                repete++;
            }
        }
        return -1;
    }
}
