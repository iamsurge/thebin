vvi mul(vvi a, vvi b) {
	vvi c = a;
	int n = 9;
	rep(i, n) rep(j, n) {
		ll s = 0;
		rep(k, n)
			s += (ll)a[i][k] * (ll)b[k][j] % mod;
		c[i][j] = s % mod;
	}
	re c;
}

vvi st(vvi x, ll p) {
	vvi ans = x;
	rep(i, 9) rep(j, 9)
		ans[i][j] = (i == j);
	while (p) {
		if (p & 1)
			ans = mul(ans, x);
		p /= 2;
		x = mul(x, x);
	}
	re ans;
}

void solve() {
	ll n;
	cin >> n;

	vvi v(9, vi(9));
	rep(i, 9) rep(j, 9) {
		if (check((i + 1) * 10 + (j + 1)))
			v[i][j] = 1;
	}

	v = st(v, n - 1);
	ll ans = 0;
	rep(i, 9) rep(j, 9)
		ans += v[i][j];
	cout << ans % mod;
}
