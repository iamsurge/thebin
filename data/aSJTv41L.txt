// Программа 2
#include <iostream>

int main()
{
    int x, y; // два числа, вводимые с клавиатуры

    std::cout << "Введите первое число: ";
    std::cin >> x;
    std::cout << "Введите второе число: ";
    std::cin >> y;

    // Рассчитаем сумму квадратов этих чисел
    int sum_squares = x*x + y*y;
    std::cout << "Сумма квадратов этих чисел: " << sum_squares << std::endl;

    return 0;
}
