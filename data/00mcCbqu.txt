function getRandomArbitrary(min, max) {
  return Math.floor(Math.random() * (max - min) + min);
}

let randomNum = getRandomArbitrary(20, 50);



console.log(randomNum);

  function* Prime(number) { 
  const infinit = !number && number !== 0;
  const re = /^.?$|^(..+?)\1+$/;  
  let actual = 1;
 
  while (infinit || number-- ) {
      if(!re.test('1'.repeat(actual)) == true) yield actual;
      actual++
  };
};

let [...primers] = Prime(randomNum); //Example
let sumPrimers = primers.reduce((a, b) => a + b, 0)
console.log(primers);
console.log(sumPrimers);
