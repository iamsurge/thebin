-- print fuel level in top right corner
term.clear()
term.setCursorPos(1,1)
f = turtle.getFuelLevel()
-- Fuel Empty
if f == 0 then
  error("NO FUEL!")
end
-- lenth of string "f"
l = math.floor(math.log10(f)+1)
write("CraftOS 1.8")
-- change color if advanced
if term.isColor() then
  fmax = 100000
  dmax = 6
  term.setTextColor(colors.orange)
else
  fmax = 20000
  dmax = 5
end
xpos = 39-dmax-l
term.setCursorPos(xpos,1)
write(f .. "/" .. fmax)
if term.isColor() then
  term.setTextColor(colors.white)
end
term.setCursorPos(1,2)
-- end
print("")
print("Place turtle at bottom left")
write("Steps: ")
steps = read()

local count1 = 1

for i=1, steps do
  turtle.dig()
  turtle.forward()
  turtle.digUp()
  turtle.turnRight()
  turtle.dig()
  turtle.forward()
  turtle.digUp()
  count1 = count1 + 1
  if count1 == 8 then
    count1 = 1
    turtle.placeUp()
  end
  turtle.back()
  turtle.turnLeft()
end
