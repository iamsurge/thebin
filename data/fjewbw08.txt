

//   Пример, который совместим с языком Си:


#include <stdio.h>




///////////////////////////////////////////////////////////////
struct  TT                                                   //  
{
	
    int a;	
	int b;	
};




/////////////////////////////////////////////////////////////// 
int main()                                                   // 
{
	
	TT    x;
		
	x.a = 3;
	x.b = 2;
	
	int S = x.a + x.b;	
	
	printf("S = %d\n", S);
}
