#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    
    vector<int> v;
    for(int i = 0; i < n; i++) {
        int broj;
        cin >> broj;
        v.push_back(broj);
    }
    int x;
    cin >> x;
    sort(v.begin(), v.end());
    
    int pozicija = lower_bound(v.begin(), v.end(), x) - v.begin();
    
    cout << pozicija + 1 << endl;
   return 0;
}
