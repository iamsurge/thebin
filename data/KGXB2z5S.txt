#include <iostream>

using namespace std;

int main()
{
    // tablica dynamiczna:
    int* tablica = new int[10];     // stworzenie tablicy dynamicznej o rozmiarze 10
    for(int i = 0; i < 10; ++i)
        cout << tablica[i] << endl;
    delete[] tablica;               // zwolnienie zaalokowanej pamięci

    // tablica dynamiczna dwuwymiarowa:
    int** tab2 = new int*[5];
    for(int i = 0; i < 5; ++i)
        tab2[i] = new int[6];

    for(int i = 0; i < 5; ++i)
    {
        for(int j = 0; j < 6; ++j)
            cout << tab2[i][j] << '\t';
        cout << endl;
    }

    // zwolnienie pamięci:
    for(int i = 0; i < 5; ++i)
        delete[] tab2[i];
    delete[] tab2;

    return 0;
}
