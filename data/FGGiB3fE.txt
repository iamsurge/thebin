#include <stdio.h>
#include <conio.h>
int
main (void)
{

  struct carro			//Definindo informacoes de Cadastro
  {
    char modelo[10];
    char marca[20];
    char combustivel[15];
	char cor [10]
	char placa [7]
    int ano;
  };

  struct carro carro;		// Definindo a variavel

  printf ("\n::::.... Cadastro carros ....::::\n\n\n");

  printf ("Informe o modelo do carro:");
  fflush (stdin);
  fgets (carro.modelo, 10, stdin);

  printf ("Informe a marca do carro: ");
  fflush (stdin);
  fgets (carro.marca, 20, stdin);

  printf ("Informe a cor do carro: ");
  fflush (stdin);
  fgets (carro.cor, 10, stdin);

  printf ("Informe a placa do carro: ");
  fflush (stdin);
  fgets (carro.placa, 7, stdin);

  printf ("Informe combustivel: ");
  fflush (stdin);
  fgets (carro.combustivel, 10, stdin);

  printf ("Informe o ano de fabricacao: ");
  scanf ("%d", &carro.ano);

  printf ("\n\n --------- informações registradas ---------\n\n");
  printf ("Modelo ...............: %s\n", carro.modelo);
  printf ("Marca ................: %s\n", carro.marca);
  printf ("Cor ..................: %s\n", carro.cor);
  printf ("Placa ................: %s\n", carro.placa);
  printf ("Combustivel ..........: %s\n", carro.combustivel);
  printf ("Ano de fabricacaoo ...: %d\n", carro.ano);

  getch ();
  return (0);
}
