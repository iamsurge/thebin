import java.util.Scanner;
import java.util.Arrays;
import java.util.Collections;

class HelloWorld {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Input the number of family: ");
        int numbers_of_family = scanner.nextInt();

        Integer []families = new Integer[numbers_of_family];

        System.out.print("Input the number of members in the family (seperated by a space): ");

        for(int i = 0; i < numbers_of_family; ++i) {
            families[i] = scanner.nextInt();

            if(families[i] > 4) {
                System.out.println("input must be equal with count of family");

                return;
            }
        }

        Arrays.sort(families, Collections.reverseOrder());

        int count = 0;

        for(int i = 0; i < numbers_of_family; ++i) {
            if(families[i] == 0)
                continue;

            count += 1;

            int pair_index = -1;

            for(int j = 0; j < numbers_of_family; ++j) {
                if(j == i || families[j] == 0) 
                    continue;

                if(families[j] > 0 && families[i] + families[j] < 5)
                    pair_index = j;
            }

            if(pair_index != -1)
                families[pair_index] = 0;
            
            families[i] = 0;
        }

        System.out.println("Minimum bus required is: " + count);
    }
}
