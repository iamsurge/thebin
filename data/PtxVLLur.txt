#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n, m;
    cin >> n;
    int podaroci[n];
    for(int i = 0; i < n; i++)
    {
        cin >> podaroci[i];
    }
    cin >> m;
    int kutii[m];
    for(int i = 0; i < m; i++)
    {
        cin >> kutii[i];
    }
    int najmala = 2000000000;
    int broj = 0;
    sort(kutii, kutii + m);
    for(int i = 0; i < n; i++)
    {
        int index = lower_bound(kutii, kutii + m, podaroci[i]) - kutii;
        if(index >= 0 and index < m)
        {
            if(najmala > (kutii[index] - podaroci[i]))
            {
                najmala = kutii[index] - podaroci[i];
                broj = podaroci[i];
            }
            else if(najmala == kutii[index] - podaroci[i])
            {
                if(broj < podaroci[i])
                {
                    broj = podaroci[i];
                }
            }
        }
    }
    if(broj == 0) {
        cout << -1 << endl;
    }
    
    else
    {
        cout << broj << endl;
        
    }
    return 0;
}
