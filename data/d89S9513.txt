#include <iostream>
using namespace std;

void quickSort(int * arr, int from, int to) {
    int i, j, etalon;
    i = from, j = to;
    etalon = arr[(from + to) / 2];
    cout << " Копие, граници на дяла от  " << i << " до " << j << endl;
    cout << "Еталон= " << etalon << endl;
    do
    {
        while (arr[i] < etalon) {
            i++;
        }
        while (arr[j] > etalon) {
            j--;
        }
        if (i <= j)
        {
            cout << "Размяна на  " << arr[i] << " и " << arr[j] << endl;
            swap(arr[i], arr[j]);
            i++;
            j--;
        }
    } while (i <= j);
    cout << "Край на размените" << endl;

    if (from < to) {
        cout << "Потъване наляво" << endl;
        quickSort(arr, from, j);
        cout << "Потъване надясно" << endl;
        quickSort(arr, i, to);
    }
    else {
        cout << "- ДЪНО. -" << endl << "Изплуване" << endl << endl;
    }
}

void Print(int * arr) {
    for (int i = 0; i < 14; i++) { cout << arr[i] << " "; }
    cout << endl;
}
int main() {
    setlocale(LC_ALL, "Bulgarian");

    int arr[14] = { 10,4,7,70,2,5,0,25,11,1,30,3,5,24 };
    cout << "Начално състояние на масива :" << endl;
    Print(arr);
    quickSort(arr, 0, 14 - 1);
    cout << endl << "Крайно състояние на масива :" << endl;
    Print(arr);
    return 0;
}
