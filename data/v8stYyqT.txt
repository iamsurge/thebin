local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/shlexware/Orion/main/source')))()
local isAutoOpenEnabled
local player = game.Players.LocalPlayer
local Window = OrionLib:MakeWindow({Name = "Random Rooms script", HidePremium = true, SaveConfig = false, ConfigFolder = "OrionTest"})
local Tab = Window:MakeTab({
	Name = "Local player",
	Icon = "rbxassetid://4483345998",
	PremiumOnly = false
})
local Section = Tab:AddSection({
	Name = "Room"
})
Tab:AddButton({
	Name = "Go to next room",
	Callback = function()
		if game.Workspace:FindFirstChild("CurrentRoomKey") then
			player.Character:PivotTo(CFrame.new(game.Workspace.CurrentRoomKey.Position) + Vector3.new(0, 2.5, 0))
		end
		if game.Workspace:FindFirstChild("CurrentRoomKey") then
			player.Character:PivotTo(CFrame.new(game.Workspace.CurrentRoomKey.Position) + Vector3.new(0, 2.5, 0))
		end
		wait(0.6)
		for _,Door in workspace:GetChildren() do 
	      if (Door.Name == "CurrentRoomDoor") then
    	        local DoorMain = Door:FindFirstChild("DoorMain")
    	        if (DoorMain) then 
    	            player.Character:PivotTo(CFrame.new(DoorMain.Position))
    	        end
    	    end
		end
  	end
})
Tab:AddToggle({
	Name = "Auto open doors",
	Default = false,
	Callback = function(Value)
		isAutoOpenEnabled = Value
		while isAutoOpenEnabled == true do
			if game.Workspace:FindFirstChild("CurrentRoomKey") then
				player.Character:PivotTo(CFrame.new(game.Workspace.CurrentRoomKey.Position) + Vector3.new(0, 2.5, 0))
			end
			if game.Workspace:FindFirstChild("CurrentRoomKey") then
				player.Character:PivotTo(CFrame.new(game.Workspace.CurrentRoomKey.Position) + Vector3.new(0, 2.5, 0))
			end
			wait(0.2)
			for _,Door in workspace:GetChildren() do 
    		  if (Door.Name == "CurrentRoomDoor") then
   			        local DoorMain = Door:FindFirstChild("DoorMain")
   			        if (DoorMain) then 
   			            player.Character:PivotTo(CFrame.new(DoorMain.Position))
   			        end
   			    end
			end
		end
	end    
})
OrionLib:Init()
