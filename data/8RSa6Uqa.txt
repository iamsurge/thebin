public class WithArgumentsSetProperty {
     public IContainer Container;

     public WithArgumentsSetProperty() {
         Container = new Container(x => x.For<IClassA>().Use<ClassA>()
                                               .Ctor<int>().Is(5)
                                               .SetProperty(p => {
                                                         p.A = 8;
                                                         p.B = 20;
                                             }));
    }
}
