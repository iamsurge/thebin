def get_max_sbstr(s,c1 ,c2):  
    n = len(s)
    prefix_front = [0 for i in s]
    prefix_back = [0 for i in s]
    for i in range(n):   
        if s[i]==c1:
            if i==0:
                prefix_front[0] = 1
            else:
                prefix_front[i] = prefix_front[i-1]+1
        else:
            if i==0:
                prefix_front[0] = 0
            else:
                prefix_front[i] = prefix_front[i-1]
    
    for i in range(n-1, -1, -1):   
        if s[i]==c2:
            if i==n-1:
                prefix_back[-1] = 1
            else:
                prefix_back[i] = prefix_back[i+1]+1
        else:
            if i==n-1:
                prefix_back[-1] = 0
            else:
                prefix_back[i] = prefix_back[i+1]
    total =0
    for i in range(len(s)):
        if s[i]== c1:
            total+=prefix_back[i]
    
    result = 0
    for i in range(len(s)):
        result = max(result, (prefix_back[i]))
        result = max(result, (prefix_front[i]))
    return total+result

print(get_max_sbstr("bcedecd", "b", "d"))
print(get_max_sbstr("abababab", "a", "b"))
print(get_max_sbstr("cccccc", "c", "b"))
