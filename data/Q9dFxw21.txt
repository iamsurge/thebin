frame = script.Parent
buttonPlus = frame.Plus
buttonMinus = frame.Minus
buttonDiv = frame.Div
buttonMul = frame.Mul
buttonRun = frame.Run
number1 = frame.Number1
number2 = frame.Number2
numberAnswer = frame.Number1


function sum(x, y)
	return x + y
end


function sub(x, y)
	return x - y
end


function div(x, y)
	return y == 0 and "На ноль делить нельязя" or x / y -- а вот и тернарная конструкция
end


function mul(x, y)
	return x*y
end


operation = sum -- переменная которая хранит функцию, которая определяет как совершать вычисления между числами


buttonPlus.MouseButton1Click:Connect(function()
	operation = sum
end)


buttonMinus.MouseButton1Click:Connect(function()
	operation = sub
end)


buttonDiv.MouseButton1Click:Connect(function()
	operation = div
end)


buttonMul.MouseButton1Click:Connect(function()
	operation = mul
end)


buttonRun.MouseButton1Click:Connect(function()
	local x = tonumber(number1.Text) -- а вот и преобразование типов (Downcasting) и необходимость писать локальные переменные
	local y = tonumber(number2.Text)
	numberAnswer.Text = operation(x, y) -- преобразование типов из number в text (Upcasting)
end)
