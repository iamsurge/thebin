class Planet {
    String name,type;
    int nMoons;

    Planet(String name,String type,int nMoons) {
        this.name = name;
        this.type = type;
        this.nMoons = nMoons;
        System.out.println("Planet "+name+ " created.");
    }
    //NAME Getter & Setter
    String getPlanetName() {
        return this.name;
    }
    void setPlanetName(String name) {
        System.out.print("!!! Planet "+this.name+" renamed to ");
        this.name = name;
        System.out.println(this.name+" !!!");
    }

    //TYPE Getter & Setter
    String getPlanetType() {
        return this.type;
    }
    void setPlanetType(String type) {
        System.out.print("!!! "+this.type+" changed to ");
        this.type = type;
        System.out.println(this.type+" !!!");
    }

    //nMoons Getter & Setter
    int getPlanetnMoons() {
        return this.nMoons;
    }
    void setPlanetnMoons(int nMoons) {
        System.out.print("!!! "+this.nMoons+" changed to ");
        this.nMoons = nMoons;
        System.out.println(this.nMoons+" !!!");
    }
}
