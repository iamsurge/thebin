#include <stdio.h>
#include <stdlib.h>


int main() {
    int N;
    scanf("%d", &N);
    int proizvod = 1;
    while(N > 0) {
        int posledna_cifra = N % 10;
        proizvod *= posledna_cifra;
        N /= 10;
    }
    printf("%d\n", proizvod);
    return 0;
}
