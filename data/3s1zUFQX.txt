Private Sub c1_Click()
    If c2.Caption = "Jalankan!" Then
        c2.Caption = "Stop"
        Timer1.Enabled = True
    Else
        c2.Caption = "Jalankan!"
        Timer1.Enabled = False
        p1.Left = 3500
        p1.Top = 150
        Timer1.Enabled = False
    End If
    
End Sub
Private Sub Form_Load()
    Timer1.Enabled = False
End Sub

Private Sub Timer1_Timer()
    If p1.Left > 1050 Then
        p1.Move p1.Left - 10, p1.Top + 5
    End If
End Sub
