package main

import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Print("Enter text: ")
	text, _ := reader.ReadString('\n')
	contain := strings.Contains(text, "help")
	if contain {
		fmt.Println("Вы видите справку.")
	} else {
		fmt.Println("Результат уравнения")
	}
}
