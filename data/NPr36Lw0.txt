-- Define the names of the files to be combined
local file1 = "file1.txt"
local file2 = "file2.txt"

-- Define the name of the combined file
local combinedFile = "combinedFile.txt"

-- Open the first file in read mode
local f1 = fs.open(file1, "r")

-- Read the contents of the first file
local content1 = f1.readAll()

-- Close the first file
f1.close()

-- Open the second file in read mode
local f2 = fs.open(file2, "r")

-- Read the contents of the second file
local content2 = f2.readAll()

-- Close the second file
f2.close()

-- Open the combined file in write mode
local f3 = fs.open(combinedFile, "w")

-- Write the contents of the first file to the combined file
f3.write(content1)

-- Write the contents of the second file to the combined file
f3.write(content2)

-- Close the combined file
f3.close()

-- Print a success message
print("Combined file created successfully!")
