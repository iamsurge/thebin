#!/bin/python3
import random
set_days = {'mon','tue','wed','thr','fri','sat','sun'}
dict_temp = {}
dict_days_temp = {}

for d in set_days:
    for h in range(0,24,2):
        dict_temp[h] = float("{:.1f}".format(random.random()*15-5))

    dict_days_temp[d] = dict_temp.copy()

for d in set_days:
    for h,t in dict_days_temp[d].items():
       print("Teplota v {} ve {}:00 byla {}".format(d,h,t))
