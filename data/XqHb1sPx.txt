#Imports
import math

#Base Code
ver = "3.0.0 (Stable Release)"
print("Welcome to the Python Calculator, you are using Version 3 (Stable), report any bugs to the creator\n")

num1 = input("Enter your first number: ")

if num1 == "DEV" or "Dev" or "dev":
    print("You are using" + ver)

num2 = input("Enter your second number (For sqaure root just say \'None\'): ")
operation = input("Enter your chosen operation:")

if num2 == "None":
    num2 = 0

def main():
    #Addition/Subtraction/Multiplication/Division
    if operation == "Add" or "add":
        ans = int(num1) + int(num2)
        print(ans)
    elif operation == "Subtract" or "subtract":
        ans = int(num1) - int(num2)
        print(ans)
    elif operation == "Multiply" or "multiply":
        ans = int(num1) * int(num2)
        print(ans)
    elif operation == "Divide" or "divide":
        ans = int(num1) / int(num2)
        print(ans)
    #Exponentiation and square roots
    elif operation == "Power" or "power":
        ans = int(num1) ** int(num2)
        print(ans)
    elif operation == "Square Root" or "Square root" or "square Root" or "square root":
        ans = math.sqrt(int(num1))
        print(ans)
    else:
        ans = "Error: Expected \'Add\' \'Subtract\' \'Multiply\' \'Divide\' \'Power\' \'Sqaure Root\', user argument was invalid."
        print(ans)

main()
