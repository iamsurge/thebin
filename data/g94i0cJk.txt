menu menubar {
  URL
  .$submenu($show_urls($1))
}
alias ulc { return $replace($1-,$chr(58),$chr(1417)) } 
alias show_urls {
  var %t = $url(0), %c = 1
  if (%t == 0) && ($1 == 1) return $style(2) Empty :$null
  while (%c <= %t) {
    var %desc = $url(%c).desc, %url = $url(%c)
    if ($1 == %c) && (%c <= 20) return $ulc(%desc) :url -n %url 
    if ($1 == 21) && (%t >= 20) return More... :url show
    inc %c
  }
}
