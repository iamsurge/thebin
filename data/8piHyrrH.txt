file = './input.txt'

def parse_input(file):
    return open(file, 'r').read().splitlines()


def map_input(list):
    rucksacks = []
    for line in list:
        rucksacks.append(map_rucksack(line))
    return rucksacks

def map_rucksack(line):
    length = int(len(line) / 2)
    items1 = map_items(line[0:length])
    items2 = map_items(line[length:2*length])
    return Rucksack(items1, items2)

def map_items(characters):
    items = []
    for char in characters:
        items.append(Item(char))
    return items

class Rucksack:

    def __init__(self, items1, items2):
        self.items1 = items1
        self.items2 = items2

    def common_item(self):
        for item1 in self.items1:
            for item2 in self.items2:
                if item1.character == item2.character:
                    return item1

class Item:

    def __init__(self, character):
        self.character = character

    def priority(self):
        ascii_code = ord(self.character)
        if self.character.islower():
            return ascii_code - 96
        else:
            return ascii_code - 64 + 26


def sum_priorities_common_items(rucksacks):
    sum = 0
    for rucksack in rucksacks:
        sum += rucksack.common_item().priority()
    return sum

if __name__ == '__main__':
    list = parse_input(file)
    rucksacks = map_input(list)
    sum = sum_priorities_common_items(rucksacks)
    print(sum)
