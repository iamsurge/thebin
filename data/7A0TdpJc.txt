namespace Seabattle
{
    public enum CellStatus
    {
        Empty,
        Ship,
        Miss,
        Damaged,
        Killed
    }

    public class Cell
    {
        public CellStatus Status;
    }

    public class Field
    {
        public Cell[,] Cells;
        public int Width;
        public int Height;
    }

    public class Player
    {
        public Field Field;
        public Field ForeignField; // чтобы отмечать промахи и попадания
    }

    public class Game
    {
        public Player[] Players = new Player[2];
        public int Turn = 0;
    }
}
