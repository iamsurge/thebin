# Script que implementa un pequeño diccionario de palabras: debe
# poseer un menu con opciones de agregar palabra, consultar significado,
# modificar significado
dicc = {}

print("DICCIONARIO")
print("-----------")

while True:
    print("""
    Menu de opciones:
    1. Agregar palabra
    2. Consultar significado
    3. Modificar significado
    4. Salir
    """)
    opcion = input("Seleccione su opción: ")
    
    if opcion == "1":
        palabra = input("Ingrese la palabra: ")
        if palabra not in dicc:
            significado = input("Ingrese el significado: ")
            dicc[palabra] = significado
        else:
            print("Ya existe la palabra")
        
                
    elif opcion == "2":
        palabra = input("Ingrese la palabra: ")
        if palabra in dicc:
            print(dicc[palabra])
        else:
            print("No existe la palabra")
        
    elif opcion == "3":
        palabra = input("Ingrese la palabra: ")
        if palabra in dicc:
            significado = input("Ingrese el nuevo significado: ")
            dicc[palabra] = significado
        else:
            print("No existe la palabra")
        
           
    
    elif opcion == "4":
        print("Hasta pronto...")
        break
        
    else:
        print("Opción incorrecta")
