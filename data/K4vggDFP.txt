#include<bits/stdc++.h>
using namespace std;
map<long long ,int > cnt;
int main(){
	cin.tie(0)->sync_with_stdio(0);
	cin.exceptions(cin.failbit);
	int q;
	cin >> q;
	while(q--){
		int n;
		long long mx = -1;
		cin >> n;
		for(int i=1;i<=n;i++){
			long long num;
			cin >> num;
			mx = max(mx,num);
			cnt[num]++;
			while(cnt[num] > 1){
				cnt[num] = 0;
				cnt[num+1]++;
				mx = max(mx,num+1);
				num++;
			}
		}
		int sum = 0;
		for(auto x:cnt)
			sum+=x.second;
		cout << mx+1-sum << '\n';
		cnt.clear();
	}
	return 0;
}
