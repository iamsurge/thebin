using System;
namespace LearnRSUE
{
    class firstPers2
    {
        static void Main()
        {
            int size = 5;
            int[] masiv = new int[size];

            for (int i = 0; i < size; i++)
            {
                Console.WriteLine($"Введите элемент номер {i+1}");
                masiv[i] = Convert.ToInt32(Console.ReadLine());
            }

            int testUp = masiv[0], countUp = 0;
            int testDown = masiv[0], countDown = 0;

            for (int i = 0; i < size; i++)
            {
                if (masiv[i] > testUp)
                {
                    testUp = masiv[i];
                    countUp++;
                }

                if (masiv[i] < testDown)
                {
                    testDown = masiv[i];
                    countDown++;
                }
            }

            if (countUp == 4)
            {
                Console.WriteLine("Целые числа упорядоченны, возрастают");
            }
            else if (countDown == 4)
            {
                Console.WriteLine("Целые числа упорядоченны, убывают");
            }
            else
            {
                Console.WriteLine("Целые числа не упорядоченны");
            }


        }
    }
