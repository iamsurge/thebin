#include <bits/stdc++.h>
#define int long long 
using namespace std;

int build(int n, int k) {
    int res = 0;
    vector<vector<int>> b;
    b.resize(n + 1);
    for (int i = 0; i < n + 1; ++i) {
        b[i].resize(n + 1);
    }
    for (int i = 0; i <= n; ++i) {
        b[i][0] = 1;
        b[i][i] = 1;
        for (int t = 1; t < i; ++t) {
            b[i][t] = b[i - 1][t - 1] + b[i - 1][t];
        }
    }
    res = b[n][k];

    return res;
}

int32_t main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, k;
    cin >> n >> k;
    int res = build(n, k);
    cout << res << '\n';

    return 0;
}
