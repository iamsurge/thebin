/* Create a program that calculates the Voltage, Resistance, and Circuit using Ohm's Law Formula*/

#include <iostream>
using namespace std;

int main() {
	char Loop = 'n'; //Declare the Variables
	int Menu, Resist, Volt, Curr;
	do {
		printf("====================\n");
		printf("Ohm's Law Main Menu\n");
		printf("[1] Voltage\n");
		printf("[2] Resistance\n");
		printf("[3] Circuit\n");
		printf("[4] Exit\n");
		printf("Select your choice: "); cin >> Menu;
		printf("====================\n");
		
		switch (Menu){
			case 1:
				printf("Enter the current's ampere: "); cin >> Curr;
				printf("Enter how many ohms of the resistor: "); cin >> Resist;
				
				printf("====================\n");
				cout << "Voltage: " << Curr * Resist << endl; // Directly calculates and outputs the result
				printf("====================\n");
				printf("Exit? "); cin >> Loop;
				break;
			case 2:
				printf("Enter how many volts: "); cin >> Volt;
				printf("Enter the current's ampere: "); cin >> Curr;
				
				printf("====================\n");
				cout << "Resistance: " << (Volt/Curr) << endl;
				printf("====================\n");
				printf("Exit? "); cin >> Loop;
				break;
			case 3:
				printf("Enter how many volts: "); cin >> Volt;
				printf("Enter how many ohms of the resistor: "); cin >> Resist;
				
				printf("====================\n");
				cout << "Current: " << (Volt/Resist) << endl;
				printf("====================\n");
				printf("Exit? "); cin >> Loop;
				break;
			case 4:
				Loop = 'y'; //Returns 'y' to stop the loop
		}
	} while (Loop == 'n');
}
