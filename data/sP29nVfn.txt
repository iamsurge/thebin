function format_phone_number(array)
{
	let result = "";
	for (i in array)
	{
		if (i == 0)
		{
			result += "(";
		}
		if (i == 6)
		{
			result += "-";
		}
		result += array[i];
		if (i == 2)
		{
			result += ") "
		}

	}
	console.log(result);
	return result;
}
