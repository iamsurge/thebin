    stage('Prepare branch as Update'){
        when {
            expression {return params.DEPLOY_BRANCH}
            expression {return params.WIPE_BRANCH}
          }
        steps{
          withEnv(["DEPLOY_BRANCH=${DEPLOY_BRANCH}"]) {
            checkout([
              $class: 'GitSCM',
              branches: [[name: "master"]],
              exstensions: [],
              submoduleCfg: [],
              userRemoteConfigs: [
                    [
                      credentialsId: "${GIT_CREDENTIALS_ID}",
                      url:"${GIT_URL_REPO}" 
                    ]
              ]
            ])
            withCredentials([sshUserPrivateKey(credentialsId: "${GIT_CREDENTIALS_ID}")]){
            script{
              sh """
                git push origin --delete ${DEPLOY_BRANCH} 
                git checkout -b ${DEPLOY_BRANCH}

				SOME ACTIONS!


                echo "push to new ${DEPLOY_BRANCH} branch"
                git config --global user.email "$EMAIL"
                git config --global user.name "$NAME"
                git add -A'
                git commit -m "TEXT"
                git config --global push.default simple
                git push origin ${DEPLOY_BRANCH}
                """                
            }
          }
          }
      }
    }
