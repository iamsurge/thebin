#include <iostream>
 
using namespace std;
const DECIMAL = 5.50; // const 不能redeclare，要换成double
const string blanks = " "
 
int main() {
  double height, weight;
  double 10%discount;
  double billingAmount$; // billingAmount
  double bonus;
 
  int hoursworked = 45;
  height = 6.2;
  weight = 156;
 
  cout << height << " " << weight << endl;
 
  discount = (2 * height + weight) % 10.0; // modulo (%) 只能整数跟整数 可能可以换成/ 或者
  price = 49.99; // undeclared variable, 要在前面放double or 在line 12 放double price
  billingAmount = price * (1 ? discount) - DECIMAL; // undeclared variable, ？ operator 要 ? & : 一起用，不过这个情况下应该不需要用到这个 ternary operator
  DECIMAL = 7.55;
  cout << price << blanks << $ billingAmount << endl; // 应该是 "$ "<< billingAmount
  bonus = hoursworked * PAY_RATE / 50; // 你没有define PAY_RATE, either 在line 6 or line 12 放 double PAY_RATE = xx.xx eg: double PAY_RATE = 12.35
  cout << "Bonus = " << bonus << endl;
  return 0;
}

#include <iostream>
 
using namespace std;
double DECIMAL = 5.50;
const string blanks = " ";
 
int main() {
  double height, weight;
  double TenPercentdiscount=10.0;
  double billingAmount;
  double bonus;
 
  int hoursworked = 45;
  height = 6.2;
  weight = 156;
 
  cout << height << blanks << weight << endl;
 
  double discount = (2 * height + weight) / TenPercentdiscount;
  double price = 49.99;
  double PAY_RATE = 10;
  billingAmount = price * (1 / discount) - DECIMAL;
  DECIMAL = 7.55;
  cout << price << blanks << "$ " << billingAmount << endl;
  bonus = hoursworked * PAY_RATE / 50;
  cout << "Bonus = " << bonus << endl;
  return 0;
}
