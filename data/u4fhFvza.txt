def factorialo():
    new = 1
    a = int(input('Введите число для факторила: '))
    if a < 0:
        print('Факториала для негативных чисел не существует')
        prod = input('Чтобы начать программу заново напишите слово "prod": ')
        if prod == prod:
            factorialo()
    elif a == 0:
        print('Факториал нуля равен 1')
        prod = input('Чтобы начать программу заново напишите слово "prod": ')
        if prod == prod:
            factorialo()
    else:
        for i in range(1, a + 1):
            new = new*i
        print('Факториал числа ',a, 'равен: ',new)
        factorialo()
factorialo()
