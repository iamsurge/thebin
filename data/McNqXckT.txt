tardis = peripheral.wrap("top")
refueling = false
 
while true do
    if tardis.getSpeed() > 0.0 and tardis.getSpeed() < 1.0 then
        tardis.setSpeed(1.0)
    end
    if tardis.getTimeLeft() ~= 0 then
        if tardis.getDoors() ~= "CLOSED" then
            tardis.setDoors("CLOSED")
        end
        refueling = false
    else
        if not refueling then
            tardis.setRefuel(true)
            refueling = true
        end
    end
        
    sleep(0.1)
end
