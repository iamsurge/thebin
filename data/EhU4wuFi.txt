from random import randint
from english_words import english_words_set


for i in range(50):
    t1 = "B"
    t2 = "B"
    while ord("Z") >= ord(t1[0]) >= ord("A"):
        t1 = list(english_words_set)[randint(1, 25487)]
    while ord("Z") >= ord(t2[0]) >= ord("A"):
        t2 = list(english_words_set)[randint(1, 25487)]
    print(t1, t2)
