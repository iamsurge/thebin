import java.util.Scanner;

public class TP06 {
    //TP 01
    private static float mySuhu(float suhuCelcius){
        float result;
        result = (suhuCelcius - 32) * 5/9 ;
        return result;
    }
    //TP 02
    private static void myString(String kalimat){

        kalimat.toLowerCase();
        kalimat = kalimat.replaceAll("a", "");
        kalimat = kalimat.replaceAll("i", "");
        kalimat = kalimat.replaceAll("u", "");
        kalimat = kalimat.replaceAll("e", "");
        kalimat = kalimat.replaceAll("o", "");
        System.out.println(kalimat);
    }
    private static void myBangunDatar(String bangunDatar, int number, int number2){

        int luas, keliling;
        float luasLingkaran, kelilingLingkaran;
        switch (bangunDatar){
            case "Persegi" :
                luas = number * number;
                keliling = 4 * number;
                System.out.println(luas + " " + keliling);
                break;
            case "Persegi Panjang" :
                luas = number * number2;
                keliling = 2 * (number + number2);
                System.out.println(luas + " " + keliling);
                break;
            case "Segitiga" :
                luas = number * number2 / 2;
                keliling = number2 * 3;
                System.out.println(luas + " " + keliling);
                break;
            case "Lingkaran" :
                luasLingkaran = (float) (3.14 * number * number / 4);
                kelilingLingkaran = (float) (3.14 * number);
                System.out.println(luasLingkaran + " " + kelilingLingkaran);
                break;
        }

    }

    //TP03
    public static void main(String[] args) {
        //Todo : Scanner

        //Todo : TP 01

        float suhu1, suhu2, suhu3, inputan;

        inputan = input.nextFloat();
        suhu2 = mySuhu(inputan);

        suhu3 = mySuhu(inputan);
        System.out.printf("%.1f %.1f %.1f",suhu1 , suhu2 , suhu3);
        System.out.println("\n");

        //Todo : TP 02
        String contohKalimat;
        contohKalimat = input.nextLine();
        myString(contohKalimat);
        //Todo : TP 03
        String namaBangunDatar;
        int numberMain, numberMain2;
        namaBangunDatar = input.nextLine();
        switch (namaBangunDatar){
            case "Persegi" :
                namaBangunDatar = "Persegi";
                numberMain = input.nextInt(); numberMain2 = 0;
                myBangunDatar(namaBangunDatar, numberMain, numberMain2);

            case "Persegi Panjang" :
                namaBangunDatar = "Persegi Panjang";
                numberMain = input.nextInt(); numberMain2 = input.nextInt();
                myBangunDatar(namaBangunDatar, numberMain, numberMain2);
                break;
            case "Segitiga" :
                namaBangunDatar = "Segitiga";
                numberMain = input.nextInt(); numberMain2 = input.nextInt();
                myBangunDatar(namaBangunDatar, numberMain, numberMain2);
                break;

            case "Lingkaran" :
                namaBangunDatar = "Lingkaran";
                numberMain = input.nextInt(); numberMain2 = 0;
                myBangunDatar(namaBangunDatar, numberMain, numberMain2);
                break;
        }        
    }
}
