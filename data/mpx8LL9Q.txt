using System;

namespace CleverLily
{
	class Program
	{
		static void Main(string[] args)
		{
		    string destination = Console.ReadLine();
		    decimal budget = 0M;
		    
		    decimal saveAmount = 0M;
		    decimal tottalSaveAmount = 0M;
		    
		    while(destination != "End")
		    {
		        if(destination == "End")
		        {
		            break;
		        }
		        
		        budget = decimal.Parse(Console.ReadLine());
		        
		        while(tottalSaveAmount <= budget)
		        {
		            saveAmount = decimal.Parse(Console.ReadLine());
		            
		            tottalSaveAmount += saveAmount;
		        }
		        
		        Console.WriteLine($"Going to {destination}!");
		        
		        tottalSaveAmount = 0M;
		        
		        destination = Console.ReadLine();
		    }
		}
	}
}
