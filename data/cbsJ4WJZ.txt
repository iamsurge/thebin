#include <stdio.h>
#include <math.h>

int main(int argc, char const *argv[])
{
    int T, A, N, B, C, D, sqrtA;
    int power;
    int M = 1000000007;
    int i, j = 1;

    scanf("%d", &T);

    for (i = 0; i < T; i++)
    {
        scanf("%d %d", &A, &N);
        sqrtA = (int)sqrt((double)A);
        // printf("%d\n", sqrtA);
        if (sqrtA * sqrtA == A)
        {
            B = sqrtA;
            C = sqrtA;
        }
        else
        {
            while (j <= sqrtA)
            {
                if (A % j == 0)
                {
                    B = j;
                    C = A / j;
                }
                j++;
            }
        }

        N = N % M;
        B = B % M;
        C = C % M;
        //printf("B: %d C: %d N: %d\n", B, C, N);
        D = B * C;
        power = (int)pow((double)D,(double) N);
        power %= M;
        printf("Case #%d: %d\n", i + 1, power);
    }

    return 0;
}
