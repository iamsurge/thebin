function isFiller(block)
    return block.name == "minecraft:stone"
end

function digahole()
    local distance = 0;
    
    while true do
        if not turtle.digDown() and not turtle.down() then break end
        distance = distance + 1
    end
    
    for i=0,distance do
        turtle.up()
        distance = distance - 1
    end
end

			turtle.turnLeft()

			turtle.forward()
			turtle.turnRight()
		else

		end

end
