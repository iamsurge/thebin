public class JavaComSQL {

    public static void main(String[] args) throws ParseException {
        Scanner ler = new Scanner(System.in);
        Movimento m = new Movimento();
        m.listar();
        

        System.out.printf("Informe a descrição:");
        m.setDescricao(ler.next());
        
        System.out.printf("Informe o valor:");
        m.setValor(ler.nextDouble());
        
        System.out.printf("Informe o local:");
        m.setLocal(ler.next());
        
        System.out.printf("Informe a data:");
        String lerData = ler.next();
        
        DateFormat format = new SimpleDateFormat("dd/MM/yyyy", Locale.US);
        java.util.Date date = format.parse(lerData);
        java.sql.Date sqlDate = new java.sql.Date(date.getTime());
        
        m.setData(sqlDate);
  
        Categorias c = new Categorias();
        System.out.printf("Selecione o código da categoria:");
        c.listar();
        m.setCodCategoria(ler.nextInt());
        
        m.inserir();
        
        System.out.println("---");
        
        m.listar();
		
/* abertura da tela FormMovimentos
        FormMovimentos f = new FormMovimentos();

        f.setLocationRelativeTo(f);
        f.setVisible(true);*/


}
}
