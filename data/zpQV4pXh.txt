using System;

namespace Barcode_Generator_2
{
	class BarcodeGenerator2
	{
		static void Main(string[] args)
		{

			int beginningRange = int.Parse(Console.ReadLine());
			int endRange = int.Parse(Console.ReadLine());

			int startN4 = beginningRange % 10;
			beginningRange = beginningRange / 10;

			int startN3 = beginningRange % 10;
			beginningRange = beginningRange / 10;

			int startN2 = beginningRange % 10;
			beginningRange = beginningRange / 10;

			int startN1 = beginningRange % 10;


			int endN4 = endRange % 10;
			endRange = endRange / 10;

			int endN3 = endRange % 10;
			endRange = endRange / 10;

			int endN2 = endRange % 10;
			endRange = endRange / 10;

			int endN1 = endRange % 10;

			for (int i = startN1; i <= endN1; i++)
			{

				if ((i % 2) != 0)
				{

					for (int j = startN2; j <= endN2; j++)
					{

						if ((j % 2) != 0)
						{

							for (int k = startN3; k <= endN3; k++)
							{

								if ((k % 2) != 0)
								{

									for (int l = startN4; l <= endN4; l++)
									{

										if ((l % 2) != 0)
										{

											Console.Write($"{i}{j}{k}{l} ");

										}

									}

								}

							}

						}

					}

				}

			}
		}
	}
}

//•	Втори ред – четирицифрено число – края на обхвата. Цяло число в интервала [1000…9999]
