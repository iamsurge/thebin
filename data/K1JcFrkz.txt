#include <stdlib.h>
#include <stdio.h>

int main(void) {
    long long n = 0;
    scanf("%lld", &n);
    char * s = (char *)malloc((n + 1) * sizeof(char));
    char * p1 = NULL, * p2 = s + n - 1;
    for (int i = 0; i < n; ++i) {
        scanf("%c", s + i);
    }
    for (int i = n - 1; i >= 1; --i) {
        if (s[i - 1] < s[i]) {
            p1 = s + i - 1;
        } else {
            if (p1 != NULL) {
                break;
            }
            p2 = s + i - 1;
        }
    }
    int i = 0;
    while (p1 + i != p2) {
        printf("%c", *(p1 + i));
        i++;
    }
    return 0;
}
