wanted_profit = float(input())
flag = False

total_income = 0
income = 0
price = 0
input_line = input()
while input_line != "Party!":
    name = input_line
    count = int(input())
    price = len(name)
    if price % 2 != 0:
        price = price - (price * 0.25)
    income = price * count
    total_income = total_income + income
    if flag:
        print("Target acquired.")
        print(f"Club income - {total_income:.2f} leva.")
        break
    input_line = input()

if input_line == "Party":
    diff = abs(wanted_profit - total_income)
    print("We need {diff:.2f} leva more.")
    print(f"Club income - {total_income:.2f} leva.")
