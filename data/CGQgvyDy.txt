local lenHole = 0

write("Enter len tunnel: ")
lenHole = tonumber(read())

turtle.refuel(64)

turtle.dig()
turtle.forward()

while lenHole > 0 do   
    turtle.digUp()
    turtle.up()
    turtle.digUp()
    turtle.up()    
    
    turtle.turnRight()
   
    turtle.dig()
    turtle.forward()
    turtle.digDown()
    
    turtle.dig()
    turtle.forward()

	turtle.digDown()
    turtle.down()
	turtle.digDown()
    turtle.down()    
    
    turtle.turnLeft()    
    turtle.turnLeft()  
    
    turtle.dig()
    turtle.forward()
    turtle.dig()
    turtle.forward()    
    
    turtle.turnRight()
    
    turtle.dig()
    turtle.forward()    
    lenHole = lenHole-1
end
