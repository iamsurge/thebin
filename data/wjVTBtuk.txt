
package nillas_insertionsort;

import java.io.BufferedReader;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;

public class Nillas_InsertionSort {
    
    private static void insertionSort(int[] array) {
        for (int i = 1; i < array.length; i++) {
            System.out.println("Step: " + i + " " + Arrays.toString(array));
            int current = array[i];
            int j = i - 1;
            while (j >= 0 && array[j] > current) {
                array[j + 1] = array[j];
                j--;
            }
            array[j + 1] = current;
        }
    }
    
    public static void main(String[] args) throws FileNotFoundException, IOException {
         // Reading from file input
        FileInputStream fis = new FileInputStream("input.txt");
        BufferedReader br = new BufferedReader(new InputStreamReader(fis));
        
        String line;
        int[] array = null;
        while ( (line = br.readLine()) != null) {
            String[] tempArr = line.split(" ");
            array = new int[tempArr.length];
            
            for (int i = 0; i < tempArr.length; i++)
                array[i] = Integer.parseInt(tempArr[i]);
        }
        
        insertionSort(array);
        System.out.print("Final Output: ");
        System.out.println(Arrays.toString(array));
    }
}
