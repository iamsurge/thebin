#include<bits/stdc++.h>
using namespace std;

// Time Complexity: O(n^2)
// For ODD Length String
int main() {
    int L, R;
    string s;
    cin >> s;
    int max_len = 0;
    for(int center = 0; center < s.size(); center++) {
        int cur_len = 1;
        for(int k=1; center-k >= 0 && center+k <=s.size(); k++) {
            if(s[center-k] != s[center+k]) break;
            else {
                cur_len += 2;
            }
        }
        if(cur_len > max_len) {
            max_len = cur_len;
            L = center - cur_len/2;
            R = center + cur_len/2;
        }

    }
    cout << max_len << endl;
    for(int i=L;i<=R;i++) {
        cout << s[i];
    }

    return 0;
}
