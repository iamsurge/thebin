#include<bits/stdc++.h>
using namespace std;

string s1, s2;

int isSubString() {
    int flag;
    for(int i=0; i<s1.size(); i++) {
        flag = 1;
        for(int j=0; j<s2.size(); j++) {
            if(s1[i+j] != s2[j]) {
                flag = 0;
            }
        }
        if(flag) return 1;
    }
}

int main() {
    cin >> s1 >> s2;

    cout << isSubString();

    return 0;
}
