import java.util.Scanner;
public class TP03 {
	private static void myBangunDatar(String bangunDatar, int number, int number2){
		int luas, keliling;
		float luasLingkaran, kelilingLingkaran;
		switch (bangunDatar){
		    case "Persegi" :
			luas = number * number;
			keliling = 4 * number;
			System.out.println(luas + " " + keliling);
			break;
		    case "Persegi Panjang" :
			luas = number * number2;
			keliling = 2 * (number + number2);
			System.out.println(luas + " " + keliling);
			break;
		    case "Segitiga" :
			luas = number * number2 / 2;
			keliling = number2 * 3;
			System.out.println(luas + " " + keliling);
			break;
		    case "Lingkaran" :
			luasLingkaran = (float) (3.14 * number * number / 4);
			kelilingLingkaran = (float) (3.14 * number);
			System.out.println(luasLingkaran + " " + kelilingLingkaran);
			break;
		}

    	}
	public static void main(final String[] args) {
		//Todo : Scanner
		Scanner input = new Scanner(System.in);
		//Todo : TP 03
		String namaBangunDatar;
		int numberMain, numberMain2;
		namaBangunDatar = input.nextLine();
		switch (namaBangunDatar){
		    case "Persegi" :
			namaBangunDatar = "Persegi";
			numberMain = input.nextInt(); numberMain2 = 0;
			myBangunDatar(namaBangunDatar, numberMain, numberMain2);
			break;
		    case "Persegi Panjang" :
			namaBangunDatar = "Persegi Panjang";
			numberMain = input.nextInt(); numberMain2 = input.nextInt();
			myBangunDatar(namaBangunDatar, numberMain, numberMain2);
			break;
		    case "Segitiga" :
			namaBangunDatar = "Segitiga";
			numberMain = input.nextInt(); numberMain2 = input.nextInt();
			myBangunDatar(namaBangunDatar, numberMain, numberMain2);
			break;

		    case "Lingkaran" :
			namaBangunDatar = "Lingkaran";
			numberMain = input.nextInt(); numberMain2 = 0;
			myBangunDatar(namaBangunDatar, numberMain, numberMain2);
			break;
		}
	}
}
