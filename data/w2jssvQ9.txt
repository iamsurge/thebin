# for cmd color
from os import system

# for one second delay
from time import sleep



# loop 10 times - from 10 to 0 with step -1
for i in range(10, 0, -1):

    # formatted string
    print(f"Deleting C:\Windows\system32 in {i} seconds")
    
    # time to wait, in seconds
    sleep(1.0)
    
    
print("See you soon :)")
sleep(0.6) 
    
    
# clear screen
system("cls") 

# simulate blackout. Actually, changing colors to white-on-black
system("color 0F") 
sleep(0.8)


# simulate BSOD. Actually, changing colors to white-on-blue
system("color 1F") 
sleep(0.15)


# Available colors:
#    0 = Black       8 = Gray
#    1 = Blue        9 = Light Blue
#    2 = Green       A = Light Green
#    3 = Aqua        B = Light Aqua
#    4 = Red         C = Light Red
#    5 = Purple      D = Light Purple
#    6 = Yellow      E = Light Yellow
#    7 = White       F = Bright White
