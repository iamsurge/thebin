from math import fabs, ceil, floor
nested_money, year, k = float(input()), int(input()), 19
# ver 1 - традиционното решение, което предполагам се очаква
ost= nested_money
for i in range (1800, year+2):
    if i % 2 == 0:
        outgo = 12000
        ost = ost- outgo
    else:
        outgo = 12000+ 50*k
        ost =  ost- outgo
        k +=1
if ost >= 0:
    print(f'Yes! He will live a carefree life and will have {ost:.2f} dollars left.')
else:
    print(f'He will need {fabs(ost):.2f} dollars to survive.')
