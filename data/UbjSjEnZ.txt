max_multiple_7 = 0
max_multiple_1 = 0

n = 0
a = int(input())
while a != 0:
    n += 1
    if a % 7 == 0 and a % 49 != 0:
            max_multiple_7 = max(max_multiple_7, a)
    else:
        max_multiple_1 = max(max_multiple_1, a)
    a = int(input())

r = int(input())

print('input:', n)
print('reference value:', r)
if max_multiple_7 == 0 or max_multiple_1 == 0:
    c = 1
else:
    c = max_multiple_7 * max_multiple_1
print('calculated value:', c)
if r == c:
    print('values true')
else:
    print('values false')
