local args = {...}

local saveFile = function (name, file) 
  local f = fs.open(name, "w")
  f.write(file)
  f.close()
end

local update = function() 
  print("Updating Bootloader")
  local res = http.get("http://atm3.mc.hzuccon.com/scipts/bootloader").readAll()

  saveFile("update", res)
  print("Updated Bootloader")
end

local install = function (script)
  print("Updating "..script)
  local res = http.get("http://atm3.mc.hzuccon.com/scipts/"..script).readAll()
  
  if not res then
    print("Could not download any script")
    return
  end

  print("Updated "..script)
end

local run = function () 
    if not args[2] then
      update()
      return
    end

    if args[2] == "install" then
      install(args[3])
    end
end
