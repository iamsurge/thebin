void Citire(int a[], int &n)
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
}
void Sort(int a[],int st,  int dr, int d)///1-cresc -1-desc
{
    bool ord=0;
    while(ord==0)
    {
        ord=1;
        for(int i=st;i<dr;i++)
            if(a[i]*d>a[i+1]*d)
        {
            swap(a[i],a[i+1]);
            ord=0;
        }
    }
}
void Afisare(int a[], int n)
{
    for(int i=1;i<=n;i++)
        cout<<a[i]<<" ";
    cout<<"\n";
}
