import datetime
date = datetime.datetime.strptime(input(), "%d %B %Y")

temperature = int(input())
rain = int(input())
winter = int(input()) #35 = 5

def isleap(n):
    if n%4==0 and n%100!=0:
        if n%400==0:
            return True
    elif n%4!=0:
        return False


# calculate winter days
winterDays = winter // 7
date += datetime.timedelta(days=winterDays)

# Handle temperature drift
if isleap(date.year): temperature += 5  # leap year mod
dt = temperature - 20
date += datetime.timedelta(days=-dt)
 
# Handle rain levels
dt = (rain - 30) // 3
date += datetime.timedelta(days=-dt)


print( date.strftime("%d %B %Y") )
