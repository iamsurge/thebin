#!/usr/bin/env python

# TEMPLATE
"""
# Script que pide una edad y responde la condicion frente al voto
# EJ:
edades menores a 16: prohibido
16 cumplidos y menos de 18: optativo
18 cumplidos y menos de 70: obligatorio
70 cumplidos o mas: optativo
"""
while True:
    
    #################################################
    edad = int(input("Ingrese su edad: "))

    if 0 <= edad <= 15:
        print("Voto: Prohibido")

    elif 16 <= edad <= 17 or edad >= 70:
        print("Voto: Optativo")
    
    elif 18 <= edad <= 69:
        print("Voto: Obligatorio")
    
    else:
        print(f"Error, ingrese edad valida: {edad}")

    ################################################

    opcion = input("Presione cualquier tecla para continuar, o '-1' para salir: ")
    if opcion == "-1":
        print("Gracias por utilizar este programa...")
        break
