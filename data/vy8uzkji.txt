    <script>
		$(window).scroll(function () {
            window.requestAnimationFrame(function() {
                // Animação da Barra de Progresso
                let ScrollArea = $(document).scrollTop();
                let PageHeight = $(document).height() - $(window).height();
                var Progress = 100 * ScrollArea / PageHeight;
                // Atribui os valores da página à barra de progresso
                $('.PROGRESS').css('width', Progress + '%');
                // Animação de Imersão
                if ($(window).scrollTop() > 1) {
                    $('.ONE').addClass('FADE@2PUV');
                    $('.RT02').removeClass('FADE@33VO');
                    $('.RT03').addClass('FADE@KR4G');
                    $('.CT02').removeClass('FADE@3IJM');
                } else {
                    $('.ONE').removeClass('FADE@2PUV');
                    $('.RT02').addClass('FADE@33VO');
                    $('.RT03').removeClass('FADE@KR4G');
                    $('.CT02').addClass('FADE@3IJM');
                }
                // Animação de Indexação dos Menus Secundários
                let Scroll = $(window).scrollTop() + $(window).height();
                let Selector = $('.REVEAL').offset().top;
                // Seleciona o valor ao chegar no fim da página
                if (Scroll > Selector) {
                    let MenuItems = [
                                      '.MAKEMEVISIBLEONE',
                                      '.MAKEMEVISIBLETWO',
                                      '.MAKEMEVISIBLETHREE'
                                    ];
                    for (let i = 0; i < MenuItems.length; i++) {
                        setTimeout(function(){
                            $(MenuItems[i]).addClass('INNEROBJECTVISIBLE');
                        }, 50 * i);
                    }
                }
            });
        });
        // Função de Áudio
        function PlayAudio(AudioElement, Volume){
            AudioElement.pause();
            AudioElement.currentTime = 0;
            AudioElement.volume = Volume;
            AudioElement.play();
        }
        $('.CLICK').click(function(){
            let AudioElement = $('.CKSOUND')[0];
            PlayAudio(AudioElement, 0.25);
        });
    </script>
