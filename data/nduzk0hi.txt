#include <bits/stdc++.h>
int pickCoupons(int n, vector<int> &coupons)
{
    int l = 0;
    unordered_map<int, int> mp;
    int ans = INT_MAX;
    for (int r = 0; r < n; r++) {
        mp[coupons[r]]++;
        while (mp[coupons[r]] > 1) {
            ans = min(ans, r - l + 1);
            mp[coupons[l]]--;
            l++;
        }
    }
    return (ans == INT_MAX ? -1 : ans);
}
