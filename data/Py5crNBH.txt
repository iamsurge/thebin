namespace Exam
{
	using System;

	class Program
	{
		static void Main(string[] args)
		{
			int n = int.Parse(Console.ReadLine());

			int mod2 = 0;
			int mod3 = 0;
			int mod4 = 0;

			for (int i = 0; i < n; i++)
			{
				int input = int.Parse(Console.ReadLine());

				if (input % 2 == 0)
				{
					mod2++;

					if (input % 4 == 0)
					{
						mod4++;
					}
				}

				if (input % 3 == 0)
				{
					mod3++;
				}
			}

			Console.WriteLine($"{1.0 * mod2 * 100 / n}:F2");
			Console.WriteLine($"{1.0 * mod3 * 100 / n}:F2");
			Console.WriteLine($"{1.0 * mod4 * 100 / n}:F2");
		}
	}
}
