local w, h = term.getSize()

function okno(nazwa, a, b, x, y)
tc()
a = tonumber(a)
b = tonumber(b)
x = tonumber(x)
y = tonumber(y)
pfb(1,2,w,h-1,colors.black)
pfb(a,b,x,y,colors.blue)
scp(a,b)
stc(colors.white)
p(nazwa)
pdp(x,b,colors.red)
pdp(x-1,b,colors.orange)
pfb(6,h,w,h,colors.blue)
pfb(1,h,5,h,colors.white)
scp(1,h)
sbc(colors.white)
stc(colors.black)
write("Start")
pdp(w,h,colors.red)
pdp(w-1,h,colors.orange)
end

function system()
tc()
scp(1,1)
write("")
n = read()
write("")
n1 = read()
write("")
n2 = read()
write("")
n3 = read()
write("")
n4 = read()

okno(n, n1, n2, n3, n4)
s(5)
end

system()
