

#include  <stdio.h>
#include <stdlib.h>
 
int      arr[123];


///////////////////////////////////////////////////////////
struct TT
{
	
	TT            *p;
	//---------------
	char szName[20];
	
	int          old;
	
} TT_Head,   //  Элемент для служебного использования (нулевой элемент) 
    *Tail;   //             Указатель хранит адрес последнего элемента.




int TT_size = 0;


//////////////////////////////////////////////////////////
int add_El(int x)                                       //  
{
	
	TT *p = (TT*)malloc(sizeof(TT));

	p->old = x;
		
	if(TT_size == 0)
	{  
		
	   TT_Head.p = p;	
	   Tail	     = p;
	} else 
	{
	   Tail->p   = p;	
	   Tail	     = p; 	
	}
	
	TT_size ++;
}


//////////////////////////////////////////////////////////
void TT_Print()                                         //   
{

    TT *p = TT_Head.p;
	
	for(int i = 1; i <= TT_size; i++)
	{
			
		printf("TT[%d].old = %d \n", i, p->old);
		
		p = p->p;
	}	
	
}

////////////////////////////////////////////////////////////////////////////// 
int main()                                                                  // 
{
	

	
	for(int i = 1; i <= 7; i++)
	{
		
		
		add_El(i+100);
		
	}
	
	printf("TT_size = %d \n", TT_size);
	
	TT_Print();
}
