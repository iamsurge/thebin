#define DIR_ASC 0
#define DIR_DESC 1

void wstawianie(int* a, int n, bool print) {

    int sum_por = 0, por = 0;
    int sum_prz = 0, prz = 0;

    for (int i = 1; i < n; i++) {
        prz = 0;
        por = 0;
        int x = a[i];
        int j = i - 1;

        sum_por++;
        por++;
        while (j >= 0 && a[j] > x) {
            sum_prz++;
            prz++;
            a[j + 1] = a[j];
            --j;
            sum_por++;
            por++;
        }
        if (a[j + 1] != x) {
            sum_prz++;
            prz++;
        }
        a[j + 1] = x;
//        if (print)
//            std::cout << "i=" << i << ", por=" << por << ", prz=" << prz << std::endl;
    }
    if (print)
        std::cout << "sum_por=" << sum_por << ", sum_prz=" << sum_prz << std::endl;
}

void wybieranie(int* a, int n, bool print) {

    int sum_por = 0, por = 0;
    int sum_prz = 0, prz = 0;

    for (int i = 0; i < n - 1; i++) {
        prz = 0;
        por = 0;
        int k = i;
        for (int j = i + 1; j < n; j++) {
            sum_por++;
            por++;
            if (a[j] < a[k]) {
                k = j;
            }
        }
        if (a[k] != a[i]) {
            sum_prz++;
            prz++;
        }
        int x = a[k];
        a[k] = a[i];
        a[i] = x;

//        if (print)
//            std::cout << "i=" << i << ", por=" << por << ", prz=" << prz << std::endl;
    }
    if (print)
        std::cout << "sum_por=" << sum_por << ", sum_prz=" << sum_prz << std::endl;
}

void bubble(int* a, int n, int dir, bool print) {

    int sum_por = 0, por = 0;
    int sum_prz = 0, prz = 0;

    if (dir == DIR_ASC) {
        for (int i = 1; i < n; i++) {
            prz = 0;
            por = 0;
            for (int j = n - 1; j >= i; j--) {
                sum_por++;
                por++;
                if (a[j - 1] > a[j]) {
                    if (a[j - 1] != a[j]) {
                        sum_prz++;
                        prz++;
                    }
                    int x = a[j - 1];
                    a[j - 1] = a[j];
                    a[j] = x;
                }
            }
//            if (print)
//                std::cout << "i=" << i << ", por=" << por << ", prz=" << prz << std::endl;
        }
        if (print)
            std::cout << "sum_por=" << sum_por << ", sum_prz=" << sum_prz << std::endl;
    }
    else {
        for (int i = 1; i < n; i++) {
            for (int j = n - 1; j >= i; j--) {
                if (a[j - 1] < a[j]) {
                    int x = a[j - 1];
                    a[j - 1] = a[j];
                    a[j] = x;
                }
            }
        }
    }
}
