var thing = document.getElementsByClassName('inputArea');
for (let i = 0; i < thing.length; i++) {
    for (let textarea of thing[i].querySelectorAll('[id^="input"]').values()) {
        for (let answer of thing[i].querySelectorAll('.solvebtn').values()) {
            textarea.value = answer.getAttribute('data-content');
        }  
    }  
}
