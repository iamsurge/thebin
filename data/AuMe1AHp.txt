#Export Database From Mac Shell Script Save "ANY_NAME.sh"
# Set Permission "sudo chmod 755"
#!/usr/bin/env bash

USERNAME=$(id -un)
DATE=$(date +"%Y_%m_%d_%H_%M_%S")
/Applications/xampp/xamppfiles/bin/mysqldump -u root -p --databases EXPORT_DATABASE_NAME > /Users/$USERNAME/Downloads/export_db_$DATE.sql
