def process_table_data(table_data):
    lines = table_data.splitlines()
    test_spaces_amount = lines[0].count(' ')
    for line in lines:
        if 'x' in line:
            previous_x_line = line
        else:
           line = previous_x_line[:previous_x_line.index('x')+2] + line
        print(line)
        current_spaces_count = line.count(' ')

        if current_spaces_count != test_spaces_amount:
            raise Exception(f'Line: {line} have {current_spaces_count} spaces. First line had {test_spaces_amount} spaces.')
    
        

table_data = """165.1 x 5.4 CHS 21.3 0.519 24.4 30.6 2710 8.65 105 138 56.5 17.3 209 1.00 C 138
5.0 CHS 19.7 0.519 26.3 33.0 2510 8.07 97.7 128 56.6 16.1 195 1.00 C 128"""
process_table_data(table_data)
