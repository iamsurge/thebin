                    {

                    "value": 5, "left": None, "right": None

def get_leafes(tree):
    if tree == None:
        return

    get_leafes(tree["left"])
    if tree['left'] == None and tree['right'] == None:
        print(tree['value'])
    get_leafes(tree["right"])
