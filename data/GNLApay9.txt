using System;

namespace Sum
{
	internal class Program
	{
		static void Main(string[] args)
		{
			for (int i = 0; i < 25; i++)
			{
				Console.Write(100 - 2 * i + ", ");
				Console.Write(2 * i + 1 + ", ");
				Console.Write(2 * i + 2 + ", ");
				Console.Write(100 - ((2 * i) + 1) + ", ");
			}
		}
	}
}
