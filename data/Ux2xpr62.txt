#include <stdio.h>
int N;

void interchange(int M[][N], int A[], int L) {
    for (int i = 0; i < N; i++) {
        M[L][i] = A[i];
    }
}

int main()
{
    printf("Type in the size of the matrix: ");
    scanf("%d", &N);
    
    printf("\nType in the elements of the matrix (%d elements): ", N*N);
    int i, j, M[N][N], A[N], L;
    for (i = 0; i < N; i++) {
        for (j = 0; j < N; j++) {
            scanf("%d", &M[i][j]);
        }
    }
    
    printf("\nType in the elements of the vector (%d elements): ", N);
    for (j = 0; j < N; j++) {
        scanf("%d", &A[j]);
    }
    
    printf("\nType in the line you want to change (counting starts from 0): ");
    scanf("%d", &L);
    
    interchange(M, A, L);

    printf("\nThe elements of the matrix after the change:\n");
    for (i = 0; i < N; i++) {
        for (j = 0; j < N; j++) {
            printf("%d ", M[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}
