using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Xml.Schema;

namespace MetodaTangenteiMod
{

    class program
    {   //x0 = 2, d0 = 11, eps = 10^(-4, -8,-12) ;ex 1
        public static double x0, d0, eps, xn;
        public static List<double> x = new List<double>();
        public static int exemplu = 1;
        

        public static double f(double x)
        {
            return x *x * x - x - 1;
        }

        static void Main(string[] args)
        {
           
            Console.Write("x0 = ");
            x0 = double.Parse(Console.ReadLine(), System.Globalization.CultureInfo.InvariantCulture);
            Console.Write("d0 = ");
            d0 = double.Parse(Console.ReadLine(), System.Globalization.CultureInfo.InvariantCulture);
            Console.Write("eps = ");
            eps = double.Parse(Console.ReadLine(), System.Globalization.CultureInfo.InvariantCulture);
            Console.Write("exemplu (1, 2, 3, 4, 5) = ");
            exemplu = int.Parse(Console.ReadLine());
            x.Add(x0);
            x.Add(x0 - (f(x0) / d0));
            while (Math.Abs(x[x.Count - 1] - x[x.Count - 2]) >= eps)
            {
                xn = x[x.Count - 1];
                x.Add(xn - (f(xn) / d0));
            }

            Console.Write($"n = {x.Count - 1}, xn = {x[x.Count - 1]}");
            Console.ReadKey();
        }
    }
}
