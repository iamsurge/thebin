package com.example.four;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.content.SharedPreferences;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

import com.android.volley.Request;
import com.android.volley.RequestQueue;
import com.android.volley.Response;
import com.android.volley.VolleyError;
import com.android.volley.toolbox.JsonObjectRequest;
import com.android.volley.toolbox.Volley;

import org.json.JSONException;
import org.json.JSONObject;

public class MainActivity extends AppCompatActivity {
    EditText et;
    Button button;
    String city;
    TextView tv;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        button = findViewById(R.id.weatherCheck);
        et = findViewById(R.id.cityInput);
        tv = findViewById(R.id.incorrectCity);
        loadData();

        if(tv.getVisibility() == View.VISIBLE) {
            tv.setVisibility(View.INVISIBLE);
        }


        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                tv.setVisibility(View.INVISIBLE);
                city = et.getText().toString();
                saveData();
                changeActivity();
            }
        });
    }
    protected void saveData() {
        SharedPreferences sharedPreferences = getSharedPreferences("shared preferences", MODE_PRIVATE);
        SharedPreferences.Editor editor = sharedPreferences.edit();
        editor.putString("city", et.getText().toString());
        editor.apply();
    }

    protected void loadData() {
        SharedPreferences sharedPreferences = getSharedPreferences("shared preferences", MODE_PRIVATE);
        String getCity = sharedPreferences.getString("city", "");
        et.setText(getCity);
    }

    public void changeActivity() {
        RequestQueue queue = Volley.newRequestQueue(getApplicationContext());
        String url = "https://api.openweathermap.org/data/2.5/weather?q="+city+"&units=metric&APPID=749561a315b14523a8f5f1ef95e45864";
        JsonObjectRequest request = new JsonObjectRequest(Request.Method.GET, url,null, new Response.Listener<JSONObject>() {

            @Override
            public void onResponse(JSONObject response) {
                try {
                    JSONObject array= (JSONObject) response.get("main");
                    String temp = array.getString("temp");
                    String pressure = array.getString("pressure");
                    String humidity = array.getString("humidity");
                    String tempMin = array.getString("temp_min");
                    String tempMax = array.getString("temp_max");
                    WeatherActivity(temp, pressure, humidity, tempMin, tempMax);
                } catch (JSONException e) {
                    tv.setVisibility(View.VISIBLE);
                }
            }
        }, new Response.ErrorListener() {
            @Override
            public void onErrorResponse(VolleyError error) {
                tv.setVisibility(View.VISIBLE);
            }
        });
        queue.add(request);
    }

    public void WeatherActivity(String temp, String pressure, String humidity, String tempMin, String tempMax){
        Intent intent = new Intent(this, WeatherActivity.class);
        intent.putExtra("city", city);
        intent.putExtra("temp", temp);
        intent.putExtra("pressure", pressure);
        intent.putExtra("humidity", humidity);
        intent.putExtra("temp_max", tempMax);
        intent.putExtra("temp_min", tempMin);
        startActivity(intent);
    }
}
