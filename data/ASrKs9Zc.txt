Program Final;

const
	MaxF = 10;
	MaxC = 10;

type tipoMatriz = array [1..MaxF; 1..MaxC] of integer;


//Dada una matriz, una fila y dos columnas, intercambia las celdas.
Procedure Intercarcambiar(var Matriz:tipoMatriz; fila,col1,col2:integer);
Var
	aux:integer;
Begin
	aux:=Matriz(fila,col1);
	Matriz(fila,col1):=Matriz(fila,col2);
	Matriz(fila,col1):=aux;
End;


//Dada una matriz y una columna, la coloca al principio, desplazando las demÃ¡s.
Procedure DesplazarAlInicio(var Matriz:tipoMatriz; columna:integer);
Var
	col,fil:integer;
Begin
	if (columna > 1) then
		for col:=columna downto 2 do //Corta en la segunda columna para intercambiar con la primera
			for fil:=1 to MaxF do				
				Intercarcambiar(Matriz,fil,col,(col-1));
End;


//Remplaza una columna por ceros
Procedure ReemplazarConCeros(var Matriz; columna:integer);
Var
	fil:integer;
Begin
	//Camino del for
	for fil:=1 to MaxF do
		Matriz(fil,columna):=0;
		
	//Camino del while
{	fil:=1;
	while (fil<=MaxF) do
		begin
			Matriz(fil,columna):=0;
			fil:=fil+1;
		end;
}
End;


//Dada una columna, devuelve la suma de sus elementos
function Sumatoria(Matriz:tipoMatriz; columna:integer):integer;
Var
	fil,suma:integer;
Begin
	suma:=0;
	for fil:=1 to MaxF do
		suma:=suma+Matriz(fil,columna);
	Sumatoria:=suma;
End;


//Dada una matriz, devuelve la columna cuya suma de elementos sea la mas alta
function ColumnaDeMayorSuma(Matriz:tipoMatriz):integer;
Var
	MaxEncontrado,colMax,suma,col:integer
Begin
	MaxEnontrado:=Sumatoria(Matriz,1); //Asumo que el mas grande es el primero;
	colMax:=1;	
	for col:=1 to MaxC do
		begin
			suma:=Sumatoria(Matriz,col);
			if (suma>MaxEncontrado) then
				begin
					MaxEncontrado:=suma;
					colMax:=col;
				end;
		end;
End;




VAR
	MATENT : tipoMatriz;
	columnaMax:integer;
BEGIN
	CargarMatriz(MATENT);
	columnaMax:=ColumnaDeMayorSuma(MATENT);
	ReemplazarConCeros(MATENT,columnaMax);
	DesplazarAlInicio(MATENT,columnaMax);
	MostrarMatriz(MATENT);
END.
