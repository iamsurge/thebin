# Delete Partitions on Android with Parted and TWRP

#get parted
https://archive.org/details/parted_20221204


adb push /path/to/parted /sbin/parted
adb shell
chmod 777 /sbin/parted
parted /dev/block/mmcblk0

#view all partitions - might want to keep a log of this
p free

#delete partition with rm command in parted

rm 42
rm 41
rm 40
rm 39
