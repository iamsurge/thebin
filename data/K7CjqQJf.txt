local pos = 0

function main()
  local timer = os.time()
  local difference = math.fmod(timer, 500)
  print('time ' .. timer)
  print('diff ' .. difference)
  os.sleep(difference * 0.05)
 
  while true do
    move()
	os.sleep(25)
  end
end

function move()
	while pos < 16 do
	  turtle.forward()
	  pos = pos + 1
	end
	
	turtle.up()
	turtle.up()

	while pos > 0 do
	  turtle.back()
	  pos = pos - 1
	end
	
	turtle.down()
	turtle.down()
end

main()
