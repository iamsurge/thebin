n = int(input())
s = 0
min_not_multiple_4 = 10001

for i in range(n):
    a = int(input())
    s += a
    if a % 4 != 0:
        min_not_multiple_4 = min(min_not_multiple_4, a)

if s % 4 != 0:
    print(n, s)
else:
    if min_not_multiple_4 == 10001:
        print(0)
    else:
        print(n - 1, s - min_not_multiple_4)
