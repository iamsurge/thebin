#!/usr/bin/env python
# -*- coding: utf-8 -*-

# Programa con menu

# Script que calcula superficies de figuras geomètricas

print("Programa que calcula superficies de figuras geométricas")
while True:
    print("""
    Menu de opciones:
    -----------------
    1. Círculo
    2. Cuadrado
    3. Rectángulo
    4. Triángulo
    5. Salir
    """)
    
    opcion = input("Seleccione una opción: ")
    
    if opcion == "1":
        radio = float(input("Ingrese el radio: "))
        print(f"Area: {3.1416 * radio ** 2:.2f}")
        
    elif opcion == "2":
        lado = float(input("Ingrese el lado: "))
        print(f"Area: {lado ** 2:.2f}")
        
    elif opcion == "3":
        lado = float(input("Ingrese el lado menor: "))
        Lado = float(input("Ingrese el lado mayor: "))
        print(f"Area: {lado * Lado:.2f}")
        
    elif opcion == "4":
        base = float(input("Ingrese la base: "))
        altura = float(input("Ingrese la altura: "))
        print(f"Area: {base * altura / 2:.2f}")
        
    elif opcion == "5":
        print("Gracias por usar este programa...")
        break
        
    else:
        print("Opción incorrecta")
