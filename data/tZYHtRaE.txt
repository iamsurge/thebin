#include<bits/stdc++.h>
using namespace std;

vector<int> sieve()
{
    vector<int> sv(4000,0);
    for(int i=2;i<4000;i++)
    {
        if(sv[i]==0){
        for(int j=i;j<4000;j+=i) 
        if(sv[j]==0) sv[j]=i;
        }
    }
    return sv;
}
int main()
{
    vector<int> sv=sieve();
    int x,n,ans=0;
    cin>>n;
    for(int i=2;i<=n;i++){
        x=i;
        set<int> st;
        while(x>1){
            st.insert(sv[x]);
            x/=sv[x];
        }
        if(st.size()==2) ans++;
    }
    cout<<ans<<"\n";
    return 0;
}
