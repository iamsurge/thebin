using System;

namespace Stream_Of_Letters
{
	class StreamOfLetters
	{
		static void Main(string[] args)
		{

			bool isLoopExit, isC, isO, isN;

			isLoopExit = isC = isO = isN = false;;
			string resultString = "";
			
			while (!isLoopExit)
			{
				string s = Console.ReadLine();

				switch (s)
				{
					case "End":
						isLoopExit = true;
						break;
					default:
						{
							if ((s == "c") && (!isC))
							{
								isC = true;								
							}
							else if ((s == "o") && (!isO))
							{
								isO = true;
							}
							else if ((s == "n") && (!isN))
							{
								isN = true;
							}
							else
							{
								char key = char.Parse(s);
								if (char.IsLetter(key))
									resultString =$"{resultString}{s}";
							}
							if (isC && isO && isN)
							{								
								Console.Write($"{resultString} ");								
								resultString =string.Empty;
								isC = isO = isN = false;
							}
						}
						break;
				}//switch (s)
			}//while (!inLoopExit)
			
		}
	}
}
