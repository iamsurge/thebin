// Программа 3
#include <iostream>

int main()
{
    int nums_watermelons; // количество арбузов
    double total_weight;  // общий вес арбузов (в килограммах)

    std::cout << "Введите количество арбузов: ";
    std::cin >> nums_watermelons;
    std::cout << "Введите общий вес арбузов (в килограммах): ";
    std::cin >> total_weight;

    // Рассчитаем средний вес одного арбуза
    double avg_weight = total_weight / nums_watermelons;
    std::cout << "Средний вес одного арбуза: " << avg_weight << std::endl;

    return 0;
}
