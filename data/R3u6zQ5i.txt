#!/bin/bash

video="https://www.youtube.com/watch?v=Nl6l63AYg7s"

adb shell wm size || exit 1
rez="$(adb shell wm size|awk '{print $3}')"
x="$(echo $rez|cut -d\x -f1)"
y="$(echo $rez|cut -d\x -f2)"

mkdir bootani || exit 1
cd bootani
mkdir part0
cd part0
youtube-dl "https://www.youtube.com/watch?v=Nl6l63AYg7s" -o "video.%(ext)s"
ffmpeg -i video.* Bootup_%05d.jpg
rm video.*

mogrify -rotate "90" -resize $x -background black -gravity center -extent $rez *.jpg
