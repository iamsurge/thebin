/*******************************************************************************
 * Scrivere un programma che usando l'iterazione stampi la tavola pitagorica.  *
 *                                                  Simone Flavio P.           *
 ******************************************************************************/
#include <stdio.h>

int main()
{
    int row = 1;
    int col = 1;
    
    while(row<=10){
        while(col<=10){
            printf("%d\t",row*col);
            col++;
        }
        col = 1;
        printf("\n");
        row++;
    }
    return 0;
}
