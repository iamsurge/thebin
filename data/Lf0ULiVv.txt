{#
    get the correct field, but this time from the current_user_id
    we use 'tb.get_current_user_id()', tb is a Toolbox proxy class so we can
    call php functions more elegantly.
    ~ is the symbol for concatenation in Twig (whereas it's . in PHP)
    
#}
{% set group = tb.get_field( 'user' , 'user_' ~ tb.get_current_user_id() ) %}
{#
    check if the group field has any length at all (if not, no point in trying to display a logo)    
#}
{% if group|length %}
    {% set image = TimberImage( group.imageuser ) %}
    <img src="{{ image.src('large') }}" width="100%" alt="{{ image.alt }}" aria-label="{{ image.title }}" />
{% else %}
    <h5>There is no field value or user not logged in</h5>
{% endif %}
