public void onStart() {
        if(session.getProduct() != null){
            getView().setProductList(session.getProduct());
        } else {
            session.setProduct(getProduct());
        }
    }

    List<PPOBProductResponse> getProduct(){
        getView().showLoading(true);
        dataPackageProductInteractor.execute(new Interactor.Listener<List<PPOBProductResponse>>() {
            @Override
            public void onResponse(List<PPOBProductResponse> response) {
                session.setProduct(response);
                getView().setProductList(response);
                getView().showLoading(false);

            }

            @Override
            public void onError(String message) {
                getView().showLoading(false);

            }
        });
        return session.getProduct();
    }

void getProductNominal(String prodId){
        getView().showLoading(true);

        dataPackageProductNominalInteractor.execute(prodId, new Interactor.Listener<List<NominalResponse>>() {
            @Override
            public void onResponse(List<NominalResponse> response) {
                session.setNominal(response);
                getView().setProductNominalList(response);
                getView().showLoading(false);
            }

            @Override
            public void onError(String message) {
                getView().showLoading(false);
            }
        });
    }
