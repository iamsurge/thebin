public class CSharpLearning
{
    static void Main()
    {
        Random rand = new Random();
        int N = 10;
        int dotCount = 30;
        int[,] matrix = new int[N, N];

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                matrix[i, j] = rand.Next(10, 99);
            }
        }

        Console.WriteLine("Матрица\n" + String.Concat(Enumerable.Repeat(".", dotCount)));
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                Console.Write(matrix[i, j] + " ");
            }
            Console.WriteLine();
        }
        Console.WriteLine(String.Concat(Enumerable.Repeat(".", dotCount)));


        for (int k = 0; k < 5; k++)
        {
            Console.WriteLine("Какую диагональ матрицы(относительно левого угла) вы хотите получить: ");
            int numberOfDiagonal; int.TryParse(Console.ReadLine(), out numberOfDiagonal);
            for (int i = numberOfDiagonal, j = 0; i < N; i++, j++)
            {
                Console.Write(matrix[i, j] + " ");
            }
            Console.WriteLine("\n");
        }

        Console.WriteLine(String.Concat(Enumerable.Repeat(".", dotCount)));
        Console.WriteLine("\n");
        for (int k = 0; k < 5; k++)
        {
            Console.WriteLine("Какую диагональ матрицы(относительно правого угла) вы хотите получить: ");
            int numberOfDiagonal; int.TryParse(Console.ReadLine(), out numberOfDiagonal);
            for (int i = 0, j = numberOfDiagonal; j < N; i++, j++)
            {
                Console.Write(matrix[i, j] + " ");
            }
            Console.WriteLine("\n");
        }
    }
}
