#include <iostream>
#include <limits>

using namespace std;


int main()
{
	for (int i = 3; i >= -3; i--)
	{
		if (i >= 0)
			cout << numeric_limits<int>::min() << "+" << i << " = " << numeric_limits<int>::min() + i << endl;
		else if(i < 0)
			cout << numeric_limits<int>::min() << i << " = " << numeric_limits<int>::min() - i << endl;
	}

	return 0;
}
