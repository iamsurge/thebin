#include <iostream>
using namespace std;
 
void quickSort(int * arr, int from, int to) {
    //zapochvash s (arr,0,sz-1);
    int i, j, etalon;
    i = from, j = to;
    etalon = arr[(from + to) / 2];
    do {
        while (arr[i] < etalon) {
            i++;
        }
        while (arr[j] > etalon) {
            j--;
        }
        if (i <= j)
        {
            swap(arr[i], arr[j]);
            i++;
            j--;
        }
    } while (i <= j);
 
    if (from < to) {
        quickSort(arr, from, j);
        quickSort(arr, i, to);
    }
}
 
void Print(int * arr, int sz) {
    for (int i = 0; i < 14; i++) { cout << arr[i] << " "; }
    cout << endl;
}
int main() {
    int arr[14] = { 10,4,7,70,2,5,0,25,11,1,30,3,5,24 };
    int sz = sizeof(arr)/sizeof(arr[0]);
    Print(arr,sz);
    quickSort(arr, 0, sz - 1);
    Print(arr);
    return 0;
}
