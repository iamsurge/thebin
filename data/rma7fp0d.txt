#include <stdio.h>

int main(int argc, char const *argv[])
{
    unsigned long long T, i, A, B, N, hasil, j = 2;
    unsigned long long M = 1000000007;
    scanf("%llu", &T);

    for (i = 0; i < T; i++)
    {
        scanf("%llu %llu", &A, &N);
        hasil = 1;
        while (N > 0)
        {
            B = A;
            while (j <= B)
            {
                if (B % j == 0)
                {
                    hasil = (hasil * j) % M;
                    B /= j;
                }
                else
                {
                    j++;
                }
            }
            N--;
        }
        printf("Case #%d: %llu\n", i + 1, hasil);
    }

    return 0;
}
