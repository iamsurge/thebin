#include<stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
	int n, flag = 0;
	printf("n =");
	scanf("%d", &n);				// вводим порядок матрицы
	int **a = (int **)malloc(n*sizeof(int *));   // задаём матрицы порядка n
		for(int i = 0; i < n; i++) {
    		a[i] = (int *)malloc(n*sizeof(int));
		}
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < n; j++) {
			scanf("%d", &a[i][j]);
		}
	} 
	for(int i = 0; i < n; i++) {		// вывод матрицы
		for(int j = 0; j < n; j++) {
			printf("%d ", a[i][j]);
		}
		printf("\n");
	}
	for(int i = 0; i < n-1 && flag == 0; i++) {    // проверка условия симметрия относительно побочной диагонали
		for(int j = 0;j < n-1 && flag == 0; j++) {
			if(a[i][j] != a[n-(j+1)][n-(i+1)]) {
				flag = 1;
			}
		}
	}
	
	if(flag == 0) {    // вывод ответа
		printf("YES");
	}
	else {
		printf("NO");
	}
	
	for(int i = 0; i < n; i++) {  // очищаем память
    	free(a[i]);
	}
	free(a);
	
}
