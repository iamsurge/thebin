#include <iostream>

using namespace std;


int main() {
    int n;
    cin >> n;
    
    int niza[n];
    
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
    }
    
    for(int i = 0; i < n; i++) {
        if(niza[i] % 2 == 0) {
            cout << niza[i] << " ";
        }
    }
    for(int i = 0; i < n; i++) {
        if(niza[i] % 2 != 0) {
            cout << niza[i] << " ";
        }
    }
    
    return 0;
}
