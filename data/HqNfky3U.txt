#include <bits/stdc++.h>
#define int long long
using namespace std;

vector<int> a;

int calc() {
    int sum = 0;
    for (size_t i = 0; i < a.size() - 1; ++i) {
        if (a[i] < a[i + 1]) {
            sum += a[i];
        }
        else {
            sum += a[i + 1];
            ++i;
        }
    }
    return sum;
}

int32_t main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        int ai;
        cin >> ai;
        a.push_back(ai);
    }
    cout << calc() << '\n';
    

    return 0;
}
