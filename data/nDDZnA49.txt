class AlarmClock {
    
    static String alarmClock4(int day, boolean vacation) {
        if (day >= 1 && day <= 5 && !vacation) {
            return "7:00";
        }
        if (day >= 1 && day <= 5 && vacation) {
            return "10:00";
        }
        if ((day == 0 || day == 6) && !vacation) {
            return "10:00";
        }
        return "off";
    }

    
    
    
    
    
    
    
    static String alarmClock(int day, boolean vacation) {
        String result;
        if (day >= 1 && day <= 5 && !vacation) {
            result = "7:00";
        } else if (day >= 1 && day <= 5 && vacation) {
            result = "10:00";
        } else if ((day == 0 || day == 6) && !vacation) {
            result = "10:00";
        } else {
            result = "off";
        }
        return result;
    }
    
    static String alarmClock2(int day, boolean vacation) {
        if (day >= 1 && day <= 5) {
            if (vacation) {
                return "10:00";
            } else {
                return "7:00";
            }
        } else {
            if (vacation) {
                return "off";
            } else {
                return "10:00";
            }
        }
    }

    static String alarmClock1(int day, boolean vacation) {
        return isWeekDay(day) ? (vacation ? "10:00" : "7:00") : (vacation ? "off" : "10:00");
    }
    
    static boolean isWeekDay(int day) {
        return day >= 1 && day <= 5;
    }
    
    public static void main(String[] args) {
        System.out.println(alarmClock(1, false));
    }

}
