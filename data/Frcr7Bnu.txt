function solve(input) {
    let butget = Number(input.shift());
    let videoCard = Number(250);
    let videoCardPrice = Number(input[0]) * videoCard;
    let procesorPrice = videoCardPrice - (videoCardPrice * 0.65);
    let priceProcesor = Number(input[1]) * procesorPrice;
    let ramPrice = videoCardPrice - (videoCardPrice * 0.90);
    let priceRam = Number(input[2]) * ramPrice;
    let total = priceRam + priceProcesor + videoCardPrice;

    let video = Number(input.shift());
    let procesor = Number(input.shift());


    if (video >= procesor) {
        let sum = total * 0.85;

        if (sum <= butget) {
            let sum1 = butget - sum
            console.log(`You have ${sum1.toFixed(2)} leva left!`);
        } else {
            console.log(`Not enough money! You need ${(sum - butget).toFixed(2)} leva more!`);
        }


    }
}
