# 2.1
# Definición de clase.
class Auto:
    def __init__(self, com, an, ca, pa, pe, pr):
        self.combustible = com
        self.anno_de_fabricacion = an
        self.cantidad_pasajeros = ca 
        self.patente = pa
        self.peso = pe
        self.precio = pr

#2.2 
# Ingreso de datos por terminal. ## PROF: no usamos la palabra terminal ##
autos = []
contador = 1
saltar = False
while True:
    print("Ingrese número de patente (ingrese 0 para salir).")
    patente = int(input(f"Patente auto {contador}> ")) ## PROF:no usamos input en la formainput(f" ##
    if patente == 0:
        break ## NO USAMOS BREAK ##
    for a in autos: # 2.5
        if patente == a.patente:
            print("Error: No puede ingresar dos autos con la misma patente.")
            break ## NO USAMOS BREAK ##
    if saltar:
        continue
    print("Ingrese tipo de combustible (bencina, petróleo, gas, electricidad).")
    combustible = input(f"Tipo de combustible auto {contador}> ") ## PROF: no usamos input en la forma input(f" ##
    if combustible.lower() not in ["bencina", "petróleo", "gas", "electricidad"]:
        print("Error: El tipo de combustible tiene que serer uno de los mencionados")
        continue
    anno = int(input(f"Año de fabricación auto {contador}> "))
    cant = int(input(f"Cantidad de pasajeros auto {contador}> "))
    peso = int(input(f"Ingrese peso auto {contador} en Kilogramos> "))
    precio = int(input(f"Ingrese precio auto {contador}> "))

    autos.append(Auto(combustible, anno, cant, patente, peso, precio))
    contador += 1

# 2.3
# Ordenar por peso (menor a mayor). ## PROF: en clases ordenamos con sort(), no de esta forma ##
for i in range(0, len(autos)):
    for j in range(0, len(autos)-i-1):
        if (autos[j].peso > autos[j+1].peso):
            temp = autos[j]
            autos[j] = autos[j+1]
            autos[j+1] = temp

# Mostrar a pantalla
for a in autos: ## PROF: no usamos print en la forma print(f" ##
    print(f"Peso: {a.peso},")
    print(f"\tTipo combustible {a.combustible},")
    print(f"\tAño de fabricacion: {a.anno_de_fabricacion},")
    print(f"\tCantidad de pasajeros {a.cantidad_pasajeros},")
    print(f"\tPatenete {a.patente},")
    print(f"\tPrecio {a.patente}.")

# 2.4 
# Precio promedio por año usando Diccionarios. ## PROF: en clases no pasamos diccionarios ##
annos_dict = {}
for a in autos:
    if a.anno_de_fabricacion not in annos_dict.keys():
        annos_dict.update({a.anno_de_fabricacion : []})

for a in autos:
    af = a.anno_de_fabricacion
    va = annos_dict.get(af)
    va.append(a.precio)
    annos_dict.update({af : va})

for a in annos_dict.items():
    promedio = sum(a[1])/len(a[1])
    print(f"El precio promedio del año {a[0]} es: {promedio}.")
