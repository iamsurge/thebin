#! /usr/bin/env python3

def multiple(number, factor):
    if number == 0:
        return factor
    if factor == 0:
        return 0
    if number == factor:
        return number
    if number < factor:
        return factor
    if number > factor:
        return (number + factor - 1) // factor * factor
