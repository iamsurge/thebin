#include<bits/stdc++.h>
using namespace std;


int main() {
    int a[] = {1, 5, 10, 12, 8, 16};

    int prefixsum[6];

    for(int i=0; i<6; i++) {
        if(i==0) prefixsum[i] = a[i];
        else prefixsum[i] = prefixsum[i-1] + a[i];
    }

    int q, l,r ;
    cin >> q;
    while(q--) {
        cin >> l >> r;
        if(l==0) cout << prefixsum[r] << endl;
        else cout << prefixsum[r] - prefixsum[l-1] << endl;
    }
    return 0;
}
