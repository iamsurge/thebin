//Въведете цели числа от клавиатурата. Изведете четните числа, подредени в нарастващ ред. Принтирайте на конзолата по два 
//начина.

public class Program
    {
        static void Main(string[] args)
        {
            
            List<int> numbers = 
                Console.ReadLine()
                .Split(" ")
                .Select(int.Parse)
                .Where(n => n%2==0)
                .OrderBy(n => n)
                .ToList();

            foreach (int num in numbers)
            {
                Console.WriteLine(num);
            }

            Console.WriteLine(String.Join(", ", numbers));

        }
    }
