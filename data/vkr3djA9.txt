void selection_sort(int* arr, int sz)
{
    for (int i = 0; i < sz - 1; i++)
    {
        int mini, index;
        mini = arr[i];
        index = i;
        for (int j = i + 1; j < sz; j++)
        {
            if (arr[j] < mini)
            {
                mini = arr[j];
                index = j;
            }
        }
        arr[index] = arr[i];
        arr[i] = mini;
    }
}
 
void bubble_sort(int* arr, int sz)
{
    int tmp;
    bool ima = true;
    int i = 0;
    while (ima)
    {
        ima = false;
        for (int j = 0; j < sz - (i + 1); j++)
        {
            if (arr[j] > arr[j + 1])
            {
                tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
                ima = true;
            }
        }
        i++;
    }
}
 
void insertion_sort(int* arr, int sz)
{
    int itemToInsert, j;
    for (int i = 1; i < sz; ++i)
    {
        j = i - 1;
        itemToInsert = arr[i];
 
        while (j >= 0 && arr[j] > itemToInsert)
        {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = itemToInsert;
    }
}
