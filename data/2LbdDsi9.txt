function digAll()
  turtle.dig()
  turtle.digDown()
  turtle.digUp()
end

function tunnel(length, width, height)
  
  -- set defaults if unset --
  if !length then
    length = 100
  end
  if !width then
    width = 3
  end
  if !height then
    height = 3
  end


  fuel = turtle.getFuelLevel()

  currentLength = 0
  while currentLength < length then
    digAll()
    turtle.forward()
  end

end
