package com.corsojava.esercizicorso;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.BeforeEach;

/**
 *
 * @author lucio
 */
public class CalculatorTest {

    private Calculator calculator;
    
    @BeforeEach
    public void setup() {
        this.calculator = new Calculator(10, 3);
    }
    
    @Test
    public void testSomma() {
        assertEquals(13, calculator.somma());
    }

    @Test
    public void testDifferenza() {
        assertEquals(7, calculator.differenza());
    }

    @Test
    public void testDivisioneIntera() {
        assertEquals(3, calculator.divisioneIntera());
    }

    @Test
    public void testDivisione() {
        assertEquals(3.33f, calculator.divisione(), 0.01f);
    }

    @Test
    public void testRestoDivisioneIntera() {
        assertEquals(1, calculator.restoDivisioneIntera());
    }

    @Test
    public void testMedia() {
        assertEquals(6.5f, calculator.media(), 0.1f);
    }

    @Test
    public void testMax() {
        assertEquals(10, calculator.max());
        Calculator calculator2 = new Calculator(3, 10);
        assertEquals(10, calculator2.max());
    }

    @Test
    public void testMin() {
        assertEquals(3, calculator.min());
        Calculator calculator2 = new Calculator(3, 10);
        assertEquals(3, calculator2.min());
    }
    
}
