const int MAXN = 1e15 + 5;
int MOD = 1e9 + 7;

int fastpow(int x, int y) {
    if (!y) return 1;
    if (y & 1) return ((x * fastpow(x, y - 1) % MOD) % MOD) % MOD;
    int ans = fastpow(x, y >>= 1);
    ans %= MOD;
    ans *= ans;
    ans %= MOD;
    return ans;
}
