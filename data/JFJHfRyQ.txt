private static void InArray()
    {
        var excelApp = new Application();
        var workbook = excelApp.Workbooks.Open(ExcelPath);
        var sheets = workbook.Sheets[1] as _Worksheet;
        var range = sheets?.UsedRange;

        if (range is null) return;

        var rowCount = range.Rows.Count; // values.GetLength(0)
        var colCount = range.Columns.Count; // values.GetLength(1)

        object[,] values = range.Value2; //row-column
        var names = new string[rowCount];
        
        for (var row = 1; row < rowCount; row++)
        {
            var result = new StringBuilder();
            for (var column = 1; column < colCount; column++)
            {
                if(column is 1 or 2 or 3 or 4 or 5 or 6) // if string
                {
                    var value = Convert.ToString(values[row, column]) ?? "";
                    result.Append(value + " ");
                }
                if (column is 7 or 8 or 9 or 10 or 11 or 12) // if byte[]
                {
                   //do stuff
                }
            }
            names[row-1] = result.ToString();
            row++;
        }
    }
