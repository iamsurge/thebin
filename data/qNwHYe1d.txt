function solve (arg){

        let rekortSec = Number(arg[0]);
        let raztoyniMetri = Number(arg[1]);
        let vremeSec = Number(arg[2]);
  
        let prepluva = raztoyniMetri * vremeSec;
  
        let evry15 = Math.floor(raztoyniMetri / 15) * 12.5;
        let totalSec = prepluva + evry15;
  
  
  
      if(totalSec < rekortSec){
          
          console.log(`Yes, he succeeded! The new world record is ${totalSec.toFixed(2)} seconds.`);
  
      } else {
          
          console.log(`No, he failed! He was ${(totalSec - rekortSec).toFixed(2)} seconds slower.`);
      }
  
  
  


    
}
