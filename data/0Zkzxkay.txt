#include <stdio.h>

int main(){
    int i,res = 0;
    char str[80],sym = "L";
    printf("Text:");
    gets(str);
    printf("%c\n",sym);
    for(i = 0;i<strlen(str);i++){
        printf("%c\n",str[i]);
        if(str[i] == sym){
            res = i + 1;
            break;
        }
    }
    printf("Result - %d",res);
}
