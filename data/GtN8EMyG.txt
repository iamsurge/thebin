#include <iostream>
#include <fstream>
#define N 1005
using namespace std;
ifstream fin("bac.txt");
/**
  sortare prin bule
*/
int main()
{
   int a[N],n,i,j,aux;
   cin>>n;
   for(int i=1;i<=n;i++)cin>>a[i];

   ///sortarea
   bool ord=0;
   m=n;
   while(ord==0)
   {
       ord=1;
       for(i=1;i<m;i++)
        if(a[i]>a[i+1])
       {
           aux=a[i];a[i]=a[i+1]; a[i+1]=aux;
           ord=0;
       }
       m--;
   }

  for(int i=1;i<=n;i++)
    cout<<a[i]<<" ";

  return 0;
}
