const ll mod = 1e9+7;

ll rec(ll i, ll j){
    if(!valid(i,j)) return 0;
    if(i == n-1 and j == n-1) return 1;
    if(dp[i][j] != -1) return dp[i][j];
    ll ans = 0;
    ans = (ans + rec(i+1,j))%mod;
    ans = (ans + rec(i,j+1))%mod;
    return dp[i][j] = ans%mod;
}
