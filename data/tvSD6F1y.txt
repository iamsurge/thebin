function FormatText($i){
  $i = str_replace("[b]","<b>",$i);
  $i = str_replace("[/b]","</b>",$i);
  $i = str_replace("[p]","<li>",$i);
  $i = str_replace("[/p]","</li>",$i);
  $i = str_replace("[tr]","<font style='color: #cc0000; font-weight: bold;'>",$i);
  $i = str_replace("[/tr]","</font>",$i);
  $i = str_replace("[tg]","<font style='color: gold; font-weight: bold;'>",$i);
  $i = str_replace("[/tg]","</font>",$i);
  $i = str_replace("[tc]","<font style='color: cyan; font-weight: bold;'>",$i);
  $i = str_replace("[/tc]","</font>",$i);
  $i = str_replace("[cs]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted cyan; width: 80%';'><legend style='text-align: left;'>[CHORUS]</legend><div class='song show-chords'>",$i);
  $i = str_replace("[/cs]","</div></fieldset>",$i);
  $i = str_replace("[pc]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted #00ffff; width: 80%';'><legend style='text-align: left;'>[PRE-CHORUS]</legend><div class='song show-chords'>",$i);
  $i = str_replace("[/pc]","</div></fieldset>",$i);
  $i = str_replace("[br]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted #ff0000; width: 80%'; '><legend style='text-align: left;'>[BRIDGE]</legend><div class='song show-chords'>",$i);
  $i = str_replace("[/br]","</div></fieldset>",$i);
  $i = str_replace("[r]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted magenta; width: 80%'; '><legend style='text-align: left;'>[REFRAIN]</legend><div class='song show-chords'>",$i);
  $i = str_replace("[/r]","</div></fieldset>",$i);
  $i = str_replace("[v]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted #999; width: 80%';'><legend style='text-align: left;'>[VERSE]</legend><div class='song show-chords'>",$i);
  $i = str_replace("[/v]","</div></fieldset>",$i);
  $i = str_replace("[s]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted #ffff00; width: 80%'; '><legend style='text-align: center;'>[SOLO]</legend><center><div class='song show-chords'>",$i);
  $i = str_replace("[/s]","</div></fieldset>",$i);
  $i = str_replace("[in]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted #00ff00; width: 80%'; '><legend style='text-align: center;'>INTRO</legend><center><div class='song show-chords'>",$i);
  $i = str_replace("[/in]","</div></fieldset>",$i);
  $i = str_replace("[otr]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted gold; width: 80%'; '><legend style='text-align: center;'>OUTRO</legend><div class='song show-chords'>",$i);
  $i = str_replace("[/otr]","</div></fieldset>",$i);
  $i = str_replace("[n]","<fieldset style='font-family: monospace; margin-left: 30px; font-size: {$_SESSION['lyric_font_size']}vw; text-align: left; border: 1px dotted yellow; width: 80%'; '><legend style='text-align: center;'>[NOTES]</legend><center>",$i);
  $i = str_replace("[/n]","</fieldset>",$i);
  $i = str_replace("[c]","<span class='chord'>",$i);
  $i = str_replace("[/c]","</span>",$i);
return nl2br($i);
}
