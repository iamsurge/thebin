int countXX_1(String str) {
  int counter = 0;
  int i = 0;
  while (i < str.length() - 1) {
    if (str.charAt(i) == 'x' && str.charAt(i+1) == 'x') {
      counter++;
    }
    i++;
  }
  return counter;
}

int countXX(String str) {
  int counter = 0;
  
  for(int i = 0; i < str.length() - 1; i++) {
    if (str.charAt(i) == 'x' && str.charAt(i+1) == 'x') {
      counter++;
    }
  }
  
  return counter;
}


int countXX_2(String str) {
  int counter = 0;
  int i = str.indexOf("xx");
  while (i != -1) {
    counter++;
    i = str.indexOf("xx", i + 1);
  }
  return counter;
}
