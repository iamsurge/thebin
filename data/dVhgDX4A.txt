for i,ImageButtons in pairs(player.PlayerGui.TitlesShopUI.ShopFrame.ScrollingFrame.Frame:GetChildren()) do
						if ImageButtons:IsA("ImageButton") then
							if ImageButtons:FindFirstChild("NametagPrice") then
								if player.TitlesFolder:FindFirstChild(ImageButtons.NametagName.Value).Value == true then
									if player.EquippedTitle.Value == ImageButtons:FindFirstChild("NametagName").Value then
										ImageButtons:FindFirstChild("TitlePrice").Text = "Equipped"
									else
										ImageButtons:FindFirstChild("TitlePrice").Text = "Owned"
									end
								elseif player.TitlesFolder:FindFirstChild(ImageButtons.NametagName.Value).Value == false then
									ImageButtons:FindFirstChild("TitlePrice").Text = ImageButtons:FindFirstChild("NametagPrice").Value .. " Gems"
								end
							end
						end
					end

					for i,check in pairs(player.PlayerGui.TitlesShopUI.ShopFrame.ScrollingFrame.Frame:GetChildren()) do
						if check:FindFirstChild("NametagRequirments") and player.leaderstats.Wins.Value >= check:FindFirstChild("NametagRequirments").Value then
							if player.EquippedTitle.Value ~= check:FindFirstChild("NametagName").Value then
								check.TitlePrice.Text = "Owned"
							elseif player.EquippedTitle.Value == check:FindFirstChild("NametagName").Value then
								check.TitlePrice.Text = "Equipped"
							end
						elseif check:FindFirstChild("NametagRequirments") and player.leaderstats.Wins.Value < check:FindFirstChild("NametagRequirments").Value then
							check.TitlePrice.Text = check:FindFirstChild("NametagRequirments").Value .. " Wins" 
						end	
					end
