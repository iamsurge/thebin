//Biblioteka

#include <iostream>

using namespace std;

int main()
{
    int n, liczba, liczba1, liczba2, liczba3;

    cin>>n;
    int tab[n];

    for(int i = 0; i < n; i++)
    {
        cin>>liczba;
        tab[i] = liczba;
    }
    
    int i1=n-3, i2=n-2, i3=n-1;
    
    liczba1 = tab[i1];
    liczba2 = tab[i2];
    liczba3 = tab[i3];
    
    int licznik = 0;

    while(i3 >= 2)
    {
        if((liczba3 > liczba2) && (liczba2 > liczba1))
        {
            licznik++;
        }
        
        if(i1 > 0)
        {
            i1--;
        }
        
        else 
        {
            if(i2 > 1)
            {
                i2--;
                i1 = i2-1;
            }
        
            else 
            {
                if(i3 > 2)
                {
                    i3--;
                    i2 = i3-1;
                    i1 = i2-1;
                }
                else
                {
                    break;
                }
            }
        }
    
        liczba3 = tab[i3];
        liczba2 = tab[i2];
        liczba1 = tab[i1];
    
}
    cout<<licznik;
    
    return 0;
}
