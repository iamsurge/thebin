# Read Fasta with one sequence
from Bio.SeqIO import parse
for sequence_to_blast in parse("path_to_fasta","fasta"):

	# Blast sequence online
	from Bio.Blast import NCBIWWW
	result_blast = NCBIWWW.qblast("blastn", "nt", sequence_to_blast.seq)

	print(result_blast.read())

	# Save results to a file
	with open('path_to_result_folder/results.xml', 'w+') as save_file:
		results = result_blast.read()
		save_file.write(results)
