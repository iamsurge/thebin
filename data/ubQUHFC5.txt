using System;
using static System.Console;

namespace Exercise
{
    class City
    {
        static int Id = 0;
        string country_code;
        string name;
        int phone_code;
        int residents;

        public string Country_code { get { return country_code; } set { country_code = value; } }
        public string Name { get { return name; } set { name = value; } }
        public int Phone_code { get { return phone_code; } set { phone_code = value; } }
        public int Residents
        {
            get { return residents; }
            set
            {
                if (value < 1)
                {
                    residents = 0;
                    WriteLine($"В городе не может быть {value} жителей!");
                }
                else
                    residents = value;
            }
        }

        public void Print()
        {
            WriteLine($"ID: {++Id} Тел. код {Phone_code} Страна: {Country_code} Город: {Name} Население: {Residents}");
        }
        public City(int phone_code, string country_code, string name, int residents)
        {
            Phone_code = phone_code;
            Country_code = country_code;
            Name = name;
            Residents = residents;
        }
        static void Main()
        {
            City Moscow = new City(7, "RU", "Москва", 13010112);
            Moscow.Print();
            City Beijing = new City(86, "CN", "Пекин", 0);
            Beijing.Print();
        }
    }
}
