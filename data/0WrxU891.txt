textutils.slowPrint("initializing Simonv2.1...")
textutils.slowPrint("Please input the name of the turtle")
name = read()
local function inet()
    lel = nil
    while lel == nil do
        event, hah = os.pullEvent("char")
        if hah ~= nil then
            lel = hah
            print("Simon Says " .. lel)
        end
    end
    return lel
end
--gets the inputs of the person.

local function Simon()
    rednet.open("back")
    while true do
        input = inet()
        if input == 'w' then
            rednet.broadcast(name .. " w")
        elseif input == 'a' then
            rednet.broadcast(name .. " a")
        elseif input == 'd' then
            rednet.broadcast(name .. " d")
        elseif input == 's' then
            rednet.broadcast(name .. " s")
        elseif input == 'e' then
            rednet.broadcast(name .. " e")
        elseif input == 'q' then
            rednet.broadcast(name .. " q")
        elseif input == 'r' then
            rednet.broadcast(name .. " r")
        elseif input == 'f' then
            rednet.broadcast(name .. " f")
		elseif input == "v" then
			rednet.broadcast(name .. " v")
        end
    end
    rednet.close("back")
end
Simon()
