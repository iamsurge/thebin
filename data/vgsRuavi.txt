Write C++ program that computes eigenvalues from scratch of given N dimensional matrix

#include<iostream>
#include<math.h>
#include<vector>

using namespace std;
 
// Function to calculate the eigenvalues of a given N-dimensional matrix
vector<double> eigenValues(int N, double matrix[][N])
{
    double eigenvalue[N], eigenvector[N][N];
    vector<double> result;
    int i, j, k;
 
    // Calculate the eigenvalues of the matrix
    for(i = 0; i < N; i++)
    {
        eigenvalue[i] = 0;
    }
    for(i = 0; i < N; i++)
    {
        // Initialize the eigenvector matrix to the identity matrix
        for(j = 0; j < N; j++)
        {
            eigenvector[i][j] = 0;
            if(i == j)
                eigenvector[i][j] = 1;
        }
 
        // Iterate until the eigenvector converges to a stable value
        for(k = 0; k < 1000; k++)
        {
            // Calculate the new eigenvector
            for(j = 0; j < N; j++)
            {
                eigenvector[i][j] = 0;
                for(int l = 0; l < N; l++)
                {
                    eigenvector[i][j] += matrix[j][l] * eigenvector[i][l];
                }
            }
            // Normalize the eigenvector
            double norm = 0;
            for(j = 0; j < N; j++)
            {
                norm += eigenvector[i][j] * eigenvector[i][j];
            }
            norm = sqrt(norm);
            for(j = 0; j < N; j++)
            {
                eigenvector[i][j] = eigenvector[i][j] / norm;
            }
            // Calculate the new eigenvalue
            eigenvalue[i] = 0;
            for(j = 0; j < N; j++)
            {
                eigenvalue[i] += matrix[j][j] * eigenvector[i][j];
            }
        }
        result.push_back(eigenvalue[i]);
    }
    return result;
}

// Main function
int main()
{
    int N;
    cout << "Enter the dimension of the matrix: ";
    cin >> N;
    double matrix[N][N];
    cout << "Enter the elements of the matrix: " << endl;
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < N; j++)
            cin >> matrix[i][j];
    }
    vector<double> result = eigenValues(N, matrix);
    cout << "The eigenvalues of the matrix are: " << endl;
    for(int i = 0; i < N; i++)
    {
        cout << result[i] << " ";
    }
    cout << endl;
    return 0;
}
