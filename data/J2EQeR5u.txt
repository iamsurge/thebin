function solve(input) {

    let name = input[0];
    let poins = Number(301);
    let win = 0;
    let unWin = 0;

    let i = 1;
    while (true) {
        let comand = input[i];
        i++;
        let commandPoint = Number(input[i]);
        i++;


        switch (comand) {
            case 'Single':
                if (poins >= 0) {
                    let sum = commandPoint * 1;
                    poins = poins - sum;
                    win++;
                } else {
                    unWin++;
                }
                break;

            case 'Double':
                if (poins >= 0) {
                    let sum = commandPoint * 2;
                    poins = poins - sum;
                    win++;
                } else {
                    unWin++;
                }
                break;

            case 'Triple':
                if (poins >= 0) {
                    let sum = commandPoint * 3;

                    poins = poins - sum;
                    win++;
                } else {
                    unWin++;
                }

            default:
                break;

        }
        if (comand === 'Retire') {
            console.log(`${name} retired after ${unWin} unsuccessful shots.`);
            break;

        }
        if (poins === 0) {
            console.log(`${name} won the leg with ${win} shots.`);
            break;

        }

    }


}
solve([
    'Michael van Gerwen', 'Triple',
    '20', 'Triple',
    '19', 'Double',
    '10', 'Single',
    '3', 'Single',
    '1', 'Triple',
    '20', 'Triple',
    '20', 'Double',
    '20', ''
]

)
