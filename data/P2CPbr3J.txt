#include <bits/stdc++.h>
using namespace std;

const int sqN=44723; //sqrt(2000000000) == 44721.4...
bool sieve[sqN];
vector<int> prime;

int main() {
	ios::sync_with_stdio(false); cin.tie(0); memset(sieve, 0, sizeof(sieve));
	for(int i=2;i<sqN;i++) {
		if(!sieve[i]) {
			prime.push_back(i);
			for(int j=i+i;j<sqN;j+=i)
				sieve[j]=true;
		}
	}
	int t; cin >> t;
	while(t--) {
		int n, ans=0; cin >> n;
		for(int i=0;i<prime.size();i++) {
			if(n==1||prime[i]*prime[i]>n) break;
			while(n%prime[i]==0) {
				n/=prime[i];
				++ans;
			}
		}
		if(n>1) ++ans;
		cout <<ans <<'\n';
	}
	return 0;
}
