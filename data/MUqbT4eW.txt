#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n, x;
    cin >> n >> x;
    int niza[n];
    for(int i = 0; i < n; i++)
    {
        cin niza[i];
    }
    sort(niza, niza + n);
    int s = 0, e = n - 1;
    while(s < e)
    {
        if(niza[s] + niza[e] == x)
        {
            cout << niza[s] << niza[e] << endl;
        }
        else if(niza[s] + niza[e] < x)
        {
            s++;
        }
        else{e--;}
    }

    return 0;
}
