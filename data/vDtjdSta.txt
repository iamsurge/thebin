#include <iostream>

using namespace std;

int main()
{
    cout << "Podaj liczbe naturalna: ";
    int x;
    cin >> x;

    cout << "Naturalne dzielniki " << x << ":\n";
    cout << 1 << endl << x << endl;

    int dzielnik;

    for(dzielnik = 2; dzielnik * dzielnik < x; ++dzielnik)
    {
        if(x % dzielnik == 0)
            cout << dzielnik << endl << x / dzielnik << endl;
    }
    if(dzielnik * dzielnik == x)
        cout << dzielnik << endl;
    return 0;
}
