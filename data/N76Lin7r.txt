using System;
using System.Collections.Generic;

namespace ConsoleApplication1
{
    internal class Program
    {
        public static void Main(string[] args)
        {
            Dictionary <double, double> poly1 = new Dictionary<double, double>();
            poly1.Add(1, 2);
            poly1.Add(2, 3);
            Func f1 = new Func(poly1);
            Dictionary <double, double> poly2 = new Dictionary<double, double>();
            poly2.Add(-1, 2);
            poly2.Add(0, 4);
            Func f2 = new Func(poly2);
            Console.WriteLine(f1.ToString());
            Console.WriteLine(f2.ToString());
            Func f = f1 + f2;
            Console.WriteLine(f.ToString());

            List<Func> funList = new List<Func>();
            Quadratic q = new Quadratic(4, 2, 0);
            Linear l = new Linear(3, 4);
            Hyperbole h = new Hyperbole(4, -8);
            funList.Add(q);
            funList.Add(l);
            funList.Add(h);
            foreach (var fun in funList)
            {
                
                Console.WriteLine(fun.ToString() + " = " + fun.findValue(5) + " (x = 5)");
            }
        }
    }
