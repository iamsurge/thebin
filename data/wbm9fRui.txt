declare
    cursor c1 is select id_przedmiot, nazwa_przedmiot from przedmioty ;
    function fCountStudents(pidp number) return number is
            x number ;
            begin
                select count(distinct id_student) into x from egzaminy where id_przedmiot = pidp ;
                return x ;
    end fCountStudents;
begin
        for vc1 in c1 loop
            dbms_output.put_line(vc1.nazwa_przedmiot || ' - ' || fCountStudents(vc1.id_przedmiot) || ' os√≥b') ;
        end loop ;
end ;
