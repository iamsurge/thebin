<script>
(function() {
    var WAITING_TIMER_COUNTER = 60;
    var HALF_OF_THE_SECOND = 500;

    function waitPageSelector() {

        WAITING_TIMER_COUNTER--;
        var nativePageSelector = document.querySelector('[id*="shopify-section-template"] .main-content .heading > p:nth-child(2)');
        
        if ((!nativePageSelector && WAITING_TIMER_COUNTER > 0) || !window.adoric) {
                setTimeout(waitPageSelector, HALF_OF_THE_SECOND);
            } else if (window.adoric) {
                window.adoric.trigger('showAdoricCleaningForm');
            }
    }
    waitPageSelector();
})();
</script>
