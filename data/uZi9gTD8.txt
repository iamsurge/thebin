anim={}
anim["idle"] = {9,10}
anim["right"] = {1,2}
anim["left"] = {3,4}
anim["up"] = {5,6}
anim["down"] = {7,8}

curAnim="right"
curFrame=1
timerAnim=0

function changeAnim(pAnim)
	if curAnim~=pAnim then
		curAnim=pAnim
		curFrame=1
	end
end

function update()
	local dt=1
	timerAnim=timerAnim+dt
	if timerAnim>=10 then
		timerAnim=0
		curFrame=curFrame+1
		if curFrame > #anim[curAnim] then
			curFrame=1
		end
	end

 -- up
 local bMove=false
	if btn(0) then
		changeAnim("up")
		bMove=true
	end
	if btn(1) then
		changeAnim("down")
		bMove=true
	end
	if btn(2) then
		changeAnim("left")
		bMove=true
	end
	if btn(3) then
		changeAnim("right")
		bMove=true
	end
	
	if bMove==false and curAnim~="idle" then
		curAnim="idle"
		curFrame=1
	end
end

function draw()
	cls(13)
	print(curAnim)
	print(curFrame,0,10)
	
	spr(anim[curAnim][curFrame],50,50)
	
end

function TIC()
	update()
	draw()
end
