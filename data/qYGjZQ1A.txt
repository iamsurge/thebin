def saisir_nb():
    nb = 2
    while not (3 < nb < 20):
        nb = int(input("Nombre d'élèves [4, 19] ? "))
    return nb

def saisir_moy(msg):
    moy = -1
    while not (0 <= moy <= 20):
        moy = float(input(msg))
    return moy

def remplir(n):
    t = [0.0]*n
    for i in range(n):
        t[i] = saisir_moy(f"Note élève n°{i+1} ? ")
    return t

def rank(v, t, n):
    r = 0
    for i in range(n):
        if t[i] > v:
            r += 1
    return r

def sort(t, n):
    r = [0.0]*n
    for i in range(n):
        rnk = rank(t[i], t, n)
        while rnk < n and r[rnk] != 0:
            rnk += 1
        r[rnk] = t[i]
    return r

def existe(v, t, n):
    i = 0
    trouve = False
    while not trouve and i < n:
        trouve = t[i] == v
        i += 1
    return trouve

def affiche(t, n):
    for i in range(n):
        print(t[i], end=", ")
    print()
    
n = saisir_nb()
t = remplir(n)
affiche(t, n)
r = sort(t, n)
affiche(r, n)
moy = saisir_moy("Moyenne à chercher ? ")
if not existe(moy, r, n):
    print(f"{moy} est introuvable!")
else:
    rnk = rank(moy, r, n) + 1
    print(f"{moy} est la {rnk}ème de la classe.")
