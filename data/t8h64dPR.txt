mass = []
ans = []
queue1 = []
queue2 = []
while True:
    try:
        person = str(input())
        mass.append(person)    
    except EOFError:
        break

for i in mass:
    k = i.split()
    if k[0] == '10':
        queue1.append(i)
    elif k[0] == '11':
        queue2.append(i)
    else:
        print(i)
print(*queue1, *queue2, sep='\n')
