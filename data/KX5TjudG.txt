Program

const 
	MAXFIL=;
	MAXCOL=;
	
type tipoMatriz = array [1..MAXFIL; 1..MAXCOL] of char;


function LongitudSecuencia(Matriz:tipoMatriz; fila,columna:integer):integer;
Var
	longitud,i:integer;
Begin
	i:=fila;
	longitud:=1;
	while (i<=MAXFILA) and (Matriz(i,columna)<>' ') do
		begin
			long:=long+1;
			i:=i+1;
		end;
	LongitudSecuencia:=long;
End;


function SiguienteSecuencia(Matriz:tipoMatriz; fila,columna:integer):integer;
Var
	i:integer;
Begin
	i:=fila;
	
	while (i<=MAXFILA) and (Matriz(i,columna)<>' ') do
		i:=i+1;
	
	while (i<=MAXFILA) and (Matriz(i,columna)=' ') do
		i:=i+1;
	
	if (i>MAXFILA) then
		i:=0;
	
	SiguienteSecuencia:=i;
End;


Procedure EncontrarMenorSecuencia(Matriz:tipoMatriz; var filMenor: integer; var colMenor:integer);
Var
	fila,columna,longMenor,longitud:integer;
Begin
	filMenor:=1;
	colMenor:=1;
	longMenor:=LongitudSecuencia(Matriz,filMenor,colMenor);
	
	for columna:=1 to MAXCOL do
		begin
			fila:=1;
			while (fila <> 0) do
				begin
					longitud:=LongitudSecuencia(Matriz,fila,columna);
					if (longitud <= longMenor) then
						begin
							longMenor:=longitud;
							filMenor:=fila;
							colMenor:=columna;
						end;
					fila:=SiguienteSecuencia(Matriz,fila,columna);
				end;
		end;
End;


Procedure BurbujearAlInicio(var Matriz:tipoMatriz; fila: integer; columna:integer);
Var
	i:integer;
	aux:char;
Begin
	for i:=fila downto 2 do
		begin
			aux:=Matriz(i,columna);
			Matriz(i,columna):=Matriz(i-1,columna);
			Matriz(i-1,columna):=aux;
		end;
End;


Procedure Ordenar(var Matriz:tipoMatriz; fila: integer; columna:integer);
Var
	i,longitud,finalSec:integer;
Begin
	while Matriz(fila-1,columna)=' ' then
		BurbujearAlInicio(Matriz,fila-1,columna);
		
	longitud:=LongitudSecuencia(Matriz,fila,columna);
	
	finalSec:=longitud+fila-1;
	
	for i:=1 to longitud do
		BurbujearAlInicio(Matriz,finalSec,columna);
end;


VAR
	MATLET,filamenor,columnamenor
BEGIN
	Cargar();
	....
	MenorSec(MATLET,filamenor,columnamenor)
	if (filamenor>1) then
		Ordenar 
	....
	imprimir();
END.
