if (fs.exists("English-Quran-plain-text.txt") == false) then
	shell.run("wget https://raw.githubusercontent.com/peace-in-mind/Quran-Truth-Edition/master/English-Quran-plain-text.txt")
end

local handle = fs.open("English-Quran-plain-text.txt", "r")
local file_lines = {}
 
for line in handle.readLine do
    file_lines[ #file_lines + 1 ] = line
end    
 
for key, value in pairs(file_lines) do
    print(value)
    sleep(.8)
end
 
handle.close()
