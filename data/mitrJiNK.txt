        private void Insert_Click(object sender, EventArgs e)
        {
            SqlConnection con = new SqlConnection("Data Source=(LocalDB)\\MSSQLLocalDB;AttachDbFilename=C:\\Users\\mihai\\source\\repos\\ChainOfStores\\ChainOfStores\\Database1.mdf;Integrated Security=True");
            con.Open();
