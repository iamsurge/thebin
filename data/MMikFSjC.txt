import java.util.Scanner;

public class NumberDigits {
    public static String avgOf4Digit(int b, int[]a) {
        double sum = 0,  colv = 0;
        for (int i = 0; i < b; i++) {
            if (a[i] >= 1000 && a[i] <= 9999) {
                sum += a[i];
                colv++;
            }
        }
        return (colv > 0) ? String.format("%.2f", (sum/colv)) : "-1.00";
    }
}
