#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main() {
    int trees[5][5], s = 16;
    for(int i = 0; i < 5; i++) {
        for(int j = 0; j < 5; j++) {
            cin >> trees[i][j];
        }
    }
    for(int i = 1; i < 4; i++) {
        for(int j = 1; j < 4; j++) {
            if(trees[i][j] > trees[i + 1][j] || trees[i][j] > trees[i - 1][j] || trees[i][j] > trees[i][j + 1] || trees[i][j] > trees[i][j - 1]) {
                s++;
            }
        }
    }
    cout << s;
    return 0;
}
