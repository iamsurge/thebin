    "ASSIGNED_BY_ID" => $_GET["mitarbeiter"],
    "UF_CRM_35_AUFTRAGGEBER" => $_GET["auftraggeber"],//Auftraggeber
    "UF_CRM_35_AUFTRAGSNUMMER" => $_GET["auftragsnummer"],//Auftragsnummer
    "UF_CRM_35_AUFTRAGSART" => $_GET["auftragsart"],//Auftragsart
    "UF_CRM_35_KFZ" => $_GET["kfz_kennzeichen"],//KFZ-Kennzeichen
    "CONTACT_ID" => $_GET["fahrer"],
    "UF_CRM_35_BESONDERHEITEN" => $_GET["besonderheiten"]
];

$item = $factory->createItem();
