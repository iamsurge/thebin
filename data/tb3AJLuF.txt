#include <iostream>

using namespace std;

/* OPERATORY ARYTMETYCZNE
    + dodawanie
    - odejmowanie
    * mnożenie
    / dzielenie
    % reszta z dzielenia

    ++ inkrementacja
    -- dekrementacja
*/

int main()
{
    cout << "Podaj dwie liczby calkowite: ";
    int a, b;
    cin >> a >> b;

    cout << a << " + " << b << " = " << a + b << endl;
    cout << a << " - " << b << " = " << a - b << endl;
    cout << a << " * " << b << " = " << a * b << endl;

    if(b != 0)
    {
        cout << a << " / " << b << " = " << 1.0 * a / b << endl;
        cout << a << " % " << b << " = " << a % b << endl;
    }
    else
        cout << "Nie mozna wykonywac dzielenia przez 0.\n";

    // uwaga: dzieląc przez siebie dwie liczby całkowite, wynik będzie całkowity

    return 0;
}
