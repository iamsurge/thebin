#pragma hdrstop

#include "PlayerUnit.h"

#pragma package(smart_init)

Player::Player()
{
	cash = 0;
	position = 0;
	numberOfPlayer = 0;
}

Player::Player(const int newCash, const int newPos, const int newNumber, const bool jail, const bool defaulter)
{
	cash = newCash;
	position = newPos;
	numberOfPlayer = newNumber;
	isJail = jail;
	isDefaulter = defaulter;
}

int Player::getCash()
{
	return cash;
}

int Player::getPosition()
{
	return position;
}

int Player::getNumberOfPlayer()
{
	return numberOfPlayer;
}

void Player::setCash(const int newCash)
{
	cash = newCash;
}

void Player::setPosition(const int newPos)
{
	position = newPos;
}

void Player::setNumberOfPlayer(const int newNumber)
{
	numberOfPlayer = newNumber;
}

void Player::setIsJail(const bool jail)
{
	isJail = jail;
}

bool Player::getIsJail()
{
	return isJail;
}

void Player::setIsDefaulter(const bool defaulter)
{
	isDefaulter = defaulter;
}

bool Player::getIsDefaulter()
{
	return isDefaulter;
}
