#include <iostream>

using namespace std;

int main()
{
    int n,fatt,num;
    bool flag = false;
    while(n != 0){cout << "inserisci un numero (inserisci 0 per terminare) " << endl;
        cin >> n;
        num=1;
        fatt=1;
        while(fatt <= n && fatt >= 0){
            num++;
            fatt = fatt * num;
            cout << fatt << endl;
            if(fatt == n){
                 flag = true;
                 break;
            }
        }
        if(flag == true){
            cout << n <<" e' il fattoriale di " << num << endl;
        } else{
           cout << n <<" non e' il fattoriale di nessun numero" << endl;
        }
    }
 }
