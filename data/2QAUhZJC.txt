function isVocal(char) {
    if(char=="a" || char=="e" || char=="i" || char=="o" || char=="u") {
        return true;
    } else {
        return false;
    }
}
function hasVocal(string) {
    for(var i=0; i<string.length; i++) {
        if(isVocal(string.charAt(i))) {
            return true;
        }
    }
    return false;
}

function getVocalGroupPosition(string) {
    if(hasVocal(string)) {
        var positions=[];
        var lastOneWasVocal=false;
        var positionCounter=0;
        for(var i=0; i<string.length; i++) {
            if(isVocal(string.charAt(i)) && !lastOneWasVocal) {
                positions[positionCounter]=i;
                lastOneWasVocal=true;
                positionCounter++;
            } else if(!isVocal(string.charAt(i))) {
                lastOneWasVocal=false;
            }
        }
        return positions;
    }
}
