#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct 
{
int idade;
char nome[100];
int max;
} Ind;


void selectionSort(Ind *v, int n)
{
    int i, j, menor;
    Ind *aux;
    aux = (Ind*)malloc(sizeof(Ind));


    for (size_t i = 0; i < n-1; i++)
    {
        for (size_t j = i+1; j < n; j++)
        {
            if (v[j].idade < v[menor].idade)
            {
                menor = j;
            }
            
        }

        if (i != menor)
        {
            aux[0] = v[menor];
            v[menor] = v[i];
            v[i] = aux[0];
        }    
        
    }
    
}

void insertionSort(Ind *v, int n)
{
    int i, j;
    Ind *aux;
    aux = (Ind*)malloc(sizeof(Ind));

    for (size_t i = 1; i < n; i++)
    {
       aux[0] = v[i];
       j = i;
       while (j>0 && strlen(v[j-1].nome) > strlen(aux[0].nome))
       {
            v[j] = v[j-1];
            j--;
       }
       v[j] = aux[0]; 
    }
    
}

int main()
{
    Ind *pessoa;
    char auxP[5][100] = {"Aarann", "Aarennnn", "Aareznnnn", "Aarman", "Aaronnnnnn"};
    pessoa = (Ind*)malloc(5*sizeof(Ind));
    
    int j = 4;
    for (size_t i = 0; i < 5; i++)
    {
        pessoa[i].idade = j;
        strcpy(pessoa[i].nome, auxP[i]);
        j--;
    }
    
    for (size_t i = 0; i < 5; i++)
    {
        printf("%d - %s\n", pessoa[i].idade, pessoa[i].nome);
    }

    insertionSort(pessoa, 5);
    printf("\n");
    for (size_t i = 0; i < 5; i++)
    {
        printf("%d - %s\n", pessoa[i].idade, pessoa[i].nome);
    }

    selectionSort(pessoa, 5);
    printf("\n");
    for (size_t i = 0; i < 5; i++)
    {
        printf("%d - %s\n", pessoa[i].idade, pessoa[i].nome);
    }

    return 0;
}
