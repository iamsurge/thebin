#include<stdio.h>

struct mahasiswa
{
	char nama[20];
	int nim;
	int tugas, kuis, uts, uas;
};

int main()
{
	FILE *tulis = fopen("mahasiswa.txt", "w");
	mahasiswa mhs;
	char jawaban;
	
	printf("Nama : ");
	scanf(" %s", &mhs.nama);
	printf("NIM : ");
	scanf(" %d", &mhs.nim);
	printf("Nilai Tugas : ");
	scanf(" %d", &mhs.tugas);
	printf("Nilai Kuis : ");
	scanf(" %d", &mhs.kuis);
	printf("Nilai UTS : ");
	scanf(" %d", &mhs.uts);
	printf("Nilai UAS : ");
	scanf(" %d", &mhs.uas);
	
	fprintf(tulis, "%s\t%d\t%d\t%d\t%d\t%d", mhs.nama, mhs.nim, mhs.tugas, mhs.kuis, mhs.uts, mhs.uas);
	fclose(tulis);
	
	printf("\nApakah anda ingin menampilkan data anda? (Y/N) ");
	scanf(" %c", &jawaban);
	
	if(jawaban == 'y' or jawaban == 'Y')
	{
		FILE *baca = fopen("mahasiswa.txt", "r");
		char data[1000];
		printf("\n--------------------------------------------\n");
		printf("Nama\tNIM\tTugas\tKuis\tUTS\tUAS\n");
		printf("--------------------------------------------\n");
		
		while(fgets(data, sizeof(data), baca))
			printf("%s", data);	
	}
}
