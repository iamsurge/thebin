# Bubble Sort.
# Sort a list into ascending order by continually swapping adjacent integers as needed

arr = [205, 499, 348, 224, 255, 324, 37, 107, 366, 121, 80, 117, 300, 432, 26, 60, 465,
       395, 199, 402, 9, 290, 450, 40, 111, 28, 319, 363, 48, 318, 259, 351, 312, 159,
       323, 139, 275, 156, 441, 247, 201, 421, 413, 159, 402, 332, 424, 11, 249, 33]

swap = True
while swap:
    swap = False
    for x in range(len(arr) - 1):
        if arr[x] > arr[x+1]:
            arr[x], arr[x+1] = arr[x+1], arr[x]
            swap = True
print(arr)
