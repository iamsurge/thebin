while (true) {
	cout << "Введите первое число: ";
	cin >> num1;

	bool isInt = true;

	for (int i = 0; i < num1.length(); i++) {
		if (num1[i] < '0' || num1[i] > '9') {
			isInt = false;
			break;
		}
	}

	if (!isInt) {
		cout << "Пожалуйста, вводите только числа!" << endl;
		num1.clear();
		continue;
	}

	// это число, выходим из цикла
	break;
}
