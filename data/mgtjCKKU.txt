using System;
using System.Collections;

namespace LearnRSUE
{
    class Program
    {
        static void Main() 
        {
            int N = 0, str = 0, numb = 0;
            string a;

            Console.Write("Введите кол-во элементов: ");    // Получаем кол-во элементов
            N = Convert.ToInt32(Console.ReadLine());        //

            ArrayList Elements = new ArrayList();   // Создаем список

            for (int i = 0; i < N; i++)    // Заполняем список
            {
                Console.Write("Введите элемент: ");
                Elements.Add(Console.ReadLine());
            }

            foreach (var i in Elements) 
            {
                try // Пытаемся преобразовать элемент в число
                {
                    Convert.ToInt32(i);
                    numb++;
                }
                catch // Если не получается, то
                {
                    str++;
                }
            }

            Console.WriteLine("\nНе чисел: " + str + "\nЧисел: " + numb);

        }
    }
}
