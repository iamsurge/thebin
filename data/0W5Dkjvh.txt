Random random = new Random();

// Возвращает случайное число типа int в диапазоне
int GetRandomIntFromRange(int min, int max) {
	return random.Next(min, max);
}

// Инициализация двухмерной матрицы случайными числами
int[,] Create2DMatrix(int rows, int cols)
{
	int[,] result = new int[rows, cols];
	for (int i = 0; i < rows; i++) {
		for (int j = 0; j < cols; j++) {
			result[i, j] = GetRandomIntFromRange(1, 9);
		}
	}
	return result;
}

// Отображает в консоли двухмерную матрицу
void Print2DMatrix(int[,] target_matrix) {
	for (int i = 0; i < target_matrix.GetLength(0); i++)
	{
		for (int j = 0; j < target_matrix.GetLength(1); j++)
		{
			Console.Write(target_matrix[i, j] + " ");
		}
		Console.WriteLine();
	}
	Console.WriteLine();
}

// Отображает в консоли значения массива
void PrintRowSum(int[] target_array) {
	for (int i = 0; i < target_array.Length; i++)
	{
		Console.Write($"[{i + 1}] " + target_array[i] + " ");
		Console.WriteLine();
	}
	Console.WriteLine();
}

// Находит индекс массива с наименьшим значением числа
int GetMinimalRowSumIndex(int[] target_array) {
	
	int min = int.MaxValue;
	
	foreach(int num in target_array) {
		if (num < min) {
			min = num;
		}
	}
	
	return Array.IndexOf(target_array, min) + 1;
}

int[] Get2DMatrixRowSum(int[,] target_matrix) {
	
	int[] row_sum = new int[target_matrix.GetLength(0)];
	
	for (int i = 0; i < target_matrix.GetLength(0); i++)
	{
		for (int j = 0; j < target_matrix.GetLength(1); j++)
		{
			row_sum[i] = row_sum[i] + target_matrix[i, j];
		}
	}
	return row_sum;
}

Console.WriteLine();

int default_matrix_rows = 4;
int default_matrix_cols = 4;

// Определяем размерность
Console.Write($"Введите количество строк матрицы ({default_matrix_rows}): ");
int matrix_rows = int.TryParse(Console.ReadLine(), out matrix_rows) ? matrix_rows : default_matrix_rows;

Console.Write($"Введите количество столбцов матрицы ({default_matrix_cols}): ");
int matrix_cols = int.TryParse(Console.ReadLine(), out matrix_cols) ? matrix_cols : default_matrix_cols;

Console.WriteLine();

// Инициализируем матрицу
int[,] matrix = Create2DMatrix(matrix_rows, matrix_cols);

// Отображаем исходную
Console.WriteLine("Исходная матрица: ");
Print2DMatrix(matrix);

// Проверка: Отображаем суммы каждой строки
Console.WriteLine("Сумма строк: ");
PrintRowSum(Get2DMatrixRowSum(matrix));

Console.WriteLine("Наименьшая сумма элементов: строка " + GetMinimalRowSumIndex(Get2DMatrixRowSum(matrix)));
