#include <iostream>
#include <set>

using namespace std;

int main()
{
    
    int poryadok = 8;
    cout << "poryadok " << poryadok << ": " << '\n';
    for(int obr = 0; obr < poryadok; obr++) {
        set<int> st;
        for(int i = 0; i < 100000; i++) {
            st.insert((obr*i)%poryadok);
        }
        cout << "obr " << obr << ": ";
        for(auto i = st.begin(); i != st.end(); i++)
            cout << *i << ' ';
        cout << '\n';
        
    }

    return 0;
}
