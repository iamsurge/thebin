#include <stdio.h>
#include <time.h>
#include <conio.h>
#include <Windows.h>
int main() {
    SetConsoleOutputCP(CP_UTF8);
    srand((int)time(NULL));
    int N=100;
    int a[N];
    int max=0;
    printf("Исходный массив:\n");
    for (int i = 0; i < N; ++i) {
        a[i]=rand()%201-100;
        if (a[i]>max) max=a[i];
        printf("%d ",a[i]);
    }
    for (int i = 0; i < N-1; ++i) {
        for (int j = i+1; j <N ; ++j) {
            if(a[i]==a[j]) {
                a[i] = max;
                a[j] = max;
            }
        }
    }
    printf("\nизменённый массив: \n");
    for (int i = 0; i < N; ++i) printf("%d ",a[i]);
    printf("\nмаксимальный элемент=%d", max);
    return 0;
}
