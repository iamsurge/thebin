constexpr int MAXW = 1e5;
ll n, w;
vector < ll > val ,weight;
vector< vector < ll > > dp;
ll knapsack(ll idx, ll v)
{
    if(idx == n || v <= 0 ) 
        return v == 0 ? 0 : 1e9;
    ll& ret = dp[idx][v];
    if(~ret)return ret;
    ret = knapsack(idx + 1, v);
    ret = min(ret, weight[idx] + knapsack(idx + 1, v - val[idx]));
    return ret;
} 
void solve()
{
    cin >> n >> w;
    val.resize(n), weight.resize(n);
    dp.assign(n + 10, vector < ll > (MAXW +5 , -1));
    for(int i = 0; i < n; i++)
        cin >> weight[i] >> val[i];
    for(ll i = MAXW ; i >=0 ;i--){
        if(knapsack(0,i) <= w){
            return cout << i << "\n" ,void();
        }
    }
}
int main()
{
    sherry();
    int t = 1;
    // cin >> t;
    while (t--)
        solve();
    return 0;
}
