using System;
class Program
{
    static void sin_x(double n, double eps)
    {
        double denominator, sinx = n, x = n, oldX;

        int i = 1;
        do
        {
            oldX = x;
            denominator = 2 * i * (2 * i + 1);
            x = -x * n * n / denominator;
            sinx = sinx + x;
            i++;
        } while (Math.Abs(oldX - x) > eps);
        Console.WriteLine(sinx);
    }

    // Driver Code
    static public void Main()
    {
        Console.Write("Введіть число(радіани): ");
        double n = double.Parse(Console.ReadLine());
        sin_x(n, 1e-5);
    }
