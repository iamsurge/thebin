using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

namespace BP_30_11_2022_switch_case_and_loops
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int number, a, b;
            double result = 0;
            char operation = ' ';
            bool isResult = false;
            START:
            Console.Write("Put the number a: ");
            a = int.Parse(Console.ReadLine());
            Console.Write("Put the number b: ");
            b = int.Parse(Console.ReadLine());
        MENU:
            Console.Clear();
            isResult = false;
            Console.WriteLine("0 - EXIT");
            Console.WriteLine("1 - addition");
            Console.WriteLine("2 - sub");
            Console.WriteLine("3 - multi");
            Console.WriteLine("4 - div");
            Console.WriteLine("5 - put new numbers");
            Console.Write("Your choice: ");
            number = int.Parse(Console.ReadLine());

            switch (number)
            {
                case 0:
                    Console.Clear();
                    Console.WriteLine("\n\n\n\n\n\t\t\t\t\t\tThank you!");
                    goto END;
                    //break;
                case 1:
                    result = a + b;
                    operation = '+';
                    isResult = true;
                    break;
                case 2:
                    result = a - b;
                    operation = '-';
                    isResult = true;
                    break;
                case 3:
                    result = a * b;
                    operation = '*';
                    isResult = true;
                    break;
                case 4:
                    if (b != 0)
                    {
                        result = (double)a / b;
                        isResult = true;
                    }
                    else
                        Console.WriteLine("Don't divide by 0!");
                    operation = '/';
                    break;
                case 5:
                    Console.Clear();
                    goto START;
                //break;
                default:
                    Console.WriteLine("Wrong choice!");
                    break;
            }
            if(isResult)
                Console.WriteLine($"{a} {operation} {b} = {result}");
            Console.ReadKey();
            goto MENU;
        END:
            System.Threading.Thread.Sleep(1000);
        }
    }
}
