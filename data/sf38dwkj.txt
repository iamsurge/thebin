import implementations.ArrayDeque;

import org.junit.Test;

import static org.junit.Assert.*;

public class ArrayDequeTest12 {
	@Test
    public void insertCorrect() {
        ArrayDeque<Integer> deque = new ArrayDeque<>();
        for (int i = 0; i < 10; i++) {
            deque.add(i);
        }
        deque.insert(2, 13);

        assertEquals(Integer.valueOf(13), deque.get(2));
        assertEquals(11, deque.size());
    }
}
