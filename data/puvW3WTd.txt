# Importanos libreria
import random
# Creamos diccinario con las opciones
jugada = {1:"Papel 📄",2:"Piedra 🪨",3:"Tijera ✂️"}
#  Pedimos al usuario que elija opción
jugador1 = 0
while not 1<= jugador1 <=3:
  # Presentamos menú
  for i in jugada:
    print(i,jugada[i])
  try:
    jugador1 = int(input("Elige jugada: "))
  except :
    print("Elige un número del 1 al 3")
# Elección aleatoria para el pc
jugador2 = random.randint(1,3)
puntuacion = jugador2 - jugador1
print("El jugador 1 ha sacado",jugada[jugador1])
print("El jugador 2 ha sacado",jugada[jugador2])
# Comprobanos la jugada
if puntuacion == 0:
  print("Habeis empatado")
elif puntuacion  == 1 or puntuacion == -2:
  print("Has ganado")
else :
  print("Has perdido")
