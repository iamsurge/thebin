class Worker:
	def __init__(self, name, lastname, patronymic, mail):
		self.name = name
		self.lastname = lastname
		self.patronymic = patronymic
		self.mail = mail
	def getData(self):
	   return [self.name, self.lastname, self.patronymic, self.mail]
	
workers = [Worker('Виктор', 'Райн', 'Валерьевич', 'p@mail.ru'), Worker('Алексей', 'Мазур', 'Артемьевич', 'a@mail.ru')]
name = str(input('Введите имя сотрудника: '))

for w in workers:
	if w.name == name:
		for data in w.getData():
			print(data)
		break
	else:
		continue
