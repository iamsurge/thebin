// This displays keys being pressed
// WASD will show UP, LEFT, DOWN, RIGHT

//Dependencies:
// sudo apt-get install libncurses5-dev libncursesw5-dev

//Compile:
// gcc main.c -o main -W -Wall -Werror -Wextra -lncurses

#include <stdio.h>
#include <curses.h>

void get_char();

int main(void)
{
  get_char();
  return 0;
}

void get_char()
{
  while (1){
    initscr();  // mandatory curses init
    cbreak();   // unbuffered input
    int chr;

    do {
      chr = getch();
      clear();
    } while (chr == ERR);

    if(chr == 97){
      printw("LEFT\n");
    }
    else if(chr == 100){
      printw("RIGHT\n");
    }
    else if(chr == 115){
      printw("DOWN\n");
    }
    else if(chr == 119){
      printw("UP\n");
    }
    else
    {
      printw("chr = %i\n", chr);
    }
    refresh();  // show the printf stuff

  }
  endwin();   // curses cleanup
}
