capacity = float(input())

num_baggage = 0
free_volume = capacity
input_line = input()
while input_line != "End":
    volume_per_baggage = float(input_line)
    num_baggage += 1

    if num_baggage % 3 == 0:
        volume_per_baggage += volume_per_baggage * 0.1
    free_volume -= volume_per_baggage

    if free_volume <= volume_per_baggage:
        print("No more space!")
        print(f"Statistic: {num_baggage} suitcases loaded.")
        break

    input_line = input()

if input_line == "End":
    print("Congratulations! All suitcases are loaded!")
    print(f"Statistic: {num_baggage} suitcases loaded.")
