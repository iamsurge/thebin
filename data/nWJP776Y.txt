#include<iostream>
#include <algorithm>
#define MAXX 1005
using namespace std;
//sortirame a nai malka, c nai-golyama 
//ako a2+b2 > c2 otstrouguglen ili tupougulen - proveri
//bez pow() che zabavya zadachata
unsigned d[MAXX];
int n;
unsigned n1, n2, n3;
int dd, hh;

void case1()
{
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++)
            for (int k = j + 1; k < n; k++)
                if (d[i] + d[j] > d[k])
                {
                    dd = d[i]*d[i] + d[j]*d[j];
                    hh = d[k]*d[k];
                    if (dd == hh) n1++;
                    else if (dd > hh) n2++;
                    else n3++;
                }
}

void case2()
{
    bool flag;
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++)
        {
            flag = true;
            for (int k = j + 1; k < n && flag; k++)
            {
                if (d[i] + d[j] > d[k])
                {
                    dd = d[i]*d[i] + d[j]*d[j];
                    hh = d[k]*d[k];
                    if (dd == hh) n1++;
                    else if (dd > hh) n2++;
                    else n3++;
                }
                else flag = false;
            }
        }
}

int main()
{
//    unsigned x = 0; cout << ~x << endl;
    while (cin >> n)
    {
        n1 = n2 = n3 = 0;
        for (int i = 0; i < n; i++) cin >> d[i];
        sort(d, d + n);
//        for (int i = 0; i < n; i++) cout << d[i] << " ";
//        cout << endl;
        case2();
        cout << n1 << " " << n2 << " " << n3 << endl;
    }
    return 0;
}
