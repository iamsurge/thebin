/*******************************************************************************
 *                                                                             *
 * Scrivi un programma che preso in ingresso un numero di giorni intero lo     *
 * divida in anni, mesi e giorni. Si assuma che gli anni durino 365 giorni     *
 * e i mesi 30 giorni.                                                         *
 *                                                  Simone Flavio P.           *
 ******************************************************************************/
#include <stdio.h>

int main()
{
    // Fase di inizializzazione delle variabili necessarie
    int anni = 0;
    int mesi = 0;
    int giorni = 0;
    int input;
    int bck;
    
    
    printf("Inserisci il numero di giorni: ");
    scanf("%d", &input);
    bck = input;        // backup del valore di input per stamparlo alla fine
    
    // fino a che l'input non è azzerato
    while(input != 0){
        if (input >= 365){      // se l'input è maggiore o uguale a 365 allora
            anni++;             // incrementa di un anno il contatore
            input -= 365;       // rimuovi i 365 giorni dall'input
        }else if (input >= 30){ // altrimenti se l'input è maggiore o uguale a 30 
            mesi++;             // incrementa di un mese il contatore
            input -= 30;        // rimuovi i 30 giorni dall'input
        }else{                  // altrimenti rimangono meno di 30 giorni, quindi:
            giorni = input;     // i giorni sono uguali all'input 
            input -= input;     // l'input deve essere azzerato.
        }
    }
    printf("%d sono %d anni, %d mesi e %d giorni!\n", bck, anni, mesi, giorni);
}
