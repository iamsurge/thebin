const
	MAXLETRAS=;
	MAXMOD=;

type tipoLetras=array [1..MAXLETRAS] of char;
type tipoMod=array [1..MAXMOD] of char;


function EstaLaSecuencia(LETRAS:tipoLetras;MODELO:tipoMod;posicion:integer):boolean;
Var
	i,k:integer; //i corresponde a LETRAS; k corresponde a MODELO
	puedeestar,aux:boolean;
Begin
	puedeestar:=true;
	i:=posicion;
	k:=1;
	while (i<=MAXLETRAS) and (k<=MAXMOD) and puedeestar do
		begin
			if LETRAS[i]=MODELO[k] then
				begin
					i:=i+1;
					k:=k+1;
					puedeestar:=true;
				end
			else
				puedeestar:=false;
		end;
		
	//forma 1
	if not puedeestar then
		aux:=false
	else
		if (k>MAXMOD) then
			aux:=true;
	
	//forma 2			
	if puedeestar and (k>MAXMOD) then
		aux:=true
	else
		aux:=false;
	
	EstaLaSecuencia:=aux;
End;

Procedure LlenarConAsteriscos (var LETRAS:tipoLetras; posicion:integer);
Var
	i:integer;
Begin
	//Forma 1
	for i:=posicion to (posicion+MAXMOD-1) do
		LETRAS[i]:='*';
	 
	//Forma 2
	for i:=1 to MAXMOD do
		LETRAS[posicion+i-1];
End;
