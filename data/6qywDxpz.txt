# continue_seed.py

import random

def rnd():
	return random.randint(99999, 9999999999)

def rnd_gen():
	
	for i in range(10):
		num1 = random.randint(100, 1000)
		print(num1, end=' ')
	print('')

# set the initial seed value
continue_seed = rnd()

# set the seed value using the initial seed value
random.seed(continue_seed)

rnd_gen() # outputs random numbers between 1 and 10

# save the current seed value in a variable
continue_seed = rnd()

# set the seed value to a different value
random.seed(12345)

rnd_gen()  # outputs random numbers between 1 and 10 that are not the same as num1

# set the seed value to the saved value
random.seed(continue_seed)

rnd_gen()  # outputs random numbers between 1 and 10 that are not the same as num1 and num2

# set the seed value back to the 2nd seed
random.seed(12345)

rnd_gen()  # outputs the same random numbers as num2

# save the current seed value in a variable
continue_seed = rnd()

# set the seed value to the saved value
random.seed(continue_seed)

rnd_gen()  # outputs random numbers between 1 and 10 that are not the same as the previous 4 cycles
