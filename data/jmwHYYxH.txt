#!/bin/bash

CONTROLADOR=0

#MENU, EN CASO QUE NO SEA UNA OPCION CORRECTA VUELVA A INTENTARLO.

while [ $CONTROLADOR -eq 0 ];
do
	clear
	echo
	echo "     MENU - $(cat usuarioensistema.txt | cut -f4 -d :)" #Busca en usuarioensistema.txt el nombre de usuario.
	echo
	echo " 1 - Alta"
	echo " 2 - Baja"
	echo " 3 - Modificacion"
	echo " 4 - Busqueda"
	echo " 5 - Lista de docentes"
	echo " 6 - Salir"
	echo
	read -p " Ingrese una opcion: " OPCION

	case $OPCION in
		1) bash alta.bash
		CONTROLADOR=1
		;;
		2) bash baja.bash
		CONTROLADOR=1
		;;
		3) bash modificacion.bash
		CONTROLADOR=1
		;;
		4) bash busqueda.bash
		CONTROLADOR=1
		;;
		5) bash docentes.bash
		CONTROLADOR=1
		;;
		6) echo "" > usuarioensistema.txt
		exit
		;;
		*) echo
		echo " La opcion que selecciono no es correcta.."
		read -p " Presione 'ENTER' para continuar..."
		;;
	esac
done
