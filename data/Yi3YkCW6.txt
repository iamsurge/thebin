#include <iostream>
#include <cstring>
#include <stdlib.h>
using namespace std;

struct data_penduduk {
	string nama,alamat,agama,status;
	char gol_darah;
	int umur;
};

data_penduduk data[100];

int main(){
	char selesai;
	int n = 0;
	do{
		int pilih;
		
		string cari;
		
		
		cout<<"MENU\n1. Tambah data\n2. Cari data\n3. Tampilkan data\n4. Hapus data\n\nPilih: ";
		cin>>pilih;
		
		switch(pilih){
			case 1:
				cout<<"Nama\t: ";cin>>data[n].nama;
				cout<<"Alamat\t: ";cin>>data[n].alamat;
				cout<<"Agama\t: ";cin>>data[n].agama;
				cout<<"Status\t: ";cin>>data[n].status;
				cout<<"Gol. Darah\t: ";cin>>data[n].gol_darah;
				cout<<"Umur\t: ";cin>>data[n].umur;
				n++;
				break;
				
			case 2:
				cout<<"Masukkan nama yang akan dicari: ";cin>>cari;
				for(int i=0;i<100;i++){
					if(data[i].nama == cari){
						cout<<"\nData ditemukan!"<<endl;
						cout<<"\nNama\t: "<<data[i].nama;
						cout<<"\nAlamat\t: "<<data[i].alamat;
						cout<<"\nAgama\t: "<<data[i].agama;
						cout<<"\nStatus\t: "<<data[i].status;
						cout<<"\nGol. Darah\t:"<<data[i].gol_darah;
						cout<<"\nUmur\t: "<<data[i].umur;
					}
				}
				break;
				
			case 3:
				for(int i=0;i<=n;i++){
					cout<<"\nNama\t: "<<data[i].nama;
					cout<<"\nAlamat\t: "<<data[i].alamat;
					cout<<"\nAgama\t: "<<data[i].agama;
					cout<<"\nStatus\t: "<<data[i].status;
					cout<<"\nGol. Darah\t:"<<data[i].gol_darah;
					cout<<"\nUmur\t: "<<data[i].umur;
				}	
				break;
				
			case 4:
				char hapus;
				cout<<"Apakah anda yakin ingin menghapus seluruh data? (y/n): ";cin>>hapus;
				if(hapus == 'y' || 'Y'){
					for(int i=0;i<=n;i++)
					memset(data, 0, sizeof(data));
					cout<<"Data telah dihapus!";
				}
				else
					cout<<"okok";
		}
		cout<<"Selesai? (y/n): ";cin>>selesai;
		system("CLS");
	}while(selesai == 'n' || selesai == 'N');
}
