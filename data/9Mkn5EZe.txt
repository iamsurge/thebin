#include "check.h"
#include <iostream>

using namespace std;

int getNumberForFactorial()
{
    int c;
    cout << "Введите положительное значение до 12 включительно: ";
    cin >> c;

    if ( (c < 1) || (c>12) )
        return getNumberForFactorial();
    else 
        return c;
}

unsigned int factorial(unsigned int x)
{
    return x == 0 ? 1 : x * factorial(x - 1);
}
