public class C1 {
    //Veiculo
    public String placa; //padrao: "abc-0000"
    protected String marca;
    private int kilometragem;

    public String getPlaca() {
        return placa;
    }

    public void setPlaca(String placa) {
        this.placa = placa;
    }

    public String getMarca() {
        return marca;
    }

    public void setMarca(String marca) {
        this.marca = marca;
    }

    public int getKilometragem() {
        return kilometragem;
    }

    public void setKilometragem(int kilometragem) {
        this.kilometragem = kilometragem;
    }

    public C1(String placa, String marca, int kilometragem){
        System.out.println("Chamado do construtor C1 com parametros!");
        this.setPlaca(placa);
        this.setMarca(marca);
        this.setKilometragem(kilometragem);
    }

    public C1(){
        System.out.println("Chamado do construtor C1 sem parametros!");
        this.placa = null;
        this.marca = null;
        this.kilometragem = 0;
    }

    public void mostrar_atributos(){
        System.out.println("Numero da placa: " + this.getPlaca());
        System.out.println("Marca: " + this.getMarca());
        System.out.println("Kilometragem do veiculo: " + this.getKilometragem() + " Km");
    }

    public  void mostrar_atributos_super(){
        System.out.println("Numero da placa: " + this.getPlaca());
        System.out.println("Marca: " + this.getPlaca());
        System.out.println("Kilometragem do veiculo: " + this.getKilometragem() + " Km");
    }
}
