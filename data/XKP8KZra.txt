using System;

namespace Car_Number
{
	class CarNumber
	{
		static void Main(string[] args)
		{
			int stratInterval = int.Parse(Console.ReadLine());
			int endInterval = int.Parse(Console.ReadLine());

			for (int i = stratInterval; i <= endInterval; i++)
			{
				for (int j = stratInterval; j <= endInterval; j++)
				{
					for (int k = stratInterval; k <= endInterval; k++)
					{
						bool flag = (((j + k) % 2) == 0);

						if (flag)
						{
							for (int l = stratInterval; l <= endInterval; l++)
							{
								flag = (i > l);
								if (flag)
								{
									flag = ((((i % 2) == 0) && ((l % 2) != 0)) || (((l % 2) == 0) && ((i % 2) != 0)));
									if (flag)
									{
										Console.Write($"{i}{j}{k}{l} ");
									}
								}

							}//for (int l = stratInterval; l <= endInterval; l++)

						}//if (isEven)

					}//for (int k = stratInterval; k <= endInterval; k++)

				}//for (int j = stratInterval; j <= endInterval; j++)

			}//for (int i = stratInterval; i <= endInterval; i++)			
		}
	}
}
