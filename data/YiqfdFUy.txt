// Программа 4
#include <iostream>

int main()
{
    int x, y, z; // три числа, вводимые с клавиатуры

    std::cout << "Введите первое число: ";
    std::cin >> x;
    std::cout << "Введите второе число: ";
    std::cin >> y;
    std::cout << "Введите третье число: ";
    std::cin >> z;

    // Сохраним трехзначное число в переменную "num"
    int num = x * 100 + y * 10 + z;
    std::cout << "Трехзначное число: " << num << std::endl;

    return 0;
}
