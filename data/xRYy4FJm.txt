using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace BP_23_11_2022_Game_L1
{
    internal class Program
    {
        static void Main(string[] args)
        {
            NEWGAME:
            Console.Clear();
            
            Random rand = new Random(DateTime.Now.Millisecond);
            int winNumber = Convert.ToInt32(rand.Next(1, 11)); //range 1-10
            int counter = 1;
            int userNumber;
            Console.WriteLine(winNumber);
        USER:
            Console.Write("Guess the number in range 1-10: ");
            userNumber = int.Parse(Console.ReadLine());
            counter++;
            if (userNumber == winNumber) 
            {
                Console.Clear();
                Console.WriteLine("\n\n\n\n\n\n\t\t\t\t\t\tYou Won!");
            }
            else 
            {
                if (counter < 4)
                {
                    Console.WriteLine("Try again!");
                    goto USER;
                }
                else
                { 
                    Console.Clear();
                    Console.WriteLine("\n\n\n\n\n\n\t\t\t\t\t\tYou lose!");
                }
                
            }
            Console.Write("Do you want to play again? y - yes: ");
            string decision = Console.ReadLine();
            if (decision == "y" || decision == "Y")
                goto NEWGAME;
            else
            {
                Console.Clear();
                Console.WriteLine("\n\n\n\n\n\n\t\t\t\t\t\tThank you!");
                System.Threading.Thread.Sleep(1000);
            }
            //
        }
    }
}
