#include <iostream>
#include <graphics.h>
#include <winbgim.h>

using namespace std;


int main()
{
    int x1 = 600, y1 = 600;
    initwindow(x1,y1);
    int x = 0; int y = 0;
    line(x, y + y1 / 3, x + x1, y + y1 / 3);
    line(x + x1/2, y, x + x1/2, y + y1);

    while(x + x1/2 <= x1 and y+y/3 <= y1)
    {
        line(x + x1, y + y1/3, x + x1/2, y + y1);
        x = x + 10;
        y = y + 10;
    }

    for(int y = 590; y >= 500; y -= 10)
    {
        int r = 600 - y;
        circle(150, y, r);
    }

    outtextxy(140, 100, "ABC");

    getch(); closegraph();
    return 0;
}
