//PASSTHROUGH HLSL 11 VERTEX SHADER FOR GAMEMAKER
struct ATTRIBUTE
{
	float3 pos : POSITION;
	//float3 nor : NORMAL;
	float2 tex : TEXCOORD0;
	float4 col : COLOR;
};
struct VERTEX
{
	float3 pos : POSITION;
	float2 tex : TEXCOORD0;
	float4 col : COLOR;
};
VERTEX main(ATTRIBUTE INPUT)
{
	VERTEX OUTPUT;
	OUTPUT.pos = mul(gm_Matrices[MATRIX_WORLD_VIEW_PROJECTION],	float4(INPUT.pos,1));
	OUTPUT.tex = INPUT.tex;
	OUTPUT.col = INPUT.col;
	return OUTPUT;
}

//PASSTHROUGH HLSL 11 PIXEL SHADER FOR GAMEMAKER
struct VERTEX
{
	float3 pos : POSITION;
	float2 tex : TEXCOORD0;
	float4 col : COLOR;
};

struct TARGET
{
	float4 col : COLOR0;
};

TARGET main(VERTEX INPUT) : SV_TARGET
{
	float4 sample = gm_BaseTextureObject.Sample(gm_BaseTexture,INPUT.tex);
	TARGET OUTPUT;
	OUTPUT.col = INPUT.col * sample;
	return OUTPUT;
}
