func whilePrint(data: [String]) {
    var index = 0
    while(index < data.count) {
        print(data[index])
        index += 1
    }
}

func repeatWhilePrint(data: [String]) {
    var index = 0
    
    repeat {
        print("Current index : \(index)")        
        if !data.isEmpty {
            print(data[index])
            index += 1
        }
    } while(index < data.count)
}

whilePrint(data: [])

repeatWhilePrint(data: [])
