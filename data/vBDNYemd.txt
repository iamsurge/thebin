//Wedrowki po gorach
#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int tab[n];
    
    for(int i = 0; i < n; i++)
    {
        cin>>tab[i];
    }
    
    int start = tab[0];
    int max = 0;
    
    for(int i = 0; i < n; i++)
    {
        start = tab[i];
        for(int j = i+1; j < n; j++)
        {
            if(tab[j] < tab[i-1])
                break;
            else if(tab[j] - start > max)
            {
                max = tab[j] - start;
            }
        }
    }
            
    cout<<max;   
    
    return 0;
}
