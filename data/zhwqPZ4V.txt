def mnozenie(a,b):
	return a*b
	
mnozenie(2,5)
mnozenie(a=2,b=5)
mnozenie(b=2,a=5)
mnozenie(2,b=5)

def mnozenie2(a,b,/):
	return a*b
	
mnozenie2(2,5)
mnozenie2(a=2,b=5)
mnozenie2(2,b=5)


def mnozenie3(a,b,/,c):
	return a*b*c
	
mnozenie3(2,5,5)
mnozenie3(2,5,c=11)
mnozenie3(2,b=5,c=11)
