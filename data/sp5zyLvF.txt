// ось так робиться агрегація в JAVA
class Library {
	function work(){
		System.out.println("I ready work");
	}
}

class Book {
	private Library library1;

	public static void main(Library library2) {
		this.library1 = library2;
	}
}

// це треба писати в середині якогось класу
public static void main(String[] args) {
	Library lib = new Library();
	Book book = new Book(lib);
	book.work();
}
