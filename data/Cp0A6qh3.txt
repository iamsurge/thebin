using System;

namespace ConsoleApp2
{
    class Program
    {
        static void Main(string[] args)
        {
            CreateSquare square = new CreateSquare(Figures.CreateSq);
            square(10, ConsoleColor.Cyan);

            CreateTriangle triangle = new CreateTriangle(Figures.CreateTriangle);
            triangle(10, ConsoleColor.Green);

            FiguresDel SquareAndTriangle = new FiguresDel(Figures.CreateSq);

            SquareAndTriangle += Figures.CreateTriangle;

            foreach (var item in SquareAndTriangle.GetInvocationList())
            {
                Console.WriteLine($"{item.DynamicInvoke(5, ConsoleColor.Red)}");
            }

        }
    }
}
