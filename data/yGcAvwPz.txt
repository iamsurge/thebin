#include <iostream>
#include <random>
#include <bits/stdc++.h>

typedef std::random_device RDEVICE;
typedef std::mt19937 MT;
typedef RNG::result_type RSTYP;
typedef std::uniform_int_distribution<RSTYP> DIST;

using std::cout;
using std::cin;
using std::endl;
using std::string;

int main() {
	RDEVICE Device;
    MT Mt(Device());
    DIST Dist(1, 100);

    int Number = Dist(Mt);

	cout << "Enter the number you're guess (1 - 100): ";
	int TryCount = 0;

	Outer:
	while(true) {
		int Guess;
		cin >> Guess;

		if(TryCount >= 10) {
			TryCount = 0;

			while(true) {
				cout << "Are you gonna give up?: ";
				string GiveUp;
				cin >> GiveUp;
				transform(GiveUp.begin(), GiveUp.end(), GiveUp.begin(), :: tolower);

				if(GiveUp == "no") {
					cout << "OK!" << endl;
					goto Outer;
				} else if(GiveUp == "yes") {
					cout << "Bye!" << endl;
					return 0;
				} else {
					continue;
				}
			}
		}

		if(Number == Guess) {
			cout << "Correct! You Won!" << endl;
			break;
		} else if(Number > Guess) {
			cout << "The number is higher, try again: ";
		} else {
			cout << "The number is lower, try again: ";
		}

		TryCount++;
	}

	return 0;
}
