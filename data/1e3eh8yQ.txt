#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    
    int niza[n];
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
    }
    int x;
    cin >> x;
    sort(niza, niza + n);
    
    int pozicija = lower_bound(niza, niza + n, x) - niza;
    
    cout << pozicija + 1 << endl;
   return 0;
}
