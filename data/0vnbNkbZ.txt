#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX 300
char vocale[20]={"aeiouAEIOU"};
int main()
{
    FILE* fis;
    FILE* out;
    out=fopen("output.txt", "w");
    char rand[MAX];
    if (!(fis = fopen("fisier.txt", "r")))
    {
        perror("Eroare la deschidere, cu mesajul de eroare:");
        exit(1);
    }
    while(!feof(fis))
    {
       if (!fgets(rand,100,fis))
            {  exit(1);}
        int nrvocale=0;
        for(int i=0;i<strlen(rand);i++)
            if(strchr(vocale, rand[i]))
                nrvocale++;
        if(nrvocale != 0 && nrvocale % 2 == 0)
            fprintf(out, "%s", rand);
    }
    fclose(fis);
    fclose(out);
    return 0;
}
