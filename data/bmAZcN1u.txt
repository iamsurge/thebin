public String stringSplosion_1(String str) {
  String result = "";
  for(int end = 1; end <= str.length(); end++) {
    result = result + str.substring(0, end);
  }
  return result;
}

public String stringSplosion_2(String str) {
  String result = "";
  String partial = "";
  for(int i = 0; i < str.length(); i++) {
    partial = partial + str.charAt(i);
    result = result + partial;
  }
  return result;
}

public String stringSplosion_3(String str) {
  StringBuilder result = new StringBuilder();
  StringBuilder partial = new StringBuilder();
  for(int i = 0; i < str.length(); i++) {
    partial.append(str.charAt(i));
    result.append(partial);
  }
  return result.toString();
}


public String stringSplosion(String str) {
  String result = "";
  for(int end = 1; end <= str.length(); end++) {
    String partial = "";
    for(int i = 0; i < end; i++) {
      partial = partial + str.charAt(i);
    }
    result = result + partial;
  }
  return result;
}
