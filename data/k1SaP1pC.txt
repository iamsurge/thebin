#include <iostream>
#include <string>
#include <stdio.h>
#include <stdlib.h>
#include <locale.h>


using namespace std;


class c10to3 {
public:
	string raw_data;
	int int_data;
	bool valid = true;

	c10to3(string data) { // конструктор
		raw_data = data;
		for (int i = 0; i < raw_data.length(); i++) {
			if (!isdigit(raw_data[i])) {
				valid = false;
			}
		}
		if (valid) {
			int_data = stoi(raw_data);
		}
	}

	string show() {
		if (!valid) {
			return "неверный формат числа";
		}
		string result_string;
		int temp_int_data = int_data;
		while (temp_int_data > 0)
		{
			result_string.insert(result_string.begin(), (temp_int_data % 3) + '0');
			temp_int_data /= 3;
		}
		return result_string;
	}

	string type() {
		if (!valid) {
			return "неверный формат числа";
		}
		if (int_data % 2 == 0)
			return "Чётное";
		else
			return "Нечётное";
	}
};

Вход в конструктор строка типа string
