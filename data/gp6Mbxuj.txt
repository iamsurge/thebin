    if app.ENABLE_ITEM_BLEND_SYSTEM:
        def SetItemImage(self, fileName):
            if not self.blendImage:
                self.blendImage = ui.ExpandedImageBox()
                self.blendImage.SetParent(self)
                self.blendImage.AddFlag("not_pick")
                self.blendImage.SetWindowHorizontalAlignCenter()
                self.blendImage.SetWindowVerticalAlignCenter()
                
            self.blendImage.LoadImage(fileName)
            
            if str(wndMgr.GetScreenWidth()) >= "1920":
                self.blendImage.SetScale(0.9, 0.9)
            elif str(wndMgr.GetScreenWidth()) >= "1366" and str(wndMgr.GetScreenWidth()) < "1920":
                self.blendImage.SetScale(0.75, 0.75)
            else:
                self.blendImage.SetScale(0.65, 0.65)
                
            self.blendImage.Show()
