#include <stdio.h>

unsigned add(unsigned a, unsigned b) {
  do {
    unsigned xor = a ^ b;
    unsigned carry = a & b;
    carry <<= 1;
    a = xor;
    b = carry;
  } while (b);
  return a;
}

int main() {
  unsigned a, b;
  scanf("%u", &a);
  scanf("%u", &b);
  printf("%u+%u=%u\n", a, b, add(a, b));
  return 0;
}
