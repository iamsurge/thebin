; Example: A simple input-box that asks for first name and last name:
#SingleInstance ignore

Gui, Add, Button,gButtonShutdown w200, Shut Down
Gui, Add, Button,gButtonForceShutdown w200, Force Shut Down
Gui, Add, Button,gButtonRestart w200, Restart
Gui, Add, Button,gButtonForceRestart w200, Force Restart
Gui, Add, Button,gButtonHibernate w200, Hibernate
Gui, Add, Button,gButtonSleep w200, Sleep
Gui, Add, Button,gButtonScreenOff w200, Screen Off

~SC176::Gui, Show,, Power Menu
; Gui, Show,, Power Menu
return  ; End of auto-execute section. The script is idle until the user does something.

SetTitleMatchMode, 2 ;We use this so it will look for "Notepad" anywhere in the title
#IfWinActive Power 
SC176::Gui, Hide
#IfWinActive

~Esc::Gui, Hide

ButtonShutdown:
Gui, Hide
Shutdown, 8
return

ButtonForceShutdown:
Gui, Hide
Shutdown, 12
return

ButtonSleep:
Gui, Hide
DllCall("PowrProf\SetSuspendState", "int", 0, "int", 0, "int", 0)
return

ButtonHibernate:
Gui, Hide
DllCall("PowrProf\SetSuspendState", "int", 1, "int", 0, "int", 0)
return

ButtonRestart:
Gui, Hide
ShutDown, 2
return

ButtonForceRestart:
Gui, Hide
ShutDown, 6
return

ButtonScreenOff:

Sleep 1000  ; Give user a chance to release keys (in case their release would wake up the monitor again).
; Turn Monitor Off:
SendMessage, 0x0112, 0xF170, 2,, Program Manager  ; 0x0112 is WM_SYSCOMMAND, 0xF170 is SC_MONITORPOWER.
; Note for the above: Use -1 in place of 2 to turn the monitor on.
; Use 1 in place of 2 to activate the monitor's low-power mode.
Gui, Hide
return

GuiClose:
Gui, Hide
return
;ExitApp
