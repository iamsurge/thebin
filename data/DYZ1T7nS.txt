Program

const
	MIN=1;
	MAX=15;

type tipo_arreglo=array[MIN..MAX] of char;

//Ver si se puede mejorar(que no cuente de mÃ¡s)
function CantApariciones(arreglo:tipo_arreglo;letra:char):integer;
var
	i,cantidad:integer;
begin
	cantidad:=0;
	
	for i:=MIN to MAX do
		if (arreglo[i]=letra) then
			cantidad:=cantidad+1;
			
	CantApariciones:=cantidad;		
end;


procedure EliminarLetras(var arreglo:tipo_arreglo;letra:char);
var
	i:integer;
begin
	for i:=MIN to MAX do
		if (arreglo[i]=letra) then
			arreglo[i]:=' ';
end;


procedure EnivarUnHuecoAlInicio(var arreglo:tipo_arreglo;posicion:integer);
var
	i:integer;
	aux:char;
begin
	i:=posicion;
	while (i>MIN) and (arreglo[i-1]<>' ') do
		begin
			aux:=arreglo[i-1];
			arreglo[i-1]:=arreglo[i];
			arreglo[i]:=aux;
			i:=i-1;
		end;  
end;


procedure EnviarVaciosAlInicio(var arreglo:tipo_arreglo);
var
	i:integer;
begin
	i:=MIN;
	while (i<=MAX) and (arreglo[i]=' ') do
		i:=i+1;
	while (i<=MAX) do
		begin
			if (arreglo[i]=' ') then
				EnivarUnHuecoAlInicio(arreglo,i);
			i:=i+1;	
		end;
end;


procedure EliminarRepetidas(var arreglo:tipo_arreglo; CantLetras:integer);
var
	i:integer;
begin
	for i:=MIN to MAX do
		begin
			if (CantApariciones(arreglo,arreglo[i])=CantLetras) then
				EliminarLetras(arreglo,arreglo[i]);
		end;
	EnviarVaciosAlInicio(arreglo);
end;


VAR
	MuchasLetras:tipo_arreglo;
	CantLetras:integer;
BEGIN
	CargarDatos(MuchasLetras,CantLetras);
	EliminarRepetidos(MuchasLetras,CantLetras);
	MostrarResultado(MuchasLetras,CantLetras);
END.
