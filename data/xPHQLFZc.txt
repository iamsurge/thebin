#include <iostream>
using namespace std;
 
unsigned int a, c;
 
int main()
{
    
   cin >> a;
   
   for(int i = 0; i < 100; i+=2)
   {
        a = a - 2;
        c = a;
       if((c % 2) == 0)
       {
           cout << "YES";
           return 0;
       }
   }
   
   cout << "NO";
}
