 * Scrivi un programma che prende due numeri in ingresso e stampa la           *
 * loro somma. Il programma termina solamente quando la somma è uguale a 0     *
 *                                                  Simone Flavio P.           *
 ******************************************************************************/
#include <stdio.h>

int main()
{
    // Fase di inizializzazione delle variabili necessarie
    int num1 = 0;
    int num2 = 0;
    int somma;
    
    printf("Inserisci il primo numero: ");
    scanf("%d", &num1);
    
    printf("Inserisci il secondo numero: ");
    scanf("%d", &num2);
    
    somma = num1 + num2;
    
    printf("La somma è: %d\n", somma);
    
    // fino a che l'input è diverso da zero
    while (somma != 0){
        printf("Inserisci il primo numero: ");
        scanf("%d", &num1);
        
        printf("Inserisci il secondo numero: ");
        scanf("%d", &num2);
    
        somma = num1 + num2;
        
        printf("La somma è: %d\n", somma);
    }
            /*
            ESISTE UNA MANIERA PIÙ ELEGANTE CHE SI OTTIENE USANDO IL DO WHILE
                Cerca in internet "do while C statement" se vuoi approfondire.
            */
}
