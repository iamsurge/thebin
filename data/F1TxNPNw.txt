		  <script>
			button1 = document.getElementById("getFirstResult");
			button1.onclick = function () {
				eqa1 = document.getElementById("eqa1").value;
			    eqb1 = document.getElementById("eqb1").value;
				eqc1 = document.getElementById("eqc1").value;
				let result = (eqc1 - eqb1)/eqa1;
				document.getElementById("firstresult").innerHTML = result;
			}	

			button2 = document.getElementById("getSecondResult");
			button2.onclick = function () {
				eqa2 = document.getElementById("eqa2").value;
			    eqb2 = document.getElementById("eqb2").value;
				eqc2 = document.getElementById("eqc2").value;
				let discriminant = Math.pow(eqb2, 2) - 4*eqa2*eqc2;

				console.log(discriminant)

				let x1, x2;
				if(discriminant > 0){
					// two possible solutions
					x1 = (eqb2 - Math.sqrt(discriminant, 2))/(2*eqa2);
					x2 = (eqb2 + Math.sqrt(discriminant, 2))/(2*eqa2);
				}
				else if(discriminant == 0){
					// one possible solution
					x1 = (eqb2/2*eqa2);
					x2 = (eqb2/2*eqa2);
				}
				else {
					x1 = "No root";
					x2 = "No root";
				}
				
				document.getElementById("resultx1").innerHTML = x1;
				document.getElementById("resultx2").innerHTML = x2;
			}
		  </script>
