-- Set the API endpoint URL and your API key
local api_url = "https://pastebin.com/api/api_post.php"
local api_key = "X8AI94uRWWUPaW_LxYjWcxoTN2bu4hVO"

-- Set the parameters for the API request
local params = {
    api_dev_key = api_key,
    api_option = "paste",
    api_paste_code = "This is a test. Did this work?"
}

-- Build the query string from the parameters
local query_string = ""
for key, value in pairs(params) do
    query_string = query_string .. key .. "=" .. value .. "&"
end

-- Build the curl command
local curl_command = "curl " .. api_url .. "?" .. query_string

-- Run the curl command using the os.execute() function
local response = os.execute(curl_command)

-- Print the response from the API
print(response)
