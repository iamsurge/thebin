import React, { useState } from "react";
import Script from "next/script";

  const [open, setOpen] = useState(false);

            flexDirection="column"
            gap={2}
            marginBottom="5rem"
            marginTop="-2rem"
          >
            {isConnected && (
              <Button
                variant="outlined"
                style={{ marginLeft: "1rem" }}
                color="warning"
                onClick={() => setOpen(!open)}
              >
                Mint Your NFT
              </Button>
            )}
            <ConnectButton label="Mint Your NFT" />
          </Box>
        </Box>

        {isConnected && <Mint open={open} handleClose={handleClose} />}
