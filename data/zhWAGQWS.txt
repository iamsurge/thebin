import turtle

#change 'n' and 'L' at the end where there are #-----

def koch(n,L):
    if n==0:
        turtle.forward(L)
    else:
        koch(n-1,L)
        turtle.left(60)
        koch(n-1,L)
        turtle.left(-120)
        koch(n-1,L)
        turtle.left(60)
        koch(n-1,L)

turtle.Screen().clear()
turtle.speed(0)
turtle.pensize(2)
turtle.penup()
turtle.goto(-300,0)
turtle.pendown()
n=4				#-----
L=4				#-----
for i in range(3):
    koch(n,L)
    turtle.left(-120)
turtle.done()
