Private Sub Command1_Click()
Dim unit As Integer
unit = Text1.Text

Select Case unit
Case Is < 5 And Check1.Value = 1
Label1.Caption = (Val(Text1.Text) * 50) - ((Val(Text1.Text) * 50) * 0.1)
Case Is < 5 And Check1.Value = 0
Label1.Caption = Val(Text1.Text) * 50
Case Is < 10 And Check1.Value = 1
Label1.Caption = (Val(Text1.Text) * 45) - ((Val(Text1.Text) * 45) * 0.1)
Case Is < 10 And Check1.Value = 0
Label1.Caption = Val(Text1.Text) * 45
Case Is < 20 And Check1.Value = 1
Label1.Caption = (Val(Text1.Text) * 40) - ((Val(Text1.Text) * 40) * 0.1)
Case Is < 20 And Check1.Value = 0
Label1.Caption = Val(Text1.Text) * 40
Case Is < 50 And Check1.Value = 1
Label1.Caption = (Val(Text1.Text) * 35) - ((Val(Text1.Text) * 35) * 0.1)
Case Is < 50 And Check1.Value = 0
Label1.Caption = Val(Text1.Text) * 35
Case Is > 50 And Check1.Value = 1
Label1.Caption = (Val(Text1.Text) * 30) - ((Val(Text1.Text) * 30) * 0.1)
Case Is > 50 And Check1.Value = 0
Label1.Caption = Val(Text1.Text) * 30
End Select

End Sub
