#include <stdio.h>
#include <stdlib.h>

int *array(int m[10][10], int n)
{
    int i, j, size = 0;
    int *v;
    v = malloc(((n-1)*n/2)*sizeof(int));   // aloc cata memorie ii trebuie
    for (i=0; i<n; i++)
    {
        for (j=0; j<n; j++)
        {
            if (i > j)
            {
                v[size] = m[i][j];
                size++;
            }
        }
    }
    return v;
}

int main()
{
    int *v;
    v = malloc(10*sizeof(int));
    int i, j, m[10][10], n=0;
    for (i=0; i<4; i++)
        for (j=0; j<4; j++)
        {
            m[i][j] = n;
            n++;
        }
    for (i=0; i<4; i++)
    {
        for (j=0; j<4; j++)
        {
            printf("%d ", m[i][j]);
        }
        printf("\n");
    }
    v = array(m, 4);
    printf("Vector:\n");
    for (i=0; i<6; i++)
        printf("%d ", *(v+i));
    return 0;
}
