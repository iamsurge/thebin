import turtle

def stairs(t: turtle.Turtle, steps=3, size=25):
    for _ in range(steps):
        t.forward(size)
        t.left(90)
        t.forward(size)
        t.right(90)
    t.up()
    t.right(90)
    t.forward(size*steps)
    t.down()
    t.left(90)

def draw_repeat(func, t: turtle.Turtle, count=7, **kwargs):
    for _ in range(count):
        func(t, **kwargs)

draw_repeat(stairs, turtle.Turtle())
