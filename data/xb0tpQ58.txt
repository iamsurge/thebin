def prime(n):
    if n <= 1: return False
    if n % 2 == 0:
        return (n == 2)
    d = 3
    while d * d <= n and n % d != 0:
        d += 2
    return d * d > n

n = int(input())
A = list(map(int, input().split()))

ok = 0
for i in A:
    if prime(i):
        ok = 1
        print(i, end = ' ')
if ok == 0:
    print(0)
