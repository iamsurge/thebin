function mineForward()
  turtle.dig()
  turtle.forward()
end

function mineUpDown()
  turtle.digUp()
  turtle.digDown()
end

function mineLayers()

  mineForward()
  mineUpDown()

  turtle.turnRight()

  mineForward()
  mineUpDown()

  mineForward()
  mineUpDown()

  turtle.turnLeft()

  mineForward()
  mineUpDown()

  turtle.turnLeft()

  mineForward()
  mineUpDown()

  mineForward()
  mineUpDown()

  turtle.turnRight()

end

function main()

    while true do
      mineLayers()
    end

end

main()
