l = int(input())
r = int(input())

l_tens = len(str(l))
r_tens = len(str(r))

if l_tens == r_tens:
    res = (r - l + 1) * l_tens
else:
    res = (10 ** l_tens - l) * l_tens
    if l != r:
        res += (r - 10 ** (r_tens - 1) + 1) * r_tens
        for t in range(l_tens + 1, r_tens):
            res += (10 ** (t + 1) - 10 ** t) * t
print(res)
