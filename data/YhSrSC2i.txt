from typing import List, Dict

adjmat = [[1, 2, 3, 4, 5], [2, 4, 6, 8, 10], [3, 6, 9, 12, 15]]
 
def adjmat_to_adjlist(adjmat: List[List[int]]) -> Dict[int, List[int]]:
    for i, key in enumerate(adjmat):
        adjlist = dict(zip(key, i))
    return adjlist

print (adjmat_to_adjlist(adjmat))
