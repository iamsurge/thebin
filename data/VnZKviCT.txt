#include <iostream>
using namespace std;

int main()
{
    string a, b, c;
    
    cin >> a >> b;
    
    for(int i = 0; i < a.size(); i++)
    { 
        if(a[i] == b[i]) c[i] = '0';
        else             c[i] = '1';
    }
    
    for(int i = 0; i < a.size(); i++) cout << c[i];
}
