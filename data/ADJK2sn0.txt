data = open('day2Data.txt', 'r')
data = data.read().split('\n')
rule = []
result = []

for i in range(len(data)):
    rule.append(data[i][0]) ## get needed data from puzzle input / the enemie's play and therfore my play
    result.append(data[i][2])

### SCORES
# rock=1, paper=2, scissors=3
# loss=0, draw=3, win=6
# paper=Y, rock=X, scissors=Z
# rock=A, paper=B, scissors=C

score = 0
for i in range(len(rule)):
    ## WIN CASES
    if rule[i] == 'C' and result[i] == 'X':
        score += 6 ## win
    elif rule[i] == 'A' and result[i] == 'Y':
        score += 6
    elif rule[i] == 'B' and result[i] == 'Z':
        score += 6
    ## DRAW
    elif rule[i] == 'A' and result[i] == 'X' or rule[i] == 'B' and result[i] == 'Y' or rule[i] == 'C' and result[i] == 'Z':
        score += 3
    ## IF WE LOST DO NOTHING BECAUSE NO ADDITION TO SCORE

    if result[i] == 'X':
        score += 1
    elif result[i] == 'Y':
        score += 2
    elif result[i] == 'Z':
        score += 3

print(score, '\n')

## PART TWO

# X=lose, Y=draw, Z=win
# rock=1, paper=2, scissors=3

score = 0

for i in range(len(rule)):
    if result[i] == 'Z': ## win
        score += 6
        if rule[i] == 'A':
            score += 2
        elif rule[i] == 'B':
            score += 3
        elif rule[i] == 'C':
            score += 1
    elif result[i] == 'Y': ## draw
        score += 3
        if rule[i] == 'A':
            score += 1
        elif rule[i] == 'B':
            score += 2
        elif rule[i] == 'C':
            score += 3
    elif result[i] == 'X': ## lose
        score += 0
        if rule[i] == 'A':
            score += 3
        elif rule[i] == 'B':
            score += 1
        elif rule[i] == 'C':
            score += 2


print(score)
