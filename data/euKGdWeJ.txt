SELECT  e.Id_egzaminator , e.Imie, e.Nazwisko, count(*) as "Liczba egzaminow"
FROM Egzaminatorzy e 
inner join Egzaminy egz
on e.Id_egzaminator = egz.Id_egzaminator
GROUP BY e.Id_egzaminator, e.Imie, e.Nazwisko
Having count(*) = (select max(count(*)) FROM Egzaminy e 
GROUP BY e.Id_egzaminator
)
