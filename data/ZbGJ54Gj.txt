local roundDecimals = function(num, places) -- https://devforum.roblox.com/t/rounding-numbers-on-guis-to-the-nearest-hundredth/468375/6
    places = math.pow(10, places or 0)
    num = num * places
    if num >= 0 then 
        num = math.floor(num + 0.5) 
    else 
        num = math.ceil(num - 0.5) 
    end
    return num / places
end
