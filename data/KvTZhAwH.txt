#include <iostream>
using namespace std;

int solve(int a)
{
    int k, sum = 0, s, minn;
    for (int i = 0; i < a; i++)
    {
        cin >> k; sum += k;
        if (i == 0) minn = k;
        if (minn > k) minn = k;
    }
    return s = (minn + a)*(a+1) - sum;
}

int main()
{
    int a1, a2;
    while (cin >> a1 >> a2)
        cout << solve(a1) << endl
        << solve(a2) << endl;
    return 0;
}
