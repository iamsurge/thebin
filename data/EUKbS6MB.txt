public class CharacterTable {

    public static void main(String[] args) {
        for (char c = 'a'; c <= 'z'; c++) {
            char uppercased = Character.toUpperCase(c);
            // System.out.println("'" + c + "' = " + (int) c + "\t'" + uppercased + "' = " + (int) uppercased);
            System.out.printf("'%c' = %-3d          '%c' = %d%n", c, (int)c, uppercased, (int)uppercased);
        }
    }

}
