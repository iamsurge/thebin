bool ok(int x) {
    int y, z, a, b, c;
    a = A + x;
    if (a % 2 == 1) {
        return 0;
    }
    b = a / 2;
    y = b - B;
    c = b;
    z = c - C;
    bool r = x + y + z <= n && y >= 0 && z >= 0;
    if (r) {
        ans[0] = x;
        ans[1] = y;
        ans[2] = z;
    }
    return r;
}

bool cmp(vector <int> a, vector <int> b) {
    return a[0] > b[0];
}

vector <int> his() {
    vector <vector <int> > ans;
    for (int x= 0; x <= n; ++x) {
        for (int y = 0; y <= n - x; ++y) {
            for (int z = 0; z <= n - x - y; ++z) {
                if (x + y + z > n) continue;
                int a = A + x, b = B + y, c = C + z;
                if (a == 2 * b && a == 2 * c) {
                    ans.push_back({x + y + z, x, y, z});
                }
            }
        }
    }
    vector <int> res = {-1, -1, -1};
    if (ans.empty()) {
        return res;
    }
    sort(ans.begin(), ans.end(), cmp);
    for (int i = 1; i <= 3; ++i) {
        res[i - 1] = ans[0][i];
    }
    return res;
}
