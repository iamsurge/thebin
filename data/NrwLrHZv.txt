# Variables: son "cajas" que nos sirven para guardar objetos
# 1) Nomenclatura:
# variable = valor  # guarda valor de la derecha del igual en la variable
# de la izquierda del igual
nombre = "Juan"
print(nombre)  # imprime Juan

# las variables pueden guardar cualquier tipo de dato
nombre = 25
print(nombre)  # imprime 25


# Una expresión es un fragmento de código que retorna un resultado.
# 2) Nomenclatura 
# variable = expresion # guarda EL RESULTADO DE LA EXPRESION en la 
# variable
total = 25 + 15 # suma 25+15 y guarda 40 en total
print(total)  # imprime 40

# que valor almacena el total ahora?
total = total + 10
print(total) # imprime 50
