#include <bits/stdc++.h>
using namespace std;

int main() {
	ios::sync_with_stdio(false); cin.tie(0);
	char c; cin >> c;
	int n, w, l, x; cin >> n >> w >> l;
	x=2*n+1;
	for(int i=0;i<=l;i++) {
		for(int j=0;j<x;j++) {
			for(int k=x-i;k<w-i+x;k++) {
				if(abs(j%x-n)+abs(k%x-n)<=n) cout << c;
				else cout << '.';
			}
			cout << '\n';
		}
	}
	return 0;
}
