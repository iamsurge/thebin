    
    #include <stdio.h>
int fukncija(int N) {
    int zbir = 0;
    for(int i = 1; i<N; i++) {
        if(i % 2 == 0) {
            zbir+=i;
        }
    }
    return zbir;
}
    int main()
{
        int n;
        scanf("%d", &n);
        printf("%d\n", fukncija(n));
        
        return 0;
        
    }
