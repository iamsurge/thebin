def decisao(legenda, opc1, opc2):
    opc = input(legenda+f"[{opc1}][{opc2}]")
    if opc == opc1:
        return True
    elif opc == opc2:
        return False
    else:
        print(" -> try again")
        decisao(legenda, opc1, opc2)
