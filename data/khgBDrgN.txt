--//Variables (this is a program to open the modem's chanel)
local args = {...}
local channel = assert(toNumber(args[1]), "Usage: openChannel <A number between 0 & 65535> <Modem>")
local modemLocation = assert(args[2], "Usage: openChannel <Port#> <Modem Location (Front/Back/Top/Bottom)>")
--//Start of program
modem = peripheral.wrap(modemLocation)
modem.open(channel)
if modem.isOpen(channel) then
	print("The modem is now ready to operate in the selected channel:" ..channel ..".")
	print("")
end
