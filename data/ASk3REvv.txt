const ll INF = 1e9;
const int N = 1e5 + 10;
const int S = 330;

int a[N], best[N], cnt[N], ans[N];
bool c[N];
int f[S][N];

void solve() {
	int n, k; cin >> n >> k;
	
	int num = 330;
	int block = (n + num - 1) / num;
	for(int i = 0; i < n; i++) cin >> a[i];

	a[n] = -INF;
	for(int i = 1; i <= block; i++) {
		int l = (i - 1) * num, r = min(l + num - 1, n - 1);
		for(int j = 0; j < k; j++) f[i][j] = n;
		for(int j = l; j <= r; j++) {
			if(a[f[i][j % k]] < a[j]) f[i][j % k] = j;
		}

		best[i] = f[i][0];
	}

	for(int rep = 1; rep <= n; rep++) {
		int pos = 1;
		for(int i = 1; i <= block; i++) {
			if(a[best[i]] > a[best[pos]]) pos = i;
		}

		ans[rep] = a[best[pos]];
		for(int i = pos + 1; i <= block; i++) {
			++cnt[i]; cnt[i] %= k;
			best[i] = f[i][cnt[i]];
		}

		int l = (pos - 1) * num, r = min(l + num - 1, n - 1);
		int d = 0;
		for(int i = l; i <= r; i++) {
			if(!c[i]) {
				f[pos][(i - d) % k] = n;
			} else ++d;
		}
		d = 0;
		c[best[pos]] = true;
		for(int i = l; i <= r; i++) {
			if(!c[i]) {
				if(a[f[pos][(i - d) % k]] < a[i]) f[pos][(i - d) % k] = i;
			} else ++d;
		}
		best[pos] = f[pos][cnt[pos]];
	}

	for(int i = 1; i <= n; i++) cout << ans[i] << " ";
}
