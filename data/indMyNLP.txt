#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define MAX 100
 
char stack[MAX];
int top = -1;
 
void push(char x)
{
    stack[++top]= x;
}
char pop()
{
    if(top == -1)
        return -1;
    else 
    {
        return stack[top--];    
    }   
}
 
int prc(char x)
{
    if(x ==')')
        return 0;   
    else if(x == '^')
        return 3;
    else if(x =='*' || x =='/' || x=='%')
        return 2;
    else if(x =='+' || x =='-')
        return 1; 
}

}
