void __fastcall TMainForm::BankrotPlayer(const int player)
{
	switch (player) {
		case 0:
			ImagePlayer1->Visible = false;
			LabelPlayerCash1->Caption = "Банкр.";
			ImageLose->Visible = true;
			ListBoxActions->Visible = false;
			ImagePause->Visible = false;
			return;
		case 1:
			ImagePlayer2->Visible = false;
			LabelPlayerCash2->Caption = "Банкр.";
			break;
		case 2:
			ImagePlayer3->Visible = false;
			LabelPlayerCash3->Caption = "Банкр.";
			break;
		case 3:
			ImagePlayer4->Visible = false;
			LabelPlayerCash4->Caption = "Банкр.";
			break;
	}

	arrOfPlayers[player]->setIsDefaulter(true);

	if (CheckWin())
	{
		ImageWin->Visible = true;
		ListBoxActions->Visible = false;
		ImagePause->Visible = false;
		return;
	}

	ListBoxActions->Items->Insert(0, "Игрок №" + IntToStr(player + 1) + " обанкротился и выбывает из игры");

	isDouble = false;
	UpdateStates();
}
