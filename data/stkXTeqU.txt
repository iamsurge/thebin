#include <iostream>

using namespace std;

const int D=7;

int main()
{
    int a, b, c;
    cout << "Wpisuj liczby calkowite:\n";
    cin >> b >> a;

    bool flaga = false;

    int najmniejsza;

    if(a % D == 0)
    {
        flaga = true;
        najmniejsza = a;
    }

    if(b % D == 0)
    {
        if(flaga && b < najmniejsza)
            najmniejsza = b;
        if(!flaga)
        {
            flaga = true;
            najmniejsza = b;
        }
    }

    do
    {
        c = b;
        b = a;
        cin >> a;

        if(a % D == 0)
        {
            if(flaga && a < najmniejsza)
                najmniejsza = a;
            if(!flaga)
            {
                flaga = true;
                najmniejsza = a;
            }
        }

    } while(a % 2 == 0 || a >= b || a >= c);

    if(flaga)
        cout << "Najmniejsza wpisana liczba podzielna przez " << D << " wynosi " << najmniejsza << ".\n";
    else
        cout << "Nie wpisano zadnej liczby podzielnej przez " << D << ".\n";

    return 0;
}
