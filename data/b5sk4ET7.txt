<?php

function hari($tgl = '')
{
    $hari = date('D', strtotime($tgl)); //ambil nama hari berdasarkan tanggal

    //buat sebuah array untuk nama hari
    $arr_hari = array(
        'Sun' => 'Minggu',
        'Mon' => 'Senin',
        'Tue' => 'Selasa',
        'Wed' => 'Rabu',
        'Thu' => 'Kamis',
        'Fri' => 'Jum\'at',
        'Sat' => 'Sabtu'
    );

    return $arr_hari[$hari];
}

echo 'Hari ini hari ' . hari(date('Y-m-d'));
