a = 0
turtle.refuel()
while a < 8 do
    local b = 0
    while b < 4 do
        turtle.placeUp()
        turtle.forward()
        turtle.turnRight()
        b = b + 1
    end
    turtle.down()
    turtle.forward()
    turtle.forward()
    turtle.forward()
    turtle.forward()
    turtle.forward()
    turtle.forward()
    turtle.up()
    a = a + 1
end
