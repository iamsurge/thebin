clear all;
close all;
%ładowanie sygnału
load('s1.mat');
load('s2.mat');
%parametry sygna;lu
fp=2000;
N=length(s1);

t=[0:1:N-1]/fp;

figure(1)
subplot(2,2,1);
plot(t, s1);
subplot(2,2,2);
plot(t,s2);

Nf=N;

N21=Nf/2+1

f=linspace(0,fp/2,N21);
%widmo amplitudowe
w1=abs(fft(s1,Nf));
w2=abs(fft(s2,Nf));

subplot(223)
stem(f,w1(1:N21));
xlabel('czestotliwosc [Hz]');
ylabel('modół wimda')

subplot(224)
stem(f,w2(1:N21));
xlabel('czestotliwosc [Hz]');
ylabel('modół wimda')
