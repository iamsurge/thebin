sides = {"right","left","top","bottom","back"}
l = peripheral.wrap("left")

for i = 1,5 do
    print(sides[i])
    l.setOutput(sides[i],false)
end
for k = 1,5 do
    l.setOutput(sides[k],false)
    sleep(2)
    l.setOutput(sides[k],true)
    sleep(2)
    l.setOutput(sides[k],false)
end
