#include <iostream>

int main() {
	setlocale(LC_ALL, "rus");// устанавливаем ру язык
	int chisla[10] = {};// создаем массив с 10 елементами
	printf("Введите значения чисел:\n");
	for (int i = 0; i < 10; i++)// записуваем числа
		std::cin >> chisla[i];

	system("CLS");// очищаем консоль

	printf("До:\n");// это шобы сравнить
	for (int i = 0; i < 10; i++)
		printf("%d\t", chisla[i]);
	printf("\n");

	int bigNum = 0, bigEl = 0;// создаем переменные для записи значения элемента и номера (максимальное число)
	int smalNum = 99999999, smalEl = 0;// создаем переменные для записи значения элемента и номера (минимальное число)
	for (int i = 0; i < 10; i++) {
		if (bigNum < chisla[i]) {// проверяем что число больше числа которое в переменной bigNum
			bigEl = i;// записуваем номер ячейки
			bigNum = chisla[i];// записуваем число
		}
		if (smalNum > chisla[i]) {// проверяем что число меньше числа которое в переменной smalNum
			smalEl = i;// записуваем номер ячейки
			smalNum = chisla[i];// записуваем число
		}
	}

	// меняем местами в массиве значения
	chisla[smalEl] = bigNum;
	chisla[bigEl] = smalNum;

	printf("После:\n");// это шобы сравнить
	for (int i = 0; i < 10; i++)
		printf("%d\t", chisla[i]);
	printf("\n");

	system("pause");// останавливаем консоль (для того чтобы не закрылась)
	return 0;
}
