#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n, x;
    cin >> n >> x;
    int niza[n];
    int niza2[n];
    for(int i = 0; i < n; i++)
    {
        cin >> niza[i];
        niza2[i] = niza[i];
    }
    sort(niza, niza + n);
    int s = 0, e = n - 1;
    int o = -1, p = -1;
    while(s < e)
    {
        if(niza[s] + niza[e] == x)
        {
            o = niza[s];
            p = niza[e];
            break;
        }
        else if(niza[s] + niza[e] < x)
        {
            s++;
        }
        else{e--;}
    }
    
    if(o == -1) {
        cout << "IMPOSSIBLE" << endl;
        return 0;
    }
    for(int i = 0; i < n; i++)
    {
        if(o == niza2[i])
        {
            cout << i + 1 << " ";
        }
        else if(p == niza2[i])
        {
            cout << i + 1<< " ";
        }
    }
    return 0;
}
