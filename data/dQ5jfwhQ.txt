def placeCrashPoint(img, x, y, tier, r=16):
	x = (x*k)+Ax + 512
	y = (y*k)+Ay + 512
	ImageDraw.Draw(img, 'RGBA').ellipse((x-r, y-r, x+r, y+r), tierCol[tier])

def placeCrashText(img, crash, fs=12):
	x = (crash['coords'][0]*k)+Ax + 512
	y = (crash['coords'][1]*k)+Ay + 512
	ImageDraw.Draw(img).text(
		(x, y), 
		crash['reqItem'] + str(crash['reqCount']), 
		'black', 
		font = ImageFont.truetype('arial', fs))

def addInverseShadow(img, intencity):
	r,g,b,a = img.split()
	imgRGB = Image.merge('RGB', (r,g,b))
	r,g,b = PIL.ImageOps.invert(imgRGB).split()
	blurred = Image.merge('RGBA', (r,g,b,a))
	return blurred.filter(ImageFilter.BoxBlur(intencity))
