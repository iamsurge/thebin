Program Task2(output);
    var odd, even, val: longint;
begin
    odd := 0;
    even := 0;

    readln(val);

    repeat
        if val mod 2 = 0 then
            even := even + 1
        else
            odd := odd + 1;
        val := val div 10
    until val = 0;
    
    writeln(even);
    writeln(odd)
end.
