import java.util.Scanner;
public class Jurnal01 {
	private static int jurnal01(int x, int y){
		if(y == 1){
		    return x;
		}else{
		    return x * jurnal01(x, y - 1);
		}
    	}

	public static void main(final String[] args) {
		Scanner input= new Scanner(System.in);
		int angka1, banyakPangkatnya;
		angka1 = input.nextInt();
		banyakPangkatnya = input.nextInt();
		System.out.println(jurnal01(angka1, banyakPangkatnya));
	}
}
