#include <iostream>

using namespace std;


int main() {
    int n;
    cin >> n;
    
    float niza[n];
    for(int i = 0; i < n; i++) {
        cin >> niza[i];
    }
    float najgolem = niza[0];
    int pozicija_najgolem = 0;
    
    float najmal = niza[0];
    int pozicija_najmal = niza[0];
    
    for(int i = 0; i < n; i++) {
        if(niza[i] > najgolem) {
            najgolem = niza[i];
            pozicija_najgolem = i;
        }
        
        if(niza[i] < najmal) {
            najmal = niza[i];
            pozicija_najmal = i;
        }
    }
    swap(niza[pozicija_najmal], niza[pozicija_najgolem]);
    
    for(int i = 0; i < n; i++) {
        cout << niza[i] << " ";
    }
 
    return 0;
}
