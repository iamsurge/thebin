#include <iostream>
using namespace std;
void showArray(int width, int height, char M[][70]) {
    for (int i = 0; i < height; i++) {
        for (int j = 0; j < width; j++) {
            cout << M[i][j];
        }
        
        cout << endl;
    }
}

int main()
{
    const char C[71] = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz1234567890 .,-!?()";
    char M[70][70];
    
    int dir = 0;
    int numRow, numCol, topBound, bottomBound, leftBound, rightBound, pointRow, pointCol;
    
    cout << "H W" << endl;
    cin >> numRow >> numCol;
    
    topBound = 1;
    bottomBound = numRow;
    leftBound = 0;
    rightBound = numCol;
    pointRow = 0;
    pointCol = 0;
    
    // Set Array M 
    for (int i = 0; i < 70; i++) {
        // Put the character at the pointer cell
        M[pointRow][pointCol] = C[i];
        
        // Move the pointer
        // dir 0: rightwards
        // dir 1: downwards
        // dir 2: leftwards
        // dir 3: upwards
        // Before moving, check if it is the boundary cell
        // If it is, change direction
        
        if (dir == 0) {
           if (pointCol <= rightBound - 2) {
               pointCol++;
           } else {
               dir = 1;
               pointRow++;
               rightBound--;
           }
        } else if (dir == 1) {
            if (pointRow <= bottomBound - 2) {
                pointRow++;
            } else {
                dir = 2;
                pointCol--;
                bottomBound--;
            }
        } else if (dir == 2) {
            if (pointCol >= leftBound + 1) {
                pointCol--;
            } else {
                dir = 3; 
                pointRow--;
                leftBound++;
            }
        } else if (dir == 3) {
            if (pointRow >= topBound + 1) {
                pointRow--;
            } else {
                dir = 0;
                pointCol++;
                topBound++;
            }
        }
    }
    
    showArray(numCol, numRow, M);
    
    
    return 0;
}
