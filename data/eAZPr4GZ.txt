//  Возвращает:
//  = 0, если дробь неправильная
//  = 1, если правильная	
////////////////////////////////////////////////////  k - это числитель
int checking_the_correct_fraction(int k, int p)   //  p -   знаменатель
{
	
	if(k >= p)  return 0;
	
	
	while (p > 0) 
	{
        int c = k % p;
        k = p;
        p = c;
    }
	
	if(k != 1) return 0; 	
    
return 1;    
}
