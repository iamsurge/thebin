#lista de la compra v2
import os
lista = []
pregunta1 = None
while pregunta1 != "F":
    pregunta1 = input("¿Qué quieres añadir a la lista de la compra?\n[F] Para finalizar\n[M] + Opciones.\nProducto: ")
    if "F" in pregunta1: #sale del programa si se introduce la letra F.
        q = "F"
        print("¡Hasta pronto!.")
    elif "M" in pregunta1:
        opciones1 = input("[V] Ver lista\n[E] Eliminar producto\n[F] Volver: ")
        if "V" in opciones1:
            print(f"Esta es la lista actual:\n{lista}")
        elif "E" in opciones1:
            eliminar1 = None
            while eliminar1 not in lista:
                eliminar1 = input("¿Qué producto quieres eliminar? [F] Salir: ") #pregunta el producto a eliminar
                if eliminar1 in lista:
                    lista.remove(f"{eliminar1}")
                    os.system("cls")
                    print(f"Hemos eliminado {eliminar1}.")
                elif "F" in eliminar1:
                    os.system("cls")
                    print(f"Salimos.")
                    break #fuerza el cierre del while.
                else:
                    print(f"{eliminar1} no está en la lista.") #informa y vuelve a preguntar.
        elif "F" in opciones1:
            pass
    elif "" in pregunta1: #si no se introduce nada, informa y vuelve a repetir la pregunta.
        print("Debes introducir un producto, para salir pulsa [F]")
    else:
        confirmacion1 = None
        while confirmacion1 != "S" and confirmacion1 != "N": #confirma el producto que queremos añadir.
            confirmacion1 = input(f"¿Seguro que deseas añadir '{pregunta1}' a la lista? [S/N].")
            if "S" in confirmacion1:
                lista.append(f"{pregunta1}")
                os.system("cls")
                print(f'Hemos añadido "{pregunta1}" a la lista.')
            elif "N" in confirmacion1:
                print(f'No añadiré "{pregunta1}" a la lista.')
            else:
                print("Debes elegir [S] o [N]")
print(f"Esta es la lista: {lista}") #muestra la lista al finalizar.
