FileInstall("E:\NAJEEB-NEW-SCRIPTS\AUTOIT\FFMPEG\VIDEO-CUTTER.au3", @ScriptDir & "\VIDEO-CUTTER.au3")

FileSetAttrib(@ScriptDir & "\ffmpeg.exe", "+H")

$start = inputbox( "Start Time" , "Enter Capture Start Time in Seconds - 0 is the beginning", "00:00:00")
$length = inputbox ("Length of Capture" , "Enter the desired duration (in seconds) of the output video", "00:00:00")

run("ffmpeg -ss " & $start & " -t " & $length & " -i " & $file & " -c copy Cutoutput.mp4")
