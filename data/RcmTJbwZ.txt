function getList(inTableName, inColumn, searchByValue, returnValues = 'sys_id') {
    const record = new SimpleRecord(inTableName);
    record.selectAttributes([inColumn, returnValues]);
    record.addQuery(inColumn, searchByValue);
    record.query();
    const companyList = [];
    while (record.next()) {
        companyList.push(record.getValue(returnValues))
    }
    return companyList;
}
