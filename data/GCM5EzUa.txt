package ppa1;

import java.util.Scanner;

public class ppa1u062 {
	
	public static Scanner sc;
	public static boolean jePrvocislo(int cislo) {
		boolean prvocislo = false;
		int i = 2;
	    while (i <= cislo / 2) {
	      if (cislo % i == 0) {
	        prvocislo = true;
	        return prvocislo;
	      }
	      ++i;
	    }
	    
	    return prvocislo;
	}
	
	public static int jeSoucetPrvocisel (int cislo) {
		
		for(int i = 2; i <= cislo / 2; i++) {
			if (!jePrvocislo(i) && !jePrvocislo(cislo - i)) {
				int hodnota = cislo - i;
				return hodnota;
			}
		}
		
		return 0;
	}
	
	public static int nactiPrirozeneCislo(Scanner sc) {
		int hranice = 0;
		int x = 0;
		
		while(x <= hranice) {
			System.out.println("Zadej prirozene cislo: ");
			x = sc.nextInt();
		}
		
		return x;
	}
	
	public static void main(String[] args) {
		sc = new Scanner(System.in);
		int n = nactiPrirozeneCislo(sc);
		
		int hodnota = jeSoucetPrvocisel(n);
		int vysledek = n - hodnota;
		if (hodnota > 0) {
			System.out.println("Lze rozlozit: " + vysledek + " + " + hodnota);
		}
		else {
			System.out.println("Nelze rozlozit.");
		}
	}
}
