#include <bits/stdc++.h>
#define int long long
using namespace std;

vector<int> a;

int calc(int n) {
    int res = 0;
    int ms = 1;
    for (size_t i = 2; i < a.size(); ++i) {
        a[i] = a[i - 1] * 3 - ms;
        ms = a[i - 1] - ms;
    }
    res = a[n];

    return res;
}

int32_t main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    a.resize(n + 1);
    a[0] = 0;
    a[1] = 3;
    cout << calc(n) << '\n';

    return 0;
}
