budget = float(input())


left_budget = budget
input_line = input()
while input_line != "ACTION":
    name = str(input_line)
    if len(name) <= 15:
        payment = float(input())
        left_budget -= payment
    elif len(name) > 15:
        payment = left_budget * 0.2
        left_budget -= payment
    if left_budget == 0:
        break

    input_line = input()

diff = abs(left_budget - payment)
if left_budget >= payment:
    print(f"We are left with {diff:.2f} leva.")
else:
    print(f"We need {diff:.2f} leva for our actors.")
