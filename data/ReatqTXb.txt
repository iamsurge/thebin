string s; int n;
vector<string> t; vector<int> id;

bool lencheck(string s, string t){
    if(s.length() == t.length()){
        int cnt = 0;
        for(int i = 0; i < s.length(); i++){
            if(s[i] != t[i]) cnt++;
        }
        return cnt <= 2;
    } else return false;
}



signed main(){
    sync_in_out
    if(fopen(tasks".inp", "r")){
        freopen(tasks".inp", "r", stdin);
        freopen(tasks".out", "w", stdout);
    }
    cin >> s;
    cin >> n;
    t.resize(n+1);
    for(int i = 1; i <= n; i++) {
        cin >> t[i];
        if(lencheck(s, t[i])){
            id.push_back(i);
        }
    }
    if(id.size() >= 1) for(auto &i : id) cout << i << " ";
    else cout << 0;
    return 0;
}
