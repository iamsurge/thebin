bool isDivisible(int a, int b)
{
    if (b == 1)
        return true;
    int gcd = __gcd(a, b);
    if (gcd == 1)
        return false;
    return isDivisible(a, b / gcd);
}
void solve()
{
    int a, b;
    cin >> a >> b;
    if (isDivisible(a, b))
        cout << "YES\n";
    else
        cout << "NO\n";
}

    while (tc--)
    {
        solve();
