n, m = map(int, input().split())
A = list(map(int, input().split()))

start = 0
for i in range(1, m):
    if A[i] != A[i - 1] + 1:
        if i - 1 > start:
            print(A[start], '-', A[i - 1], ',', sep='', end = '')
        else:
            print(A[i - 1], ',', sep='', end = '')
        start = i

if m - 1 > start:
    print(A[start], '-', A[m - 1], sep='', end = '')
else:
    print(A[m - 1], sep='', end = '')
