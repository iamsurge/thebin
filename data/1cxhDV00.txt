import os
from collections.abc import Iterable
def gen_files_path(folder: str, path: str) -> Iterable[str]:
    for i_elem in os.listdir(path):
        try:
            current_path = os.path.join(path, i_elem)
            if os.path.isdir(current_path) and i_elem == folder:
                yield f'Каталог {os.path.join(path, i_elem)} найден.\nСписок файлов этого каталога: {next(os.walk(i_elem)[2])}'
                return
            else:
                yield from gen_files_path(folder, current_path)

        except PermissionError:
            continue

user_folder = input('Введите имя искомого каталога: ')
abs_path = input('Введите рабочую директории: ')
if abs_path == '':
    abs_path = os.path.abspath(os.path.join(os.path.sep))
result = gen_files_path(folder=user_folder, path=abs_path)
for i_path in result:
    print(i_path)
