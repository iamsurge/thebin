#include <iostream>
using namespace std;

string a, b;
int       c;

/////////////////////////////////
int main()                    //
{
    cin >> a >> b;
    
    for(int i = 0; i < a.size(); i++)
    for(int y = 0; y < b.size(); y++)  if(a[i] == b[y]) c++; 
        
    cout << c;   
}
