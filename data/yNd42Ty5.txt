window.__getPFCustomATCData = function(pId) {
const variantEl = document.querySelector(`[data-product-id="${pId}"][data-pf-type="ProductVariantSwatches"]`)
	let variantId
	if(variantEl.querySelector('input')){
		variantId = variantEl.querySelector('input').value
	} else {
		variantId = variantEl.querySelector('select').value
	}
	const selling_plan = __pageflyProducts[pId].variants.find(variant => variant.id === parseInt(variantId))?.selling_plan_allocations[0]?.selling_plan_id
	if(!selling_plan) return {}
	return {selling_plan}
}
