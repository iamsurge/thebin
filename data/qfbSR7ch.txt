IEnumerator WelcomeRoom()
    {

        float A = Input.GetAxisRaw("Horizontal");
        float B = Input.GetAxisRaw("Vertical");

        float step = SpeedMove / (A - B);
        float progress = 0;

        while ((transform.position).sqrMagnitude > 0.01f)
        {
            progress += step * Time.deltaTime;
            //transform.position = Vector3.Lerp(A, B, progress);
            transform.Translate(new Vector3(A, B, 0f) * progress);
            yield return null;
        }

        //transform.position = B;
    }
