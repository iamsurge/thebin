<!-- Basic JavaScript onclick method -->
<h1>Display my name below</h1>
 
<form>
    <label for="fName">Name</label><br>
    <input type="text" name="fName" id="firstName" placeholder="John">
    <br>
    <input type="button" value="Display my Name" onclick="replaceText()">
</form>
 
<p id="output">Your Name will be displayed here.</p>
 
<script>
 
    function replaceText() {
 
        var myName = document.getElementById("firstName").value;
        document.getElementById("output").innerHTML = "Your input Name is: " + myName;
 
    }
 
</script>
