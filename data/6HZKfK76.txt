#include <iostream>
export module Convert;


export int YearsToDays(int x)
{
	return x * 365;
}

export int DaysToHour(int x) // Конвертируем дни в часы
{
	return 24 * x;
}

export int HourToMin(int x)
{
	return 60 * x;
}

export int DaysToMinutes(int x)
{
	return DaysToHour(HourToMin(x));
}

export int MinToSec(int x)
{
	return 60 * x;
}

export int HourToSec(int x)
{
	return HourToMin(MinToSec(x));
}
