    }

//mahane na element
int pop(Element** B){
    Element * temp;
    temp = (*B)->next;
    int d = (*B)->data;
    delete[](*B);
    (*B) = temp;
    return d;
}

struct Element{
    int data;
    struct Element* next;
};

int main() {
    Element* begin = NULL;
    Element* end = NULL;
    
    //purvi element
    Element* temp = new Element;
    temp->data = 1;
    
    temp->next = NULL;
    
    begin = temp;
    end= temp;
    
    //dobavyane na element
    temp = new Element;
    temp->data = 2;
    
    temp->next = NULL;
    end->next = temp;
    end = temp;

    temp = begin->next;
    delete begin;
    begin = temp;
    
    return 0;
}
