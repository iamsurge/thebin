-- Constants --
newMessageEvent = game.ReplicatedStorage.DefaultChatSystemChatEvents.OnMessageDoneFiltering
sayMessageEvent = game.ReplicatedStorage.DefaultChatSystemChatEvents.SayMessageRequest
commandSymbol = ';'

-- Functions --
function getValueByKey(list, keyName)
	for key,value in pairs(list) do
		if string.lower(key) == string.lower(keyName) then
			return value
		end
	end
	for key,value in pairs(list) do
		if string.sub(string.lower(key), 1, #keyName) == string.lower(keyName) then
			return value
		end
	end
end
function sendMessage(message, channel)
	channel = channel or 'All'
	sayMessageEvent:FireServer(message, channel)
end

_G.msgGet = newMessageEvent.OnClientEvent:Connect(function(args, channel)
	local message = args.Message
	if string.sub(message, 1, #commandSymbol) == commandSymbol then
		message = string.sub(message, #commandSymbol+1, #message)
		sendMessage(getValueByKey(messages, message))
	end
end)
