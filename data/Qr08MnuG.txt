from random import choice, randint


class KillError(Exception):
    pass


class DrunkError(Exception):
    pass


class CarCrashError(Exception):
    pass


class GluttonyError(Exception):
    pass


class DepressionError(Exception):
    pass

def one_day():
    day = 0
    total_carma = 0
    while total_carma < 500:
        day += 1
        if randint(1, 10) == 1:
            error = choice([KillError('KillError'), DrunkError('DrunkError'), CarCrashError('CarCrashError'), GluttonyError('GluttonyError'), DepressionError('DepressionError')])
            with open('karma.log', 'a', encoding='utf-8') as file:
                file.write(f'День: {day} Исключение: {error}\n')
            print(f'День: {day} Исключение: {error}')
        else:
            carma = randint(1, 7)
            total_carma += carma
            print(f'День: {day} Выпало кармы: {carma} Всего кармы: {total_carma}')

one_day()
