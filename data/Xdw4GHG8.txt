using System;
 
namespace Walking
{
    class Program
    {
        static void Main(string[] args)
        {
            int totalSteps = 10000; 
            int inputSteps;     
            bool isLoopExit = false;            
 
            do
            {
                string s = Console.ReadLine();
 
                switch (s)
                {
                    case "Going home":
                        isLoopExit = true;
                        inputSteps = int.Parse(Console.ReadLine());
                        totalSteps -= inputSteps;
                        break;
                    default:
                        inputSteps = int.Parse(s);
                        totalSteps -= inputSteps;
                        if (totalSteps <= 0)
                            isLoopExit = true;
                        break;
                }               
            } while (!isLoopExit);
 
            if(totalSteps <= 0)
                Console.WriteLine($"Goal reached! Good job!\n{(-1) * totalSteps} steps over the goal!");
            else
                Console.WriteLine($"{totalSteps} more steps to reach goal.");
        }
    }
}
