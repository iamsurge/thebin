import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt(), m = in.nextInt();
        int[][] arrn = new int[n][m];
        int k = 0;
        boolean res = true;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i % 2 == 0) {
                    arrn[i][j] = k;
                } else {
                    arrn[i][m - 1 - j] = k;
                }
                k++;
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (arrn[i][j] < 10) {
                    System.out.print("  " + arrn[i][j]);
                } else {
                    System.out.print(" " + arrn[i][j]);
                }
            }
            System.out.println("");
        }
    }
}
