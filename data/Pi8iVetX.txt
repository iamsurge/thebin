import os
from collections.abc import Iterable
def gen_files_path(folder: str, path: str) -> Iterable[str]:
    for i_elem in os.listdir(path):
        try:
            current_path = os.path.join(path, i_elem)
            print(current_path)
            print(os.path.join(path, folder))
            print(path)
            print(os.path.isdir(current_path))
            print(current_path == os.path.join(path, folder))
            if os.path.isdir(current_path) and current_path == os.path.join(path, folder):
                return 'Каталог {} найден.\nСписок файлов этого каталога:\n {}'.format(os.path.join(path, i_elem), '\n'.join(os.listdir(i_elem)))
                #raise StopIteration
            else:
                yield from gen_files_path(folder, current_path)
        except (PermissionError, NotADirectoryError, FileNotFoundError):
            continue

user_folder = input('Введите имя искомого каталога: ')
abs_path = input('Введите рабочую директории: ')
if abs_path == '':
    abs_path = os.path.abspath(os.path.join(os.path.sep))
result = gen_files_path(folder=user_folder, path=abs_path)
for i_path in result:
    print(i_path)
