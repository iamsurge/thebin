using System;
using System.Collections.Generic;

namespace ConsoleApp24
{
    class Program
    {
        static void Main(string[] args)
        {
            string urav = args[0];
            char operant = urav[1];
            Console.WriteLine(urav);
            int x = 0;
            List<int> numbers = new List<int>();

            foreach (char sim in urav)
            {
                int tempnumber;

                if (int.TryParse(sim.ToString(),out tempnumber))
                    numbers.Add(tempnumber);
            }

            switch (operant)
            {
                case '-':
                    if (urav[0] == 'x') //x-a=b
                        x = numbers[1] + numbers[0];
                    else if (urav[2] == 'x') //3-x=5
                        x = numbers[0] - numbers[1];
                    else if (urav[4] == 'x') //a-b=x
                        x = numbers[0] - numbers[1];
                    break;
                case '+':
                    if (urav[0] == 'x') //3+a=7
                        x = numbers[1] - numbers[0];
                    else if (urav[2] == 'x') //3+x=5
                        x = numbers[1] - numbers[0];
                    else if (urav[4] == 'x') //a+b=x
                        x = numbers[0] + numbers[1];
                    break;
                default:
                    Console.WriteLine("[0-9,x][+-][0-9,x]=[0-9,x]");
                    break;
            }
            Console.WriteLine(x);
            Console.ReadKey();
        }
    }
}
