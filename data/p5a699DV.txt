class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int maxPossibleProfit = 0;
        int bestBuyPrice = prices[0];

        for(int stockPrice: prices) {
            bestBuyPrice = min(bestBuyPrice, stockPrice);
            maxPossibleProfit = max(maxPossibleProfit, stockPrice - bestBuyPrice);
        }

        return maxPossibleProfit;
    }
};
