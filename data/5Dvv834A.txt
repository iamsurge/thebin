turtle.turnLeft()
while turtle.getItemCount() ~= 64 do
    turtle.suck()
    if turtle.getItemDetail().name ~= "kubejs:substrate_blaze" then
    	turtle.drop()
	end
end
turtle.turnRight()
turtle.drop()
local place = 0
turtle.select(3)
repeat
    turtle.placeUp()
    place = place + 1
until place == 10
turtle.select(1)
turtle.suck()
turtle.suck()
if turtle.getItemDetail().name == "kubejs:substrate_chaos" then
    turtle.drop()
end
if turtle.getItemDetail(2) ~= nil and turtle.getItemDetail(2).name ~= "kubejs:substrate_silicon" then
    turtle.select(2)
    turtle.drop()
    turtle.select(1)
end
os.reboot()
