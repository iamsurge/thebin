
#include <iostream>
using namespace std;

unsigned int a, b, c;

int main()
{
    
   cin >> a;
   
   b = a / 2;
   c = a % 2;
  
   if(c == 0)
   {
      
       for(int r = 0; r <= 100; r+=2)
       {
            if(b == r)
            {
                cout << "YES";
                return 0;
            }
       }
   
   } 
   
   cout << "NO";
}
