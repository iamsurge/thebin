#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
vector<ll>d, m, a;
vector<ll>res, p;
ll sum;
ll bebra=0;
void get(ll n){
  if((ll)p.size() == n){
    sum = 0;
    for(ll i = 0; i<5; i++)sum+=d[p[i]];
    for(ll i = 5; i<8; i++)sum+=m[p[i]];
    for(ll i = 8; i<10; i++)sum+=a[p[i]];
    if(sum>=bebra){
      bebra = sum;
      res = p;
    }
    return;
  }
  for(ll i = 0; i<n; i++){
    ll f = false;
    for(ll j:p){
      if(i == j){
        f = true;
      }
    }
    if(!f){
      p.push_back(i);
      get(n);
      p.pop_back();
    }
  }
}
signed main(){
  ios::sync_with_stdio(0);
  ll n;
  cin>>n;
  d.resize(n);
  m.resize(n);
  a.resize(n);
  for(ll i  = 0; i<n; i++){
    cin>>d[i]>>m[i]>>a[i];
  }
  get(n);
  for(ll i = 0; i<5; i++)cout<<res[i]+1<<" ";
  cout<<"\n";
  for(ll i = 5; i<8; i++)cout<<res[i]+1<<" ";
  cout<<"\n";
  for(ll i = 8; i<10; i++)cout<<res[i]+1<<" ";
}
