"""
agenda telefonica
ejercicio propuesto

Opciones:
1. Agregar contacto
2. Modificar contacto
3. Consultar numero 
4. Salir


# supongamos una persona, un telefono
# agenda = {"Juan P":11225, "Ana G":44555}
"""

agenda = {}

################   funciones   ###########################3##

def menu():
    return """
    Opciones:
    ---------------------
    1. Agregar contacto
    2. Modificar numero
    3. Modificar nombre
    ---------------------
    4. Consultar numero
    5. Consultar nombre
    6. Imprimir agenda
    ---------------------
    7. Eliminar contacto
        
    8. Salir
    """


def agregar_contacto(agenda):
    nombre = input("Ingrese el nombre del contacto: ")
    numero = input("Ingrese el nro de teléfono: ")
    agenda[nombre] = numero
    return "Contacto agregado"

    
    
def modificar_numero(agenda):
    nombre = input("Ingrese el nombre del contacto: ")
    nombres = list(agenda.keys())
    if nombre in nombres:
        numero = input("Ingrese el nuevo nro de teléfono: ")
        agenda[nombre] = numero
        return "Numero modificado"
    else:
        return f"No existe el nombre {nombre}"


def modificar_nombre(agenda):
    numero = input("Ingrese el numero del contacto: ")
    # borro el par nombre-viejo:numero
    borrado = False
    for k,v in agenda.items():
        if v == numero:
            del agenda[k]
            borrado = True
            break
    if borrado:
        nombre = input("Ingrese el nuevo nombre del contacto: ")
        agenda[nombre] = numero
        return "Nombre modificado"
    else:
        return f"No existe el numero {numero}"
    

def consultar_numero(agenda):
    nombre = input("Ingrese el nombre del contacto: ")
    nombres = list(agenda.keys())
    if nombre in nombres:
        return f"Teléfono: {agenda.get(nombre)}"
    else:
        return f"No existe el contacto {nombre}"
            
    
    
def consultar_nombre(agenda):
    numero = input("Ingrese el numero del contacto: ")
    hallado = False
    for k,v in agenda.items():
        if v == numero:
            return f"Nombre: {k}"
            hallado = True
            break
        
    if not hallado:
        return f"No existe el numero {numero}"

    
    
def imprimir(agenda):
    if agenda:
        for k,v in agenda.items():
            print(f"nombre: {k} - telefono: {v}")
    else:
        print("Agenda vacía")


def eliminar_contacto(agenda):
    nombre = input("Ingrese el nombre del contacto: ")
    nombres = list(agenda.keys())
    if nombre in nombres:
        del agenda[nombre]
        return "Contacto eliminado"
    else:
        return f"No existe el contacto {nombre}"
    


    
##################################################################    
        
print("Agenda telefónica")

while True:
    print(menu())
    
    opcion = input("Ingrese su opción: ")
    
    if opcion == "1":
        print(agregar_contacto(agenda))
        
            
    elif opcion == "2":
        print(modificar_numero(agenda))
            
            
    elif opcion == "3":
        print(modificar_nombre(agenda))
        
           
    elif opcion == "4":
        print(consultar_numero(agenda))
    
        
    elif opcion == "5":
        print(consultar_nombre(agenda))
        
        
    elif opcion == "6":
        imprimir(agenda)
                
            
    elif opcion == "7":
        print(eliminar_contacto(agenda))
        
         
    
    elif opcion == "8":
        print("Adiós...")
        break
    
    
    else:
        print("Opción incorrecta...")
