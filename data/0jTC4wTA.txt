task.wait(5) -- to compensate for the server doing a million things on startup and being slow

local function test(func)
	local iterations = 1000000
	local sum = 0
	for i = 1, 1000000 do
		local startTime = tick()
		func()
		local endTime = tick()
		sum += endTime - startTime
	end
	
	return sum / iterations
end

print("Four args average:", test(function()
	UDim2.new(0, 0, 0, 0)
end))

task.wait()

print("Two args average:", test(function()
	-- There are 3 constructors being called here!
	UDim2.new(UDim.new(), UDim2.new())
end))

task.wait()

print("No args average:", test(function()
	UDim2.new()
end))
