#define isnot 0=

open "input3.txt" for input as #1

dim as string sLine
dim as integer iSum

while isnot eof(1)
  line input #1, sLine
  var iLen = len(sLine)
  if iLen = 0 then continue while
  
  dim as byte b(127)
  for N as long = 0 to (iLen\2)-1
    b(sLine[N]) or= 1
    b(sLine[iLen-(N+1)]) or= 2
  next N      
  
  for C as long = asc("A") to asc("z")
    if b(C) = 3 then
      var iVal = iif(C>asc("Z"),C-asc("a")+1,C-asc("A")+27)      
      iSum += iVal
    end if
  next C
  
wend
close #1

print iSum
sleep
