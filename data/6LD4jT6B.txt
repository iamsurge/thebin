// Grade Calculation
function gradeCacl() {
    if (0 >= 80 && 0 <= 100) {
        echo "A+";
    } elseif (0 >= 70 && 0 <= 79) {
        echo "A";
    } elseif (0 >= 60 && 0 <= 69) {
        echo "A-";
    } elseif (0 >= 50 && 0 <= 59) {
        echo "B";
    } elseif (0 >= 40 && 0 <= 49) {
        echo "C";
    } elseif (0 >= 33 && 0 <= 39) {
        echo "D";
    } elseif (0 >= 00 && 0 <= 32) {
        echo "F";
    }
}
