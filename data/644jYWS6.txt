#include<fstream>
#include<algorithm>
#include<vector>
#include<cstring>
#define N 100005
#define MOD 100003
using namespace std;
ifstream fin ("coborare.in");
ofstream fout ("coborare.out");

int n,m,v,c;
vector<int> g[N];
int dp[N];
int DFS(int nod)
{
    if(nod==c) return 1;
    if(dp[nod]) return dp[nod];
    int d=0;
    for(auto i:g[nod])
        d=(d+DFS(i))%MOD;
    return dp[nod]=d;
}
int main()
{
   fin>>n>>m>>v>>c;
   while(m--)
   {
       int x,y;
       fin>>x>>y;
       g[x].push_back(y);
   }
   int sol=DFS(v);
   fout<<sol<<"\n";
   return 0;
}
