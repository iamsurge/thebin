-- Set up the HTTP server and specify the port to listen on
local server = http.createServer(80)

-- Handle incoming requests and send back a response
server:on("request", function(request, response)
  -- Print the request data to the console for debugging
  print(request)

  -- Set the response status code and header
  response:setStatusCode(200)
  response:setHeader("Content-Type", "text/html")

  -- Set the response body to a message
  response:setBody("<h1>Hello, World!</h1>")

  -- Send the response back to the client
  response:send()
end)

-- Start the server
server:listen()
