#include <stdio.h>
#include <stdlib.h>

void funkcija(int A, int B) {
    for(int i = A; i <= B; i++) {
        int broj_na_deliteli = 0;
        for(int j = 1; j <= i; j++) {
            if(i % j == 0) {
                broj_na_deliteli++;
            }
        }
        if(broj_na_deliteli == 2) {
            printf("%d ", i);
        }
    }
}

int main() {
    int A, B;
    scanf("%d%d", &A, &B);
    funkcija(A, B);
    
    
    return 0;
}
