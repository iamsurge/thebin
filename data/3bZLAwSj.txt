def find_circle_A(B):
    n = len(B)
    tot_sum = 0
    for i in B:
        tot_sum += i
    tot_sum= tot_sum//3
    A = [0 for i in range(n)]
    if n%3 == 1:
        temp_sum = 0
        for i in range(2,n,3):
            temp_sum += B[i]
        A[0] = temp_sum-tot_sum
        temp_sum = 0
        for i in range(3,n,3):
            temp_sum+=B[i]
        A[1] = temp_sum-tot_sum
    if n%3 == 2:
        temp_sum = 0
        for i in range(1,n,3):
            temp_sum+=B[i]
        A[0] = temp_sum-tot_sum
        temp_sum = 0
        for i in range(5,n,3):
            temp_sum+=B[i]
        A[1] = B[0]+B[2]+temp_sum-tot_sum
    for i in range(2, n):
        A[i] = B[i-1]-A[i-1]-A[i-2]
    
    return A
    
    
print(find_circle_A([179,129,62,144,182]))
