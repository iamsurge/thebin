#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>

int main()
{
    FILE* fin, * fout;
    if ((fin = fopen("input.txt", "rt")) == NULL)
        exit(1);
    if ((fout = fopen("output.txt", "wt")) == NULL)
        exit(1);

    char N, K;
    fscanf(fin, "%hhd %hhd", &N, &K);
    char tmp = fgetc(fin);
    char commands[257];
    fgets(commands, 256, fin);
    short int str_len = strlen(commands);
    commands[str_len-1] = 0;

    char* cur_command = NULL;
    cur_command = strtok(commands, " ");
    while (cur_command != NULL) {
        if (strcmp(cur_command, "up") == 0) {
            if (K < N) {
                K++;
            }
        }
        else if (strcmp(cur_command, "down") == 0) {
            if (K > 1) {
                K--;
            }
        }
        cur_command = strtok(NULL, " ");
    }
    fprintf(fout, "%hhd\n", K);
    fclose(fin);
    fclose(fout);
    return 0;
}
