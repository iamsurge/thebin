#include <stdio.h>


void rekurzija(int x) {
    if(x > 10) {
        return;
    }
    printf("%d ", x);
    rekurzija(x + 1);
}
int main() {
    rekurzija(1);
    return 0;
}
/*
 
 rekurzija(1) --> rekurzija(2)
 rekurzija(2) --> rekurzija(3)
 rekurzija(3) --> rekurzija(4)
 
 */
