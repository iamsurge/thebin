#include "CellUnit.h"

#pragma package(smart_init)

Cell::Cell(Player* newOwner, const int newCost, const int newIncome, const Position newPosition, const String newName, const int num) : owner(newOwner)
{
    owner = newOwner;
	cost = newCost;
	income = newIncome;
	position = newPosition;
	name = newName;
	number = num;
}

Player* Cell::getOwner()
{
	return owner;
}

void Cell::setOwner(Player* newOwner)
{
	owner = newOwner;
}

int Cell::getCost()
{
	return cost;
}

void Cell::setCost(const int newCost)
{
	cost = newCost;
}

int Cell::getIncome()
{
	return income;
}

void Cell::setIncome(const int newIncome)
{
	income = newIncome;
}

Position Cell::getPositions()
{
	return position;
}

void Cell::setPositions(const Position newPosition)
{
	position = newPosition;
}

String Cell::getName()
{
	return name;
}

void Cell::setName(const String newName)
{
	name = newName;
}

int Cell::getNumber()
{
    return number;
}

void Cell::setShape(TShape* newShape)
{
	shape = newShape;
}

void Cell::setShapeColor(TColor color)
{
	shape->Brush->Color = color;
}
