# como usar un bucle indefinido
# Script que pide un nro entero y muestra su cuadrado por pantalla. La carga 
# de numeros debe concluir cuando se ingresa un cero

while True:
    numero = int(input("Ingrese un entero: "))
    if numero == 0:
        break
    print(f"Su cuadrado es {numero**2}")
