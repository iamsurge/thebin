public String repeatFront_1(String str, int n) {
  StringBuilder result = new StringBuilder();
  
  for(int currLength = n; currLength >= 1; currLength--) {
    result.append(str.substring(0, currLength));
  }
  
  return result.toString();
}

public String repeatFront_2(String str, int n) {
  StringBuilder result = new StringBuilder();
  StringBuilder partial = new StringBuilder(str);
  partial.setLength(n);
  while(n > 0) {
    result.append(partial);
    n--;
    partial.setLength(n);
  }

  return result.toString();
}

public String repeatFront(String str, int n) {
  StringBuilder result = new StringBuilder();
  StringBuilder partial = new StringBuilder(str);
  partial.setLength(n);
  while(partial.length() > 0) {
    result.append(partial);
    partial.setLength(partial.length() - 1);
  }

  return result.toString();
}
