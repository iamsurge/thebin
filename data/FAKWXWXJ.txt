public static void main(String[] args) throws IOException {
        cdfilm film = new cdfilm();
        
        System.out.println("==========");
        System.out.println("Masukkan Inputan Anda");
        System.out.println("===========");
        
        
        //disini kita rekam masukkan dari keyboard user
        
        Scanner inputan= new Scanner(System.in);
        Scanner inputan1 = new Scanner(System.in);
        
        System.out.println("Masukkan Jumlah Data");
        int jumlahData;
        char again;
        jumlahData= inputan1.nextInt();
        int i=0;
        //looping
        do{
            do{
            System.out.println("Anda Memilih CD FILM");
            System.out.println("Masukkan Judul Film");

            film.judul =  inputan.nextLine();
            System.out.println("Masukkan Nama Pemain");
            film.pemain = inputan.nextLine();
            System.out.println("Masukkan Sutradara");
            film.sutradara= inputan.nextLine();
            System.out.println("Masukkan Publisher");
            film.publisher = inputan.nextLine();
            
            
            System.out.println("Kategory:");
            System.out.println("1. Semua Umur");
            System.out.println("2. Remaja");
            System.out.println("3. Dewasa");
            System.out.println("4. Anak-anak");
            System.out.println("Masukkan Pilihan: ");
            int category;
            category= inputan1.nextInt();
            switch(category){
                case 1:
                    film.category = "Semua Umur";
                    break;
                case 2:
                    film.category = "Remaja";
                    break;
                case 3: 
                    film.category = "Dewasa";
                    break;
                case 4:
                    film.category = "Anak-anak";
                    break;
                default:
                    System.out.print("Pilihan Snda Tidak Sesuai");
            }
            
            
            System.out.println("Stok: ");
            film.stok = inputan.nextInt();
           
                System.out.println("Judul Film     :" + film.judul);
                System.out.println("Pemain Film    :" + film.pemain);
                System.out.println("Sutradara Film :" + film.sutradara);            
                System.out.println("Publisher Film :" + film.publisher);            
                System.out.println("Category Film  :" + film.category);
                System.out.println("Stok Film      :" + film.stok); 
                
                System.out.println("Anda Telah Memasukkan Data Sebanyak: "+(i++ + 1));
                System.out.println("Masukkan Data Lagi?");
                again = inputan1.next().charAt(0);
            }while(again == 'Y');           
           
            if(again == 'N'){
                clearScreen();
                System.exit(0);
            }
             
        }while(i<=jumlahData);
    }
    
    public static void clearScreen(){
        System.out.println("\f");
        System.out.print("Everything on the console will cleared");
        System.out.print("\033[H\033[2J");
        System.out.flush();
    }
