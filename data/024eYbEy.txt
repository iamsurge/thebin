#include <iostream>
#include <string>
using namespace std;

bool charIsAlp(char s1) {
    string alp = "aeiouyAEIOUY";
    for (int j = 0; j < alp.length(); ++j) {
        if (s1 == alp[j])
            return true;
    }
    return false;
}

int main() {

    string s1;
    s1 = "babcakomul";
    string result = "";
    int count = 0, k = 0;
    for (int i = 0; i < s1.length(); ++i) {
        if (charIsAlp(s1[i]))
            count++;
    }
    for (int i = 0; i < s1.length(); ++i) {
        bool f = charIsAlp(s1[i]);
        result += s1[i];
        if (i+1 != s1.length() && f && k < count - 1 && s1[i+1] != ' ') {
            result += '-';
            ++k;
        }
    }
    result += " ";
    cout << result;
    return 0;
}
