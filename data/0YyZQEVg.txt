import pyautogui as pg
import time

wait_time = 5 #seconds

print("what do you want to spam?")
msg = input(">> ")

print("How many times?")
times = int(input(">> "))

print("waiting "+str(wait_time)+" seconds...")
time.sleep(wait_time)

for i in range(times):
	pg.write(msg)
	pg.press("enter")
