# FUNCIONES PROPIAS CON ARGUMENTOS DE LONG. VARIABLE

def sumar(*args):
    total = 0
    for n in args:
        total += n
    return total
    
    
print(sumar(1,2,3))




def sumar2(**kwargs):
    total = 0
    for n in kwargs:
        total += kwargs[n]
    return total

print(sumar2(a=12, b=152, c=140, d=100, e=14, f=-418))

# orden para usar dif argumentos
def f(a, b, *args, c=999, **kwargs):
    print(a)
    print(b)
    print(args)
    print(c)
    print(kwargs)
    
    
f(1,2,100,200,300,400,c=4,x=23,y=2,z=21)
