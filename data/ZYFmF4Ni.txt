package general;

public class Utils {
    public static String linkedListToString(Node front) {
        Node tmp = front.next;
        StringBuilder resStr = new StringBuilder("[");
        while (tmp != null) {
            resStr.append(tmp.getData().toString());
            resStr.append(" ");
            tmp = tmp.next;
        }
        resStr.append("]");
        return resStr.toString();
    }

    public static String DoubleLinkedToString(Node head, Node tail) {
        Node tmp = head.next;
        StringBuilder resStr = new StringBuilder("[");
        while (tmp != tail) {
            resStr.append(tmp.getData().toString());
            resStr.append(" ");
            tmp = tmp.next;
        }
        resStr.append("]");
        return resStr.toString();
    }
}
