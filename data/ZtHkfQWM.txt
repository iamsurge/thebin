#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main() {
    int n, score, hs = 0, as = 0, totaltime = 1920, minute, seconds;
    char team, points;
    cin >> n;
    for(int i = 0; i < n; i++) {
        cin >> team >> score >> minute >> points >> seconds;
    }
    for(int i = 0; i < n; i++) {
        if(team == 'H') {
            hs+= score;
        } else {
            as+= score;
        }
    }
    if(hs > as) {
        cout << "H" << " ";
    } else {
        cout << "A" << " ";
    }
    return 0;
}
