#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    int T;
    cin >> T;

    while(T--)
    {
        int N;
        cin >> N;

        vector<unsigned long long> Vec(N); //po burz nachin ot push.back()
        for(int i=0;i<N;i++) {
            cin >> Vec[i];
        }

        sort(Vec.begin(), Vec.end());


        if(N%2==0) {
            cout << min(Vec[N/2-1],Vec[N/2]) << endl;
        }
        else {
            cout << Vec[N/2] << endl;
        }
    }


    return 0;
}
