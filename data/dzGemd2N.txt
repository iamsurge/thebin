local r = peripheral.wrap("BiggerReactors_Reactor_0")
local capacity = r.battery().capacity()
local storedEnergy
local rodLevel

write("Starting up...")
sleep(0.5)
 
while(true) do
    storedEnergy = r.battery().stored()
    rodLevel = (storedEnergy/capacity) * 100
    r.setAllControlRodLevels(rodLevel)
    term.setCursorPos(1, 1)
    write("Rod Level: " .. rodLevel)
    term.setCursorPos(1, 3)
    write("RF Out:    " .. r.battery().producedLastTick())
	term.setCursorPos(1, 5)
    write("Burned:    " .. r.fuelTank().burnedLastTick() / 1000)
    sleep(0.5)
end
