LOW_FUEL_LEVEL_LIMIT = 160
REFUEL_LEVEL = LOW_FUEL_LEVEL_LIMIT * 2

function check_fuel_level()
    if turtle.getFuelLevel() < LOW_FUEL_LEVEL_LIMIT then

        turtle.select(1)
        fuelcheck = turtle.getFuelLevel()
        print("Fuel Level: "..fuelcheck)        
        print("Please Refill")
        while turtle.getFuelLevel() < REFUEL_LEVEL
        do
            turtle.refuel()
        end
    end
    term.clear()
    term.setCursorPos(1,1) 
end
