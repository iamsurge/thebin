function everest(input) {

    let index = 0;
    let command = input[index];
    index++;
    let days = 1;
    let meeters = 5364;

    while (command !== `END`) {
        if (command === `Yes`) {
            days++;

            if (days > 5) {
                break;
            }
        }

        let dailyMeeters = Number(input[index]);
        index++;
        meeters += dailyMeeters;

        if (meeters >= 8848) {
            break;
        }

        command = (input[index]);
        index++;
    }

    if (meeters >= 8848) {
        console.log(`Goal reached for ${days} days!`);
    } else {
        console.log(`Failed!`);
        console.log(`${meeters}`);
    }

}
everest(["Yes", "1254", "Yes", "1402", "No", "250", "Yes", "635"])
console.log("-------------------------------------------");
everest((["Yes", "1000", "Yes", "945", "No", "1200", "END"]))
console.log("-------------------------------------------");
everest(["Yes", "535", "Yes", "849", "Yes", "499", "Yes", "400", "Yes", "500"])
console.log("-------------------------------------------");
everest(["Yes", "700", "END"])
