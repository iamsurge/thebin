//Matrix

#include <iostream>

using namespace std;
 
int main()
{
    int ny, nx;
    cin>>ny;
    cin>>nx;
    int tab[ny][nx];
    
    for(int i = 0; i < ny; i++)
    {
        for(int j = 0; j < nx; j++)
        {
            cin>>tab[i][j];
        }
    }
    
    int min = 10000000;
    int max=0;
   
    
    //ponizej przek
    for(int x=0; x<nx;x++)
    {
        int y=ny-1;
        int suma=0;
        int tempx=x;
       for(y=ny-1;y>=0;y--)
       {
           suma+=tab[y][tempx];
           if(suma>max)
           max=suma;
           if(suma<min)
           min=suma;
            
          if(tempx<=0 || y<=0)
          {
            break; 
          }
          else
          {
             tempx--;
          }
       }
    }
    for(int x=nx-ny+1; x<nx; x++)
    {
        int suma=0;
        int tempx=x;

        for(int y=0;y<ny;y++)
       {
            suma+=tab[y][tempx]; 
            if(suma>max)
            max=suma;
            if(suma<min)
            min=suma;
           if(tempx>=nx-1 || y>=ny)
           {
               break;
           }
           else
           {
               tempx++;
           }
       }

    }
    
    cout<<endl<<max<<" "<<min;
    return 0;
}
