def cubo():
    """Función que calcula el cubo de un nro"""
    y = ingresar_numero()
    return y**3
    
    
    
def ingresar_numero():
    """Funciòn que valida el ingreso de un nro"""
    while True:
        try:
            n = float(input("Ingrese un nro: "))
        except ValueError:
            print("No ha ingresado un nro")
        else:
            return n
            
if __name__ == "__main__":
    x = cubo()
    print(f"El cubo es {x:.6f}")
