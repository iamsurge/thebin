#include <iostream>

using namespace std;

int main()
{
    int a, b;
    cout << "Podaj dwie liczby naturalne: ";
    cin >> a >> b;
    if(a * b == 0)
    {
        cout << "Liczby te nie maja wspolnego dzielnika ani wspolnej wielokrotnosci.\n";
    }

    int x = a;
    int y = b;

    while(a != b)
    {
        if(a > b)
            a -= b;
        if(b > a)
            b -= a;
    }
    cout << "Najwiekszy wspolny dzielnik liczb " << x << " i " << y << " wynosi " << a << endl;
    cout << "Najmniejsza wspolna wielokrotnosc tych liczb wynosi " << x * y / a << endl;
    return 0;
}
