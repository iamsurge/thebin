#include <iostream>

using namespace std;

int main()
{
    cout << "Podaj liczbe naturalna: ";
    int x;
    cin >> x;

    int suma = 1;

    int dzielnik;

    for(dzielnik = 2; dzielnik * dzielnik < x; ++dzielnik)
    {
        if(x % dzielnik == 0)
            suma += (dzielnik + x / dzielnik);
    }
    if(dzielnik * dzielnik == x)
        suma += dzielnik;

    cout << "Suma dzielnikow wlasciwych liczby " << x << " wynosi " << suma << endl;
    return 0;
}
