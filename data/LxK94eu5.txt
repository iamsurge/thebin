using System;
using static System.Console;

namespace Exercise
{
    class Website
    {
        static int Id = 0;
        string name, link, ip, description;

        public string Name { get { return name; } set { name = value; } }
        public string Link { get { return link; } set { link = value; } }
        public string Ip { get { return ip; } set { ip = value; } }
        public string Description { get { return description; } set { description = value; } }

        public void Print()
        {
            WriteLine($"ID:{++Id} Название: {Name} Ссылка: {Link} IP: {Ip} Описание: {Description}");
        }
        public Website(string name, string link, string ip, string description)
        {
            Name = name;
            Link = link;
            Ip = ip;
            Description = description;
        }
        static void Main()
        {
            Website Ranepa = new Website("ЗФ РАНХиГС", "https://zf.ranepa.ru/", "185.71.67.123", "сайт академии");
            Ranepa.Print();
            Website Vk = new Website("Oops", "https://vk.com/", "87.240.137.164", "социальная сеть") { name = "Вконтакте" };
            Vk.Print();
        }
    }
}
