#include <iostream>
using namespace std;
 
long long a, b, c[1000], w;
 
int main()
{
    cin >> a >> b;
    
    for(int i = 0; i < a; i++) cin >> c[i];
    for(int i = 0; i < a; i++) 
    {
             if(c[i] <=  b) w++;
        else if(c[i] >= b) w+=2;
    }
    
    cout << w;
}
