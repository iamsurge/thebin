const convertAsNumbers = bills => {
  return bills.map((bill) => {
    const [month, ...rest] = bill;
    return [month, ...rest.map(Number)]
  })
}
 
const convertAsNumbers2 = bills => { 
  return bills.reduce((acc, bill) => {
    const converted = bill.map((item) => {
      const parsed = +item;
      return !isNaN(parsed) ? parsed : item;
    })
 
    return [...acc, converted]
  }, [])
}
