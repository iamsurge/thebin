#include <iostream>

using namespace std;

int main()
{
    double sum = 0;
    double n;
    int i;
    
    cout << "Enter the number of terms: " << endl;
    cin >> n;
    
    for (i = 1; i <= n; i = i + 2) // нечётные
    {
    sum = sum + (1.0 / i);
    }
    
    for (i = 2; i <= n; i = i + 2) // чётные
    {
    sum = sum - (1.0 / i);
    }

    cout << "The function equals: " << sum;
    
    return 0;
}
