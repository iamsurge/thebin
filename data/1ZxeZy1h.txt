#include <iostream>

using namespace std;

int ilosc_liczb();
int suma_liczb(int ile);

int main()
{
  cout<<suma_liczb(ilosc_liczb());
}

int ilosc_liczb()
{
    int max_liczb;
    cout<<"Podaj ilosc liczb: ";
    cin>>max_liczb;
    return max_liczb;
}
int suma_liczb(int ile)
{
    int suma = 0;
    for (int liczba = 0; liczba <= ile; liczba++)
    {
        if(liczba % 2 != 0)
        {
            suma += liczba;
        }
    }
     return suma;
}
