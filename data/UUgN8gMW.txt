import java.util.Random;


public class Main {

    public static void main(String[] args){

       Random randomNumber = new Random();
        int scoreCpu = 0;
        int scorePlayer = 0;



       for (int i = 0; i < 10; i++)
       {
           int randomDiceCpu = randomNumber.nextInt(6) + 1;
           int randomDicePlayer = randomNumber.nextInt(6) + 1;

           System.out.println("CPU dice: " + randomDiceCpu);
           System.out.println("Player dice: " + randomDicePlayer);

           if (randomDiceCpu > randomDicePlayer)
           {
               scoreCpu++;
               System.out.println("CPU wins");
               System.out.println(" ");
           }
           else if (randomDiceCpu == randomDicePlayer)
           {
               System.out.println("Tie, no one wins");
               System.out.println(" ");
           }
           else
           {
               scorePlayer++;
               System.out.println("Player wins");
               System.out.println(" ");
           }

       }

        System.out.println(" ");

        if (scoreCpu > scorePlayer)
        {
            System.out.println("CPU's a winner with score: " + scoreCpu);
        }
        else
        {
            System.out.println("Player's a winner with score: " + scorePlayer);
        }
    }
}
