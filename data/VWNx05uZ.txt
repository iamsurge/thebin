#include <iostream>

using namespace std;

class Lampka
{
    // pola:
    string kolor;
    int moc;
    bool wlaczona;
    // działania
public:
    // konstruktor:
    Lampka(string k, int power, bool onoff);
    // pozostałe metody:
    void wlacz();
    void wylacz();
    void wypisz_stan();
    // metody dostępowe:
    void ustaw_kolor(string k);
    void ustaw_moc(int power);
    void czy_wlaczona(bool onoff);
};

Lampka::Lampka(string k, int power, bool onoff) : kolor(k), moc(power), wlaczona(onoff) {}

void Lampka::wlacz()
{
    wlaczona = true;
}

void Lampka::wylacz()
{
    wlaczona = false;
}

void Lampka::wypisz_stan()
{
    cout << "Lampka koloru " << kolor << " o mocy " << moc << "W jest obecnie ";
    if(wlaczona)
        cout << "wlaczona.\n";
    else
        cout << "wylaczona.\n";
}

void Lampka::ustaw_kolor(string k)
{
    kolor = k;
}

void Lampka::ustaw_moc(int power)
{
    moc = power;
}

void Lampka::czy_wlaczona(bool onoff)
{
    wlaczona = onoff;
}

int main()
{
    Lampka lampka1("zielony", 150, false);
    lampka1.wypisz_stan();
    lampka1.ustaw_kolor("czerwony");
    lampka1.ustaw_moc(100);
    lampka1.czy_wlaczona(false);
    lampka1.wypisz_stan();
    lampka1.wlacz();
    lampka1.wypisz_stan();
    return 0;
}
