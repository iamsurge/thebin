ll n;
    cin >> n;
    ll a[n], x = 0, sumO = 0, sumE = 0;
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
        if (i >= 0)
            x += a[i];
    }
    for (int i = 1; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (a[i] <= a[j])
                swap(a[i], a[j]);
        }
    }
    if (x % 2 == 0)
        sumE == x;
    else
        sumO == x;
