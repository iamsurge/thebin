

m = peripheral.find("modem")

function listen(channel)
    m.open(channel)
    _,_,_,_,data,_ = os.pullEvent("modem_message")
    m.close(channel)
    return data
end

while true do
	data = listen(4)
	if data.done then
		term.clear()
		term.write("MINING DONE. Enter Y to CONTINUE")
		r = read()
		if string.match(r, "y") then
			transmit(3, {continue=true})
			os.sleep(60)
		end
	else
		term.clear()
		term.write("Hole"..data.hole.." of 51")
	end
end
