chest = peripheral.wrap("top")
orb = peripheral.wrap("left")

while(true)
do
    if(chest.getItemDetail(1) == nil and orb.getItemDetail(1) == nil and orb.getItemDetail(2) == nil)
    then
        break
    else
        if (orb.getItemDetail(2) == nil and orb.getItemDetail(1) == nil)
        then
            orb.pullItems(peripheral.getName(chest), 1, 1)
            orb.pullItems(peripheral.getName(chest), 2, 1)
        else
            if (orb.getItemDetail(1) ~= nil)
            then
                if (orb.getItemDetail(1).name == "powah:steel_energized")
                then
                    chest.pullItems(peripheral.getName(orb), 1)
                else
                    if(orb.getItemDetail(2) < 1 and orb.getItemDetail(3) < 1)
                    then
                        orb.pullItems(peripheral.getName(chest), 1, 1)
                        orb.pullItems(peripheral.getName(chest), 2, 1)
                    end
                end
            end
        end
    end
end
