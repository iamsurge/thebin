#include <stdio.h>
#include <stdlib.h>

int main() {
    int N;
    scanf("%d", &N);
    
    int cetvrta_cifra = N % 10;
    int treta_cifra = (N / 10) % 10;
    int vtora_cifra = (N / 100) % 10;
    int prva_cifra = N / 1000;
    
    int brojac = 0;
    if(prva_cifra == 5) {
        prva_cifra = 6;
        brojac++;
    }
    if(vtora_cifra == 5) {
        vtora_cifra = 6;
        brojac++;
    }
    if(treta_cifra == 5) {
        treta_cifra = 6;
        brojac++;
    }
    if(cetvrta_cifra == 5) {
        cetvrta_cifra = 6;
        brojac++;
    }
    
    if(brojac < 2) {
        printf("ERROR");
        return 0;
    }
    int kolega = 1000 * prva_cifra + 100 * vtora_cifra + 10 * treta_cifra + cetvrta_cifra;
    printf("%d\n", kolega);
    
    float procent = (100 * kolega - 100 * N) / N;
    printf("%f%%\n", procent);
    return 0;
}
