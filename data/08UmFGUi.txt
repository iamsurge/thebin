local block, data = turtle.inspect()
if block then
    if data.name == "minecraft:acacia_slab" then
        turtle.turnRight()
    end
end
local stop = false
while stop ~= true do
    local blockd, datad = turtle.inspectDown()
    if blockd then
        turtle.digDown()
    end
    turtle.forward()
    local block, data = turtle.inspect()
    if block then
        if data.name == "minecraft:chest" then
            for i = 1,16 do
                turtle.select(i)
                local start = turtle.drop()
                if start == false then
                    stop = true
                    break
                end
            end
            if stop == true then break end
            turtle.turnRight()
            turtle.turnRight()
        elseif data.name == "minecraft:acacia_planks" then
            turtle.turnRight()
            turtle.turnRight()
        end
    end
end
