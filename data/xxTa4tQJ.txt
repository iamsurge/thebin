# Proyecto integrador
"""
Etapa 1
Una universidad desea crear un programa para contabilizar los cursos que
tiene cada alumno. Para ello se debe realizar primero una aplicación
de consola la cual debe solicitar el nombre de un alumno y la cantidad de
cursos en la que se encuentra inscripto. Estos dos valores deben almacenarse 
como una lista de dos elementos (el nombre y la cantidad de cursos como 
un número entero) en una lista de alumnos.

Una vez hecho esto, se debe hacer que el programa, al iniciar, pregunte 
cuál de las siguientes dos operaciones se debe realizar: ingresar un
alumno o ver la lista de alumnos ingresados

Esto debe preguntarse infinitamente hasta que el usuario escriba “3”. 

Ingrese el número de la operación que desea ejecutar:
1 - Ver la lista de alumnos.
2 - Añadir un alumno a la lista.
3 - Salir.
>>> 2
Ingrese el nombre del alumno: Pablo
Ingrese la cantidad de cursos: 3
¡El alumno fue añadido a la lista!
Ingrese el número de la operación que desea ejecutar:
1 - Ver la lista de alumnos.
2 - Añadir un alumno a la lista.
3 - Salir.
>>> 1
Lista de alumnos:
Pablo - 3 cursos

Ingrese el número de la operación que desea ejecutar:
1 - Ver la lista de alumnos.
2 - Añadir un alumno a la lista.
3 - Salir.
>>> 4
La opción ingresada no es correcta, vuelva a
intentarlo.
Ingrese el número de la operación que desea ejecutar:
1 - Ver la lista de alumnos.
2 - Añadir un alumno a la lista.
3 - Salir.
>>> 3
¡Gracias por utilizar el programa!

"""
# modelo de datos:
# alumnos = [ ["Juan",3], ["Andrea",4], ["Roque",5] ]

alumnos = []

print("Administración de alumnos")

while True:
    
    print("""
                  Menu
    --------------------------------
    1 - Ver la lista de alumnos.
    2 - Añadir un alumno a la lista.
    3 - Salir.
    --------------------------------
    """)
    opcion = input("Ingrese el número de la operación que desea ejecutar: ")
    
    if opcion == "1":
        if alumnos:
            print("Lista de alumnos:")
            for alumno in alumnos:
                print(f"{alumno[0]} - {alumno[1]} cursos")
        else:
            print("No hay alumnos inscriptos")
        
    elif opcion == "2":
        nombre = input("Ingrese el nombre del alumno: ")
        while True:
            cursos = input("Ingrese la cantidad de cursos: ")
            if cursos.isdecimal() and int(cursos) != 0:
                cursos = int(cursos)
                break
            print("Error, debe ingresar un numero entero mayor a cero")
        
        alumno = [nombre, cursos]
        alumnos.append(alumno)
        print("¡El alumno fue añadido a la lista!")
        
    elif opcion == "3":
        print("¡Gracias por utilizar el programa!")
        break
        
    else:
        print("Opción incorrecta")
