// PIR Demo Sketch

const int OPin = 7;		// Pin 7 is OUTPUT
const int IPin = 2;		// Pin 2 is INPUT

int i;					// state of sensor

void setup()
{
  pinMode(OPin,OUTPUT);		// initialises digital pins for
  pinMode(IPin,INPUT);		// output & input
}

void loop()
{
  i = digitalRead(IPin);	// checks for movement on Pin 2

  if (i==1)			// movement detected, state = 1
  {
    digitalWrite(OPin,HIGH);	// turns on LED on Pin 7 when detected
  }
  
else

  {
    digitalWrite(OPin,LOW);	// turns off LED on Pin 7 when no movement
  }				// is detected
}
