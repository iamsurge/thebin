#include <iostream>
using namespace std;

struct list {
    string Data; list* Next;
};

void addElement(list** L, string a) {
    list* temp = new list;
    temp->Data = a;
    temp->Next = nullptr;

    list* right = nullptr;
    list* left = nullptr;

    if ((*L) == nullptr) {
        (*L) = temp;
    }
    else {
        right = (*L);
        left = (*L);
        //namirane na poziciya na temp
        while (right != nullptr && temp->Data > right->Data) {
            left = right;
            right = right->Next;
        }
        //postavyane na temp
        temp->Next = right;
        if (right == (*L)) (*L) = temp;
        else left->Next = temp;
    }
}

int main() {
    string s;
    list* L = nullptr;
    list* temp = nullptr;

    for(int i = 0; i < 3; i++){
        cin >> s;
        addElement(&L, s);
    }
    
    temp = L;
    while (temp != nullptr) {
        cout << temp->Data << " ";
        temp = temp->Next;
    }
    return 0;
}
