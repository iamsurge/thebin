#include<bits/stdc++.h>
using namespace std;

int isSubSequence(string s1, string s2) {
    int i = 0, j = 0;
    while(j < s2.size()) {
        while(i < s1.size()) {
            if(s2[j] == s1[i]) {
                i++;
                j++;
                break;
            } else i++;
        }
        if(i==s1.size()) break;
    }
    if(j==s2.size()) return 1;
    return 0;
}

int main() {
    string s1, s2;
    cin >> s1 >> s2;
    cout << isSubSequence(s1, s2);
    return 0;
}
