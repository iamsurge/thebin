int mayorclave(arbolN A, posicion p, unsigned short int G){
    int ,clavehij,clavemax=-1;
    unsigned short int grado=0;
    posicion c;
    
    if(!Nulo(p)){
        c=HijoMasIzq(p,A);
        while(!Nulo(c)){
            grado++;
            clavehij=mayorclave(A,c,G);
            if(clavehij>clavemax)
                clavemax=clavehij;
            c=HermanoDer(c,A);
        }
        if(grado==G || grado ==2*G){
            if(info(p,A)>clavemax)
                return info(p,A);
            else
                return clavemax;
        }
        else
            return clavemax;
    }
    else
        return -1;
}
