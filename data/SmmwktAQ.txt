SELECT
  DATE_TRUNC(DATETIME(paymentDateTime), DAY) paymentYearMonthDay,
  COUNT (DISTINCT userID) usersWithRenewalSubsPaymentCount

FROM
  `uptimerobot-001.ur.payments`

WHERE 
  paymentNewUpgradeRenewal = 3
  AND paymentPeriod IN (1,12)
  AND paymentStatus = 1 
  AND userID IS NOT NULL

GROUP BY
  paymentYearMonthDay

ORDER BY 
  paymentYearMonthDay DESC
