#include <iostream>
#include <vector>
using namespace std;

vector<int>used;
bool found = true;
int n = 5;
	
vector<vector<int>>g {
  {2, 4}, //1
  {1, 3}, //2
  {2, 4}, //3
  {1, 3, 5}, //4
  {4}, //5  
};
 
void dfs(int x){
 
    used[x] = 1;
    
    for(int i = 0;i<g[x].size();++i){
        
        int num = g[x][i];
        
        if(used[num] == 1)found = 1;
        if(used[num] == 0)dfs(num);
        
    } 
    
    used[x] = 2;
}


int main() {
	for(int i = 0;i<n;++i){
        if(!used[i]) dfs(i);
    }
    if(found)cout << "Circle";
    else cout << "No circle";
    
    return 0;
}
