#include <iostream>
#include <math.h>
export module Uravnenie;


export double GetDisc(int a, int b, int c)
{
    return b * b - 4 * a * c;
}

export double x1(int a, int b, int c, double D)
{
    return (-b + sqrt(GetDisc(a, b, c))) / 2 * a;
}

export double x2(int a, int b, int c, double D)
{
    return (-b - sqrt(GetDisc(a, b, c))) / 2 * a;
}
