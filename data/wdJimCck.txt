from openpyxl import Workbook

# creo un libro de excel
libro = Workbook()

# activo el libro al crearle una hoja
hoja = libro.active

# Agrego el nro 42 a la celda A1
hoja['A1'] = 42

# Agreo una lista de valores en las celdas A2,B2 y C2
hoja.append([1, 2, 3])

# Python convierte autom√†ticamente los tipos de datos de frcha y hora
import datetime
hoja['A3'] = datetime.datetime.now()

# Grabo el libro
libro.save("sample.xlsx")
