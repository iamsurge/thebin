package com.corsojava.esercizicorso;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

/**
 *
 * @author lucio
 */
public class StaticCalculatorTest {
    
    @Test
    public void testSomma() {
        assertEquals(13, StaticCalculator.somma(10, 3));
    }

    @Test
    public void testDifferenza() {
        assertEquals(7, StaticCalculator.differenza(10, 3));
    }

    @Test
    public void testDivisioneIntera() {
        assertEquals(3, StaticCalculator.divisioneIntera(10, 3));
    }

    @Test
    public void testDivisione() {
        assertEquals(3.33f, StaticCalculator.divisione(10, 3), 0.01f);
    }

    @Test
    public void testRestoDivisioneIntera() {
        assertEquals(1, StaticCalculator.restoDivisioneIntera(10, 3));
    }

    @Test
    public void testMedia() {
        assertEquals(6.5f, StaticCalculator.media(10, 3), 0.1f);
    }

    @Test
    public void testMax() {
        assertEquals(10, StaticCalculator.max(10, 3));
        assertEquals(10, StaticCalculator.max(3, 10));
    }

    @Test
    public void testMin() {
        assertEquals(3, StaticCalculator.min(10, 3));
        assertEquals(3, StaticCalculator.min(3, 10));
    }
    
}
