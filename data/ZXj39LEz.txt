#include <stdio.h>
#include <stdlib.h>
#define rgbceil 255

int main (int argc, char *argv[])
{
  // Pedir os valores do RGB, Alpha e quantas vezes calcular
  float a;
  int t;
  int rgb[3] = {0,0,0};
  printf("Dite os valores respectivamente: R, G, B, Alpha, Repetições.\n");
  scanf("%i %i %i %f %i",&rgb[0],&rgb[1],&rgb[2],&a,&t);

  printf("  \tR\tG\tB\n");
  // Baseado no valor de t, repetir os cálculos:
  for (size_t i = 0; i < t; i++) {
    for (size_t y = 0; y < 3; y++) { // itera a array do rgb
      // caso um dos elementos seja maior que o teto (255)
      // usando a ideia do negativo para lidar com as idas e retornos do incremento
      // else condição regular (mais frequente)
      rgb[y] = (rgb[y] + a > rgbceil) ? -255 + (rgb[y] + a) - rgbceil: rgb[y] + a;
    }
    printf("%lu)\t%i\t%i\t%i\n",i+1,abs(rgb[0]),abs(rgb[1]),abs(rgb[2]));
  }
  
  return 0;
}
