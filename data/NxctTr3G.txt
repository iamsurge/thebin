using System;

namespace MyApp
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int hourOfExam = int.Parse(Console.ReadLine());
            int minuteOfExam = int.Parse(Console.ReadLine());
            int hourOfArrival = int.Parse(Console.ReadLine());
            int minuteOfArrival = int.Parse(Console.ReadLine());

            int timeDifference = (hourOfArrival - hourOfExam) * 60 + minuteOfArrival - minuteOfExam;

            if (timeDifference > 0)
                Console.WriteLine("Late");
            else if (-30 <= timeDifference && timeDifference <= 0)
                Console.WriteLine("On time");
            else if (timeDifference < -30)
                Console.WriteLine("Early");

            if (timeDifference != 0)
            {
                if (Math.Abs(timeDifference) < 60 && Math.Abs(timeDifference) != 60)
                    Console.WriteLine($"{Math.Abs(timeDifference)} minutes " + (timeDifference < 0 ? "before" : "after") + " the start");
                else
                    Console.WriteLine($"{Math.Abs(timeDifference / 60)}:{(Math.Abs(timeDifference % 60)).ToString("00")} " + (Math.Abs(timeDifference) >= 60 ? "hours " : "") + (timeDifference < 0 ? "before" : "after") + " the start");
            }
        }
    }
}
