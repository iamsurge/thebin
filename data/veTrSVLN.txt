using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Practice
{
    class Vocabuary
    {
        private SortedList<string, string> voc;
        public Vocabuary()
        {
            voc = new SortedList<string, string>()
            {
                ["peace"] = "мир",
                ["sun"] = "сонце",
                ["fish"] = "риба",
                ["carpet"] = "килим",
                ["bed"] = "ліжко"

            };
        }

        public void Print()
        {
            foreach (var pair in voc)
            {
                Console.WriteLine($"{pair, 10}");
            }

            Console.WriteLine("\n\n");
        }

        public void AddWord (string eng, string uk)
        {
            if(voc.TryAdd(eng, uk))
            {
                Console.WriteLine("________SUCCESSFULLY_ADDED________");
            }
            else
            {
                Console.WriteLine("__________ERROR___________");
            }
        }

        public void CorrectUkTranslation (string eng)
        {
            voc[eng] = Console.ReadLine();
        }
        public void CorrectEngTranslation(string eng, string neweng)
        {
            if(voc.ContainsKey(eng))
            {
                string uk = voc[eng];

                voc.Remove(eng);

                voc.Add(neweng, uk);
            }
        }

        public void DeleteUkWord(string uk)
        {
            int idx = voc.IndexOfValue(uk);

            if(idx >= 0)
            {
                voc.RemoveAt(idx);
            }
        }

        public void DeleteEngWord(string eng)
        {
            int idx = voc.IndexOfValue(eng);

            if (idx >= 0)
            {
                voc.RemoveAt(idx);
            }
        }
    }
}
