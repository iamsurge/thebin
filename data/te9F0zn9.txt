turtle.up()
turtle.turnRight()

local function deployTurtle()
    for i = 2, 16 do
        turtle.forward()
        turtle.turnLeft()
        turtle.select(i)
        turtle.placeDown()
        turtle.turnRight()
    end
end

local function goBack()
    for i = 1, 15 do
        turtle.back()
    end
    turtle.down()
    turtle.turnLeft()
end

deployTurtle()
goBack()
