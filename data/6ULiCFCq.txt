
import java.io.*;
import java.util.StringTokenizer;

public class Gyvunas {
    private String pavadinimas;
    private int amzius;
    private double svoris;
    private String k_salis;

    Gyvunas() {}
    public Gyvunas(String pav, int amzius, double svoris, String salis) {
        pavadinimas = pav;
        this.amzius = amzius;
        this.svoris = svoris;
        k_salis = salis;
    }

    public String gautiPav () {
        return pavadinimas;
    }

    public int gautiAmziu () {
        return amzius;
    }

    public double gautiSvori () {
        return svoris;
    }

    public String gautiSali () {
        return k_salis;
    }
}
