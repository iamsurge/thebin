using System;
using System.Text;

namespace Data_Type_Finder
{
	class DataTypeFinder
	{
		static void Main(string[] args)
		{
			const string ConstInteger = "integer";
			const string ConstFloat = "floating point";
			const string ConstChar = "character";
			const string ConstBool = "boolean";
			const string ConstString = "string";

			bool isExitLoop = false;

			double floatValue;
			int integerValue;
			bool booleanValue;
			char charValue;

			while (!isExitLoop)
			{
				string input = Console.ReadLine();
				StringBuilder sb1 = new StringBuilder();

				if (input == "END")
				{
					isExitLoop = true;
				}
				else
				{
					try
					{
						integerValue = Convert.ToInt32(input);
						sb1.Append($"{input} is {ConstInteger} type");
					}
					catch (Exception)
					{
						try
						{
							floatValue = Convert.ToDouble(input);
							sb1.Append($"{input} is {ConstFloat} type");

						}
						catch (Exception)
						{
							try
							{
								booleanValue = Convert.ToBoolean(input);
								sb1.Append($"{input} is {ConstBool} type");
							}
							catch (Exception)
							{
								try
								{
									charValue = Convert.ToChar(input);
									sb1.Append($"{input} is {ConstChar} type");
								}
								catch (Exception)
								{
									sb1.Append($"{input} is {ConstString} type");
								}
							}
						}
					}

					Console.WriteLine(sb1.ToString());
				}
			}
		}
	}
}
//You will receive an input until you receive "END". Find what data type is the input. Possible data types are:
//•	Integer 
//•	Floating point 
//•	Characters
//•	Boolean
//•	Strings

//•         Знаци
