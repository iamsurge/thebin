local chat = peripheral.find("chatBox")
local playerD = peripheral.find("playerDetector")
local ar = peripheral.find("arController")

ar.clear()
while(true) do
	local x = 20
	local allPlayers = playerD.getPlayersInRange(200)

	for k,loopPlayer in pairs(allPlayers) do
		local pos = playerD.getPlayerPos(loopPlayer)
		local print =  " y: " .. pos.y .. " z: " .. pos.z
		
		local part = loopPlayer
		local id = loopPlayer .. 1
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 20, x, 0xCC4C4C)

		local part = "'s position is "
		local id = loopPlayer .. 2
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 40, x, 0xCCff4C)

		local part = "x: "
		local id = loopPlayer .. 3
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 60, x, 0xCCffff)

		local part = string(pos.x)
		local id = loopPlayer .. 4
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 65, x, 0xffffff)

		local part = "y: "
		local id = loopPlayer .. 5
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 80, x, 0xCCffff)

		local part = string(pos.y)
		local id = loopPlayer .. 6
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 85, x, 0xffffff)

		local part = "z: "
		local id = loopPlayer .. 7
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 100, x, 0xCCffff)

		local part = string(pos.y)
		local id = loopPlayer .. 8
		ar.clearElement(id)
		ar.drawStringWithId(id, part, 105, x, 0xffffff)

		x = x + 10
	end
end
