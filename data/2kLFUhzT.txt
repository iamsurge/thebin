function sue_cok(peruntukan) {
    const point_count = 10;
    let prefix;
    switch (peruntukan) {
        case "matkul":
            prefix = "MK";
        break;
        case "dosen":
            prefix = "DO";
        break;
        default:
            throw new Error(`peruntukan="${peruntukan}" tidak valid!\n    Yang dapat diterima: "matkul" dan "dosen"`);
    }
    for (var i = 1; i <= point_count; i++) {
        const bullet_target = document.getElementById(
            `${prefix}${i}3`
        );
        if (!bullet_target) {
            throw new Error("Galat! Pastikan berada di halaman kuisioner " + peruntukan);
        }
        bullet_target.click();
    }
    document.getElementById("txtKomentar").value = " NULL";
    document.getElementById("chkPermanent").click();
    document.querySelector(`input[name="button"]`).click();
}

// Di halaman kuisioner mata kuliah
sue_cok("matkul")

// Di halaman kuisioner dosen
sue_cok("dosen")
