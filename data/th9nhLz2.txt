#include <iostream>
#include <vector>

using namespace std;

// https://informatics.msk.ru/mod/statements/view.php?id=813&chapterid=3089#1

int dp[101][10001], w[101], p[101];
vector<int> result;

void findAns(int k, int s){
    if(dp[k][s] == 0){
        return;
    }
    if(dp[k][s] == dp[k - 1][s]){
        findAns(k - 1, s);
    }
    else{
        findAns(k - 1, s - w[k]);
        result.push_back(k);
    }
}

 int main()
 {
     int n, m;
     cin>> n >> m;

     for(int i = 0; i <= n; i++){
        dp[i][0] = 0;
     }
     for(int i = 0; i <= m; i++){
        dp[0][i] = 0;
     }

     for(int i = 1; i <= n; i++){
        cin >> w[i];
     }

     for(int i = 1; i <= n; i++){
        cin >> p[i];
     }

     for(int k = 1; k <= n; k++){
        for(int s = 1; s <= m; s++){
            if(s >= w[k]){
                dp[k][s] = max(dp[k - 1][s], dp[k - 1][s - w[k]] + p[k]);
            }
            else{
                dp[k][s] = dp[k - 1][s];
            }
        }
     }

     findAns(n, m);
     for(int i = 0; i < result.size(); i++){
        cout << result[i] << '\n';
     }
 }
