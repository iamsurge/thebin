#include <stdio.h>
#define MAXSIZE 10
int stack[MAXSIZE] , top = -1;

void push();
void pop();
void display();

void main(){
	int n , i , tmp; 
	scanf("%d" , &n);
	for(i = 0; i < n; i++){
		push();
	}
	display();
}

void push(){
	int n; 
	if(top == MAXSIZE){
		printf("\nStack Overflow\n");
	}
	else{
		scanf("%d" , &n);
		top++;
		stack[top] = n;
	}
}
void display(){
	int i;
	if(top == -1){
		printf("\nStack Underflow\n");
	}
	else{
		printf("\n");
		for(i = 0; i <= top; i++){
			printf("%d " , stack[i]);
		}
		printf("\n");
	}
}
