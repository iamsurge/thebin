
var arr = [100,25,6,71,-83,4,7];
var len = arr.length;
var i;
var first_max = arr[0];
var second_max = '';
for (i = 0; i < len; i++){
	if(arr[i] > first_max){
		second_max = first_max;
		first_max = arr[i];
	}else if(arr[i] < first_max && arr[i] > second_max){
		second_max = arr[i];
	};
};

alert(first_max +'\r\n'+ second_max);
