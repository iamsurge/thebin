#include <iostream>
#include <vector>
using namespace std;

vector<int> simple = {};

void write_number (int n)
{
    cout << " " << n;
}

void sieve(int n)
{
    vector<bool> is_prime(n+1, true);
    for (int i = 2; i<=n; i++)
    {
        if (is_prime[i])
        {
            simple.push_back(i);
            for (int j = i; j <= n/i; j++)
            {
                is_prime[i*j] = false;
            }
        }
    }
}



int get_input()
{
    int n;
    cout << "This program lists all primes up to a given integer.\n"
         << "Enter an integer: \n";
    cin >> n;
    return n;
}

int main() {
    int n = get_input();
    if (n < 2)
    {
        cout << "There is no prime numbers less then 2.\n";
    }
    else
    {
        cout << "The prime numbers up to " << n << " are: ";
        sieve(n);
        for (auto x: simple) {
            cout<<x<<" ";
        }
        cout << ".\n";
    }
    return 0;
}
