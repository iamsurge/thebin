/*
C++ program that takes a n number and ask n times for numbers and outputs the even/odd numbers with their sum and average

nmbr = Number
qstn = Total Number to be asked to user
ctr = Counter for loop
smvn/smdd = Sum of even/ Sum of odd
ctrvn/ctrdd = Counter of even/ Counter of odd
vnvrg/ddvrg = Even average/ Odd average
strvn/strdd = Store even/ Store odd
*/

#include <iostream>

using namespace std;

int main(){
    int nmbr, qstn, ctr = 0, smvn = 0, smdd = 0; //Declaration of all variables
    int ctrvn = 0, ctrdd = 0;
    float vnvrg = 0, ddvrg = 0;

    cout << "Even/Odd Sorter\n";
    cout << "------------------------\n";
    printf("How many numbers? ");
    cin >> qstn;

    int strvn[qstn]; //Declaration of array for storing even/odd numbers
    int strdd[qstn];

    while (true){
        if (ctr == qstn) //Stops the loop when the ctr is same as qstn
            break;
        printf("Enter number: ");
        cin >> nmbr;

        if (nmbr % 2 == 0){ //Checks nmbr if even or odd
            smvn += nmbr;
            strvn[ctrvn] = nmbr;
            ctrvn++;
        }else{
            smdd += nmbr;
            strdd[ctrdd] = nmbr;
            ctrdd++;
        }
        ctr++;
    }

    vnvrg = float(smvn) / float(ctrvn); //Calculates the average
    ddvrg = float(smdd) / float(ctrdd);

    cout << "------------------------\n";
    cout << "Even numbers: ";
    for(int vn = 0; vn < ctrvn; vn++){ //Outputs the even numbers
        cout << strvn[vn] << " ";
    }
    cout << "\nSum: " << smvn << endl;
    cout << "Average: " << vnvrg << endl;

    cout << "------------------------\n";
    cout << "Odd numbers: ";
    for(int dd = 0; dd < ctrdd; dd++){ //Outputs the odd numbers
        cout << strdd[dd] << " ";
    }
    cout << "\nSum: " << smdd << endl;
    cout << "Average: " << ddvrg << endl;
}
