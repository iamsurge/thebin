local function tDigDown()
    turtle.digDown()
    turtle.down()
    turtle.digDown()
end

local function tPlace()
    turtle.select(1)
    turtle.placeDown()
    turtle.up()
    turtle.placeDown()
end

local function detectPlace()
    while turtle.detect() == false do
        local success, data = turtle.inspectDown()
            if success then
                if data.name ~= "minecraft:grass_block" then
                    tDigDown()
                    tPlace()
                end
            end
                turtle.forward()
    end
end

local args = {...}
    times = args[1]
        for i = 1, times do
            detectPlace()
end
