int main()
{   float a=0,n,m=0,qn,qp,pn=0,pp,cm=0;//em ordem, número do usuário, média, quantidade negativos, quantidade positivos, percentual negativo,percentual positivo.
    char b;
    //Repetição
    do{
        //Obtendo o número 
        printf("informe seu número\n");
        scanf  ("%f",&n);
        b=n;
        //Definindo contador para média
        if(n>=0){cm=cm+1;}else {cm=cm+1;}
       
        //Definindo se o número é positivo ou negativo
        if(n>=0){qp=qp+1;}else{qn=qn+1;}
   
    //Soma                    
    a=a+n;
    
    //Média
    m=a/cm;
        // definindo porcentagem de positivos e negativos
    pp=qp*100/cm;
    pn=qn*100/cm;
    printf ("A quantidade de números informados são: ""%.2f\n",cm);
    printf ("A média atual é: ""%.2f\n",m);
    printf ("A quantidade de números positivos é atual é: ""%.2f\n",qp);
    printf ("A quantidade de números positivos é atual é: ""%.2f\n",qn);
    printf ("A porcentagem de números positivos é atual é: ""%.2f\n",pp);
    printf ("A porcentagem de números negativos é atual é: ""%.2f\n",pn);
    }while(b!='p');
    return 0;
}
