Program Task2(output);
    var odd, even, val: longint;
begin
    odd := 0;
    even := 0;
 
    readln(val);

    while val > 0 do
        begin
            if val mod 2 = 0 then
                even := even + 1
            else
                odd := odd + 1;
            val := val div 10;
        end;
    
    writeln(even);
    writeln(odd)
end.
